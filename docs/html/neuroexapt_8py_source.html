278 def _analyze_entropy(
279 self,
280 dataloader: torch.utils.data.DataLoader,
281 num_batches: Optional[int] = None
282 ) -> Dict[str, float]:
283 """Analyze entropy metrics."""
284 entropy_values = []
285 layer_entropies = {}
286
287 self.model.eval()
288 with torch.no_grad():
289 for batch_idx, (inputs, _) in enumerate(dataloader):
290 if num_batches and batch_idx >= num_batches:
291 break
292
293 inputs = inputs.to(self.device)
294 outputs = self.model(inputs)
295
296 # Measure entropy
297 entropy = self.entropy_ctrl.measure(outputs)
298 entropy_values.append(entropy)
299
300 # Get layer entropies
301 for name, value in self.entropy_ctrl.get_layer_entropies().items():
302 if name not in layer_entropies:
303 layer_entropies[name] = []
304 layer_entropies[name].append(value)
305
306 # Average metrics
307 metrics = {
308 'average_entropy': np.mean(entropy_values),
309 'entropy_std': np.std(entropy_values),
310 'layer_entropies': {
311 name: np.mean(values)
312 for name, values in layer_entropies.items()
313 }
314 }
315
316 return metrics

318 def evolve_structure(
319 self,
320 performance_metrics: Dict[str, float],
321 force_action: Optional[str] = None
322 ) -> Tuple[bool, Dict[str, Any]]:
323 """
324 Evolve model structure based on current metrics.
325
326 Args:
327 performance_metrics: Current performance metrics
328 force_action: Force specific action (prune/expand/mutate)
329
330 Returns:
331 Tuple of (structure_changed, evolution_info)
332 """
333 if self.model is None:
334 raise ValueError("No model wrapped. Call wrap_model() first.")
335
336 # Get current metrics
337 current_entropy = self.entropy_ctrl.entropy_history[-1] if self.entropy_ctrl.entropy_history else 0.5
338 threshold = self.entropy_ctrl.threshold
339
340 # Prepare metrics for evolution
341 info_metrics = {
342 'layer_importances': self.info_theory._activation_cache.get('layer_importances', {}),
343 'layer_entropies': self.entropy_ctrl.get_layer_entropies()
344 }
345
346 entropy_metrics = {
347 'current_entropy': current_entropy,
348 'threshold': threshold,
349 'layer_entropies': info_metrics['layer_entropies']
350 }
351
352 # Determine action
353 if force_action:
354 action = force_action
355 else:
356 if self.entropy_ctrl.should_prune(current_entropy):
357 action = 'prune'
358 elif self.entropy_ctrl.should_expand(current_entropy):
359 action = 'expand'
360 elif self.current_epoch % 10 == 0: # Periodic mutation
361 action = 'mutate'
362 else:
363 action = 'none'
364
365 # Apply evolution
366 if action != 'none':
367 # Use structural evolution
368 evolved_model, evolution_step = self.struct_optimizer.evolve_step(
369 self.model,
370 info_metrics,
371 entropy_metrics,
372 performance_metrics
373 )
374
375 # Update model
376 self.model = evolved_model
377 self.wrapped_model.model = evolved_model
378
379 # Re-register hooks
380 self.info_theory.clear_hooks()
381 self.info_theory.register_hooks(evolved_model)
382
383 # Track evolution
384 self.evolution_history.append(evolution_step)
385
386 if self.verbose:
387 self.logger.info(
388 f"Evolved structure: {action} - "
389 f"Parameters: {evolution_step.parameters_before} -> {evolution_step.parameters_after}"
390 )
391
392 return True, evolution_step.__dict__
393
394 return False, {'action': 'none'}
