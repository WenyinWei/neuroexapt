20 Args:

23 log_file: Optional file to log to
24
25 Returns:
26 Configured logger

31 # Avoid duplicate handlers

33 return logger

36 console_handler = logging.StreamHandler()

38
39 # Format
40 formatter = logging.Formatter(

43 )
44 console_handler.setFormatter(formatter)
45 logger.addHandler(console_handler)

47 # File handler if specified
48 if log_file:
49 os.makedirs(os.path.dirname(log_file), exist_ok=True)
50 file_handler = logging.FileHandler(log_file)
51 file_handler.setLevel(level)
52 file_handler.setFormatter(formatter)
53 logger.addHandler(file_handler)
54
