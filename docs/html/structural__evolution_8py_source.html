102 def compute_structural_entropy(self, model: nn.Module) -> float:
103 """
104 Compute structural entropy of the network.
105
106 Implements: S = -Σ p_i * log(p_i)
107 where p_i is the normalized importance of layer i.
108
109 Args:
110 model: Neural network model
111
112 Returns:
113 Structural entropy value
114 """
115 if not self.layer_importance:
116 return 0.0
117
118 # Normalize importance values to probabilities
119 total_importance = sum(self.layer_importance.values())
120 if total_importance == 0:
121 return 0.0
122
123 probabilities = [imp / total_importance for imp in self.layer_importance.values()]
124
125 # Calculate entropy
126 entropy = 0.0
127 for p in probabilities:
128 if p > 0:
129 entropy -= p * np.log(p)
130
131 return entropy

133 def calculate_entropy_gradient(
134 self,
135 current_entropy: float,
136 target_entropy: float,
137 layer_importances: Dict[str, float]
138 ) -> Dict[str, float]:
139 """
140 Calculate entropy gradient for each layer.
141
142 Implements: ∂S/∂t = -α * I(L_i;O) + β * KL(p_old||p_new)
143
144 Args:
145 current_entropy: Current structural entropy
146 target_entropy: Target structural entropy
147 layer_importances: Layer importance values
148
149 Returns:
150 Entropy gradients for each layer
151 """
152 gradients = {}
153
154 for layer_name, importance in layer_importances.items():
155 # Information retention term
156 retention_term = -self.alpha * importance
157
158 # Structure variation term (simplified KL divergence)
159 old_p = self.layer_importance.get(layer_name, 0.5)
160 new_p = importance
161 kl_term = self.beta * (new_p * np.log((new_p + 1e-10) / (old_p + 1e-10)))
162
163 # Combined gradient
164 gradients[layer_name] = retention_term + kl_term
165
166 return gradients

174 """
175 Prune layers based on entropy threshold.
176
177 Args:
178 model: Model to prune
179 layer_entropies: Entropy values for each layer
180 threshold: Entropy threshold for pruning
181
182 Returns:
183 Pruned model and list of pruned layer names
184 """
185 # Sort layers by entropy (ascending)
186 sorted_layers = sorted(layer_entropies.items(), key=lambda x: x[1])
187
188 # Determine layers to prune

190 current_layer_count = len(self.layer_registry)
191 max_prune_count = int(current_layer_count * self.prune_ratio)
192
193 for layer_name, entropy in sorted_layers:
194 if entropy < threshold and len(layers_to_prune) < max_prune_count:
195 # Check if we maintain minimum layers
196 if current_layer_count - len(layers_to_prune) > self.min_layers:
197 layers_to_prune.append(layer_name)
198
199 if not layers_to_prune:
200 return model, []
201
202 # Create pruned model
203 pruned_model = self._remove_layers(model, layers_to_prune)

207 def expand_with_mi(
208 self,
209 model: nn.Module,
210 layer_importances: Dict[str, float],
211 gamma: float = 0.1
212 ) -> Tuple[nn.Module, List[str]]:
213 """
214 Expand network guided by mutual information.
215
216 Args:
217 model: Model to expand
218 layer_importances: Importance values for each layer
219 gamma: Expansion threshold factor
220
221 Returns:
222 Expanded model and list of new layer names
223 """
224 # Sort layers by importance (descending)
225 sorted_layers = sorted(layer_importances.items(), key=lambda x: x[1], reverse=True)
226
227 # Determine expansion points
228 expansion_points = []
229 current_layer_count = len(self.layer_registry)
230 max_expand_count = int(current_layer_count * self.expand_ratio)
231
232 # Find high-importance layers as expansion candidates
233 avg_importance = np.mean(list(layer_importances.values()))
234 expansion_threshold = avg_importance * (1 + gamma)
235
236 for layer_name, importance in sorted_layers:
237 if importance > expansion_threshold and len(expansion_points) < max_expand_count:
238 if current_layer_count + len(expansion_points) < self.max_layers:
239 expansion_points.append(layer_name)
240
241 if not expansion_points:
242 return model, []
243
244 # Create expanded model
245 expanded_model, new_layers = self._add_layers(model, expansion_points)
246
247 return expanded_model, new_layers

289 def evolve_step(
290 self,

293 entropy_metrics: Dict[str, Any],
294 performance_metrics: Dict[str, float]

296 """
297 Perform one evolution step based on current metrics.

300 model: Current model

302 entropy_metrics: Entropy-related metrics
303 performance_metrics: Performance metrics (accuracy, loss, etc.)
304
305 Returns:
306 Evolved model and evolution step record

308 self.current_epoch += 1
309
310 # Extract relevant metrics
311 layer_importances = info_theory_metrics.get('layer_importances', {})
312 layer_entropies = entropy_metrics.get('layer_entropies', {})
313 current_entropy = entropy_metrics.get('current_entropy', 0.0)

316 # Update internal state
317 self.layer_importance = layer_importances
318 self.layer_entropy = layer_entropies
319
320 # Count parameters before evolution
321 params_before = sum(p.numel() for p in model.parameters() if p.requires_grad)
322
323 # Decide evolution action
324 action = self._decide_action(current_entropy, threshold, performance_metrics)
325
326 # Execute evolution
327 if action == 'prune':
328 evolved_model, affected_layers = self.prune_by_entropy(
329 model, layer_entropies, threshold
330 )
331 elif action == 'expand':
332 evolved_model, affected_layers = self.expand_with_mi(
333 model, layer_importances
334 )
335 elif action == 'mutate':
336 evolved_model = self.mutate_discrete_parameters(model)
337 affected_layers = ['discrete_parameters']
338 else:
339 evolved_model = model
340 affected_layers = []
341
342 # Count parameters after evolution
343 params_after = sum(p.numel() for p in evolved_model.parameters() if p.requires_grad)
344
345 # Calculate information gain
346 info_gain = self._calculate_info_gain(model, evolved_model, layer_importances)

349 step = EvolutionStep(
350 epoch=self.current_epoch,
351 action=action,
352 target_layers=affected_layers,
353 metrics_before=performance_metrics.copy(),
354 metrics_after=performance_metrics.copy(), # Will be updated after training
355 info_gain=info_gain,

364 def _decide_action(
365 self,
366 current_entropy: float,
367 threshold: float,
368 performance_metrics: Dict[str, float]
369 ) -> str:
370 """Decide which evolution action to take."""
371 # Check performance degradation
372 if len(self.evolution_history) > 0:
373 last_step = self.evolution_history[-1]
374 perf_drop = performance_metrics.get('accuracy', 0) - last_step.metrics_before.get('accuracy', 0)
375
376 if perf_drop < -0.05: # Significant performance drop
377 return 'expand' # Recover by expanding
378
379 # Entropy-based decision
380 if current_entropy < threshold * 0.8:
381 return 'prune'
382 elif current_entropy > threshold * 1.5:
383 return 'expand'
384 elif self.current_epoch % 10 == 0: # Periodic mutation
385 return 'mutate'
386 else:
387 return 'none'
