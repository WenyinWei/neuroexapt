42class PruneByEntropy(StructuralOperator):
43 """
44 Prune layers based on entropy threshold.
45
46 Low entropy indicates redundant or uninformative layers.
47 """
48
49 def __init__(
50 self,
51 threshold: float = 0.3,
52 min_layers_to_keep: int = 3,
53 prune_ratio: float = 0.1,
54 layer_types: Optional[List[type]] = None
55 ):
56 """
57 Initialize entropy-based pruning operator.
58
59 Args:
60 threshold: Entropy threshold below which layers are pruned
61 min_layers_to_keep: Minimum number of layers to retain
62 prune_ratio: Maximum ratio of layers to prune in one step
63 layer_types: Types of layers eligible for pruning
64 """
65 self.threshold = threshold
66 self.min_layers_to_keep = min_layers_to_keep
67 self.prune_ratio = prune_ratio
68 self.layer_types = layer_types or [nn.Conv2d, nn.Linear, nn.Conv1d, nn.Conv3d]
69
70 def apply(self, model: nn.Module, metrics: Dict[str, Any]) -> Tuple[nn.Module, Dict[str, Any]]:
71 """Apply entropy-based pruning."""
72 layer_entropies = metrics.get('layer_entropies', {})
73 if not layer_entropies:
74 return model, {'pruned_layers': [], 'message': 'No entropy metrics available'}
75
76 # Get eligible layers
77 eligible_layers = self._get_eligible_layers(model)
78
79 # Sort by entropy (ascending)
80 entropy_items = [(name, layer_entropies.get(name, float('inf')))
81 for name in eligible_layers]
82 entropy_items.sort(key=lambda x: x[1])
83
84 # Determine layers to prune
85 max_prune = int(len(eligible_layers) * self.prune_ratio)
86 layers_to_prune = []
87
88 for layer_name, entropy in entropy_items:
89 if entropy < self.threshold and len(layers_to_prune) < max_prune:
90 if len(eligible_layers) - len(layers_to_prune) > self.min_layers_to_keep:
91 layers_to_prune.append(layer_name)
92
93 if not layers_to_prune:
94 return model, {'pruned_layers': [], 'message': 'No layers meet pruning criteria'}
95
96 # Create pruned model
97 pruned_model = self._prune_layers(model, layers_to_prune)
98
99 info = {
100 'pruned_layers': layers_to_prune,
101 'num_pruned': len(layers_to_prune),
102 'threshold_used': self.threshold,
103 'lowest_entropy': entropy_items[0][1] if entropy_items else None
104 }
105
106 return pruned_model, info
107
108 def can_apply(self, model: nn.Module, metrics: Dict[str, Any]) -> bool:
109 """Check if pruning can be applied."""
110 layer_entropies = metrics.get('layer_entropies', {})
111 if not layer_entropies:
112 return False
113
114 eligible_layers = self._get_eligible_layers(model)
115 if len(eligible_layers) <= self.min_layers_to_keep:
116 return False
117
118 # Check if any layer has entropy below threshold
119 for name in eligible_layers:
120 if name in layer_entropies and layer_entropies[name] < self.threshold:
121 return True
122
123 return False
124
125 def _get_eligible_layers(self, model: nn.Module) -> List[str]:
126 """Get list of layers eligible for pruning."""
127 eligible = []
128
129 for name, module in model.named_modules():
130 if any(isinstance(module, layer_type) for layer_type in self.layer_types):
131 if len(list(module.parameters())) > 0:
132 eligible.append(name)
133
134 return eligible
135
136 def _prune_layers(self, model: nn.Module, layers_to_prune: List[str]) -> nn.Module:
137 """Replace specified layers with identity mapping."""
138 pruned_model = copy.deepcopy(model)
139
140 for layer_name in layers_to_prune:
141 parts = layer_name.split('.')
142 parent = pruned_model
143
144 # Navigate to parent module
145 for part in parts[:-1]:
146 parent = getattr(parent, part)
147
148 # Replace with identity
149 setattr(parent, parts[-1], nn.Identity())
150
151 return pruned_model

154class ExpandWithMI(StructuralOperator):
155 """
156 Expand network capacity guided by mutual information.
157
158 High mutual information indicates information bottlenecks.
159 """
160
161 def __init__(
162 self,
163 gamma: float = 0.1,
164 max_layers_to_add: int = 5,
165 expand_ratio: float = 0.1,
166 expansion_factor: float = 1.5
167 ):
168 """
169 Initialize MI-guided expansion operator.
170
171 Args:
172 gamma: Threshold factor for determining expansion points
173 max_layers_to_add: Maximum layers to add in one step
174 expand_ratio: Maximum ratio of layers to expand
175 expansion_factor: Factor for new layer dimensions
176 """
177 self.gamma = gamma
178 self.max_layers_to_add = max_layers_to_add
179 self.expand_ratio = expand_ratio
180 self.expansion_factor = expansion_factor
181
182 def apply(self, model: nn.Module, metrics: Dict[str, Any]) -> Tuple[nn.Module, Dict[str, Any]]:
183 """Apply MI-guided expansion."""
184 layer_importances = metrics.get('layer_importances', {})
185 if not layer_importances:
186 return model, {'expanded_layers': [], 'message': 'No importance metrics available'}
187
188 # Calculate expansion threshold
189 importances = list(layer_importances.values())
190 if not importances:
191 return model, {'expanded_layers': [], 'message': 'No layer importances found'}
192
193 avg_importance = np.mean(importances)
194 threshold = avg_importance * (1 + self.gamma)
195
196 # Find expansion candidates
197 candidates = []
198 for name, importance in layer_importances.items():
199 if importance > threshold:
200 candidates.append((name, importance))
201
202 # Sort by importance (descending)
203 candidates.sort(key=lambda x: x[1], reverse=True)
204
205 # Select layers to expand
206 max_expand = min(
207 self.max_layers_to_add,
208 int(len(layer_importances) * self.expand_ratio)
209 )
210
211 layers_to_expand = [name for name, _ in candidates[:max_expand]]
212
213 if not layers_to_expand:
214 return model, {'expanded_layers': [], 'message': 'No layers meet expansion criteria'}
215
216 # Create expanded model
217 expanded_model = self._expand_layers(model, layers_to_expand)
218
219 info = {
220 'expanded_layers': layers_to_expand,
221 'num_expanded': len(layers_to_expand),
222 'threshold_used': threshold,
223 'highest_importance': candidates[0][1] if candidates else None
224 }
225
226 return expanded_model, info
227
228 def can_apply(self, model: nn.Module, metrics: Dict[str, Any]) -> bool:
229 """Check if expansion can be applied."""
230 layer_importances = metrics.get('layer_importances', {})
231 if not layer_importances:
232 return False
233
234 importances = list(layer_importances.values())
235 if not importances:
236 return False
237
238 avg_importance = np.mean(importances)
239 threshold = avg_importance * (1 + self.gamma)
240
241 # Check if any layer exceeds threshold
242 return any(imp > threshold for imp in importances)
243
244 def _expand_layers(self, model: nn.Module, layers_to_expand: List[str]) -> nn.Module:
245 """Add new layers at specified expansion points."""
246 expanded_model = copy.deepcopy(model)
247
248 for layer_name in layers_to_expand:
249 parts = layer_name.split('.')
250 parent = expanded_model
251
252 # Navigate to target module
253 for part in parts[:-1]:
254 parent = getattr(parent, part)
255
256 original_layer = getattr(parent, parts[-1])
257
258 # Create expanded block based on layer type
259 if isinstance(original_layer, nn.Linear):
260 expanded_block = self._create_linear_expansion(original_layer)
261 elif isinstance(original_layer, nn.Conv2d):
262 expanded_block = self._create_conv2d_expansion(original_layer)
263 else:
264 # Skip unsupported layer types
265 continue
266
267 # Replace with expanded block
268 setattr(parent, parts[-1], expanded_block)
269
270 return expanded_model
271
272 def _create_linear_expansion(self, layer: nn.Linear) -> nn.Module:
273 """Create expanded linear block."""
274 in_features = layer.in_features
275 out_features = layer.out_features
276 hidden_features = int(out_features * self.expansion_factor)
277
278 return nn.Sequential(
279 layer,
280 nn.ReLU(inplace=True),
281 nn.Linear(out_features, hidden_features),
282 nn.ReLU(inplace=True),
283 nn.Linear(hidden_features, out_features)
284 )
285
286 def _create_conv2d_expansion(self, layer: nn.Conv2d) -> nn.Module:
287 """Create expanded conv2d block."""
288 in_channels = layer.in_channels
289 out_channels = layer.out_channels
290 hidden_channels = int(out_channels * self.expansion_factor)
291
292 return nn.Sequential(
293 layer,
294 nn.ReLU(inplace=True),
295 nn.Conv2d(
296 out_channels, hidden_channels,
297 kernel_size=3, padding=1, stride=1
298 ),
299 nn.BatchNorm2d(hidden_channels),
300 nn.ReLU(inplace=True),
301 nn.Conv2d(
302 hidden_channels, out_channels,
303 kernel_size=1, stride=1
304 )
305 )

308class MutateDiscrete(StructuralOperator):
309 """
310 Mutate discrete parameters using continuous relaxation.
311
312 Implements differentiable discrete parameter optimization.
313 """
314
315 def __init__(
316 self,
317 mutation_rate: float = 0.1,
318 temperature: float = 1.0,
319 parameter_ranges: Optional[Dict[str, Tuple[int, int]]] = None
320 ):
321 """
322 Initialize discrete parameter mutation operator.
323
324 Args:
325 mutation_rate: Probability of mutating each parameter
326 temperature: Temperature for Gumbel-softmax relaxation
327 parameter_ranges: Ranges for discrete parameters
328 """
329 self.mutation_rate = mutation_rate
330 self.temperature = temperature
331 self.parameter_ranges = parameter_ranges or {
332 'kernel_size': (1, 7),
333 'stride': (1, 3),
334 'dilation': (1, 3),
335 'groups': (1, 16),
336 'num_heads': (1, 16), # For attention layers
337 'hidden_ratio': (1, 4) # For MLP expansion
338 }
339
340 def apply(self, model: nn.Module, metrics: Dict[str, Any]) -> Tuple[nn.Module, Dict[str, Any]]:
341 """Apply discrete parameter mutation."""
342 mutated_model = copy.deepcopy(model)
343 mutations = []
344
345 for name, module in mutated_model.named_modules():
346 if np.random.random() < self.mutation_rate:
347 mutation_info = self._mutate_module(name, module)
348 if mutation_info:
349 mutations.append(mutation_info)
350
351 info = {
352 'mutations': mutations,
353 'num_mutations': len(mutations),
354 'mutation_rate_used': self.mutation_rate
355 }
356
357 return mutated_model, info
358
359 def can_apply(self, model: nn.Module, metrics: Dict[str, Any]) -> bool:
360 """Mutation can always be applied."""
361 return True
362
363 def _mutate_module(self, name: str, module: nn.Module) -> Optional[Dict[str, Any]]:
364 """Mutate a single module's discrete parameters."""
365 mutation_info = {'layer': name, 'type': type(module).__name__, 'changes': {}}
366
367 if isinstance(module, (nn.Conv1d, nn.Conv2d, nn.Conv3d)):
368 # Mutate convolution parameters
369 changes = self._mutate_conv_params(module)
370 if changes:
371 mutation_info['changes'] = changes
372 return mutation_info
373
374 elif isinstance(module, nn.MultiheadAttention):
375 # Mutate attention parameters
376 changes = self._mutate_attention_params(module)
377 if changes:
378 mutation_info['changes'] = changes
379 return mutation_info
380
381 return None
382
383 def _mutate_conv_params(self, conv: nn.Module) -> Dict[str, Tuple[Any, Any]]:
384 """Mutate convolutional layer parameters."""
385 changes = {}
386
387 # Note: Actual mutation of kernel size, stride, etc. requires
388 # reconstructing the layer, which is complex. This is a simplified
389 # version that tracks what would be changed.
390
391 if 'kernel_size' in self.parameter_ranges and np.random.random() < 0.5:
392 old_kernel = conv.kernel_size
393 k_min, k_max = self.parameter_ranges['kernel_size']
394
395 # Continuous relaxation
396 theta = np.random.randn()
397 k_continuous = torch.sigmoid(torch.tensor(theta)) * (k_max - k_min) + k_min
398 new_kernel = int(k_continuous.item() + 0.5)
399
400 if isinstance(old_kernel, tuple):
401 new_kernel = (new_kernel, new_kernel)
402
403 changes['kernel_size'] = (old_kernel, new_kernel)
404
405 return changes
406
407 def _mutate_attention_params(self, attn: nn.MultiheadAttention) -> Dict[str, Tuple[Any, Any]]:
408 """Mutate attention layer parameters."""
409 changes = {}
410
411 if 'num_heads' in self.parameter_ranges and np.random.random() < 0.5:
412 old_heads = attn.num_heads
413 h_min, h_max = self.parameter_ranges['num_heads']
414
415 # Ensure new number of heads divides embed_dim
416 embed_dim = attn.embed_dim
417 valid_heads = [h for h in range(h_min, h_max + 1) if embed_dim % h == 0]
418
419 if valid_heads:
420 theta = np.random.randn()
421 h_continuous = torch.sigmoid(torch.tensor(theta)) * len(valid_heads)
422 new_heads_idx = int(h_continuous.item())
423 new_heads = valid_heads[min(new_heads_idx, len(valid_heads) - 1)]
424
425 changes['num_heads'] = (old_heads, new_heads)
426
427 return changes

430class CompoundOperator(StructuralOperator):
431 """
432 Combine multiple operators with conditional logic.
433 """
434
435 def __init__(self, operators: List[StructuralOperator]):
436 """
437 Initialize compound operator.
438
439 Args:
440 operators: List of operators to combine
441 """
442 self.operators = operators
443
444 def apply(self, model: nn.Module, metrics: Dict[str, Any]) -> Tuple[nn.Module, Dict[str, Any]]:
445 """Apply operators sequentially based on conditions."""
446 current_model = model
447 all_info = {'operations': []}
448
449 for i, operator in enumerate(self.operators):
450 if operator.can_apply(current_model, metrics):
451 current_model, info = operator.apply(current_model, metrics)
452 all_info['operations'].append({
453 'operator': type(operator).__name__,
454 'index': i,
455 'info': info
456 })
457
458 all_info['num_operations'] = len(all_info['operations'])
459
460 return current_model, all_info
461
462 def can_apply(self, model: nn.Module, metrics: Dict[str, Any]) -> bool:
463 """Check if any operator can be applied."""
464 return any(op.can_apply(model, metrics) for op in self.operators)

467class AdaptiveOperator(StructuralOperator):
468 """
469 Operator that adapts its parameters based on training progress.
470 """
471
472 def __init__(
473 self,
474 base_operator: StructuralOperator,
475 schedule: str = 'linear',
476 initial_scale: float = 1.0,
477 final_scale: float = 0.1
478 ):
479 """
480 Initialize adaptive operator.
481
482 Args:
483 base_operator: Base operator to adapt
484 schedule: Adaptation schedule ('linear', 'cosine', 'exponential')
485 initial_scale: Initial parameter scale
486 final_scale: Final parameter scale
487 """
488 self.base_operator = base_operator
489 self.schedule = schedule
490 self.initial_scale = initial_scale
491 self.final_scale = final_scale
492 self.current_epoch = 0
493 self.total_epochs = 100 # Default, should be set externally
494
495 def update_epoch(self, epoch: int, total_epochs: int):
496 """Update current epoch for adaptation."""
497 self.current_epoch = epoch
498 self.total_epochs = total_epochs
499 self._adapt_parameters()
500
501 def _adapt_parameters(self):
502 """Adapt operator parameters based on schedule."""
503 progress = self.current_epoch / self.total_epochs
504
505 if self.schedule == 'linear':
506 scale = self.initial_scale + (self.final_scale - self.initial_scale) * progress
507 elif self.schedule == 'cosine':
508 scale = self.final_scale + 0.5 * (self.initial_scale - self.final_scale) * \
509 (1 + np.cos(np.pi * progress))
510 elif self.schedule == 'exponential':
511 scale = self.initial_scale * (self.final_scale / self.initial_scale) ** progress
512 else:
513 scale = self.initial_scale
514
515 # Apply scale to base operator parameters
516 if isinstance(self.base_operator, PruneByEntropy):
517 self.base_operator.threshold *= scale
518 elif isinstance(self.base_operator, ExpandWithMI):
519 self.base_operator.gamma *= scale
520 elif isinstance(self.base_operator, MutateDiscrete):
521 self.base_operator.mutation_rate *= scale
522
523 def apply(self, model: nn.Module, metrics: Dict[str, Any]) -> Tuple[nn.Module, Dict[str, Any]]:
524 """Apply base operator with adapted parameters."""
525 return self.base_operator.apply(model, metrics)
526
527 def can_apply(self, model: nn.Module, metrics: Dict[str, Any]) -> bool:
528 """Check if base operator can be applied."""
529 return self.base_operator.can_apply(model, metrics)
