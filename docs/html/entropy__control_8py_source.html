41class AdaptiveEntropy:
42 """
43 Adaptive Entropy Controller for dynamic threshold adjustment.
44
45 Implements adaptive entropy thresholding based on:
46 τ = τ₀ · exp(-γ · Epoch) · (1 + δ · TaskComplexity)
47 """
48

63 decay_rate: Entropy decay rate γ

90 def update_threshold(
91 self,
92 epoch: int,
93 task_complexity: Optional[float] = None
94 ) -> float:
95 """
96 Update entropy threshold based on current epoch and task complexity.
97
98 Implements: τ = τ₀ · exp(-γ · Epoch) · (1 + δ · TaskComplexity)
99
100 Args:
101 epoch: Current training epoch
102 task_complexity: Optional explicit task complexity (0-1)
103
104 Returns:
105 Updated threshold value
106 """
107 self.current_epoch = epoch
108
109 # Use provided complexity or estimate
110 if task_complexity is not None:
111 self.estimated_complexity = task_complexity
112 elif self.complexity_estimator is not None:
113 self.estimated_complexity = self.complexity_estimator(epoch)
114
115 # Calculate adaptive threshold
116 decay_factor = np.exp(-self.decay_rate * epoch)
117 complexity_factor = 1 + self.task_complexity_factor * self.estimated_complexity
118
119 self.threshold = self.initial_threshold * decay_factor * complexity_factor
120 self.threshold = max(self.min_threshold, self.threshold)
121
122 # Record history
123 self.threshold_history.append(self.threshold)
124
125 return self.threshold

127 def measure(
128 self,
129 outputs: torch.Tensor,
130 layer_outputs: Optional[Dict[str, torch.Tensor]] = None
131 ) -> float:
132 """
133 Measure current entropy from model outputs.
134
135 Args:
136 outputs: Model outputs
137 layer_outputs: Optional dict of layer outputs
138
139 Returns:
140 Current entropy value
141 """
142 # Calculate output entropy
143 if outputs.dim() > 1 and outputs.size(1) > 1:
144 # Multi-class output - use softmax entropy
145 probs = torch.softmax(outputs, dim=-1)
146 log_probs = torch.log(probs + 1e-10)
147 entropy = -torch.sum(probs * log_probs, dim=-1).mean()
148 current_entropy = entropy.item()
149 else:
150 # Single output - use variance as proxy for entropy
151 current_entropy = outputs.var().item()
152
153 # Calculate layer entropies if provided
154 if layer_outputs:
155 self._layer_entropy_cache.clear()
156 for name, layer_out in layer_outputs.items():
157 if layer_out.dim() > 1:
158 layer_probs = torch.softmax(layer_out.view(layer_out.size(0), -1), dim=-1)
159 layer_log_probs = torch.log(layer_probs + 1e-10)
160 layer_entropy = -torch.sum(layer_probs * layer_log_probs, dim=-1).mean()
161 self._layer_entropy_cache[name] = layer_entropy.item()
162 else:
163 self._layer_entropy_cache[name] = layer_out.var().item()
164
165 # Update history
166 self.entropy_history.append(current_entropy)
167 if len(self.entropy_history) > self.window_size * 10:
168 self.entropy_history = self.entropy_history[-self.window_size * 10:]
169
170 return current_entropy

172 def should_prune(self, current_entropy: Optional[float] = None) -> bool:
173 """
174 Determine if pruning should occur based on entropy threshold.
175
176 Args:
177 current_entropy: Current entropy value (uses last measured if None)
178
179 Returns:
180 True if entropy is below threshold and pruning should occur
181 """
182 if current_entropy is None:
183 if not self.entropy_history:
184 return False
185 current_entropy = self.entropy_history[-1]

187 return current_entropy < self.threshold

189 def should_expand(self, current_entropy: Optional[float] = None) -> bool:
190 """
191 Determine if expansion should occur based on entropy analysis.
192
193 Args:
194 current_entropy: Current entropy value
195
196 Returns:
197 True if expansion is recommended
198 """
199 if current_entropy is None:
200 if not self.entropy_history:
201 return False
202 current_entropy = self.entropy_history[-1]
203
204 # Expand if entropy is significantly above threshold
205 # or if entropy has been increasing
206 expand_threshold = self.threshold * 1.5
207
208 if current_entropy > expand_threshold:
209 return True
210
211 # Check if entropy is increasing (potential capacity limit)
212 if len(self.entropy_history) >= self.window_size:
213 recent_entropy = self.entropy_history[-self.window_size:]
214 entropy_trend = np.polyfit(range(len(recent_entropy)), recent_entropy, 1)[0]
215 if entropy_trend > 0.01: # Positive trend
216 return True
217
218 return False

236 def adapt_threshold(self, performance_metric: float, target_metric: float):
237 """
238 Adapt threshold based on performance feedback.
239
240 Args:
241 performance_metric: Current performance (e.g., accuracy)
242 target_metric: Target performance
243 """
244 # If performance is below target, increase threshold (less pruning)
245 # If performance is above target, decrease threshold (more pruning)
246 performance_gap = target_metric - performance_metric
247
248 adjustment = self.adapt_rate * performance_gap
249 self.threshold *= (1 + adjustment)
250 self.threshold = max(self.min_threshold, self.threshold)

252 def estimate_task_complexity(
253 self,
254 dataset_size: int,
255 num_classes: int,
256 input_dim: int,
257 model_depth: int
258 ) -> float:
259 """
260 Estimate task complexity based on dataset and model characteristics.
261
262 Args:
263 dataset_size: Number of training samples
264 num_classes: Number of output classes
265 input_dim: Input dimensionality
266 model_depth: Model depth
267
268 Returns:
269 Estimated complexity (0-1)
270 """
271 # Normalize factors
272 size_factor = np.log10(dataset_size + 1) / 6 # Assume max 1M samples
273 class_factor = np.log2(num_classes + 1) / 10 # Assume max 1024 classes
274 dim_factor = np.log10(input_dim + 1) / 5 # Assume max 100k dimensions
275 depth_factor = model_depth / 100 # Assume max depth 100
276
277 # Weighted combination
278 complexity = (
279 0.3 * size_factor +
280 0.3 * class_factor +
281 0.2 * dim_factor +
282 0.2 * depth_factor
283 )
284
285 # Clamp to [0, 1]
286 self.estimated_complexity = max(0, min(1, complexity))
287
288 return self.estimated_complexity
