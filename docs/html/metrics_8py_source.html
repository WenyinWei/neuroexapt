18def calculate_entropy(
19 tensor: torch.Tensor,
20 method: str = 'histogram',
21 bins: int = 30,
22 epsilon: float = 1e-10
23) -> float:
24 """
25 Calculate Shannon entropy of a tensor.
26
27 Args:
28 tensor: Input tensor
29 method: Method for entropy calculation ('histogram', 'kde', 'discrete')
30 bins: Number of bins for histogram method
31 epsilon: Small value to avoid log(0)
32
33 Returns:
34 Entropy value in bits
35 """
36 data = tensor.detach().cpu().numpy().flatten()
37
38 if method == 'histogram':
39 # Histogram-based entropy estimation
40 hist, _ = np.histogram(data, bins=bins)
41 hist = hist + epsilon
42 hist = hist / hist.sum()
43 return -np.sum(hist * np.log2(hist))
44
45 elif method == 'kde':
46 # Kernel density estimation
47 try:
48 kde = stats.gaussian_kde(data)
49 # Sample from KDE for entropy estimation
50 samples = kde.resample(1000).flatten()
51 hist, _ = np.histogram(samples, bins=bins)
52 hist = hist + epsilon
53 hist = hist / hist.sum()
54 return -np.sum(hist * np.log2(hist))
55 except:
56 # Fallback to histogram if KDE fails
57 return calculate_entropy(tensor, method='histogram', bins=bins)
58
59 elif method == 'discrete':
60 # For discrete distributions (e.g., after softmax)
61 if tensor.dim() > 1:
62 # Apply softmax if needed
63 probs = F.softmax(tensor.view(-1), dim=0)
64 else:
65 probs = tensor.view(-1)
66 probs = probs.clamp(min=epsilon)
67 return -torch.sum(probs * torch.log2(probs)).item()
68
69 else:
70 raise ValueError(f"Unknown entropy calculation method: {method}")

73def calculate_mutual_information(
74 x: torch.Tensor,
75 y: torch.Tensor,
76 method: str = 'histogram',
77 bins: int = 30
78) -> float:
79 """
80 Calculate mutual information I(X;Y).
81
82 Args:
83 x: First variable
84 y: Second variable
85 method: Method for MI calculation
86 bins: Number of bins for discretization
87
88 Returns:
89 Mutual information in bits
90 """
91 x_data = x.detach().cpu().numpy().flatten()
92 y_data = y.detach().cpu().numpy().flatten()
93
94 if len(x_data) != len(y_data):
95 raise ValueError("X and Y must have the same number of samples")
96
97 if method == 'histogram':
98 # 2D histogram for joint probability
99 hist_2d, _, _ = np.histogram2d(x_data, y_data, bins=bins)
100 hist_2d = hist_2d + 1e-10
101
102 # Joint probability
103 p_xy = hist_2d / hist_2d.sum()
104
105 # Marginal probabilities
106 p_x = p_xy.sum(axis=1)
107 p_y = p_xy.sum(axis=0)
108
109 # MI = H(X) + H(Y) - H(X,Y)
110 h_x = -np.sum(p_x * np.log2(p_x + 1e-10))
111 h_y = -np.sum(p_y * np.log2(p_y + 1e-10))
112 h_xy = -np.sum(p_xy * np.log2(p_xy + 1e-10))
113
114 return h_x + h_y - h_xy
115
116 else:
117 raise ValueError(f"Unknown MI calculation method: {method}")

120def calculate_conditional_entropy(
121 x: torch.Tensor,
122 y: torch.Tensor,
123 bins: int = 30
124) -> float:
125 """
126 Calculate conditional entropy H(X|Y).
127
128 Args:
129 x: Variable X
130 y: Conditioning variable Y
131 bins: Number of bins for discretization
132
133 Returns:
134 Conditional entropy in bits
135 """
136 # H(X|Y) = H(X,Y) - H(Y)
137 x_data = x.detach().cpu().numpy().flatten()
138 y_data = y.detach().cpu().numpy().flatten()
139
140 # Joint entropy
141 hist_2d, _, _ = np.histogram2d(x_data, y_data, bins=bins)
142 hist_2d = hist_2d + 1e-10
143 p_xy = hist_2d / hist_2d.sum()
144 h_xy = -np.sum(p_xy * np.log2(p_xy + 1e-10))
145
146 # Marginal entropy of Y
147 p_y = p_xy.sum(axis=0)
148 h_y = -np.sum(p_y * np.log2(p_y + 1e-10))
149
150 return h_xy - h_y

153def calculate_kl_divergence(
154 p: torch.Tensor,
155 q: torch.Tensor,
156 epsilon: float = 1e-10
157) -> float:
158 """
159 Calculate Kullback-Leibler divergence KL(P||Q).
160
161 Args:
162 p: Distribution P
163 q: Distribution Q
164 epsilon: Small value to avoid log(0)
165
166 Returns:
167 KL divergence in bits
168 """
169 # Ensure distributions
170 if p.dim() > 1:
171 p = F.softmax(p.view(-1), dim=0)
172 q = F.softmax(q.view(-1), dim=0)
173
174 p = p.clamp(min=epsilon)
175 q = q.clamp(min=epsilon)
176
177 return torch.sum(p * torch.log2(p / q)).item()

180def calculate_js_divergence(
181 p: torch.Tensor,
182 q: torch.Tensor
183) -> float:
184 """
185 Calculate Jensen-Shannon divergence.
186
187 Args:
188 p: Distribution P
189 q: Distribution Q
190
191 Returns:
192 JS divergence in bits
193 """
194 # JS(P||Q) = 0.5 * KL(P||M) + 0.5 * KL(Q||M), where M = 0.5 * (P + Q)
195 m = 0.5 * (p + q)
196
197 js = 0.5 * calculate_kl_divergence(p, m) + 0.5 * calculate_kl_divergence(q, m)
198
199 return js

202def calculate_layer_complexity(
203 layer: nn.Module,
204 include_bias: bool = True
205) -> Dict[str, float]:
206 """
207 Calculate complexity metrics for a layer.
208
209 Args:
210 layer: Neural network layer
211 include_bias: Whether to include bias in parameter count
212
213 Returns:
214 Dictionary of complexity metrics
215 """
216 metrics = {}
217
218 # Parameter count
219 total_params = 0
220 trainable_params = 0
221
222 for name, param in layer.named_parameters():
223 if not include_bias and 'bias' in name:
224 continue
225 total_params += param.numel()
226 if param.requires_grad:
227 trainable_params += param.numel()
228
229 metrics['total_parameters'] = total_params
230 metrics['trainable_parameters'] = trainable_params
231
232 # Layer-specific metrics
233 if isinstance(layer, nn.Conv2d):
234 metrics['kernel_size'] = layer.kernel_size
235 metrics['in_channels'] = layer.in_channels
236 metrics['out_channels'] = layer.out_channels
237 metrics['groups'] = layer.groups
238 metrics['computational_cost'] = (
239 layer.kernel_size[0] * layer.kernel_size[1] *
240 layer.in_channels * layer.out_channels / layer.groups
241 )
242 elif isinstance(layer, nn.Linear):
243 metrics['in_features'] = layer.in_features
244 metrics['out_features'] = layer.out_features
245 metrics['computational_cost'] = layer.in_features * layer.out_features
246
247 return metrics

250def calculate_network_complexity(
251 model: nn.Module,
252 input_shape: Optional[Tuple[int, ...]] = None
253) -> Dict[str, Any]:
254 """
255 Calculate comprehensive complexity metrics for entire network.
256
257 Args:
258 model: Neural network model
259 input_shape: Input shape for FLOPs calculation
260
261 Returns:
262 Dictionary of network complexity metrics
263 """
264 metrics = {
265 'total_parameters': 0,
266 'trainable_parameters': 0,
267 'layer_complexities': {},
268 'depth': 0,
269 'width_stats': {}
270 }
271
272 # Count parameters and analyze layers
273 layer_widths = []
274
275 for name, module in model.named_modules():
276 if len(list(module.children())) == 0: # Leaf module
277 layer_metrics = calculate_layer_complexity(module)
278 if layer_metrics['total_parameters'] > 0:
279 metrics['layer_complexities'][name] = layer_metrics
280 metrics['total_parameters'] += layer_metrics['total_parameters']
281 metrics['trainable_parameters'] += layer_metrics['trainable_parameters']
282 metrics['depth'] += 1
283
284 # Track width
285 if 'out_features' in layer_metrics:
286 layer_widths.append(layer_metrics['out_features'])
287 elif 'out_channels' in layer_metrics:
288 layer_widths.append(layer_metrics['out_channels'])
289
290 # Width statistics
291 if layer_widths:
292 metrics['width_stats'] = {
293 'mean': np.mean(layer_widths),
294 'std': np.std(layer_widths),
295 'min': min(layer_widths),
296 'max': max(layer_widths)
297 }
298
299 # Estimate FLOPs if input shape provided
300 if input_shape is not None:
301 try:
302 metrics['estimated_flops'] = estimate_flops(model, input_shape)
303 except:
304 warnings.warn("Could not estimate FLOPs")
305
306 return metrics

309def estimate_flops(
310 model: nn.Module,
311 input_shape: Tuple[int, ...]
312) -> int:
313 """
314 Estimate FLOPs (floating point operations) for a model.
315
316 This is a simplified estimation focusing on major operations.
317
318 Args:
319 model: Neural network model
320 input_shape: Input shape (excluding batch dimension)
321
322 Returns:
323 Estimated FLOPs
324 """
325 total_flops = 0
326
327 # Create dummy input
328 dummy_input = torch.randn(1, *input_shape)
329
330 # Hook to capture layer inputs/outputs
331 def hook_fn(module, input, output):
332 nonlocal total_flops
333
334 if isinstance(module, nn.Conv2d):
335 # Conv2d FLOPs
336 out_h, out_w = output.shape[2:]
337 kernel_ops = module.kernel_size[0] * module.kernel_size[1]
338 bias_ops = 1 if module.bias is not None else 0
339
340 flops = (kernel_ops * module.in_channels / module.groups + bias_ops) * \
341 module.out_channels * out_h * out_w
342 total_flops += flops
343
344 elif isinstance(module, nn.Linear):
345 # Linear FLOPs
346 flops = module.in_features * module.out_features
347 if module.bias is not None:
348 flops += module.out_features
349 total_flops += flops
350
351 # Register hooks
352 hooks = []
353 for module in model.modules():
354 if isinstance(module, (nn.Conv2d, nn.Linear)):
355 hooks.append(module.register_forward_hook(hook_fn))
356
357 # Forward pass
358 with torch.no_grad():
359 model(dummy_input)
360
361 # Remove hooks
362 for hook in hooks:
363 hook.remove()
364
365 return int(total_flops)

368def calculate_structural_entropy(
369 importances: Dict[str, float]
370) -> float:
371 """
372 Calculate structural entropy based on layer importances.
373
374 Args:
375 importances: Dictionary of layer importances
376
377 Returns:
378 Structural entropy in bits
379 """
380 if not importances:
381 return 0.0
382
383 # Normalize to probabilities
384 values = list(importances.values())
385 total = sum(values)
386
387 if total == 0:
388 return 0.0
389
390 probs = [v / total for v in values]
391
392 # Calculate entropy
393 entropy = 0.0
394 for p in probs:
395 if p > 0:
396 entropy -= p * np.log2(p)

398 return entropy

401def calculate_redundancy_score(
402 layer_importances: List[float],
403 output_entropy: float,
404 depth: int,
405 depth_decay: float = 0.03
406) -> float:
407 """
408 Calculate network redundancy score.
409
410 Implements: R = 1 - I(L_i;O) / (H(O) exp(- Depth))
411
412 Args:
413 layer_importances: List of layer importance values
414 output_entropy: Output entropy H(O)
415 depth: Network depth
416 depth_decay: Decay factor
417
418 Returns:
419 Redundancy score between 0 and 1
420 """
421 if not layer_importances or output_entropy == 0:
422 return 0.0
423
424 total_importance = sum(layer_importances)
425 normalization = output_entropy * np.exp(-depth_decay * depth)
426
427 redundancy = 1 - (total_importance / (normalization + 1e-10))
428
429 return max(0, min(1, redundancy))

432def calculate_discrete_parameter_gradient(
433 theta: torch.Tensor,
434 k_min: int,
435 k_max: int,
436 temperature: float = 1.0
437) -> Tuple[int, torch.Tensor]:
438 """
439 Calculate gradient for discrete parameter optimization.
440
441 Implements: k = floor(() * (k_max - k_min) + 0.5)
442
443 Args:
444 theta: Continuous parameter
445 k_min: Minimum discrete value
446 k_max: Maximum discrete value
447 temperature: Temperature for sigmoid
448
449 Returns:
450 Discrete value and gradient tensor
451 """
452 # Apply temperature-scaled sigmoid
453 sigma = torch.sigmoid(theta / temperature)
454
455 # Map to discrete range
456 k_continuous = sigma * (k_max - k_min) + k_min
457 k_discrete = int(torch.floor(k_continuous + 0.5).item())
458
459 # Gradient flows through continuous relaxation
460 return k_discrete, k_continuous

463def calculate_information_gain(
464 old_importance: Dict[str, float],
465 new_importance: Dict[str, float]
466) -> float:
467 """
468 Calculate information gain between two network states.
469
470 Args:
471 old_importance: Layer importances before modification
472 new_importance: Layer importances after modification
473
474 Returns:
475 Information gain (can be negative for information loss)
476 """
477 # Get common layers
478 common_layers = set(old_importance.keys()) & set(new_importance.keys())
479
480 if not common_layers:
481 return 0.0
482
483 # Calculate gain for common layers
484 gain = 0.0
485 for layer in common_layers:
486 gain += new_importance[layer] - old_importance[layer]
487
488 # Penalize removed layers
489 removed_layers = set(old_importance.keys()) - set(new_importance.keys())
490 for layer in removed_layers:
491 gain -= old_importance[layer]
492
493 # Reward new layers (conservative estimate)
494 new_layers = set(new_importance.keys()) - set(old_importance.keys())
495 if new_layers and common_layers:
496 avg_importance = float(np.mean([new_importance[l] for l in common_layers]))
497 gain += len(new_layers) * avg_importance * 0.5 # Conservative factor
498
499 return gain

534
535 # Weighted combination
536 complexity = (
537 weights[0] * class_complexity +
538 weights[1] * dim_complexity +
539 weights[2] * size_complexity
540 )
