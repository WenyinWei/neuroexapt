12def plot_evolution_history(
13 evolution_history: List[Any],
14 save_path: Optional[str] = None
15):
16 """
17 Plot evolution history showing structural changes over time.
18
19 Args:
20 evolution_history: List of evolution steps
21 save_path: Optional path to save plot
22 """
23 if not evolution_history:
24 return
25
26 # Extract data
27 epochs = [step.epoch for step in evolution_history]
28 params_before = [step.parameters_before for step in evolution_history]
29 params_after = [step.parameters_after for step in evolution_history]
30 actions = [step.action for step in evolution_history]
31
32 # Create figure
33 fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 8))
34
35 # Plot parameter evolution
36 ax1.plot(epochs, params_before, 'o-', label='Before', alpha=0.7)
37 ax1.plot(epochs, params_after, 's-', label='After', alpha=0.7)
38
39 # Add action annotations
40 for i, (epoch, action) in enumerate(zip(epochs, actions)):
41 if action != 'none':
42 ax1.annotate(
43 action,
44 xy=(epoch, params_after[i]),
45 xytext=(epoch, params_after[i] + 1000),
46 arrowprops=dict(arrowstyle='->', alpha=0.5),
47 fontsize=8,
48 ha='center'
49 )
50
51 ax1.set_xlabel('Epoch')
52 ax1.set_ylabel('Number of Parameters')
53 ax1.set_title('Parameter Evolution During Training')
54 ax1.legend()
55 ax1.grid(True, alpha=0.3)
56
57 # Plot efficiency gains
58 efficiency_gains = [step.efficiency_gain() for step in evolution_history]
59 colors = ['green' if g > 0 else 'red' for g in efficiency_gains]
60
61 ax2.bar(epochs, efficiency_gains, color=colors, alpha=0.7)
62 ax2.axhline(y=0, color='black', linestyle='-', linewidth=0.5)
63 ax2.set_xlabel('Epoch')
64 ax2.set_ylabel('Efficiency Gain')
65 ax2.set_title('Structural Evolution Efficiency')
66 ax2.grid(True, alpha=0.3)
67
68 plt.tight_layout()
69
70 if save_path:
71 plt.savefig(save_path, dpi=150, bbox_inches='tight')
72 else:
73 plt.show()
74
75 plt.close()

78def plot_entropy_history(
79 entropy_history: List[float],
80 threshold_history: List[float],
81 save_path: Optional[str] = None
82):
83 """
84 Plot entropy evolution with adaptive threshold.
85
86 Args:
87 entropy_history: List of entropy values
88 threshold_history: List of threshold values
89 save_path: Optional path to save plot
90 """
91 plt.figure(figsize=(10, 6))
92
93 # Plot entropy
94 plt.plot(entropy_history, label='Entropy', alpha=0.8, linewidth=2)
95
96 # Plot threshold
97 if len(threshold_history) > len(entropy_history):
98 threshold_history = threshold_history[:len(entropy_history)]
99 elif len(threshold_history) < len(entropy_history):
100 # Extend threshold history
101 last_threshold = threshold_history[-1] if threshold_history else 0.5
102 threshold_history.extend([last_threshold] * (len(entropy_history) - len(threshold_history)))
103
104 plt.plot(threshold_history, '--', label='Threshold', color='red', alpha=0.8)
105
106 # Add shaded regions
107 plt.fill_between(
108 range(len(entropy_history)),
109 0,
110 threshold_history,
111 alpha=0.2,
112 color='red',
113 label='Pruning Zone'
114 )
115
116 plt.xlabel('Training Step')
117 plt.ylabel('Entropy')
118 plt.title('Adaptive Entropy Control')
119 plt.legend()
120 plt.grid(True, alpha=0.3)
121
122 if save_path:
123 plt.savefig(save_path, dpi=150, bbox_inches='tight')
124 else:
125 plt.show()
126
127 plt.close()

130def plot_layer_importance_heatmap(
131 layer_importances: Dict[str, List[float]],
132 save_path: Optional[str] = None
133):
134 """
135 Plot heatmap of layer importances over time.
136
137 Args:
138 layer_importances: Dictionary mapping layer names to importance history
139 save_path: Optional path to save plot
140 """
141 if not layer_importances:
142 return
143
144 # Prepare data
145 layer_names = list(layer_importances.keys())
146 importance_matrix = []
147
148 max_len = max(len(hist) for hist in layer_importances.values())
149
150 for name in layer_names:
151 history = layer_importances[name]
152 # Pad if necessary
153 if len(history) < max_len:
154 history = history + [history[-1]] * (max_len - len(history))
155 importance_matrix.append(history)
156
157 importance_matrix = np.array(importance_matrix)
158
159 # Create heatmap
160 plt.figure(figsize=(12, 8))
161
162 sns.heatmap(
163 importance_matrix,
164 xticklabels=False,
165 yticklabels=layer_names,
166 cmap='YlOrRd',
167 cbar_kws={'label': 'Importance'},
168 vmin=0
169 )
170
171 plt.xlabel('Training Step')
172 plt.ylabel('Layer')
173 plt.title('Layer Importance Evolution')
174
175 if save_path:
176 plt.savefig(save_path, dpi=150, bbox_inches='tight')
177 else:
178 plt.show()
179
180 plt.close()

183def plot_information_metrics(
184 metrics_history: Dict[str, List[float]],
185 save_path: Optional[str] = None
186):
187 """
188 Plot information-theoretic metrics over training.
189
190 Args:
191 metrics_history: Dictionary of metric histories
192 save_path: Optional path to save plot
193 """
194 n_metrics = len(metrics_history)
195 if n_metrics == 0:
196 return
197
198 fig, axes = plt.subplots(
199 (n_metrics + 1) // 2,
200 2,
201 figsize=(12, 4 * ((n_metrics + 1) // 2))
202 )
203
204 if n_metrics == 1:
205 axes = [axes]
206 else:
207 axes = axes.flatten()
208
209 for idx, (metric_name, values) in enumerate(metrics_history.items()):
210 ax = axes[idx]
211
212 ax.plot(values, alpha=0.8)
213 ax.set_title(metric_name.replace('_', ' ').title())
214 ax.set_xlabel('Step')
215 ax.set_ylabel('Value')
216 ax.grid(True, alpha=0.3)
217
218 # Add moving average
219 if len(values) > 10:
220 window = min(50, len(values) // 10)
221 ma = np.convolve(values, np.ones(window)/window, mode='valid')
222 ax.plot(range(window-1, len(values)), ma, 'r--', alpha=0.8, label='MA')
223 ax.legend()
224
225 # Hide unused subplots
226 for idx in range(n_metrics, len(axes)):
227 axes[idx].axis('off')
228
229 plt.tight_layout()
230
231 if save_path:
232 plt.savefig(save_path, dpi=150, bbox_inches='tight')
233 else:
234 plt.show()
235
236 plt.close()

250 fig = plt.figure(figsize=(16, 12))
251
252 # Create grid
253 gs = fig.add_gridspec(3, 3, hspace=0.3, wspace=0.3)
254
255 # 1. Evolution history (top-left, 2x1)
256 ax1 = fig.add_subplot(gs[0:2, 0])
257 if ne_instance.evolution_history:
258 epochs = [step.epoch for step in ne_instance.evolution_history]
259 params = [step.parameters_after for step in ne_instance.evolution_history]
260 ax1.plot(epochs, params, 'o-')
261 ax1.set_title('Parameter Evolution')
262 ax1.set_xlabel('Epoch')
263 ax1.set_ylabel('Parameters')
264 else:
265 ax1.text(0.5, 0.5, 'No evolution history', ha='center', va='center')
266

268 ax2 = fig.add_subplot(gs[0, 1])

271 ax2.set_title('Recent Entropy')
272 ax2.set_xlabel('Step')
273 ax2.set_ylabel('Entropy')
274 else:
275 ax2.text(0.5, 0.5, 'No entropy data', ha='center', va='center')
276
277 # 3. Current metrics (top-right)
278 ax3 = fig.add_subplot(gs[0, 2])
279 ax3.axis('off')
280 metrics_text = f"""Current Status:
281 Epoch: {ne_instance.current_epoch}
282 Entropy Threshold: {ne_instance.entropy_ctrl.threshold:.3f}
283 Evolution Steps: {len(ne_instance.evolution_history)}
284 """
285 ax3.text(0.1, 0.8, metrics_text, va='top', fontsize=12)
286
287 # 4. Layer importance (middle row, full width)
288 ax4 = fig.add_subplot(gs[1, :])
289 layer_importances = ne_instance.info_theory._activation_cache.get('layer_importances', {})
290 if layer_importances:
291 names = list(layer_importances.keys())[:20] # Limit to 20 layers
292 values = [layer_importances[n] for n in names]
293 ax4.barh(names, values)
294 ax4.set_xlabel('Importance')
295 ax4.set_title('Current Layer Importances')
296 else:
297 ax4.text(0.5, 0.5, 'No layer importance data', ha='center', va='center')
298
299 # 5. Evolution summary (bottom row)
300 ax5 = fig.add_subplot(gs[2, :])
301 if ne_instance.evolution_history:
302 actions = {}

304 actions[step.action] = actions.get(step.action, 0) + 1

312
313 # Add title

316 fontsize=16
317 )
318

320 plt.savefig(save_path, dpi=150, bbox_inches='tight')
321 else:
322 plt.show()
323
324 plt.close()
