24class SimpleCNN(nn.Module):
25 """Simple CNN for CIFAR-10 classification."""
26
27 def __init__(self, num_classes=10):
28 super(SimpleCNN, self).__init__()
29
30 # Convolutional layers
31 self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
32 self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
33 self.conv3 = nn.Conv2d(64, 128, kernel_size=3, padding=1)
34
35 # Pooling layers
36 self.pool = nn.MaxPool2d(2, 2)
37
38 # Dropout for regularization
39 self.dropout = nn.Dropout(0.25)
40
41 # Fully connected layers
42 self.fc1 = nn.Linear(128 * 4 * 4, 256)
43 self.fc2 = nn.Linear(256, 128)
44 self.fc3 = nn.Linear(128, num_classes)
45
46 def forward(self, x):
47 # Convolutional layers with ReLU and pooling
48 x = self.pool(F.relu(self.conv1(x)))
49 x = self.pool(F.relu(self.conv2(x)))
50 x = self.pool(F.relu(self.conv3(x)))
51
52 # Flatten for fully connected layers
53 x = x.view(-1, 128 * 4 * 4)
54
55 # Fully connected layers
56 x = F.relu(self.fc1(x))
57 x = self.dropout(x)
58 x = F.relu(self.fc2(x))
59 x = self.fc3(x)
60

64def get_cifar10_dataloaders(batch_size=128, num_workers=2):
65 """Get CIFAR-10 data loaders."""
66
67 # Data transformations
68 transform_train = transforms.Compose([
69 transforms.RandomCrop(32, padding=4),
70 transforms.RandomHorizontalFlip(),
71 transforms.ToTensor(),
72 transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2023, 0.1994, 0.2010)),
73 ])
74
75 transform_test = transforms.Compose([
76 transforms.ToTensor(),

78 ])
79
80 # Datasets
81 trainset = torchvision.datasets.CIFAR10(
82 root='./data', train=True, download=True, transform=transform_train
83 )
84 testset = torchvision.datasets.CIFAR10(
85 root='./data', train=False, download=True, transform=transform_test
86 )
87
88 # Data loaders
89 train_loader = DataLoader(
90 trainset, batch_size=batch_size, shuffle=True, num_workers=num_workers
91 )
92 test_loader = DataLoader(
