16class InformationBottleneck:
17 """
18 Information Bottleneck engine for measuring and optimizing information flow
19 through neural network layers.
20
21 Based on the principle: I(X;Y) - β·I(Y;Z)
22 where X is input, Y is layer representation, Z is output.
23 """
24
25 def __init__(
26 self,
27 beta: float = 1.0,
28 n_bins: int = 30,

30 device: Optional[torch.device] = None

32 """
33 Initialize Information Bottleneck engine.
34
35 Args:
36 beta: Trade-off parameter between compression and prediction
37 n_bins: Number of bins for entropy estimation
38 normalize: Whether to normalize mutual information values
39 device: Torch device for computations
40 """
41 self.beta = beta
42 self.n_bins = n_bins
43 self.normalize = normalize
44 self.device = device or torch.device("cuda" if torch.cuda.is_available() else "cpu")

50 def calculate_entropy(self, tensor: torch.Tensor, dim: Optional[int] = None) -> float:
51 """
52 Calculate Shannon entropy of a tensor.
53
54 Args:
55 tensor: Input tensor
56 dim: Dimension along which to calculate entropy
57
58 Returns:
59 Entropy value
60 """
61 if dim is not None:
62 # Calculate entropy along specific dimension
63 probs = F.softmax(tensor, dim=dim)
64 log_probs = torch.log(probs + 1e-10)
65 entropy = -torch.sum(probs * log_probs, dim=dim)
66 return entropy.mean().item()
67
68 # Flatten tensor and estimate entropy using histogram
69 data = tensor.detach().cpu().numpy().flatten()
70
71 # Create histogram
72 hist, _ = np.histogram(data, bins=self.n_bins)
73 hist = hist + 1e-10 # Avoid log(0)
74 hist = hist / hist.sum()
75
76 return float(scipy_entropy(hist))

78 def calculate_mutual_information(
79 self,
80 x: torch.Tensor,
81 y: torch.Tensor,
82 normalize: Optional[bool] = None
83 ) -> float:
84 """
85 Calculate mutual information I(X;Y) using binning method.
86
87 Args:
88 x: First variable
89 y: Second variable
90 normalize: Whether to normalize (overrides instance setting)
91
92 Returns:
93 Mutual information value
94 """
95 normalize = self.normalize if normalize is None else normalize
96
97 # Ensure tensors are on CPU for numpy operations
98 x_np = x.detach().cpu().numpy().flatten()
99 y_np = y.detach().cpu().numpy().flatten()
100
101 # Create 2D histogram
102 hist_2d, _, _ = np.histogram2d(x_np, y_np, bins=self.n_bins)
103 hist_2d = hist_2d + 1e-10
104
105 # Normalize to get joint probability
106 p_xy = hist_2d / hist_2d.sum()
107
108 # Marginal distributions
109 p_x = p_xy.sum(axis=1)
110 p_y = p_xy.sum(axis=0)
111
112 # Calculate MI: I(X;Y) = H(X) + H(Y) - H(X,Y)
113 h_x = float(scipy_entropy(p_x))
114 h_y = float(scipy_entropy(p_y))
115 h_xy = float(scipy_entropy(p_xy.flatten()))
116
117 mi = h_x + h_y - h_xy
118
119 if normalize:
120 # Normalize by min(H(X), H(Y))
121 mi = mi / (min(h_x, h_y) + 1e-10)
122
123 return float(mi)

125 def calculate_layer_importance(
126 self,
127 layer_output: torch.Tensor,
128 target_output: torch.Tensor,
129 task_type: str = "classification",
130 layer_name: Optional[str] = None
131 ) -> float:
132 """
133 Calculate layer importance I(L_i;O) with task-aware weighting.
134
135 Implements: I(L_i;O) = H(O) - H(O|L_i) · ψ(TaskType)
136
137 Args:
138 layer_output: Output of layer i
139 target_output: Target/final output
140 task_type: Type of task (classification, generation, regression)
141 layer_name: Optional layer identifier for caching
142
143 Returns:
144 Layer importance score
145 """
146 # Task-specific weight function ψ
147 task_weights = {
148 "classification": 1.2,
149 "generation": 0.8,
150 "regression": 1.0,
151 "detection": 1.1
152 }
153 psi = task_weights.get(task_type, 1.0)
154
155 # Calculate H(O)
156 h_output = self.calculate_entropy(target_output)
157
158 # Estimate H(O|L_i) using conditional entropy approximation
159 # This is approximated as H(O) - I(L_i;O)
160 mi = self.calculate_mutual_information(layer_output, target_output)
161 h_output_given_layer = h_output - mi
162
163 # Apply task-aware weighting
164 importance = h_output - h_output_given_layer * psi
165
166 # Cache result if layer name provided
167 if layer_name:
168 self._activation_cache[f"{layer_name}_importance"] = importance
169
170 return importance

172 def calculate_redundancy(
173 self,
174 layer_importances: List[float],
175 output_entropy: float,
176 depth: int,
177 depth_decay_lambda: float = 0.03
178 ) -> float:
179 """
180 Calculate network redundancy.
181
182 Implements: R = 1 - Σ I(L_i;O) / (H(O) · exp(-λ · Depth))
183
184 Args:
185 layer_importances: List of layer importance values
186 output_entropy: Entropy of output H(O)
187 depth: Network depth
188 depth_decay_lambda: Decay factor for depth normalization
189
190 Returns:
191 Redundancy score (0 to 1)
192 """
193 # Sum of layer importances
194 total_importance = sum(layer_importances)
195
196 # Depth-aware normalization
197 normalization = output_entropy * np.exp(-depth_decay_lambda * depth)
198
199 # Calculate redundancy
200 redundancy = 1 - (total_importance / (normalization + 1e-10))
201
202 return max(0, min(1, redundancy)) # Clamp to [0, 1]

204 def register_hooks(self, model: nn.Module, layer_names: Optional[List[str]] = None):
205 """
206 Register forward hooks to capture layer activations.
207
208 Args:
209 model: PyTorch model
210 layer_names: Specific layers to monitor (None for all)
211 """
212 self.clear_hooks()
213
214 def hook_fn(name):
215 def hook(module, input, output):
216 self._activation_cache[name] = output.detach()
217 return hook
218
219 # Register hooks for specified layers or all layers
220 for name, module in model.named_modules():
221 if layer_names is None or name in layer_names:
222 if len(list(module.children())) == 0: # Leaf module
223 handle = module.register_forward_hook(hook_fn(name))
224 self._hooks.append(handle)
225
226 def clear_hooks(self):
227 """Remove all registered hooks."""
228 for hook in self._hooks:
229 hook.remove()
230 self._hooks.clear()
231 self._activation_cache.clear()

233 def compute_information_loss(
234 self,
235 predictions: torch.Tensor,
236 targets: torch.Tensor,
237 layer_outputs: Optional[Dict[str, torch.Tensor]] = None
238 ) -> torch.Tensor:
239 """
240 Compute information-theoretic loss for training.
241
242 Args:
243 predictions: Model predictions
244 targets: Ground truth targets
245 layer_outputs: Optional dict of layer outputs
246
247 Returns:
248 Information loss tensor
249 """
250 # Use cached activations if layer_outputs not provided
251 if layer_outputs is None:
252 layer_outputs = self._activation_cache
253
254 # Base prediction loss (cross-entropy for classification)
255 if predictions.dim() > 1 and predictions.size(1) > 1:
256 # Multi-class classification
257 base_loss = F.cross_entropy(predictions, targets)
258 else:
259 # Binary classification or regression
260 base_loss = F.mse_loss(predictions, targets.float())
261
262 # Information bottleneck regularization
263 ib_loss = 0.0
264 if layer_outputs:
265 for name, output in layer_outputs.items():
266 if "importance" not in name: # Skip cached importance values
267 # Minimize I(layer;input) while preserving I(layer;output)
268 layer_entropy = self.calculate_entropy(output)
269 ib_loss += self.beta * layer_entropy
270
271 total_loss = base_loss + ib_loss
272
273 return total_loss

275 def analyze_network(
276 self,
277 model: nn.Module,
278 dataloader: torch.utils.data.DataLoader,
279 task_type: str = "classification"
280 ) -> Dict[str, Union[float, Dict[str, float]]]:
281 """
282 Comprehensive information-theoretic analysis of a neural network.
283
284 Args:
285 model: Neural network model
286 dataloader: Data loader for analysis
287 task_type: Type of task
288
289 Returns:
290 Dictionary containing analysis results
291 """
292 model.eval()
293 self.register_hooks(model)
294
295 layer_importances = {}
296 layer_entropies = {}
297 total_mi = 0.0
298
299 with torch.no_grad():
300 for batch_idx, (inputs, targets) in enumerate(dataloader):
301 if batch_idx > 10: # Analyze first 10 batches
302 break
303
304 inputs = inputs.to(self.device)
305 outputs = model(inputs)
306
307 # Analyze each layer
308 for name, activation in self._activation_cache.items():
309 if "importance" not in name:
310 # Calculate layer entropy
311 entropy = self.calculate_entropy(activation)
312 layer_entropies[name] = layer_entropies.get(name, 0) + entropy
313
314 # Calculate layer importance
315 importance = self.calculate_layer_importance(
316 activation, outputs, task_type, name
317 )
318 layer_importances[name] = layer_importances.get(name, 0) + importance
319
320 # Calculate mutual information with input
321 mi = self.calculate_mutual_information(
322 inputs.view(inputs.size(0), -1),
323 activation.view(activation.size(0), -1)
324 )
325 total_mi += mi
326
327 # Average over batches
328 n_batches = min(len(dataloader), 10)
329 layer_importances = {k: v/n_batches for k, v in layer_importances.items()}
330 layer_entropies = {k: v/n_batches for k, v in layer_entropies.items()}
331
332 # Calculate overall metrics
333 output_entropy = self.calculate_entropy(outputs)
334 redundancy = self.calculate_redundancy(
335 list(layer_importances.values()),
336 output_entropy,
337 len(layer_importances)
338 )
339
340 self.clear_hooks()
341
342 return {
343 "layer_importances": layer_importances,
344 "layer_entropies": layer_entropies,
345 "output_entropy": output_entropy,
346 "redundancy": redundancy,
347 "average_mutual_information": total_mi / (n_batches * len(layer_importances))
348 }

364 """
365 Initialize adaptive information bottleneck.
366
367 Args:
368 initial_beta: Starting beta value
369 beta_schedule: Schedule type (linear, exponential, cosine)
370 min_beta: Minimum beta value
371 max_beta: Maximum beta value
372 **kwargs: Additional arguments for parent class
373 """
374 super().__init__(beta=initial_beta, **kwargs)
375 self.initial_beta = initial_beta
376 self.beta_schedule = beta_schedule
377 self.min_beta = min_beta
378 self.max_beta = max_beta
379 self.current_epoch = 0
380
381 def update_beta(self, epoch: int, total_epochs: int):
382 """Update beta according to schedule."""
383 self.current_epoch = epoch
384 progress = epoch / total_epochs
385
386 if self.beta_schedule == "linear":
387 self.beta = self.initial_beta + (self.max_beta - self.initial_beta) * progress
388 elif self.beta_schedule == "exponential":
389 self.beta = self.initial_beta * (self.max_beta / self.initial_beta) ** progress
390 elif self.beta_schedule == "cosine":
391 self.beta = self.min_beta + 0.5 * (self.max_beta - self.min_beta) * \
392 (1 + np.cos(np.pi * (1 - progress)))
393 else:
394 self.beta = self.initial_beta
395
396 self.beta = max(self.min_beta, min(self.max_beta, self.beta))
