101 def fit(
102 self,
103 train_loader: DataLoader,
104 val_loader: Optional[DataLoader] = None,
105 epochs: int = 100,
106 loss_fn: Optional[Callable] = None,
107 eval_metric: str = "accuracy",
108 early_stopping_patience: int = 20,
109 save_best: bool = True,
110 save_path: Optional[str] = None
111 ) -> Dict[str, List[float]]:
112 """
113 Train the model with dynamic architecture optimization.
114
115 Args:
116 train_loader: Training data loader
117 val_loader: Validation data loader
118 epochs: Number of training epochs
119 loss_fn: Loss function (will use CrossEntropyLoss for classification)
120 eval_metric: Evaluation metric for model selection
121 early_stopping_patience: Patience for early stopping
122 save_best: Whether to save best model
123 save_path: Path to save best model
124
125 Returns:
126 Training history dictionary
127 """
128 # Setup loss function
129 if loss_fn is None:
130 loss_fn = nn.CrossEntropyLoss()
131
132 # Early stopping setup
133 patience_counter = 0
134 best_val_metric = 0.0 if eval_metric == "accuracy" else float('inf')
135
136 if self.verbose:
137 self.logger.info(f"Starting training for {epochs} epochs")
138 self.logger.info(f"Device: {self.device}")
139 self.logger.info(f"Model parameters: {sum(p.numel() for p in self.wrapped_model.parameters())}")
140
141 for epoch in range(epochs):
142 self.current_epoch = epoch
143
144 # Update epoch in NeuroExapt
145 self.neuro_exapt.update_epoch(epoch, epochs)
146
147 # Training phase
148 train_metrics = self._train_epoch(train_loader, loss_fn)
149
150 # Validation phase
151 if val_loader is not None:
152 val_metrics = self._validate_epoch(val_loader, loss_fn)
153 else:
154 val_metrics = {}
155
156 # Log metrics
157 self._log_epoch_metrics(epoch, train_metrics, val_metrics)
158
159 # Structure evolution
160 if epoch % self.evolution_frequency == 0 and epoch > 0:
161 self._evolve_structure(train_metrics, val_metrics)
162
163 # Learning rate scheduling
164 if self.scheduler is not None:
165 if hasattr(self.scheduler, 'step'):
166 if 'ReduceLROnPlateau' in str(type(self.scheduler)):
167 val_loss = val_metrics.get('loss', train_metrics['loss'])
168 self.scheduler.step(val_loss)
169 else:
170 self.scheduler.step()
171
172 # Early stopping check
173 current_val_metric = val_metrics.get(eval_metric, train_metrics.get(eval_metric, 0.0))
174
175 if eval_metric == "accuracy":
176 is_better = current_val_metric > best_val_metric
177 else:
178 is_better = current_val_metric < best_val_metric
179
180 if is_better:
181 best_val_metric = current_val_metric
182 patience_counter = 0
183
184 if save_best and save_path:
185 self._save_checkpoint(save_path, epoch, train_metrics, val_metrics)
186 else:
187 patience_counter += 1
188
189 if patience_counter >= early_stopping_patience:
190 if self.verbose:
191 self.logger.info(f"Early stopping at epoch {epoch}")
192 break
193
194 if self.verbose:
195 self.logger.info("Training completed")
196
197 return dict(self.training_history)

199 def _train_epoch(self, train_loader: DataLoader, loss_fn: Callable) -> Dict[str, float]:
200 """Train for one epoch."""
201 self.wrapped_model.train()
202
203 total_loss = 0.0
204 total_info_loss = 0.0
205 correct = 0
206 total = 0
207
208 progress_bar = tqdm(train_loader, disable=not self.verbose, desc=f"Epoch {self.current_epoch}")
209
210 for batch_idx, (data, target) in enumerate(progress_bar):
211 data, target = data.to(self.device), target.to(self.device)
212
213 # Zero gradients
214 self.optimizer.zero_grad()
215
216 # Forward pass
217 output = self.wrapped_model(data)
218
219 # Compute combined loss
220 loss = self.wrapped_model.compute_loss(output, target, loss_fn)
221
222 # Backward pass
223 loss.backward()
224 self.optimizer.step()
225
226 # Metrics
227 total_loss += loss.item()
228
229 # Accuracy calculation (assuming classification)
230 if len(target.shape) == 1: # Classification
231 pred = output.argmax(dim=1)
232 correct += pred.eq(target).sum().item()
233 total += target.size(0)
234
235 # Update progress bar
236 if batch_idx % 50 == 0:
237 progress_bar.set_postfix({
238 'Loss': f'{loss.item():.4f}',
239 'Acc': f'{100.0 * correct / total:.2f}%' if total > 0 else 'N/A'
240 })
241
242 # Calculate epoch metrics
243 epoch_loss = total_loss / len(train_loader)
244 epoch_accuracy = 100.0 * correct / total if total > 0 else 0.0
245
246 metrics = {
247 'loss': epoch_loss,
248 'accuracy': epoch_accuracy
249 }
250
251 # Add information-theoretic metrics
252 info_metrics = self._calculate_info_metrics()
253 metrics.update(info_metrics)
254
255 return metrics

260

263 total = 0

272 total_loss += loss.item()
273
274 # Accuracy calculation

278 total += target.size(0)

280 epoch_loss = total_loss / len(val_loader)

288 def _calculate_info_metrics(self) -> Dict[str, float]:
289 """Calculate information-theoretic metrics."""
290 metrics = {}
291
292 try:
293 # Entropy metrics
294 entropy_history = self.neuro_exapt.entropy_ctrl.entropy_history
295 if entropy_history:
296 metrics['entropy'] = entropy_history[-1]
297 metrics['entropy_threshold'] = self.neuro_exapt.entropy_ctrl.threshold
298
299 # Network complexity
300 complexity = self.neuro_exapt.info_theory._activation_cache.get('network_complexity', 0)
301 metrics['complexity'] = complexity
302
303 # Layer importance variance (measure of information distribution)
304 layer_importances = self.neuro_exapt.info_theory._activation_cache.get('layer_importances', {})
305 if layer_importances:
306 importance_values = list(layer_importances.values())
307 metrics['importance_variance'] = np.var(importance_values)
308 metrics['importance_mean'] = np.mean(importance_values)
309
310 except Exception as e:
311 if self.verbose:
312 self.logger.warning(f"Could not calculate info metrics: {e}")
313
314 return metrics
