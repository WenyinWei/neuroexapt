35 def __init__(self, preserve_ratio: float = 0.95, noise_scale: float = 1e-4):
36 """
37 Args:
38 preserve_ratio: 权重保持的比例，用于通道扩展时的权重复制
39 noise_scale: 添加的微小随机噪声规模，避免完全对称

44 def identity_layer_init(self, layer: nn.Module, input_shape: Optional[Tuple[int, ...]] = None) -> nn.Module:
45 """
46 恒等映射初始化
47
48 将新增层初始化为恒等映射，确保层的输出 = 输入
49
50 Args:
51 layer: 要初始化的层
52 input_shape: 输入张量的形状
53
54 Returns:
55 初始化后的层
56 """
57 if isinstance(layer, nn.Conv2d):
58 return self._identity_conv2d_init(layer)
59 elif isinstance(layer, nn.Linear):
60 return self._identity_linear_init(layer)
61 elif isinstance(layer, nn.BatchNorm2d):
62 return self._identity_batchnorm_init(layer)
63 else:
64 logger.warning(f"Identity initialization not implemented for {type(layer)}")
65 return layer

67 def _identity_conv2d_init(self, conv: nn.Conv2d) -> nn.Conv2d:
68 """卷积层恒等映射初始化"""
69 with torch.no_grad():
70 # 重置权重
71 conv.weight.zero_()
72
73 # 设置中心权重为1 (对于3x3卷积，中心为(1,1))
74 if conv.kernel_size == (3, 3):
75 center = (1, 1)
76 elif conv.kernel_size == (1, 1):
77 center = (0, 0)
78 else:
79 center = tuple(k//2 for k in conv.kernel_size)
80
81 # 只在输入输出通道数相同时设置恒等
82 min_channels = min(conv.in_channels, conv.out_channels)
83 for i in range(min_channels):
84 conv.weight[i, i, center[0], center[1]] = 1.0
85
86 # 如果有偏置，设为零
87 if conv.bias is not None:
88 conv.bias.zero_()
89
90 logger.debug(f"Initialized Conv2d {conv.in_channels}->{conv.out_channels} as identity")
91 return conv
92
93 def _identity_linear_init(self, linear: nn.Linear) -> nn.Linear:
94 """线性层恒等映射初始化"""
95 with torch.no_grad():
96 # 重置为零
97 linear.weight.zero_()
98
99 # 设置对角线为1
100 min_dim = min(linear.in_features, linear.out_features)
101 for i in range(min_dim):
102 linear.weight[i, i] = 1.0
103
104 # 偏置设为零
105 if linear.bias is not None:
106 linear.bias.zero_()
107
108 logger.debug(f"Initialized Linear {linear.in_features}->{linear.out_features} as identity")
109 return linear

111 def _identity_batchnorm_init(self, bn: nn.BatchNorm2d) -> nn.BatchNorm2d:
112 """BatchNorm层恒等映射初始化"""
113 with torch.no_grad():
114 # 设置为恒等变换
115 bn.weight.fill_(1.0)
116 bn.bias.zero_()
117
118 # 重置运行统计
119 if bn.running_mean is not None:
120 bn.running_mean.zero_()
121 if bn.running_var is not None:
122 bn.running_var.fill_(1.0)
123
124 logger.debug(f"Initialized BatchNorm2d as identity")
125 return bn

127 def expand_channels_preserving(self, old_layer: nn.Module, new_channels: int,
128 expansion_strategy: str = "replicate") -> nn.Module:
129 """
130 通道扩展的函数保持初始化
131
132 Args:
133 old_layer: 原始层
134 new_channels: 新的通道数
135 expansion_strategy: 扩展策略 ("replicate", "interpolate", "zero_pad")
136
137 Returns:
138 扩展后的层
139 """
140 if isinstance(old_layer, nn.Conv2d):
141 return self._expand_conv2d_channels(old_layer, new_channels, expansion_strategy)
142 elif isinstance(old_layer, nn.BatchNorm2d):
143 return self._expand_batchnorm_channels(old_layer, new_channels, expansion_strategy)
144 else:
145 logger.warning(f"Channel expansion not implemented for {type(old_layer)}")
146 return old_layer

148 def _expand_conv2d_channels(self, old_conv: nn.Conv2d, new_out_channels: int,
149 strategy: str) -> nn.Conv2d:
150 """卷积层通道扩展"""
151 old_out_channels = old_conv.out_channels
152
153 if new_out_channels <= old_out_channels:
154 logger.warning("New channel count should be larger than old count")
155 return old_conv
156
157 # 创建新的卷积层
158 new_conv = nn.Conv2d(
159 old_conv.in_channels,
160 new_out_channels,
161 old_conv.kernel_size,
162 old_conv.stride,
163 old_conv.padding,
164 old_conv.dilation,
165 old_conv.groups,
166 old_conv.bias is not None
167 )
168
169 with torch.no_grad():
170 # 复制原有权重
171 new_conv.weight[:old_out_channels] = old_conv.weight.clone()
172
173 # 扩展策略
174 if strategy == "replicate":
175 # 复制现有通道
176 expand_count = new_out_channels - old_out_channels
177 replicate_indices = torch.randperm(old_out_channels)[:expand_count]
178 new_conv.weight[old_out_channels:] = old_conv.weight[replicate_indices].clone()
179
180 # 添加微小噪声避免完全对称
181 noise = torch.randn_like(new_conv.weight[old_out_channels:]) * self.noise_scale
182 new_conv.weight[old_out_channels:] += noise
183
184 elif strategy == "zero_pad":
185 # 新通道初始化为零
186 new_conv.weight[old_out_channels:].zero_()
187
188 elif strategy == "interpolate":
189 # 使用现有通道的插值
190 for i in range(old_out_channels, new_out_channels):
191 # 在现有通道间插值
192 alpha = (i - old_out_channels) / (new_out_channels - old_out_channels)
193 idx1 = min(int(alpha * old_out_channels), old_out_channels - 1)
194 idx2 = min(idx1 + 1, old_out_channels - 1)
195 weight1, weight2 = 1 - alpha, alpha
196
197 new_conv.weight[i] = (weight1 * old_conv.weight[idx1] +
198 weight2 * old_conv.weight[idx2])
199
200 # 复制偏置
201 if old_conv.bias is not None:
202 new_conv.bias[:old_out_channels] = old_conv.bias.clone()
203 if strategy == "replicate":
204 new_conv.bias[old_out_channels:] = old_conv.bias[replicate_indices].clone()
205 else:
206 new_conv.bias[old_out_channels:].zero_()
207
208 logger.info(f"Expanded Conv2d channels: {old_out_channels} -> {new_out_channels} "
209 f"using {strategy} strategy")
210 return new_conv
211
212 def _expand_batchnorm_channels(self, old_bn: nn.BatchNorm2d, new_channels: int,
213 strategy: str) -> nn.BatchNorm2d:
214 """BatchNorm层通道扩展"""
215 old_channels = old_bn.num_features
216
217 if new_channels <= old_channels:
218 return old_bn
219
220 new_bn = nn.BatchNorm2d(new_channels, old_bn.eps, old_bn.momentum,
221 old_bn.affine, old_bn.track_running_stats)
222
223 with torch.no_grad():
224 if old_bn.affine:
225 # 复制权重和偏置
226 new_bn.weight[:old_channels] = old_bn.weight.clone()
227 new_bn.bias[:old_channels] = old_bn.bias.clone()
228
229 # 新通道初始化
230 new_bn.weight[old_channels:].fill_(1.0)
231 new_bn.bias[old_channels:].zero_()
232
233 if old_bn.track_running_stats:
234 # 复制运行统计
235 new_bn.running_mean[:old_channels] = old_bn.running_mean.clone()
236 new_bn.running_var[:old_channels] = old_bn.running_var.clone()
237
238 # 新通道统计初始化
239 new_bn.running_mean[old_channels:].zero_()
240 new_bn.running_var[old_channels:].fill_(1.0)
241
242 new_bn.num_batches_tracked = old_bn.num_batches_tracked
243
244 logger.info(f"Expanded BatchNorm channels: {old_channels} -> {new_channels}")
245 return new_bn

247 def zero_branch_init(self, branch: nn.Module) -> nn.Module:
248 """
249 分支零初始化
250
251 将新增分支的最后一层初始化为零权重，使其在诞生瞬间不影响主干流
252 """
253 # 找到分支的最后一层
254 last_layer = None
255 for module in branch.modules():
256 if isinstance(module, (nn.Conv2d, nn.Linear)):
257 last_layer = module
258
259 if last_layer is not None:
260 with torch.no_grad():
261 last_layer.weight.zero_()
262 if last_layer.bias is not None:
263 last_layer.bias.zero_()
264 logger.info(f"Zero-initialized branch output layer: {type(last_layer)}")
265
266 return branch

268 def smooth_parameter_transfer(self, old_model: nn.Module, new_model: nn.Module,
269 layer_mapping: Dict[str, str]) -> nn.Module:
270 """
271 平滑参数传递
272
273 将旧模型的参数平滑传递到新模型中，支持形状变化的情况
274
275 Args:
276 old_model: 旧模型
277 new_model: 新模型
278 layer_mapping: 层名称映射 {"new_layer_name": "old_layer_name"}
279
280 Returns:
281 参数传递后的新模型
282 """
283 old_state_dict = old_model.state_dict()
284 new_state_dict = new_model.state_dict()
285
286 transferred_count = 0
287
288 for new_name, old_name in layer_mapping.items():
289 if old_name in old_state_dict and new_name in new_state_dict:
290 old_param = old_state_dict[old_name]
291 new_param = new_state_dict[new_name]
292
293 if old_param.shape == new_param.shape:
294 # 直接复制
295 new_state_dict[new_name] = old_param.clone()
296 transferred_count += 1
297
298 elif len(old_param.shape) == len(new_param.shape):
299 # 形状不同但维度相同，尝试部分复制
300 min_shape = tuple(min(old_param.shape[i], new_param.shape[i])
301 for i in range(len(old_param.shape)))
302
303 # 创建索引切片
304 slices = tuple(slice(0, s) for s in min_shape)
305
306 with torch.no_grad():
307 new_param[slices] = old_param[slices]
308
309 transferred_count += 1
310 logger.info(f"Partially transferred {old_name} -> {new_name}: "
311 f"{old_param.shape} -> {new_param.shape}")
312 else:
313 logger.warning(f"Cannot transfer {old_name} -> {new_name}: "
314 f"incompatible shapes {old_param.shape} vs {new_param.shape}")
315
316 new_model.load_state_dict(new_state_dict)
317 logger.info(f"Successfully transferred {transferred_count} parameters")
318
319 return new_model

322def create_identity_residual_block(in_channels: int, out_channels: int) -> nn.Module:
323 """
324 创建恒等残差块
325
326 用于添加新层时的函数保持初始化
327 """
328 if in_channels == out_channels:
329 # 直接恒等连接
330 return nn.Identity()
331 else:
332 # 需要通道调整的恒等连接
