71 def _select_best_gpu(self) -> int:
72 """选择最优GPU"""
73 best_gpu = 0
74 max_memory = 0
75
76 for i in range(torch.cuda.device_count()):
77 memory = torch.cuda.get_device_properties(i).total_memory
78 if memory > max_memory:
79 max_memory = memory
80 best_gpu = i
81
82 logger.info(f"🎯 Selected GPU {best_gpu} with {max_memory / 1024**3:.1f}GB memory")
83 return best_gpu

99 def register_model(self, name: str, model: nn.Module) -> nn.Module:
100 """
101 注册并转移模型到目标设备
102
103 Args:
104 name: 模型名称
105 model: 模型实例
106
107 Returns:
108 转移到目标设备的模型
109 """
110 logger.info(f"📋 Registering model '{name}' to {self.device}")
111
112 # 转移到目标设备
113 model = model.to(self.device)
114
115 # 注册模型
116 self.registered_models[name] = model
117
118 # 记录模型参数量
119 param_count = sum(p.numel() for p in model.parameters())
120 param_size_mb = sum(p.numel() * p.element_size() for p in model.parameters()) / 1024**2
121
122 logger.info(f" Parameters: {param_count:,} ({param_size_mb:.1f}MB)")
123
124 return model

126 def to_device(self, data: Any) -> Any:
127 """
128 将数据转移到目标设备
129
130 Args:
131 data: 要转移的数据
132
133 Returns:
134 转移到目标设备的数据
135 """
136 if isinstance(data, torch.Tensor):
137 return data.to(self.device, non_blocking=True)
138 elif isinstance(data, (list, tuple)):
139 return type(data)(self.to_device(item) for item in data)
140 elif isinstance(data, dict):
141 return {key: self.to_device(value) for key, value in data.items()}
142 else:
143 return data

159 def transfer_model_state(self, source_model: nn.Module,
160 target_model: nn.Module) -> nn.Module:
161 """
162 在同一设备上转移模型状态
163
164 Args:
165 source_model: 源模型
166 target_model: 目标模型
167
168 Returns:
169 加载了状态的目标模型
170 """
171 # 确保两个模型在同一设备上
172 source_model = source_model.to(self.device)
173 target_model = target_model.to(self.device)
174
175 source_dict = source_model.state_dict()
176 target_dict = target_model.state_dict()
177
178 # 逐层转移兼容的参数
179 transferred = 0
180 for key in target_dict.keys():
181 if key in source_dict:
182 source_param = source_dict[key]
183 target_param = target_dict[key]
184
185 if source_param.shape == target_param.shape:
186 target_dict[key] = source_param.clone()
187 transferred += 1
188 else:
189 logger.warning(f"Shape mismatch for {key}: "
190 f"{source_param.shape} -> {target_param.shape}")
191
192 target_model.load_state_dict(target_dict)
193 logger.info(f"✅ Transferred {transferred} parameters between models")
194
195 return target_model

197 def optimize_memory(self):
198 """优化内存使用"""
199 if self.device.type == 'cuda':
200 # 清空GPU缓存
201 torch.cuda.empty_cache()
202
203 # 触发垃圾回收
204 gc.collect()
205
206 # 记录内存使用
207 current_memory = torch.cuda.memory_allocated(self.device) / 1024**2
208 self.memory_history.append(current_memory)
209
210 if current_memory > self.peak_memory:
211 self.peak_memory = current_memory
212
213 logger.debug(f"💾 Memory: {current_memory:.1f}MB (Peak: {self.peak_memory:.1f}MB)")
214
215 def get_memory_stats(self) -> Dict[str, Union[float, str]]:
216 """获取内存统计信息"""
217 if self.device.type == 'cuda':
218 allocated = torch.cuda.memory_allocated(self.device) / 1024**2
219 cached = torch.cuda.memory_reserved(self.device) / 1024**2
220 total = torch.cuda.get_device_properties(self.device).total_memory / 1024**2
221
222 return {
223 'allocated_mb': allocated,
224 'cached_mb': cached,
225 'total_mb': total,
226 'peak_mb': self.peak_memory,
227 'utilization': allocated / total
228 }
229 else:
230 return {'device': 'cpu'}

251 def safe_model_creation(self, model_class, *args, **kwargs) -> nn.Module:
252 """安全的模型创建，自动处理设备和内存"""
253 try:
254 # 直接在目标设备上创建模型
255 model = model_class(*args, **kwargs)
256
257 # 确保在正确设备上
258 model = model.to(self.device)
259
260 # 优化内存
261 self.optimize_memory()
262
263 return model
264
265 except RuntimeError as e:
266 if "out of memory" in str(e).lower():
267 logger.error("🚨 GPU memory insufficient for model creation")
268 self.optimize_memory()
269
270 # 尝试在CPU上创建
271 logger.warning("🔄 Falling back to CPU")
272 original_device = self.device
273 self.device = torch.device('cpu')
274 try:
275 model = model_class(*args, **kwargs)
276 return model
277 except Exception as e2:
278 # 恢复原设备设置
279 self.device = original_device
280 raise e2

282 raise e
