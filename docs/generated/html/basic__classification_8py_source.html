46class FixedCell(nn.Module):
47 """
48 Fixed architecture cell that bypasses MixedOp computation
49 """
50 def __init__(self, steps, block_multiplier, C_prev_prev, C_prev, C, reduction, reduction_prev):
51 super(FixedCell, self).__init__()
52 self.reduction = reduction
53 self.block_multiplier = block_multiplier
54 self.steps = steps
55
56 # Preprocessing layers
57 if reduction_prev:
58 self.preprocess0 = FactorizedReduce(C_prev_prev, C, affine=False)
59 else:
60 self.preprocess0 = ReLUConvBN(C_prev_prev, C, 1, 1, 0, affine=False)
61 self.preprocess1 = ReLUConvBN(C_prev, C, 1, 1, 0, affine=False)
62
63 # Fixed operations (high-performance genotype)
64 self.ops = nn.ModuleList()
65
66 # Create operations for each connection
67 for i in range(self.steps):
68 for j in range(2 + i):
69 # Choose operation type
70 if reduction:
71 if j < 2:
72 op_name = 'max_pool_3x3'
73 else:
74 op_name = 'sep_conv_3x3'
75 else:
76 if j == 0:
77 op_name = 'sep_conv_3x3'
78 elif j == 1:
79 op_name = 'sep_conv_5x5'
80 else:
81 op_name = 'sep_conv_3x3'
82
83 stride = 2 if reduction and j < 2 else 1
84 op = self._create_operation(op_name, C, stride)
85 self.ops.append(op)
86
87 def _create_operation(self, op_name, C, stride):
88 """Create a single operation"""
89 if op_name == 'none':
90 return Zero(stride)
91 elif op_name == 'avg_pool_3x3':
92 return nn.Sequential(
93 nn.AvgPool2d(3, stride=stride, padding=1, count_include_pad=False),
94 nn.BatchNorm2d(C, affine=False)
95 )
96 elif op_name == 'max_pool_3x3':
97 return nn.Sequential(
98 nn.MaxPool2d(3, stride=stride, padding=1),
99 nn.BatchNorm2d(C, affine=False)
100 )
101 elif op_name == 'skip_connect':
102 return Identity() if stride == 1 else FactorizedReduce(C, C, affine=False)
103 elif op_name == 'sep_conv_3x3':
104 return SepConv(C, C, 3, stride, 1, affine=False)
105 elif op_name == 'sep_conv_5x5':
106 return SepConv(C, C, 5, stride, 2, affine=False)
107 else:
108 return Identity()
109
110 def forward(self, s0, s1):
111 """Forward pass without architecture parameters"""
112 s0 = self.preprocess0(s0)
113 s1 = self.preprocess1(s1)
114
115 states = [s0, s1]
116 offset = 0
117
118 for i in range(self.steps):
119 s = 0
120 for j in range(2 + i):
121 op = self.ops[offset + j]
122 h = op(states[j])
123 s = s + h
124
125 offset += len(states)
126 states.append(s)
127
128 return torch.cat(states[-self.block_multiplier:], dim=1)

130class FixedNetwork(nn.Module):
131 """
132 Fixed architecture network that bypasses architecture search
133 """
134 def __init__(self, C, num_classes, layers, steps=4, block_multiplier=4):
135 super(FixedNetwork, self).__init__()
136 self._C = C
137 self._num_classes = num_classes
138 self._layers = layers
139 self._steps = steps
140 self._block_multiplier = block_multiplier
141
142 C_curr = self._block_multiplier * C
143 self.stem = nn.Sequential(
144 nn.Conv2d(3, C_curr, 3, padding=1, bias=False),
145 nn.BatchNorm2d(C_curr)
146 )
147
148 C_prev_prev, C_prev, C_curr = C_curr, C_curr, C
149 self.cells = nn.ModuleList()
150 reduction_prev = False
151
152 for i in range(layers):
153 if i in [layers // 3, 2 * layers // 3]:
154 C_curr *= 2
155 reduction = True
156 else:
157 reduction = False
158
159 cell = FixedCell(steps, block_multiplier, C_prev_prev, C_prev, C_curr, reduction, reduction_prev)
160 reduction_prev = reduction
161 self.cells.append(cell)
162 C_prev_prev, C_prev = C_prev, self._block_multiplier * C_curr
163
164 self.global_pooling = nn.AdaptiveAvgPool2d(1)
165 self.classifier = nn.Linear(C_prev, num_classes)
166
167 def forward(self, input):
168 """Fast forward pass"""
169 s0 = s1 = self.stem(input)
170
171 for cell in self.cells:
172 s0, s1 = s1, cell(s0, s1)
173
174 out = self.global_pooling(s1)
175 logits = self.classifier(out.view(out.size(0), -1))
176 return logits

178class CachedNetwork:
179 """Wrapper for Network class to provide caching functionality"""
180 def __init__(self, model):
181 self.model = model
182 self.cached_normal_weights = None
183 self.cached_reduce_weights = None
184 self.cache_valid = False
185
186 def update_weight_cache(self):
187 """Update weight cache"""
188 with torch.no_grad():
189 self.cached_normal_weights = torch.softmax(self.model.alphas_normal, dim=-1)
190 self.cached_reduce_weights = torch.softmax(self.model.alphas_reduce, dim=-1)
191 self.cache_valid = True
192
193 def forward(self, input):
194 """Forward pass with cached weights"""
195 if not self.cache_valid:
196 self.update_weight_cache()
197
198 # Use cached weights directly in Network's forward logic
199 import torch.utils.checkpoint as cp
200 from neuroexapt.core.model import GatedCell
201
202 s0 = s1 = self.model.stem(input)
203
204 for i, cell in enumerate(self.model.cells):
205 # Determine which set of precomputed weights to use
206 if isinstance(cell, GatedCell):
207 if cell.cell.reduction:
208 weights = self.cached_reduce_weights
209 else:
210 weights = self.cached_normal_weights
211 else:
212 weights = self.cached_reduce_weights if cell.reduction else self.cached_normal_weights
213
214 if self.model.use_checkpoint and self.model.training:
215 # Wrap cell forward in checkpoint to save memory
216 def _cell_forward(a, b):
217 return cell(a, b, weights)
218
219 s1_new = cp.checkpoint(_cell_forward, s0, s1)
220 else:
221 s1_new = cell(s0, s1, weights)
222
223 s0, s1 = s1, s1_new
224
225 out = self.model.global_pooling(s1)
226 logits = self.model.classifier(out.view(out.size(0), -1))
227 return logits
228
229 def __call__(self, input):
230 return self.forward(input)

232 def __getattr__(self, name):
233 # 🔧 修复：防止__getattr__无穷递归
234 # 添加黑名单和递归深度检查
235 if name.startswith('_') or name in {'model', 'cached_normal_weights', 'cached_reduce_weights', 'cache_valid'}:
236 raise AttributeError(f"'{type(self).__name__}' object has no attribute '{name}'")

238 # 防止递归：检查是否正在查找model属性

242 if name in self._getattr_in_progress:
243 raise AttributeError(f"递归调用检测：'{name}' 属性查找循环")

245 try:
246 self._getattr_in_progress.add(name)
247 # 安全地获取model属性

249 return getattr(self.model, name)
250 else:
251 raise AttributeError(f"'{type(self).__name__}' object has no attribute '{name}'")
252 finally:
253 self._getattr_in_progress.discard(name)

637def train_epoch(train_queue, valid_queue, model, architect, criterion, optimizer, epoch, args, monitor=None):
638 """Optimized train epoch with minimal output overhead"""
639 model.train()
640
641 # 重置模型的epoch计数器
642 if hasattr(model, 'reset_epoch_counters'):
643 model.reset_epoch_counters()
644
645 objs = AvgrageMeter()
646 top1 = AvgrageMeter()
647 top5 = AvgrageMeter()
648
649 # Performance tracking
650 batch_times = []
651
652 # For architecture search modes
653 if architect is not None:
654 architect.set_epoch(epoch)
655 valid_iter = iter(valid_queue)
656 arch_updates = 0
657
658 start_time = time.time()
659
660 # 极简进度条
661 total_batches = len(train_queue)
662 quiet_mode = getattr(args, 'quiet', False)
663
664 if quiet_mode:
665 # 最小化的进度条
666 pbar = tqdm(enumerate(train_queue), total=total_batches,
667 desc=f"E{epoch}",
668 bar_format='{desc}: {percentage:3.0f}%|{bar:15}| {n_fmt}/{total_fmt} {rate_fmt}',
669 mininterval=2.0) # 减少更新频率
670 else:
671 # 正常模式也简化
672 pbar = tqdm(enumerate(train_queue), total=total_batches,
673 desc=f"Epoch {epoch}",
674 bar_format='{desc}: {percentage:3.0f}%|{bar:20}| {n_fmt}/{total_fmt}',
675 mininterval=1.0)
676
677 for step, (input, target) in pbar:
678 batch_start_time = time.perf_counter()
679
680 n = input.size(0)
681 input = input.cuda(non_blocking=True)
682 target = target.cuda(non_blocking=True)
683
684 # Architecture evolution step - 简化逻辑
685 if architect is not None:
686 should_update = (step + 1) % getattr(args, 'arch_update_freq', 50) == 0 and epoch >= getattr(args, 'warmup_epochs', 5)
687
688 if should_update:
689 try:
690 input_search, target_search = next(valid_iter)
691 except StopIteration:
692 valid_iter = iter(valid_queue)
693 input_search, target_search = next(valid_iter)
694
695 input_search = input_search.cuda(non_blocking=True)
696 target_search = target_search.cuda(non_blocking=True)
697
698 # Invalidate cache if using cached mode
699 if hasattr(model, 'cache_valid'):
700 model.cache_valid = False
701
702 architect.step(input, target, input_search, target_search,
703 optimizer.param_groups[0]['lr'], optimizer, args.use_first_order)
704 arch_updates += 1
705
706 # Weight update step
707 optimizer.zero_grad()
708 logits = model(input)
709 loss = criterion(logits, target)
710 loss.backward()
711 nn.utils.clip_grad_norm_(model.parameters(), args.grad_clip)
712 optimizer.step()
713
714 # Statistics
715 prec1, prec5 = accuracy(logits, target, topk=(1, 5))
716 objs.update(loss.item(), n)
717 top1.update(prec1.item(), n)
718 top5.update(prec5.item(), n)
719
720 batch_time = time.perf_counter() - batch_start_time
721 batch_times.append(batch_time)
722
723 # 极简的进度条更新（移除所有复杂逻辑）
724 if step % 100 == 0: # 减少更新频率
725 pbar.set_postfix({
726 'Loss': f'{objs.avg:.3f}',
727 'Acc': f'{top1.avg:.1f}%'
728 })
729
730 # GPU缓存清理（减少频率）
731 if step % 500 == 0:
732 torch.cuda.empty_cache()
733
734 pbar.close()
735 epoch_time = time.time() - start_time
736
737 # 极简epoch总结
738 avg_batch_time = sum(batch_times) / len(batch_times) if batch_times else 0
739
740 if quiet_mode:
741 print(f"E{epoch}: {objs.avg:.3f} {top1.avg:.1f}% {epoch_time:.0f}s")
742 else:
743 print(f"✅ Epoch {epoch}: Loss={objs.avg:.3f} Acc={top1.avg:.1f}% Time={epoch_time:.0f}s")
744 if architect and arch_updates > 0:
745 print(f" 🧬 架构更新: {arch_updates}次")
746
747 return top1.avg, objs.avg, epoch_time
