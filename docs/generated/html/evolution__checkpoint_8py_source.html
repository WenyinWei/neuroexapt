73 def __init__(self, experiment_name: str, base_dir: str = "./evolution_checkpoints"):
74 self.experiment_name = experiment_name

96 def _generate_checkpoint_id(self, epoch: int, growth_type: Optional[str] = None) -> str:
97 """生成checkpoint ID"""
98 timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
99 if growth_type:
100 return f"epoch_{epoch:03d}_{growth_type}_{timestamp}"
101 else:
102 return f"epoch_{epoch:03d}_baseline_{timestamp}"

104 def save_checkpoint(self,
105 network: nn.Module,
106 optimizer: torch.optim.Optimizer,
107 scheduler: Optional[torch.optim.lr_scheduler._LRScheduler],
108 epoch: int,
109 training_stats: Dict,
110 growth_type: Optional[str] = None,
111 parent_id: Optional[str] = None) -> str:
112 """
113 保存架构进化检查点
114
115 Args:
116 network: 神经网络模型
117 optimizer: 优化器
118 scheduler: 学习率调度器
119 epoch: 当前epoch
120 training_stats: 训练统计信息
121 growth_type: 生长类型 ('grow_depth', 'grow_width', 'grow_branches')
122 parent_id: 父节点checkpoint ID
123
124 Returns:
125 checkpoint_id: 新创建的checkpoint ID
126 """
127
128 checkpoint_id = self._generate_checkpoint_id(epoch, growth_type)
129
130 # 获取架构摘要
131 if hasattr(network, 'get_architecture_summary'):
132 arch_summary = network.get_architecture_summary()
133 else:
134 arch_summary = {
135 'total_parameters': sum(p.numel() for p in network.parameters()),
136 'epoch': epoch
137 }
138
139 # 创建检查点
140 checkpoint = EvolutionCheckpoint(
141 checkpoint_id=checkpoint_id,
142 epoch=epoch,
143 network_state=network.state_dict(),
144 optimizer_state=optimizer.state_dict(),
145 scheduler_state=scheduler.state_dict() if scheduler else None,
146 training_stats=training_stats.copy(),
147 architecture_summary=arch_summary,
148 parent_id=parent_id or self.current_checkpoint_id,
149 growth_type=growth_type
150 )
151
152 # 保存到文件
153 checkpoint_path = self.checkpoint_dir / f"{checkpoint_id}.pt"
154 torch.save({
155 'network_state': checkpoint.network_state,
156 'optimizer_state': checkpoint.optimizer_state,
157 'scheduler_state': checkpoint.scheduler_state,
158 'metadata': checkpoint.to_dict()
159 }, checkpoint_path)
160
161 # 更新进化树
162 self.evolution_tree[checkpoint_id] = checkpoint
163 self.checkpoint_order.append(checkpoint_id)
164 self.current_checkpoint_id = checkpoint_id
165
166 # 保存元数据
167 self._save_evolution_history()
168
169 logger.info(f"✅ Checkpoint saved: {checkpoint_id}")
170 logger.info(f" Epoch: {epoch}, Growth: {growth_type or 'baseline'}")
171 logger.info(f" Parameters: {arch_summary.get('total_parameters', 0):,}")
172 logger.info(f" Best accuracy: {training_stats.get('best_accuracy', 0):.2f}%")
173
174 return checkpoint_id

176 def load_checkpoint(self, checkpoint_id: str) -> Tuple[Dict, Dict, Dict, Dict]:
177 """
178 加载检查点
179
180 Returns:
181 (network_state, optimizer_state, scheduler_state, metadata)
182 """
183
184 if checkpoint_id not in self.evolution_tree:
185 raise ValueError(f"Checkpoint {checkpoint_id} not found")
186
187 checkpoint_path = self.checkpoint_dir / f"{checkpoint_id}.pt"
188 if not checkpoint_path.exists():
189 raise FileNotFoundError(f"Checkpoint file not found: {checkpoint_path}")
190
191 checkpoint_data = torch.load(checkpoint_path, map_location='cpu')
192
193 logger.info(f"📥 Checkpoint loaded: {checkpoint_id}")
194
195 return (
196 checkpoint_data['network_state'],
197 checkpoint_data['optimizer_state'],
198 checkpoint_data['scheduler_state'],
199 checkpoint_data['metadata']
200 )

202 def get_best_checkpoint(self) -> Optional[str]:
203 """获取最佳性能的checkpoint"""
204 if not self.evolution_tree:
205 return None
206
207 best_checkpoint_id = None
208 best_accuracy = -1.0
209
210 for checkpoint_id, checkpoint in self.evolution_tree.items():
211 accuracy = checkpoint.training_stats.get('best_accuracy', 0.0)
212 if accuracy > best_accuracy:
213 best_accuracy = accuracy
214 best_checkpoint_id = checkpoint_id
215
216 return best_checkpoint_id

218 def get_recent_checkpoints(self, n: int = 5) -> List[str]:

222 def rollback_to_checkpoint(self, checkpoint_id: str) -> None:
223 """回滚到指定checkpoint（谱系管理）"""
224 if checkpoint_id not in self.evolution_tree:
225 raise ValueError(f"Checkpoint {checkpoint_id} not found")
226
227 # 找到所有在此checkpoint之后的衍生checkpoints
228 descendants = self._find_descendants(checkpoint_id)
229
230 logger.info(f"🔄 Rolling back to checkpoint: {checkpoint_id}")
231 if descendants:
232 logger.info(f" Will remove {len(descendants)} descendant checkpoints")
233
234 # 删除衍生checkpoints（可选，或者保留作为分支）
235 for desc_id in descendants:
236 if desc_id in self.evolution_tree:
237 checkpoint_path = self.checkpoint_dir / f"{desc_id}.pt"
238 if checkpoint_path.exists():
239 checkpoint_path.unlink() # 删除文件
240 del self.evolution_tree[desc_id]
241 if desc_id in self.checkpoint_order:
242 self.checkpoint_order.remove(desc_id)
243
244 # 更新当前checkpoint
245 self.current_checkpoint_id = checkpoint_id
246
247 # 保存更新后的元数据
248 self._save_evolution_history()
249
250 logger.info(f"✅ Rollback completed to {checkpoint_id}")
251
252 def _find_descendants(self, parent_id: str) -> List[str]:
253 """找到指定checkpoint的所有后代"""
254 descendants = []
255
256 def _find_children(parent: str):
257 for checkpoint_id, checkpoint in self.evolution_tree.items():
258 if checkpoint.parent_id == parent:
259 descendants.append(checkpoint_id)
260 _find_children(checkpoint_id) # 递归查找
261
262 _find_children(parent_id)
263 return descendants
264
265 def get_evolution_lineage(self) -> List[Dict]:
266 """获取进化谱系"""
267 lineage = []
268
269 for checkpoint_id in self.checkpoint_order:
270 if checkpoint_id in self.evolution_tree:
271 checkpoint = self.evolution_tree[checkpoint_id]
272 lineage.append({
273 'id': checkpoint_id,
274 'epoch': checkpoint.epoch,
275 'growth_type': checkpoint.growth_type,
276 'parent_id': checkpoint.parent_id,
277 'accuracy': checkpoint.training_stats.get('best_accuracy', 0.0),
278 'parameters': checkpoint.architecture_summary.get('total_parameters', 0),
279 'depth': checkpoint.architecture_summary.get('depth', 0),
280 'created_time': checkpoint.created_datetime
281 })
282
283 return lineage

322 def _save_evolution_history(self):

335 def _load_evolution_history(self):
336 """加载进化历史元数据"""
337 if not self.metadata_file.exists():
338 return
339
340 try:
341 with open(self.metadata_file, 'r') as f:
342 metadata = json.load(f)
343
344 # 重建进化树
345 for checkpoint_id, cp_data in metadata.get('evolution_tree', {}).items():
346 # 注意：这里只加载元数据，不加载实际的模型状态
347 checkpoint = EvolutionCheckpoint(
348 checkpoint_id=cp_data['checkpoint_id'],
349 epoch=cp_data['epoch'],
350 network_state={}, # 实际状态在需要时从文件加载
351 optimizer_state={},
352 scheduler_state={},
353 training_stats=cp_data['training_stats'],
354 architecture_summary=cp_data['architecture_summary'],
355 parent_id=cp_data.get('parent_id'),
356 growth_type=cp_data.get('growth_type')
357 )
358 checkpoint.created_time = cp_data['created_time']
359 checkpoint.created_datetime = cp_data['created_datetime']
360
361 self.evolution_tree[checkpoint_id] = checkpoint
362
363 self.checkpoint_order = metadata.get('checkpoint_order', [])
364 self.current_checkpoint_id = metadata.get('current_checkpoint_id')
365
366 logger.info(f"📚 Loaded evolution history: {len(self.evolution_tree)} checkpoints")
367
368 except Exception as e:
369 logger.warning(f"Failed to load evolution history: {e}")
370 self.evolution_tree = {}
371 self.checkpoint_order = []
372 self.current_checkpoint_id = None

374 def cleanup_old_checkpoints(self, keep_last_n: int = 10, keep_best: bool = True):
375 """清理旧的checkpoints，保留最近的N个和最佳的"""
376 if len(self.checkpoint_order) <= keep_last_n:

389 keep_ids.add(best_id)

391 # 删除其他的

404
405 deleted_count += 1
406

408 logger.info(f"🧹 Cleaned up {deleted_count} old checkpoints")
409 self._save_evolution_history()
