57def quick_evaluate(model, dataloader, device):
58 """Quick evaluation on a few batches."""
59 model.eval()
60 correct = 0
61 total = 0
62
63 with torch.no_grad():
64 for i, (data, targets) in enumerate(dataloader):
65 if i >= 3: # Only use 3 batches for speed
66 break
67 data, targets = data.to(device), targets.to(device)
68 outputs = model(data)
69 _, predicted = torch.max(outputs, 1)
70 total += targets.size(0)
71 correct += (predicted == targets).sum().item()
72
73 model.train()
74 return 100 * correct / total if total > 0 else 0.0
