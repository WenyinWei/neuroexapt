146 def _create_optimizer(self, optimizer_type: str, learning_rate: float) -> optim.Optimizer:
147 """Create optimizer based on type."""
148 optimizer_type = optimizer_type.lower()
149
150 if optimizer_type == 'adam':
151 return optim.Adam(self.model.parameters(), lr=learning_rate)
152 elif optimizer_type == 'sgd':
153 return optim.SGD(self.model.parameters(), lr=learning_rate, momentum=0.9)
154 elif optimizer_type == 'rmsprop':
155 return optim.RMSprop(self.model.parameters(), lr=learning_rate)
156 elif optimizer_type == 'adamw':
157 return optim.AdamW(self.model.parameters(), lr=learning_rate)
158 else:
159 warnings.warn(f"Unknown optimizer type: {optimizer_type}, using Adam")
160 return optim.Adam(self.model.parameters(), lr=learning_rate)
161
162 def _create_criterion(self, criterion_type: str) -> nn.Module:
163 """Create loss function based on type."""
164 criterion_type = criterion_type.lower()
165
166 if criterion_type == 'crossentropy':
167 return nn.CrossEntropyLoss()
168 elif criterion_type == 'mse':
169 return nn.MSELoss()
170 elif criterion_type == 'bce':
171 return nn.BCELoss()
172 elif criterion_type == 'bcewithlogits':
173 return nn.BCEWithLogitsLoss()
174 else:
175 warnings.warn(f"Unknown criterion type: {criterion_type}, using CrossEntropyLoss")
176 return nn.CrossEntropyLoss()

178 def evaluate(self, test_loader: DataLoader) -> Dict[str, float]:
179 """
180 Evaluate the model on test data.
181
182 Args:
183 test_loader: Test data loader
184
185 Returns:
186 Evaluation metrics
187 """
188 self.model.eval()
189 correct = 0
190 total = 0
191 test_loss = 0
192
193 criterion = nn.CrossEntropyLoss()
194
195 with torch.no_grad():
196 for data, targets in test_loader:
197 data, targets = data.to(self.device), targets.to(self.device)
198 outputs = self.model(data)
199 loss = criterion(outputs, targets)
200
201 test_loss += loss.item()
202 _, predicted = outputs.max(1)
203 total += targets.size(0)
204 correct += predicted.eq(targets).sum().item()
205
206 test_accuracy = 100. * correct / total
207 avg_test_loss = test_loss / len(test_loader)
208
209 if self.verbose:
210 print(f"📊 Test Results:")
211 print(f" - Test accuracy: {test_accuracy:.2f}%")
212 print(f" - Test loss: {avg_test_loss:.4f}")
213
214 return {
215 'test_accuracy': test_accuracy,
216 'test_loss': avg_test_loss
217 }
