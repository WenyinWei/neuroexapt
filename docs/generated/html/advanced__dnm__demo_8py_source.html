49class AdaptiveResNet(nn.Module):
50 """增强的自适应ResNet - 冲刺95%准确率"""
51
52 def __init__(self, num_classes=10):
53 super(AdaptiveResNet, self).__init__()
54
55 # 🚀 增强的初始特征提取
56 self.conv1 = nn.Conv2d(3, 64, 3, padding=1, bias=False) # 增加初始通道数
57 self.bn1 = nn.BatchNorm2d(64)
58 self.relu = nn.ReLU(inplace=True)
59
60 # 🚀 更深的特征提取网络
61 self.feature_block1 = self._make_resnet_block(64, 128, 2, 2) # 2个残差块
62 self.feature_block2 = self._make_resnet_block(128, 256, 2, 2) # 2个残差块
63 self.feature_block3 = self._make_resnet_block(256, 512, 2, 2) # 2个残差块
64 self.feature_block4 = self._make_resnet_block(512, 512, 1, 2) # 2个残差块，不降采样
65
66 # 🚀 增强的全局特征聚合
67 self.global_pool = nn.AdaptiveAvgPool2d((1, 1))
68 self.global_max_pool = nn.AdaptiveMaxPool2d((1, 1))
69
70 # 🚀 更强的分类器
71 self.classifier = nn.Sequential(
72 nn.Flatten(),
73 nn.Linear(512 * 2, 1024), # 结合avg和max pooling
74 nn.BatchNorm1d(1024),
75 nn.ReLU(inplace=True),
76 nn.Dropout(0.5),
77
78 nn.Linear(1024, 512),
79 nn.BatchNorm1d(512),
80 nn.ReLU(inplace=True),
81 nn.Dropout(0.3),
82
83 nn.Linear(512, 256),
84 nn.BatchNorm1d(256),
85 nn.ReLU(inplace=True),
86 nn.Dropout(0.2),
87
88 nn.Linear(256, num_classes)
89 )

91 def _make_resnet_block(self, in_channels, out_channels, stride, num_blocks):
92 """创建残差块组"""
93 layers = []
94
95 # 第一个块可能有降采样
96 layers.append(ResidualBlock(in_channels, out_channels, stride))
97
98 # 后续块保持相同尺寸
99 for _ in range(1, num_blocks):
100 layers.append(ResidualBlock(out_channels, out_channels, 1))
101
102 return nn.Sequential(*layers)
103
104 def forward(self, x):
105 # 🚀 增强的初始特征提取
106 x = self.relu(self.bn1(self.conv1(x)))
107
108 # 🚀 深度残差特征提取
109 x = self.feature_block1(x)
110 x = self.feature_block2(x)
111 x = self.feature_block3(x)
112 x = self.feature_block4(x)
113
114 # 🚀 双重全局池化特征聚合
115 avg_pool = self.global_pool(x)
116 max_pool = self.global_max_pool(x)
117 x = torch.cat([avg_pool, max_pool], dim=1) # 特征融合
118
119 # 分类
120 x = self.classifier(x)
121
122 return x

124class ResidualBlock(nn.Module):
125 """残差块实现"""
126
127 def __init__(self, in_channels, out_channels, stride=1):
128 super(ResidualBlock, self).__init__()
129
130 self.conv1 = nn.Conv2d(in_channels, out_channels, 3, stride=stride, padding=1, bias=False)
131 self.bn1 = nn.BatchNorm2d(out_channels)
132 self.conv2 = nn.Conv2d(out_channels, out_channels, 3, stride=1, padding=1, bias=False)
133 self.bn2 = nn.BatchNorm2d(out_channels)
134 self.relu = nn.ReLU(inplace=True)
135
136 # 残差连接
137 self.shortcut = nn.Sequential()
138 if stride != 1 or in_channels != out_channels:
139 self.shortcut = nn.Sequential(
140 nn.Conv2d(in_channels, out_channels, 1, stride=stride, bias=False),
141 nn.BatchNorm2d(out_channels)
142 )
143
144 def forward(self, x):
145 residual = self.shortcut(x)
146
147 out = self.relu(self.bn1(self.conv1(x)))
148 out = self.bn2(self.conv2(out))
149
150 out += residual
151 out = self.relu(out)
152
153 return out

197 def forward_hook(name):
198 def hook(module, input, output):

208 def hook(module, grad_input, grad_output):
209 try:
210 if grad_output[0] is not None:

212 print(f" 📉 反向钩子捕获: {name} - 形状 {grad_output[0].shape}")

275 def train_epoch(self, optimizer, epoch):
276 """训练一个epoch"""
277 self.model.train()
278 total_loss = 0
279 correct = 0
280 total = 0
281
282 for batch_idx, (data, target) in enumerate(self.train_loader):
283 data, target = data.to(self.device), target.to(self.device)
284
285 optimizer.zero_grad()
286 output = self.model(data)
287 loss = F.cross_entropy(output, target)
288 loss.backward()
289 optimizer.step()
290
291 total_loss += loss.item()
292 pred = output.argmax(dim=1, keepdim=True)
293 correct += pred.eq(target.view_as(pred)).sum().item()
294 total += target.size(0)
295
296 if batch_idx % 100 == 0:
297 print(f' Train Batch: {batch_idx:3d}/{len(self.train_loader)}, '
298 f'Loss: {loss.item():.6f}, Acc: {100.*correct/total:.2f}%')
299
300 avg_loss = total_loss / len(self.train_loader)
301 accuracy = 100. * correct / total
302
303 return avg_loss, accuracy
304
305 def test_epoch(self):
306 """测试一个epoch"""
307 self.model.eval()
308 test_loss = 0
309 correct = 0
310
311 with torch.no_grad():
312 for data, target in self.test_loader:
313 data, target = data.to(self.device), target.to(self.device)
314 output = self.model(data)
315 test_loss += F.cross_entropy(output, target, reduction='sum').item()
316 pred = output.argmax(dim=1, keepdim=True)
317 correct += pred.eq(target.view_as(pred)).sum().item()
318
319 test_loss /= len(self.test_loader.dataset)
320 accuracy = 100. * correct / len(self.test_loader.dataset)
321
322 return test_loss, accuracy
