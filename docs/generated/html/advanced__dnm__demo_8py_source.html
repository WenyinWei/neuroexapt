124class ResidualBlock(nn.Module):
125 """æ®‹å·®å—å®ç°"""
126
127 def __init__(self, in_channels, out_channels, stride=1):
128 super(ResidualBlock, self).__init__()
129
130 self.conv1 = nn.Conv2d(in_channels, out_channels, 3, stride=stride, padding=1, bias=False)
131 self.bn1 = nn.BatchNorm2d(out_channels)
132 self.conv2 = nn.Conv2d(out_channels, out_channels, 3, stride=1, padding=1, bias=False)
133 self.bn2 = nn.BatchNorm2d(out_channels)
134 self.relu = nn.ReLU(inplace=True)
135
136 # æ®‹å·®è¿æ¥
137 self.shortcut = nn.Sequential()
138 if stride != 1 or in_channels != out_channels:
139 self.shortcut = nn.Sequential(
140 nn.Conv2d(in_channels, out_channels, 1, stride=stride, bias=False),
141 nn.BatchNorm2d(out_channels)
142 )
143
144 def forward(self, x):
145 residual = self.shortcut(x)
146
147 out = self.relu(self.bn1(self.conv1(x)))
148 out = self.bn2(self.conv2(out))
149
150 out += residual
151 out = self.relu(out)
152
153 return out
