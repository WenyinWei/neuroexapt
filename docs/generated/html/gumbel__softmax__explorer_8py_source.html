108 def _sample_gumbel(self, shape: Tuple[int, ...], eps: float = 1e-20) -> torch.Tensor:

124 def _compute_exploration_stats(self, alpha_weights: torch.Tensor,

151 def _compute_diversity_score(self, samples: torch.Tensor) -> float:
152 """è®¡ç®—æ¶æ„é€‰æ‹©çš„å¤šæ ·æ€§åˆ†æ•°"""
153 # è®¡ç®—æ¯ä¸ªæ“ä½œè¢«é€‰æ‹©çš„é¢‘ç‡
154 op_frequencies = samples.sum(dim=0)
155 total_selections = op_frequencies.sum()

157 if total_selections > 0:
158 op_probs = op_frequencies / total_selections
159 # ä½¿ç”¨å½’ä¸€åŒ–ç†µä½œä¸ºå¤šæ ·æ€§æŒ‡æ ‡
160 diversity = -(op_probs * torch.log(op_probs + 1e-8)).sum().item()

162 return diversity / max_diversity if max_diversity > 0 else 0.0

328def create_annealing_schedule(initial_temp: float, min_temp: float,
329 total_epochs: int, schedule_type: str = "cosine") -> List[float]:
330 """
331 åˆ›å»ºæ¸©åº¦é€€ç«è®¡åˆ’
332
333 Args:
334 initial_temp: åˆå§‹æ¸©åº¦
335 min_temp: æœ€å°æ¸©åº¦
336 total_epochs: æ€»è½®æ•°
337 schedule_type: è®¡åˆ’ç±»å‹ ("linear", "cosine", "exponential")
338
339 Returns:
340 æ¸©åº¦è®¡åˆ’åˆ—è¡¨
341 """
342 temperatures = []
343
344 for epoch in range(total_epochs):
345 if schedule_type == "linear":
346 progress = epoch / (total_epochs - 1)
347 temp = initial_temp * (1 - progress) + min_temp * progress
348
349 elif schedule_type == "cosine":
350 progress = epoch / (total_epochs - 1)
351 temp = min_temp + (initial_temp - min_temp) * 0.5 * (1 + math.cos(math.pi * progress))
352
353 elif schedule_type == "exponential":
354 decay_rate = (min_temp / initial_temp) ** (1 / (total_epochs - 1))
355 temp = initial_temp * (decay_rate ** epoch)
356
357 else:
358 raise ValueError(f"Unknown schedule type: {schedule_type}")
359
360 temperatures.append(max(temp, min_temp))
361
362 return temperatures
