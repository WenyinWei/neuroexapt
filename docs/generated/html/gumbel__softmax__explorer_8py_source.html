34class GumbelSoftmaxExplorer:
35 """
36 Gumbel-Softmax引导式探索器
37
38 通过可控的随机性实现架构空间的有效探索
39 """
40
41 def __init__(self, initial_temp: float = 5.0, min_temp: float = 0.1,
42 anneal_rate: float = 0.98, exploration_factor: float = 1.0):
43 """
44 Args:
45 initial_temp: 初始温度，控制随机性强度
46 min_temp: 最小温度，防止完全确定性选择
47 anneal_rate: 温度退火率，每轮衰减比例
48 exploration_factor: 探索因子，调节探索强度
49 """
50 self.initial_temp = initial_temp
51 self.current_temp = initial_temp
52 self.min_temp = min_temp
53 self.anneal_rate = anneal_rate
54 self.exploration_factor = exploration_factor
55
56 # 探索历史统计
57 self.exploration_history = []
58 self.diversity_scores = []
59 self.step_count = 0
60
61 logger.info(f"🌡️ Gumbel-Softmax Explorer initialized: "
62 f"temp={initial_temp:.2f}→{min_temp:.2f}, "
63 f"anneal_rate={anneal_rate:.3f}")
64
65 def sample_architecture(self, alpha_weights: torch.Tensor,
66 hard: bool = False) -> Tuple[torch.Tensor, Dict]:
67 """
68 使用Gumbel-Softmax采样架构
69
70 Args:
71 alpha_weights: 架构参数 [num_edges, num_ops]
72 hard: 是否使用硬采样（straight-through）
73
74 Returns:
75 采样结果和统计信息
76 """
77 self.step_count += 1
78
79 # 生成Gumbel噪声
80 gumbel_noise = self._sample_gumbel(alpha_weights.shape)
81
82 # 加入温度和噪声的logits
83 logits = (alpha_weights + gumbel_noise) / self.current_temp
84
85 # Softmax采样
86 soft_samples = F.softmax(logits, dim=-1)
87
88 if hard:
89 # 硬采样：选择最大值，但保持梯度
90 hard_samples = self._straight_through_softmax(soft_samples)
91 samples = hard_samples
92 else:
93 samples = soft_samples
94
95 # 计算探索统计
96 stats = self._compute_exploration_stats(alpha_weights, samples, logits)
97
98 # 记录探索历史
99 self.exploration_history.append({
100 'step': self.step_count,
101 'temperature': self.current_temp,
102 'entropy': stats['entropy'],
103 'diversity': stats['diversity']
104 })
105
106 return samples, stats
107
108 def _sample_gumbel(self, shape: Tuple[int, ...], eps: float = 1e-20) -> torch.Tensor:
109 """采样Gumbel分布噪声"""
110 uniform = torch.rand(shape, device=torch.cuda.current_device() if torch.cuda.is_available() else None)
111 gumbel = -torch.log(-torch.log(uniform + eps) + eps)
112 return gumbel * self.exploration_factor
113
114 def _straight_through_softmax(self, soft_samples: torch.Tensor) -> torch.Tensor:
115 """Straight-through硬采样，保持梯度流"""
116 # 硬采样：选择最大概率的操作
117 hard_samples = torch.zeros_like(soft_samples)
118 max_indices = torch.argmax(soft_samples, dim=-1, keepdim=True)
119 hard_samples.scatter_(-1, max_indices, 1.0)
120
121 # Straight-through梯度：前向使用硬采样，反向使用软采样
122 return hard_samples.detach() + soft_samples - soft_samples.detach()
123
124 def _compute_exploration_stats(self, alpha_weights: torch.Tensor,
125 samples: torch.Tensor,
126 logits: torch.Tensor) -> Dict:
127 """计算探索相关统计信息"""
128 with torch.no_grad():
129 # 熵计算 - 衡量探索程度
130 probs = F.softmax(logits, dim=-1)
131 entropy = -(probs * torch.log(probs + 1e-8)).sum(dim=-1).mean().item()
132
133 # 多样性分数 - 衡量选择的多样性
134 diversity = self._compute_diversity_score(samples)
135
136 # 与贪婪选择的差异
137 greedy_selection = torch.zeros_like(alpha_weights)
138 greedy_indices = torch.argmax(alpha_weights, dim=-1, keepdim=True)
139 greedy_selection.scatter_(-1, greedy_indices, 1.0)
140
141 difference_from_greedy = torch.abs(samples - greedy_selection).sum().item()
142
143 return {
144 'entropy': entropy,
145 'diversity': diversity,
146 'temperature': self.current_temp,
147 'difference_from_greedy': difference_from_greedy,
148 'exploration_strength': entropy / math.log(alpha_weights.size(-1)) # 归一化熵
149 }
150
151 def _compute_diversity_score(self, samples: torch.Tensor) -> float:
152 """计算架构选择的多样性分数"""
153 # 计算每个操作被选择的频率
154 op_frequencies = samples.sum(dim=0)
155 total_selections = op_frequencies.sum()
156
157 if total_selections > 0:
158 op_probs = op_frequencies / total_selections
159 # 使用归一化熵作为多样性指标
160 diversity = -(op_probs * torch.log(op_probs + 1e-8)).sum().item()
161 max_diversity = math.log(len(op_frequencies))
162 return diversity / max_diversity if max_diversity > 0 else 0.0
163 return 0.0

165 def update_temperature(self, performance_gain: Optional[float] = None,
166 force_anneal: bool = False) -> float:
167 """
168 更新温度参数
169
170 Args:
171 performance_gain: 性能提升，用于自适应温度调节
172 force_anneal: 强制退火
173
174 Returns:
175 新的温度值
176 """
177 if force_anneal or self.current_temp > self.min_temp:
178 if performance_gain is not None:
179 # 自适应温度调节：性能提升时降温，性能下降时升温
180 if performance_gain > 0:
181 # 性能提升，可以适当降低探索
182 self.current_temp *= self.anneal_rate
183 else:
184 # 性能下降，增加探索
185 self.current_temp = min(self.current_temp * 1.05, self.initial_temp)
186 else:
187 # 标准线性退火
188 self.current_temp = max(self.current_temp * self.anneal_rate, self.min_temp)
189
190 logger.debug(f"Temperature updated to {self.current_temp:.3f}")
191 return self.current_temp

193 def get_exploration_schedule(self, total_steps: int) -> List[float]:
194 """
195 生成完整的探索温度计划
196
197 Args:
198 total_steps: 总步数
199
200 Returns:
201 温度计划列表
202 """
203 schedule = []
204 temp = self.initial_temp
205
206 for step in range(total_steps):
207 schedule.append(temp)
208 temp = max(temp * self.anneal_rate, self.min_temp)
209
210 return schedule

212 def adaptive_temperature_control(self, validation_loss_history: List[float],
213 window_size: int = 5) -> float:
214 """
215 基于验证损失历史的自适应温度控制
216
217 Args:
218 validation_loss_history: 验证损失历史
219 window_size: 滑动窗口大小
220
221 Returns:
222 调整后的温度
223 """
224 if len(validation_loss_history) < window_size:
225 return self.current_temp
226
227 # 计算最近窗口的损失趋势
228 recent_losses = validation_loss_history[-window_size:]
229 trend = np.polyfit(range(len(recent_losses)), recent_losses, 1)[0]
230
231 if trend > 0: # 损失上升，增加探索
232 self.current_temp = min(self.current_temp * 1.1, self.initial_temp)
233 logger.info(f"🔥 Increasing exploration: temp={self.current_temp:.3f} (loss trend: +{trend:.4f})")
234 elif trend < -0.001: # 损失显著下降，减少探索
235 self.current_temp = max(self.current_temp * 0.95, self.min_temp)
236 logger.info(f"❄️ Reducing exploration: temp={self.current_temp:.3f} (loss trend: {trend:.4f})")
237
238 return self.current_temp

240 def get_exploration_report(self) -> Dict:
241 """获取探索过程的详细报告"""
242 if not self.exploration_history:
243 return {"message": "No exploration history available"}
244
245 history = self.exploration_history
246
247 return {
248 "total_steps": len(history),
249 "current_temperature": self.current_temp,
250 "average_entropy": np.mean([h['entropy'] for h in history]),
251 "average_diversity": np.mean([h['diversity'] for h in history]),
252 "temperature_range": {
253 "initial": history[0]['temperature'],
254 "current": history[-1]['temperature'],
255 "min": min(h['temperature'] for h in history)
256 },
257 "exploration_trend": {
258 "entropy_trend": np.polyfit(range(len(history)),
259 [h['entropy'] for h in history], 1)[0],
260 "diversity_trend": np.polyfit(range(len(history)),
261 [h['diversity'] for h in history], 1)[0]
262 }
263 }

266class MultiObjectiveExplorer(GumbelSoftmaxExplorer):
267 """
268 多目标Gumbel-Softmax探索器
269
270 同时考虑准确率、延迟、参数量等多个目标的架构探索

273 def __init__(self, objectives: List[str] = ["accuracy", "latency", "params"],
274 objective_weights: Optional[List[float]] = None, **kwargs):
275 super().__init__(**kwargs)
276
277 self.objectives = objectives
278 self.objective_weights = objective_weights or [1.0] * len(objectives)
279 self.objective_history = {obj: [] for obj in objectives}
280
281 logger.info(f"🎯 Multi-objective explorer: {objectives} with weights {objective_weights}")
282
283 def sample_with_objectives(self, alpha_weights: torch.Tensor,
284 objective_scores: Dict[str, float],
285 hard: bool = False) -> Tuple[torch.Tensor, Dict]:
286 """
287 基于多目标的架构采样
288
289 Args:
290 alpha_weights: 架构参数
291 objective_scores: 各目标的当前分数
292 hard: 是否硬采样
293
294 Returns:
295 采样结果和统计信息
296 """
297 # 更新目标历史
298 for obj, score in objective_scores.items():
299 if obj in self.objective_history:
300 self.objective_history[obj].append(score)
301
302 # 计算多目标权重调整
303 objective_adjustment = self._compute_objective_adjustment(objective_scores)
304
305 # 调整架构参数权重
306 adjusted_alpha = alpha_weights + objective_adjustment
307
308 return self.sample_architecture(adjusted_alpha, hard)
309
310 def _compute_objective_adjustment(self, objective_scores: Dict[str, float]) -> torch.Tensor:
311 """基于多目标分数计算架构参数调整"""
312 # 这里可以实现复杂的多目标优化策略
313 # 简化版本：根据目标达成情况调整温度
314
315 weighted_score = sum(score * weight
316 for score, weight in zip(objective_scores.values(), self.objective_weights))
317
318 # 根据综合分数调整探索强度
319 if weighted_score > 0.8: # 高分时减少探索
320 self.current_temp *= 0.98
321 elif weighted_score < 0.5: # 低分时增加探索
322 self.current_temp *= 1.02
323
324 # 返回零调整（简化版本，实际应用中可扩展为更复杂的调整策略）
325 return torch.zeros(1, 1) * 0.1

328def create_annealing_schedule(initial_temp: float, min_temp: float,
329 total_epochs: int, schedule_type: str = "cosine") -> List[float]:
330 """
331 创建温度退火计划
332
333 Args:
334 initial_temp: 初始温度
335 min_temp: 最小温度
336 total_epochs: 总轮数
337 schedule_type: 计划类型 ("linear", "cosine", "exponential")
338
339 Returns:
340 温度计划列表
341 """
342 temperatures = []
343
344 for epoch in range(total_epochs):
345 if schedule_type == "linear":
346 progress = epoch / (total_epochs - 1)
347 temp = initial_temp * (1 - progress) + min_temp * progress
348
349 elif schedule_type == "cosine":
350 progress = epoch / (total_epochs - 1)
351 temp = min_temp + (initial_temp - min_temp) * 0.5 * (1 + math.cos(math.pi * progress))
352
353 elif schedule_type == "exponential":
354 decay_rate = (min_temp / initial_temp) ** (1 / (total_epochs - 1))
355 temp = initial_temp * (decay_rate ** epoch)
356
357 else:
358 raise ValueError(f"Unknown schedule type: {schedule_type}")
359
360 temperatures.append(max(temp, min_temp))
361
362 return temperatures
