14class EvolvableCell(nn.Module):
15 """
16 A single cell that represents a discrete architecture based on a genotype.
17 Unlike the original Cell that used a weighted sum of operations, this cell
18 builds a specific, fixed graph based on the provided genotype.
19 """
20
21 def __init__(self, genotype, C_prev_prev, C_prev, C, reduction, reduction_prev):
22 super(EvolvableCell, self).__init__()
23 print(f"Initializing EvolvableCell: C_prev_prev={C_prev_prev}, C_prev={C_prev}, C={C}, reduction={reduction}")
24
25 self.reduction = reduction
26

29 else:
30 self.preprocess0 = ReLUConvBN(C_prev_prev, C, 1, 1, 0, affine=False)
31 self.preprocess1 = ReLUConvBN(C_prev, C, 1, 1, 0, affine=False)
32
33 if reduction:
34 op_names, indices = zip(*genotype.reduce)
35 concat = genotype.reduce_concat
36 else:
37 op_names, indices = zip(*genotype.normal)

42 def _compile(self, C, op_names, indices, concat, reduction):
43 assert len(op_names) == len(indices)
44 self._steps = len(op_names) // 2
45 self._concat = concat
46 self.multiplier = len(concat)
47
48 self._ops = nn.ModuleList()
49 for name, index in zip(op_names, indices):
50 stride = 2 if reduction and index < 2 else 1
51 op = OPS[name](C, stride, True)
52 self._ops += [op]
53 self._indices = indices
54
55 def forward(self, s0, s1):
56 s0 = self.preprocess0(s0)
57 s1 = self.preprocess1(s1)
58
59 states = [s0, s1]
60 for i in range(self._steps):
61 h1 = states[self._indices[2*i]]
62 h2 = states[self._indices[2*i+1]]
63 op1 = self._ops[2*i]
64 op2 = self._ops[2*i+1]
65 h1 = op1(h1)
66 h2 = op2(h2)
67 s = h1 + h2
68 states += [s]
69
70 return torch.cat([states[i] for i in self._concat], dim=1)

73class EvolvableNetwork(nn.Module):
74 """
75 A network that is built from a discrete architecture specified by a genotype.
76 This network can be mutated by providing a new genotype.
77 """
78 def __init__(self, C, num_classes, layers, genotype, stem_multiplier=3):
79 super(EvolvableNetwork, self).__init__()
80 self._C = C
81 self._num_classes = num_classes
82 self._layers = layers
83 self.genotype = genotype
84
85 C_curr = stem_multiplier * C
86 self.stem = nn.Sequential(
87 nn.Conv2d(3, C_curr, 3, padding=1, bias=False),
88 nn.BatchNorm2d(C_curr)
89 )
90
91 C_prev_prev, C_prev, C_curr = C_curr, C_curr, C
92 self.cells = nn.ModuleList()
93 reduction_prev = False
94 for i in range(layers):
95 # Determine if current cell is a reduction cell
96 if i in [layers // 3, 2 * layers // 3]:
97 C_curr *= 2
98 reduction = True
99 else:
100 reduction = False
101
102 # Select appropriate genotype based on reduction type
103 if reduction:
104 op_names, indices = zip(*genotype.reduce)
105 if len(op_names) != len(indices):
106 raise ValueError("Genotype reduce is not valid")
107 else:
108 op_names, indices = zip(*genotype.normal)
109 if len(op_names) != len(indices):
110 raise ValueError("Genotype normal is not valid")
111
112 cell = EvolvableCell(genotype, C_prev_prev, C_prev, C_curr, reduction, reduction_prev)
113 reduction_prev = reduction
114 self.cells += [cell]
115 C_prev_prev, C_prev = C_prev, cell.multiplier * C_curr
116
117 self.global_pooling = nn.AdaptiveAvgPool2d(1)
118 self.classifier = nn.Linear(C_prev, num_classes)
119
120 def forward(self, input):
121 s0 = s1 = self.stem(input)
122 for i, cell in enumerate(self.cells):
123 s0, s1 = s1, cell(s0, s1)
124 out = self.global_pooling(s1)
125 logits = self.classifier(out.view(out.size(0), -1))
126 return logits
