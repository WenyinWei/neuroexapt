202 def _hybrid_division_strategy(self,
203 layer: nn.Module,
204 neuron_idx: int,
205 activations: Optional[torch.Tensor],
206 gradients: Optional[torch.Tensor],
207 targets: Optional[torch.Tensor]) -> Tuple[bool, Dict[str, Any]]:
208 """混合分裂策略"""
209 try:
210 # 收集所有可用信息
211 gradient_score = self._evaluate_gradient_importance(gradients, neuron_idx) if gradients is not None else 0.0
212 activation_score = self._evaluate_activation_importance(activations, neuron_idx) if activations is not None else 0.0
213 info_score = self._evaluate_information_importance(activations, targets, neuron_idx) if activations is not None and targets is not None else 0.0
214
215 # 加权组合评分
216 total_score = 0.4 * gradient_score + 0.3 * activation_score + 0.3 * info_score
217
218 # 根据综合评分选择分裂策略
219 if total_score > 0.7:
220 # 高重要性神经元 - 使用保守分裂
221 return self._conservative_division(layer, neuron_idx, total_score)
222 elif total_score > 0.3:
223 # 中等重要性神经元 - 使用标准分裂
224 return self._standard_division(layer, neuron_idx, total_score)
225 else:
226 # 低重要性神经元 - 使用激进分裂
227 return self._aggressive_division(layer, neuron_idx, total_score)
228
229 except Exception as e:
230 logger.error(f"Error in hybrid division strategy: {e}")
231 return False, {'error': str(e)}

233 def _divide_conv_neuron_gradient(self,
234 layer: nn.Conv2d,
235 neuron_idx: int,
236 gradients: torch.Tensor,
237 original_weight: torch.Tensor) -> Tuple[bool, Dict[str, Any]]:
238 """基于梯度分裂卷积神经元"""
239 out_channels, in_channels, kernel_h, kernel_w = layer.weight.shape
240
241 if neuron_idx >= out_channels:
242 return False, {'error': f'Neuron index {neuron_idx} out of range for {out_channels} channels'}
243
244 # 创建新的层
245 new_layer = nn.Conv2d(
246 in_channels=in_channels,
247 out_channels=out_channels + 1,
248 kernel_size=(kernel_h, kernel_w),
249 stride=layer.stride,
250 padding=layer.padding,
251 bias=layer.bias is not None
252 )
253
254 # 复制原始权重
255 new_layer.weight.data[:out_channels] = original_weight
256
257 # 基于梯度信息初始化新神经元
258 parent_grad = gradients[neuron_idx] if gradients.dim() > 0 else gradients
259 if isinstance(parent_grad, torch.Tensor) and parent_grad.numel() > 0:
260 # 添加梯度导向的变化
261 noise = torch.randn_like(original_weight[neuron_idx]) * self.noise_scale
262 gradient_direction = torch.sign(parent_grad.mean()) if parent_grad.numel() > 0 else 1.0
263 new_layer.weight.data[out_channels] = original_weight[neuron_idx] + gradient_direction * noise
264 else:
265 # 默认噪声初始化
266 new_layer.weight.data[out_channels] = original_weight[neuron_idx] + torch.randn_like(original_weight[neuron_idx]) * self.noise_scale
267
268 # 处理偏置
269 if layer.bias is not None:
270 new_layer.bias.data[:out_channels] = layer.bias.data
271 noise_value = (torch.randn(1, device=layer.bias.device) * self.noise_scale)
272 new_layer.bias.data[out_channels] = layer.bias.data[neuron_idx] + noise_value
273
274 # 替换层
275 self._replace_layer_in_model(layer, new_layer)
276
277 return True, {
278 'strategy': 'gradient_based',
279 'original_channels': out_channels,
280 'new_channels': out_channels + 1,
281 'neuron_idx': neuron_idx,
282 'noise_scale': self.noise_scale
283 }

324 def _analyze_activation_pattern(self, activations: torch.Tensor, neuron_idx: int) -> Dict[str, float]:
325 """分析激活模式"""
326 if activations.dim() == 4: # Conv layer: (N, C, H, W)
327 if neuron_idx < activations.shape[1]:
328 neuron_activations = activations[:, neuron_idx, :, :]
329 else:
330 neuron_activations = activations.mean(dim=1)
331 elif activations.dim() == 2: # Linear layer: (N, F)
332 if neuron_idx < activations.shape[1]:
333 neuron_activations = activations[:, neuron_idx]
334 else:
335 neuron_activations = activations.mean(dim=1)
336 else:
337 neuron_activations = activations.flatten()
338
339 return {
340 'mean': neuron_activations.mean().item(),
341 'std': neuron_activations.std().item(),
342 'sparsity': (neuron_activations == 0).float().mean().item(),
343 'max_activation': neuron_activations.max().item(),

347 def _estimate_mutual_information(self, activations: torch.Tensor, targets: torch.Tensor, neuron_idx: int) -> float:
348 """估计互信息"""
349 try:
350 # 简化的互信息估计
351 if activations.dim() == 4 and neuron_idx < activations.shape[1]:
352 neuron_activations = activations[:, neuron_idx, :, :].mean(dim=(1, 2))
353 elif activations.dim() == 2 and neuron_idx < activations.shape[1]:
354 neuron_activations = activations[:, neuron_idx]
355 else:
356 return 0.0
357
358 # 使用相关系数作为互信息的近似
359 # 注意：相关系数仅能捕捉线性关系，作为互信息的代理可能会导致误导性结论。
360 # 更健壮的信息传递估计方法（如基于分箱的互信息估计、sklearn 的 mutual_info_regression/mutual_info_classif 等）可用于更准确的评估。
361 # 具体实现可根据需求替换此处相关系数的计算。
362 if targets.dim() > 1:
363 targets_flat = targets.argmax(dim=1).float()
364 else:
365 targets_flat = targets.float()
366
367 correlation = torch.corrcoef(torch.stack([neuron_activations, targets_flat]))[0, 1]
368 return abs(correlation.item()) if not torch.isnan(correlation) else 0.0
369
370 except Exception:
371 return 0.0

373 def _estimate_neuron_entropy(self, activations: torch.Tensor, neuron_idx: int) -> float:
374 """估计神经元熵"""
375 try:
376 if activations.dim() == 4 and neuron_idx < activations.shape[1]:
377 neuron_activations = activations[:, neuron_idx, :, :].flatten()
378 elif activations.dim() == 2 and neuron_idx < activations.shape[1]:
379 neuron_activations = activations[:, neuron_idx]
380 else:
381 return 0.0
382
383 # 使用直方图估计熵
384 hist = torch.histc(neuron_activations, bins=20)
385 hist = hist / hist.sum()
386 hist = hist[hist > 0]
387
388 entropy = -torch.sum(hist * torch.log2(hist + 1e-10)).item()
389 return entropy
390
391 except Exception:
392 return 0.0

394 def _evaluate_gradient_importance(self, gradients: torch.Tensor, neuron_idx: int) -> float:
395 """评估梯度重要性"""
396 if gradients is None:
397 return 0.0
398
399 try:
400 if gradients.dim() > 0 and neuron_idx < gradients.shape[0]:
401 grad_magnitude = torch.norm(gradients[neuron_idx]).item()
402 else:
403 grad_magnitude = torch.norm(gradients).item()
404
405 # 归一化到 [0, 1]
406 return min(1.0, grad_magnitude / 10.0)

408 return 0.0

410 def _evaluate_activation_importance(self, activations: torch.Tensor, neuron_idx: int) -> float:
411 """评估激活重要性"""
412 if activations is None:
413 return 0.0
414
415 try:
416 pattern = self._analyze_activation_pattern(activations, neuron_idx)
417
418 # 基于激活统计计算重要性
419 sparsity_score = 1.0 - pattern['sparsity'] # 低稀疏性 = 高重要性
420 variance_score = min(1.0, pattern['std'] / 5.0) # 高方差 = 高重要性
421 magnitude_score = min(1.0, abs(pattern['mean']) / 5.0) # 高幅度 = 高重要性
422
423 return (sparsity_score + variance_score + magnitude_score) / 3.0
424 except Exception:
425 return 0.0

428 """评估信息重要性"""
