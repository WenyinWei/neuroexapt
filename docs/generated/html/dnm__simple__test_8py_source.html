28class SimpleCNN(nn.Module):
29 """简单但可演化的CNN模型"""
30
31 def __init__(self, num_classes=10):
32 super().__init__()
33
34 # 特征提取层
35 self.features = nn.Sequential(
36 # 第一组卷积
37 nn.Conv2d(3, 32, kernel_size=3, padding=1),
38 nn.BatchNorm2d(32),
39 nn.ReLU(inplace=True),
40 nn.MaxPool2d(2, 2),
41
42 # 第二组卷积
43 nn.Conv2d(32, 64, kernel_size=3, padding=1),
44 nn.BatchNorm2d(64),
45 nn.ReLU(inplace=True),
46 nn.MaxPool2d(2, 2),
47
48 # 第三组卷积
49 nn.Conv2d(64, 128, kernel_size=3, padding=1),
50 nn.BatchNorm2d(128),
51 nn.ReLU(inplace=True),
52 nn.MaxPool2d(2, 2),
53 )
54
55 # 分类器
56 self.classifier = nn.Sequential(
57 nn.AdaptiveAvgPool2d((1, 1)),
58 nn.Flatten(),
59 nn.Linear(128, 256),
60 nn.ReLU(inplace=True),
61 nn.Dropout(0.5),
62 nn.Linear(256, 128),
63 nn.ReLU(inplace=True),
64 nn.Dropout(0.3),
65 nn.Linear(128, num_classes)
66 )
67
68 # 权重初始化
69 self._initialize_weights()
70
71 def _initialize_weights(self):
72 for m in self.modules():
73 if isinstance(m, nn.Conv2d):
74 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
75 if m.bias is not None:
76 nn.init.constant_(m.bias, 0)
77 elif isinstance(m, nn.BatchNorm2d):
78 nn.init.constant_(m.weight, 1)
79 nn.init.constant_(m.bias, 0)
80 elif isinstance(m, nn.Linear):
81 nn.init.normal_(m.weight, 0, 0.01)
82 nn.init.constant_(m.bias, 0)
83
84 def forward(self, x):
85 x = self.features(x)
86 x = self.classifier(x)
87 return x
