62class InformationTheoryTrigger(MorphogenesisTrigger):
63 """基于信息论的触发器"""
64
65 def __init__(self, entropy_threshold: float = 0.1, mi_threshold: float = 0.05):
66 self.entropy_threshold = entropy_threshold
67 self.mi_threshold = mi_threshold
68 self.history = deque(maxlen=10)
69
70 def should_trigger(self, context: Dict[str, Any]) -> Tuple[bool, str]:
71 activations = context.get('activations', {})
72 gradients = context.get('gradients', {})
73
74 if not activations or not gradients:
75 return False, "缺少激活值或梯度信息"
76
77 # 计算信息熵变化
78 entropy_changes = self._compute_entropy_changes(activations)
79
80 # 计算互信息
81 mutual_info = self._compute_mutual_information(activations)
82
83 # 梯度方差分析
84 gradient_variance = self._analyze_gradient_variance(gradients)
85
86 self.history.append({
87 'entropy_changes': entropy_changes,
88 'mutual_info': mutual_info,
89 'gradient_variance': gradient_variance
90 })
91
92 # 信息瓶颈检测
93 if self._detect_information_bottleneck():
94 return True, f"信息瓶颈检测：熵变化={entropy_changes:.4f}, 互信息={mutual_info:.4f}"
95
96 return False, "信息论指标未达到触发条件"
97
98 def _compute_entropy_changes(self, activations: Dict[str, torch.Tensor]) -> float:
99 """计算激活值熵的变化"""
100 total_entropy_change = 0.0
101 count = 0
102
103 for name, activation in activations.items():
104 if len(activation.shape) >= 2:
105 # 计算每个神经元的熵
106 activation_flat = activation.view(activation.size(0), -1)
107 probs = F.softmax(activation_flat, dim=-1) + 1e-8
108 entropy = -torch.sum(probs * torch.log(probs), dim=-1).mean()
109
110 if len(self.history) > 0:
111 prev_entropy = self.history[-1].get('entropy_changes', 0)
112 entropy_change = abs(entropy.item() - prev_entropy)
113 total_entropy_change += entropy_change
114 count += 1
115
116 return total_entropy_change / max(count, 1)
117
118 def _compute_mutual_information(self, activations: Dict[str, torch.Tensor]) -> float:
119 """计算层间互信息"""
120 layer_names = list(activations.keys())
121 if len(layer_names) < 2:
122 return 0.0
123
124 # 简化的互信息估计
125 mi_sum = 0.0
126 pairs = 0
127
128 for i in range(len(layer_names) - 1):
129 for j in range(i + 1, min(i + 3, len(layer_names))): # 只考虑相邻层
130 act1 = activations[layer_names[i]].flatten()
131 act2 = activations[layer_names[j]].flatten()
132
133 # 使用相关系数近似互信息
134 if len(act1) == len(act2):
135 correlation = torch.corrcoef(torch.stack([act1, act2]))[0, 1]
136 mi = -0.5 * torch.log(1 - correlation**2 + 1e-8)
137 mi_sum += mi.item()
138 pairs += 1
139
140 return mi_sum / max(pairs, 1)
141
142 def _analyze_gradient_variance(self, gradients: Dict[str, torch.Tensor]) -> float:
143 """分析梯度方差"""
144 total_variance = 0.0
145 count = 0
146
147 for name, grad in gradients.items():
148 if grad is not None:
149 variance = torch.var(grad).item()
150 total_variance += variance
151 count += 1
152
153 return total_variance / max(count, 1)
154
155 def _detect_information_bottleneck(self) -> bool:
156 """检测信息瓶颈"""
157 if len(self.history) < 5:
158 return False
159
160 recent_entropies = [h['entropy_changes'] for h in list(self.history)[-5:]]
161 recent_mis = [h['mutual_info'] for h in list(self.history)[-5:]]
162
163 # 熵变化趋于稳定且互信息较低
164 entropy_stability = np.std(recent_entropies) < self.entropy_threshold
165 low_mi = np.mean(recent_mis) < self.mi_threshold
166
167 return entropy_stability and low_mi

170 return 0.8

293 def _compute_gradient_norm(self, gradients: Dict[str, torch.Tensor]) -> float:
294 """计算梯度范数"""
295 total_norm = 0.0
296 for grad in gradients.values():
297 if grad is not None:
298 total_norm += torch.norm(grad).item() ** 2

301 def _detect_gradient_pathology(self) -> bool:

313 if np.max(recent_grads) > 100:
314 return True
315
316 # 梯度振荡
317 grad_diff = np.diff(recent_grads)
318 if len(grad_diff) > 5:
319 oscillation = np.sum(np.diff(np.sign(grad_diff)) != 0) / len(grad_diff)
320 if oscillation > 0.7:

322
323 return False

473 def _analyze_network_topology(self, model: nn.Module, activations: Dict[str, torch.Tensor]) -> Dict[str, float]:
474 """分析网络拓扑特性"""
475 metrics = {}
476
477 # 计算层间连接强度
478 layer_connections = self._compute_layer_connections(model)
479 metrics['avg_connection_strength'] = np.mean(list(layer_connections.values()))
480
481 # 计算网络中心性
482 centrality = self._compute_network_centrality(activations)
483 metrics['centrality'] = centrality
484
485 # 计算聚类系数
486 clustering = self._compute_clustering_coefficient(activations)
487 metrics['clustering'] = clustering
488
489 return metrics

491 def _compute_layer_connections(self, model: nn.Module) -> Dict[str, float]:
492 """计算层间连接强度"""
493 connections = {}
494
495 for name, module in model.named_modules():
496 if isinstance(module, (nn.Linear, nn.Conv2d)):
497 if hasattr(module, 'weight') and module.weight is not None:
498 weight_norm = torch.norm(module.weight).item()
499 connections[name] = weight_norm
500
501 return connections
502
503 def _compute_network_centrality(self, activations: Dict[str, torch.Tensor]) -> float:
504 """计算网络中心性"""
505 if len(activations) < 2:
506 return 0.0
507
508 # 简化的中心性计算
509 activation_norms = {}
510 for name, activation in activations.items():
511 activation_norms[name] = torch.norm(activation).item()
512
513 norm_values = list(activation_norms.values())
514 if not norm_values:
515 return 0.0
516
517 # 计算标准化的中心性
518 max_norm = max(norm_values)
519 avg_norm = np.mean(norm_values)
520
521 return max_norm / (avg_norm + 1e-8)
522
523 def _compute_clustering_coefficient(self, activations: Dict[str, torch.Tensor]) -> float:
524 """计算聚类系数"""
525 if len(activations) < 3:
526 return 0.0
527
528 # 简化的聚类系数计算
529 layer_names = list(activations.keys())
530 correlations = []
531
532 for i in range(len(layer_names)):
533 for j in range(i+1, len(layer_names)):
534 act1 = activations[layer_names[i]].flatten()
535 act2 = activations[layer_names[j]].flatten()
536
537 if len(act1) == len(act2) and len(act1) > 1:
538 corr = torch.corrcoef(torch.stack([act1, act2]))[0, 1]
539 correlations.append(abs(corr.item()))
540
541 return np.mean(correlations) if correlations else 0.0
542
543 def _detect_network_bottleneck(self, metrics: Dict[str, float]) -> bool:
544 """检测网络瓶颈"""
545 centrality = metrics.get('centrality', 0)
546 return centrality > 3.0 # 中心性过高表明存在瓶颈
547
548 def _detect_connectivity_imbalance(self, metrics: Dict[str, float]) -> bool:

551 return False
552
553 recent_metrics = list(self.connectivity_history)[-5:]
554 connection_strengths = [m.get('avg_connection_strength', 0) for m in recent_metrics]
555
556 # 连接强度方差过大
557 strength_std = np.std(connection_strengths)
558 return strength_std > 0.5

561 return 0.7

563class NeuronDivisionExecutor:
564 """神经元分裂执行器"""
565
566 def __init__(self):
567 self.division_history = []
568
569 def execute_division(self, model: nn.Module, layer_name: str, division_type: str = 'width_expansion') -> Tuple[nn.Module, int]:
570 """执行神经元分裂"""
571 try:
572 if division_type == 'width_expansion':
573 return self._expand_layer_width(model, layer_name)
574 elif division_type == 'depth_expansion':
575 return self._expand_network_depth(model, layer_name)
576 elif division_type == 'branch_creation':
577 return self._create_branch(model, layer_name)
578 else:
579 logger.warning(f"未知的分裂类型: {division_type}")
580 return model, 0
581
582 except Exception as e:
583 logger.error(f"神经元分裂执行失败: {e}")
584 return model, 0
585
586 def _expand_layer_width(self, model: nn.Module, layer_name: str) -> Tuple[nn.Module, int]:
587 """扩展层宽度（增加神经元数量）"""
588 new_model = copy.deepcopy(model)
589 parameters_added = 0
590
591 # 找到目标层
592 target_module = None
593 for name, module in new_model.named_modules():
594 if name == layer_name:
595 target_module = module
596 break
597
598 if target_module is None:
599 logger.warning(f"未找到目标层: {layer_name}")
600 return model, 0
601
602 if isinstance(target_module, nn.Linear):
603 # 扩展全连接层
604 old_out_features = target_module.out_features
605 new_out_features = int(old_out_features * 1.2) # 增加20%
606 expansion_size = new_out_features - old_out_features
607
608 # 创建新的权重和偏置
609 new_weight = torch.zeros(new_out_features, target_module.in_features)
610 new_bias = torch.zeros(new_out_features) if target_module.bias is not None else None
611
612 # 复制原有权重
613 new_weight[:old_out_features] = target_module.weight.data
614 if new_bias is not None:

627 if target_module.bias is not None:

640 target_module.in_channels,
641 new_out_channels,
642 target_module.kernel_size,
643 target_module.stride,
644 target_module.padding,
645 target_module.dilation,
646 target_module.groups,
647 target_module.bias is not None
648 )

653 if target_module.bias is not None:
654 new_conv.bias.data[:old_out_channels] = target_module.bias.data
655
656 # 初始化新增的通道
657 nn.init.kaiming_normal_(new_conv.weight.data[old_out_channels:])
658 if new_conv.bias is not None:
659 nn.init.zeros_(new_conv.bias.data[old_out_channels:])

662 parent_name = '.'.join(layer_name.split('.')[:-1])
663 child_name = layer_name.split('.')[-1]
664
665 if parent_name:
666 parent_module = new_model
667 for part in parent_name.split('.'):
668 parent_module = getattr(parent_module, part)
669 setattr(parent_module, child_name, new_conv)
670 else:
671 setattr(new_model, child_name, new_conv)
672
673 parameters_added = expansion_size * target_module.in_channels * \
674 target_module.kernel_size[0] * target_module.kernel_size[1]
675
676 self.division_history.append({
677 'layer': layer_name,
678 'type': 'width_expansion',
679 'parameters_added': parameters_added
680 })

682 logger.info(f"执行宽度扩展: {layer_name}, 新增参数: {parameters_added}")
683 return new_model, parameters_added

738 def should_trigger_morphogenesis(self, epoch: int, train_metrics: Dict[str, float],

756 'gradients': self.gradient_cache

764 should_trigger, reason = trigger.should_trigger(context)
765 if should_trigger:

767 triggered_reasons.append(f"{trigger.__class__.__name__}: {reason}")

773 if trigger_results and trigger_results[0][0].get_priority() >= 0.8:

777 if len(trigger_results) >= 2 and all(t[0].get_priority() >= 0.7 for t in trigger_results[:2]):
778 return True, triggered_reasons

882 def _compute_division_score(self, layer_name: str, module: nn.Module) -> float:
883 """计算层的分裂分数"""
884 score = 0.0
885
886 # 基于权重分析
887 if hasattr(module, 'weight') and module.weight is not None:
888 weight = module.weight.data
889
890 # 权重方差（高方差表明神经元分化程度高）
891 weight_var = torch.var(weight).item()
892 score += weight_var * 0.3
893
894 # 权重范数（适中的范数最佳）
895 weight_norm = torch.norm(weight).item()
896 normalized_norm = weight_norm / weight.numel()
897 score += (1.0 - abs(normalized_norm - 0.1)) * 0.2
898
899 # 基于激活值分析
900 if layer_name in self.activation_cache:
901 activation = self.activation_cache[layer_name]
902
903 # 激活值多样性
904 act_std = torch.std(activation).item()
905 score += act_std * 0.3
906
907 # 激活值饱和度
908 saturation = torch.mean((activation > 0.9).float()).item()
909 score += (1.0 - saturation) * 0.2
910
911 return score
