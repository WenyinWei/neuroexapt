107 def _pearson_correlation(self, x: torch.Tensor, y: torch.Tensor) -> float:
108 """è®¡ç®—çš®å°”é€Šç›¸å…³ç³»æ•°"""

112 y_centered = y - torch.mean(y)
113
114 # è®¡ç®—ç›¸å…³ç³»æ•°
115 numerator = torch.sum(x_centered * y_centered)
116 denominator = torch.sqrt(torch.sum(x_centered ** 2) * torch.sum(y_centered ** 2))
117
118 if denominator > 1e-8:
119 correlation = numerator / denominator
120 return correlation.item()

394class ChannelAttention(nn.Module):
395 """é€šé“æ³¨æ„åŠ›æ¨¡å—"""
396
397 def __init__(self, source_channels: int, target_channels: int, reduction_ratio: int = 16):
398 super().__init__()
399
400 self.source_channels = source_channels
401 self.target_channels = target_channels
402
403 # å…¨å±€å¹³å‡æ± åŒ–
404 self.global_avg_pool = nn.AdaptiveAvgPool2d(1)
405
406 # æ³¨æ„åŠ›ç½‘ç»œ
407 hidden_channels = max(source_channels // reduction_ratio, 1)
408 self.attention_net = nn.Sequential(
409 nn.Linear(source_channels, hidden_channels),
410 nn.ReLU(inplace=True),
411 nn.Linear(hidden_channels, target_channels),
412 nn.Sigmoid()
413 )
414
415 # é€šé“é€‚é…å™¨
416 if source_channels != target_channels:
417 self.channel_adapter = nn.Conv2d(source_channels, target_channels, 1, bias=False)
418 else:
419 self.channel_adapter = nn.Identity()
420
421 def forward(self, source_features: torch.Tensor, target_features: torch.Tensor) -> torch.Tensor:
422 # è®¡ç®—æ³¨æ„åŠ›æƒé‡
423 b, c, h, w = source_features.size()
424
425 # å…¨å±€æ± åŒ–å¾—åˆ°é€šé“æè¿°ç¬¦
426 channel_descriptor = self.global_avg_pool(source_features).view(b, c)
427
428 # è®¡ç®—æ³¨æ„åŠ›æƒé‡
429 attention_weights = self.attention_net(channel_descriptor).view(b, self.target_channels, 1, 1)
430
431 # é€‚é…æºç‰¹å¾çš„é€šé“æ•°
432 adapted_source = self.channel_adapter(source_features)
433
434 # åº”ç”¨æ³¨æ„åŠ›æƒé‡
435 attended_features = adapted_source * attention_weights
436
437 # ä¸ç›®æ ‡ç‰¹å¾èåˆ
438 if attended_features.shape == target_features.shape:
439 return target_features + attended_features
440 else:
441 # å°ºå¯¸ä¸åŒ¹é…æ—¶ä½¿ç”¨è‡ªé€‚åº”æ± åŒ–
