31class GradientCorrelationAnalyzer:
32 """梯度相关性分析器"""
33
34 def __init__(self, correlation_threshold=0.1, history_length=10):
35 self.correlation_threshold = correlation_threshold
36 self.history_length = history_length
37 self.gradient_history = defaultdict(deque)
38 self.correlation_cache = {}
39
40 def collect_gradients(self, model: nn.Module) -> Dict[str, torch.Tensor]:
41 """收集模型的梯度信息"""
42 gradients = {}
43
44 for name, param in model.named_parameters():
45 if param.grad is not None and param.requires_grad:
46 # 只收集卷积和线性层的梯度
47 if any(layer_type in name for layer_type in ['conv', 'linear', 'fc']):
48 gradients[name] = param.grad.clone().detach()
49
50 return gradients
51
52 def update_gradient_history(self, gradients: Dict[str, torch.Tensor]) -> None:
53 """更新梯度历史记录"""
54 for name, grad in gradients.items():
55 # 将梯度展平并添加到历史记录
56 flat_grad = grad.view(-1)
57
58 if len(self.gradient_history[name]) >= self.history_length:
59 self.gradient_history[name].popleft()
60
61 self.gradient_history[name].append(flat_grad)
62
63 def calculate_layer_correlation(self, layer1_name: str, layer2_name: str) -> float:
64 """计算两层之间的梯度相关性"""
65
66 # 检查缓存
67 cache_key = tuple(sorted([layer1_name, layer2_name]))
68 if cache_key in self.correlation_cache:
69 return self.correlation_cache[cache_key]
70
71 # 获取梯度历史
72 grad_history1 = self.gradient_history.get(layer1_name, [])
73 grad_history2 = self.gradient_history.get(layer2_name, [])
74
75 if len(grad_history1) < 3 or len(grad_history2) < 3:
76 return 0.0
77
78 # 计算时间序列相关性
79 correlations = []
80 min_length = min(len(grad_history1), len(grad_history2))
81
82 for i in range(min_length):
83 grad1 = grad_history1[i]
84 grad2 = grad_history2[i]
85
86 # 调整到相同维度
87 min_size = min(grad1.size(0), grad2.size(0))
88 if min_size < 10: # 梯度太小，跳过
89 continue
90
91 grad1_sample = grad1[:min_size]
92 grad2_sample = grad2[:min_size]
93
94 # 计算皮尔逊相关系数
95 correlation = self._pearson_correlation(grad1_sample, grad2_sample)
96 if not math.isnan(correlation):
97 correlations.append(abs(correlation)) # 使用绝对值
98
99 # 计算平均相关性
100 avg_correlation = np.mean(correlations) if correlations else 0.0
101
102 # 缓存结果
103 self.correlation_cache[cache_key] = avg_correlation
104
105 return avg_correlation
106
107 def _pearson_correlation(self, x: torch.Tensor, y: torch.Tensor) -> float:
108 """计算皮尔逊相关系数"""
109 try:
110 # 标准化
111 x_centered = x - torch.mean(x)
112 y_centered = y - torch.mean(y)
113
114 # 计算相关系数
115 numerator = torch.sum(x_centered * y_centered)
116 denominator = torch.sqrt(torch.sum(x_centered ** 2) * torch.sum(y_centered ** 2))
117
118 if denominator > 1e-8:
119 correlation = numerator / denominator
120 return correlation.item()
121 else:
122 return 0.0
123
124 except Exception as e:
125 logger.debug(f"Correlation calculation failed: {e}")
126 return 0.0
127
128 def find_beneficial_connections(self, layer_names: List[str]) -> List[Dict]:
129 """找到有益的连接候选"""
130 beneficial_connections = []
131
132 for i, layer1 in enumerate(layer_names):
133 for j, layer2 in enumerate(layer_names):
134 if i >= j: # 避免重复和自连接
135 continue
136
137 # 计算相关性
138 correlation = self.calculate_layer_correlation(layer1, layer2)
139
140 # 判断是否为有益连接
141 if correlation > self.correlation_threshold:
142 # 分析连接类型和层间距离
143 layer_distance = abs(i - j)
144 connection_type = self._determine_connection_type(layer1, layer2, layer_distance)
145
146 beneficial_connections.append({
147 'source_layer': layer1,
148 'target_layer': layer2,
149 'correlation': correlation,
150 'layer_distance': layer_distance,
151 'connection_type': connection_type,
152 'priority': self._calculate_connection_priority(correlation, layer_distance)
153 })
154
155 # 按优先级排序
156 beneficial_connections.sort(key=lambda x: x['priority'], reverse=True)
157
158 return beneficial_connections
159
160 def _determine_connection_type(self, layer1: str, layer2: str, distance: int) -> str:
161 """确定连接类型"""
162
163 # 基于层名称和距离确定连接类型
164 if distance == 1:
165 return 'adjacent' # 相邻层，通常不需要额外连接
166 elif distance == 2:
167 return 'skip_connection' # 跳跃连接
168 elif distance >= 3:
169 if 'conv' in layer1 and 'conv' in layer2:
170 return 'feature_pyramid' # 特征金字塔连接
171 elif 'fc' in layer1 and 'fc' in layer2:
172 return 'dense_connection' # 密集连接
173 else:
174 return 'cross_stage' # 跨阶段连接
175 else:
176 return 'unknown'
177
178 def _calculate_connection_priority(self, correlation: float, distance: int) -> float:
179 """计算连接优先级"""
180
181 # 基础优先级 = 相关性强度
182 base_priority = correlation
183
184 # 距离惩罚：距离越远，优先级略微降低
185 distance_penalty = 1.0 / (1.0 + 0.1 * distance)
186
187 # 最终优先级
188 priority = base_priority * distance_penalty
189
190 return priority

295 def _get_module_by_name(self, model: nn.Module, module_name: str) -> Optional[nn.Module]:
296 """根据名称获取模块"""
297 try:
298 # 尝试直接获取
299 if hasattr(model, 'get_submodule'):
300 return model.get_submodule(module_name)
301 else:
302 # 兼容老版本
303 parts = module_name.split('.')
304 module = model
305 for part in parts:
306 module = getattr(module, part)
307 return module
308 except Exception:
309 return None

311 def _get_layer_output_shape(self, module: nn.Module) -> Optional[Tuple]:
312 """获取层的输出形状"""
313 if isinstance(module, nn.Conv2d):
314 return ('conv', module.out_channels, module.kernel_size, module.stride, module.padding)
315 elif isinstance(module, nn.Linear):
316 return ('linear', module.out_features)
317 elif isinstance(module, nn.BatchNorm2d):
318 return ('bn2d', module.num_features)
319 elif isinstance(module, nn.BatchNorm1d):
320 return ('bn1d', module.num_features)
321 else:
322 return None
323
324 def _create_shape_adapter(self, source_shape: Tuple, target_shape: Tuple) -> Optional[nn.Module]:
325 """创建形状适配器"""
326
327 # Conv2d 到 Conv2d 的适配
328 if source_shape[0] == 'conv' and target_shape[0] == 'conv':
329 source_channels = source_shape[1]
330 target_channels = target_shape[1]
331
332 if source_channels != target_channels:
333 # 通道数适配
334 return nn.Conv2d(source_channels, target_channels, kernel_size=1, bias=False)
335 else:
336 # 形状相同，使用恒等映射
337 return nn.Identity()
338
339 # Linear 到 Linear 的适配
340 elif source_shape[0] == 'linear' and target_shape[0] == 'linear':
341 source_features = source_shape[1]
342 target_features = target_shape[1]
343
344 if source_features != target_features:
345 return nn.Linear(source_features, target_features, bias=False)
346 else:
347 return nn.Identity()
348
349 # Conv2d 到 Linear 的适配 (需要全局池化)
350 elif source_shape[0] == 'conv' and target_shape[0] == 'linear':
351 source_channels = source_shape[1]
352 target_features = target_shape[1]
353
354 return nn.Sequential(
355 nn.AdaptiveAvgPool2d(1),
356 nn.Flatten(),
357 nn.Linear(source_channels, target_features, bias=False)
358 )
359
360 else:
361 logger.warning(f"Unsupported shape adaptation: {source_shape} -> {target_shape}")
362 return None
363
364 def _create_attention_module(self, source_shape: Tuple, target_shape: Tuple) -> Optional[nn.Module]:
365 """创建注意力模块"""
366
367 # 简化的注意力机制
368 if source_shape[0] == 'conv' and target_shape[0] == 'conv':
369 source_channels = source_shape[1]
370 target_channels = target_shape[1]
371
372 # 通道注意力
373 return ChannelAttention(source_channels, target_channels)
374
375 elif source_shape[0] == 'linear' and target_shape[0] == 'linear':
376 source_features = source_shape[1]
377 target_features = target_shape[1]
378
379 # 特征注意力
380 return FeatureAttention(source_features, target_features)
381
382 else:
383 return None
384
385 def _add_module_to_model(self, model: nn.Module, module_name: str, module: nn.Module) -> None:
386 """向模型添加新模块"""
387 # 将模块添加到模型的一个特殊容器中
388 if not hasattr(model, '_dnm_connections'):
389 model._dnm_connections = nn.ModuleDict()
390
391 model._dnm_connections[module_name] = module

394class ChannelAttention(nn.Module):
395 """通道注意力模块"""
396
397 def __init__(self, source_channels: int, target_channels: int, reduction_ratio: int = 16):
398 super().__init__()
399
400 self.source_channels = source_channels
401 self.target_channels = target_channels
402
403 # 全局平均池化
404 self.global_avg_pool = nn.AdaptiveAvgPool2d(1)
405
406 # 注意力网络
407 hidden_channels = max(source_channels // reduction_ratio, 1)
408 self.attention_net = nn.Sequential(
409 nn.Linear(source_channels, hidden_channels),
410 nn.ReLU(inplace=True),
411 nn.Linear(hidden_channels, target_channels),
412 nn.Sigmoid()
413 )
414
415 # 通道适配器
416 if source_channels != target_channels:
417 self.channel_adapter = nn.Conv2d(source_channels, target_channels, 1, bias=False)
418 else:
419 self.channel_adapter = nn.Identity()
420
421 def forward(self, source_features: torch.Tensor, target_features: torch.Tensor) -> torch.Tensor:
422 # 计算注意力权重
423 b, c, h, w = source_features.size()
424
425 # 全局池化得到通道描述符
426 channel_descriptor = self.global_avg_pool(source_features).view(b, c)
427
428 # 计算注意力权重
429 attention_weights = self.attention_net(channel_descriptor).view(b, self.target_channels, 1, 1)
430
431 # 适配源特征的通道数
432 adapted_source = self.channel_adapter(source_features)
433
434 # 应用注意力权重
435 attended_features = adapted_source * attention_weights
436
437 # 与目标特征融合
438 if attended_features.shape == target_features.shape:
439 return target_features + attended_features
440 else:
441 # 尺寸不匹配时使用自适应池化
442 attended_features = F.adaptive_avg_pool2d(attended_features, target_features.shape[2:])
443 return target_features + attended_features

446class FeatureAttention(nn.Module):
447 """特征注意力模块"""
448
449 def __init__(self, source_features: int, target_features: int):
450 super().__init__()
451
452 self.source_features = source_features
453 self.target_features = target_features
454
455 # 注意力权重计算
456 hidden_size = max(min(source_features, target_features) // 4, 1)
457 self.attention_net = nn.Sequential(
458 nn.Linear(source_features, hidden_size),
459 nn.ReLU(inplace=True),
460 nn.Linear(hidden_size, target_features),
461 nn.Sigmoid()
462 )
463
464 # 特征适配器
465 if source_features != target_features:
466 self.feature_adapter = nn.Linear(source_features, target_features, bias=False)
467 else:
468 self.feature_adapter = nn.Identity()
469
470 def forward(self, source_features: torch.Tensor, target_features: torch.Tensor) -> torch.Tensor:
471 # 计算注意力权重
472 attention_weights = self.attention_net(source_features)
473
474 # 适配源特征
475 adapted_source = self.feature_adapter(source_features)
476
477 # 应用注意力
478 attended_features = adapted_source * attention_weights
479
480 # 与目标特征融合
481 return target_features + attended_features

484class DNMConnectionGrowth:
485 """DNM连接生长主控制器"""
486
487 def __init__(self, config: Optional[Dict] = None):
488 self.config = config or self._default_config()
489 self.analyzer = GradientCorrelationAnalyzer(**self.config['analyzer'])
490 self.builder = ConnectionBuilder()
491 self.connection_statistics = defaultdict(int)
492 self.growth_history = []

494 def _default_config(self) -> Dict:
495 """默认配置"""
496 return {
497 'analyzer': {
498 'correlation_threshold': 0.15,
499 'history_length': 8
500 },
501 'growth': {
502 'max_new_connections': 3,
503 'min_correlation_threshold': 0.1,
504 'growth_frequency': 8, # 每8个epoch分析一次
505 'connection_types': ['skip_connection', 'attention_connection']
506 },
507 'filtering': {
508 'min_layer_distance': 2, # 最小层间距离

514 def collect_and_analyze_gradients(self, model: nn.Module) -> None:
515 """收集并分析梯度"""
516 # 收集当前梯度
517 gradients = self.analyzer.collect_gradients(model)
518
519 # 更新梯度历史
520 self.analyzer.update_gradient_history(gradients)
521
522 logger.debug(f"Collected gradients from {len(gradients)} layers")

524 def analyze_and_grow_connections(self, model: nn.Module, epoch: int) -> Dict[str, Any]:
525 """分析并生长新连接"""
526
527 # 检查是否到了生长时机
528 if epoch % self.config['growth']['growth_frequency'] != 0:
529 return {'connections_grown': 0, 'message': 'Not growth epoch'}
530
531 # 获取所有层名称
532 layer_names = []
533 for name, module in model.named_modules():
534 if isinstance(module, (nn.Conv2d, nn.Linear)) and any(t in name for t in ['conv', 'fc', 'linear']):
535 layer_names.append(name)
536
537 if len(layer_names) < 2:
538 return {'connections_grown': 0, 'message': 'Not enough layers for connections'}
539
540 # 找到有益连接
541 beneficial_connections = self.analyzer.find_beneficial_connections(layer_names)
542
543 # 过滤连接
544 filtered_connections = self._filter_connections(beneficial_connections)
545
546 # 执行连接生长
547 connections_grown = self._grow_connections(model, filtered_connections)
548
549 # 记录生长历史
550 growth_record = {
551 'epoch': epoch,
552 'candidates_found': len(beneficial_connections),
553 'connections_grown': connections_grown,
554 'grown_connections': [conn for conn in filtered_connections[:connections_grown]]
555 }
556 self.growth_history.append(growth_record)
557
558 # 更新统计
559 for conn in filtered_connections[:connections_grown]:
560 self.connection_statistics[conn['connection_type']] += 1
561
562 result = {
563 'connections_grown': connections_grown,
564 'beneficial_connections_found': len(beneficial_connections),
565 'connection_candidates': filtered_connections[:5], # 前5个候选
566 'growth_history': growth_record,
567 'message': f'Successfully grown {connections_grown} connections'
568 }
569
570 logger.info(f"DNM Connection Growth completed: {connections_grown} connections grown")
571 return result
572
573 def _filter_connections(self, connections: List[Dict]) -> List[Dict]:
574 """过滤连接候选"""
575 filtered = []
576
577 min_distance = self.config['filtering']['min_layer_distance']
578 max_distance = self.config['filtering']['max_layer_distance']
579 max_connections = self.config['growth']['max_new_connections']
580
581 for conn in connections:
582 # 距离过滤
583 if not (min_distance <= conn['layer_distance'] <= max_distance):
584 continue
585
586 # 相关性过滤
587 if conn['correlation'] < self.config['growth']['min_correlation_threshold']:
588 continue
589
590 # 避免重复连接
591 connection_id = f"{conn['source_layer']}_to_{conn['target_layer']}"
592 if self.config['filtering']['avoid_redundant_connections']:
593 if connection_id in self.builder.built_connections:
594 continue
595
596 # 添加连接ID
597 conn['connection_id'] = connection_id
598 filtered.append(conn)
599
600 # 限制数量
601 if len(filtered) >= max_connections:
602 break
603
604 return filtered
605
606 def _grow_connections(self, model: nn.Module, connections: List[Dict]) -> int:
607 """执行连接生长"""
608 grown_count = 0
609
610 for conn in connections:
611 try:
612 connection_type = conn['connection_type']
613 connection_id = conn['connection_id']
614 source_layer = conn['source_layer']
615 target_layer = conn['target_layer']
616
617 success = False
618
619 # 根据连接类型生长
620 if connection_type in ['skip_connection', 'dense_connection']:
621 success = self.builder.build_skip_connection(
622 model, source_layer, target_layer, connection_id
623 )
624 elif connection_type in ['feature_pyramid', 'cross_stage']:
625 success = self.builder.build_attention_connection(
626 model, source_layer, target_layer, connection_id
627 )
628
629 if success:
630 grown_count += 1
631 logger.info(f"Successfully grown {connection_type}: {source_layer} -> {target_layer}")
632 else:
633 logger.warning(f"Failed to grow connection: {source_layer} -> {target_layer}")
634
635 except Exception as e:
636 logger.error(f"Error growing connection {conn.get('connection_id', 'unknown')}: {e}")
637 continue
638
639 return grown_count
