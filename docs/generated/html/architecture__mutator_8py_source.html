78 def _add_layer_to_sequential(self, model: nn.Sequential, position: int,
79 layer_type: str, **kwargs) -> nn.Sequential:
80 """向Sequential模型添加层"""

89 else:
90 raise ValueError(f"Unsupported layer type: {layer_type}")
91
92 # 函数保持初始化
93 if self.preserve_function and self.initializer:
94 new_layer = self.initializer.identity_layer_init(new_layer)
95
96 # 插入新层
97 layers.insert(position, new_layer)
98 new_model = nn.Sequential(*layers)
99
100 self._record_mutation("add_layer", {
101 "position": position,
102 "layer_type": layer_type,
103 "kwargs": kwargs

316 def _integrate_branch(self, model: nn.Module, branch_point: str,
317 branch: nn.Module, merge_strategy: str) -> nn.Module:
318 """将分支集成到模型中"""
319 # 简化版本：创建包装器
320 class BranchedModel(nn.Module):
321 def __init__(self, main_model, branch, branch_point, merge_strategy):
322 super().__init__()
323 self.main_model = main_model
324 self.branch = branch
325 self.branch_point = branch_point
326 self.merge_strategy = merge_strategy
327
328 def forward(self, x):
329 # 这里需要实现分支逻辑
330 main_out = self.main_model(x)
331 branch_out = self.branch(x)
332
333 if self.merge_strategy == "add":
334 return main_out + branch_out
335 elif self.merge_strategy == "concat":
336 return torch.cat([main_out, branch_out], dim=1)
337 else:
338 return main_out # 默认只返回主路径
339
340 return BranchedModel(model, branch, branch_point, merge_strategy)

342 def _random_genotype_mutation(self, genotype: Genotype) -> Genotype:
343 """随机基因型突变"""
344 import random
345
346 # 复制当前基因型
347 normal_ops = list(genotype.normal)
348 reduce_ops = list(genotype.reduce)
349
350 # 随机突变几个操作
351 num_mutations = max(1, int(len(normal_ops) * self.mutation_strength))
352
353 for _ in range(num_mutations):
354 if random.random() < 0.5 and normal_ops: # 突变normal操作
355 idx = random.randint(0, len(normal_ops) - 1)
356 new_op = random.choice(PRIMITIVES)
357 normal_ops[idx] = (new_op, normal_ops[idx][1])
358 elif reduce_ops: # 突变reduce操作
359 idx = random.randint(0, len(reduce_ops) - 1)
360 new_op = random.choice(PRIMITIVES)
361 reduce_ops[idx] = (new_op, reduce_ops[idx][1])
362
363 return Genotype(
364 normal=normal_ops,
365 normal_concat=genotype.normal_concat,
366 reduce=reduce_ops,
367 reduce_concat=genotype.reduce_concat

433class GradualArchitectureGrowth:
434 """
435 渐进式架构生长
436
437 实现网络架构的渐进式增长，避免剧烈变化
438 """
439
440 def __init__(self, mutator: ArchitectureMutator,
441 growth_schedule: Optional[List[Dict]] = None):
442 """
443 Args:
444 mutator: 架构突变器
445 growth_schedule: 生长计划
446 """
447 self.mutator = mutator
448 self.growth_schedule = growth_schedule or self._default_growth_schedule()
449 self.current_stage = 0
450
451 logger.info(f"🌱 Gradual Architecture Growth initialized with {len(self.growth_schedule)} stages")
452
453 def _default_growth_schedule(self) -> List[Dict]:
454 """默认生长计划"""
455 return [
456 {"epoch": 10, "action": "expand_channels", "params": {"expansion_factor": 1.2}},
457 {"epoch": 20, "action": "add_layer", "params": {"layer_type": "residual"}},
458 {"epoch": 30, "action": "expand_channels", "params": {"expansion_factor": 1.5}},
459 {"epoch": 40, "action": "add_branch", "params": {"merge_strategy": "add"}}
460 ]
461
462 def should_grow(self, current_epoch: int) -> bool:
463 """判断是否应该生长"""
464 if self.current_stage >= len(self.growth_schedule):
465 return False
466
467 target_epoch = self.growth_schedule[self.current_stage]["epoch"]
468 return current_epoch >= target_epoch
469
470 def perform_growth(self, model: nn.Module, current_epoch: int) -> Optional[nn.Module]:
471 """执行生长操作"""
472 if not self.should_grow(current_epoch):
473 return None
474
475 stage_config = self.growth_schedule[self.current_stage]
476 action = stage_config["action"]
477 params = stage_config.get("params", {})
478
479 logger.info(f"🌿 Performing growth stage {self.current_stage}: {action}")
480
481 if action == "expand_channels":
482 result = self.mutator.expand_channels(model, "conv1", **params)
483 elif action == "add_layer":
484 result = self.mutator.add_layer(model, -1, **params)
485 elif action == "add_branch":
486 result = self.mutator.add_branch(model, "conv1", [{"type": "conv"}], **params)
487 else:
488 logger.warning(f"Unknown growth action: {action}")
489 result = model
490
491 self.current_stage += 1
492 return result
