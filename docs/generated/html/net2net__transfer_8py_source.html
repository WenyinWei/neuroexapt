35 def net2wider_conv(self,
36 conv_layer: nn.Conv2d,
37 next_layer: Optional[nn.Module],
38 new_width: int,
39 noise_std: float = 1e-7) -> Tuple[nn.Conv2d, Optional[nn.Module]]:
40 """
41 卷积层宽度扩展 (增加输出通道数)
42
43 Args:
44 conv_layer: 要扩展的卷积层
45 next_layer: 后续层(需要相应调整输入通道)
46 new_width: 新的输出通道数
47 noise_std: 添加的噪声标准差(用于打破对称性)
48
49 Returns:
50 (新的卷积层, 新的后续层)
51 """
52 old_width = conv_layer.out_channels
53 if new_width <= old_width:
54 raise ValueError(f"New width {new_width} must be greater than old width {old_width}")
55
56 # 创建新的卷积层
57 new_conv = nn.Conv2d(
58 in_channels=conv_layer.in_channels,
59 out_channels=new_width,
60 kernel_size=conv_layer.kernel_size,
61 stride=conv_layer.stride,
62 padding=conv_layer.padding,
63 dilation=conv_layer.dilation,
64 groups=conv_layer.groups,
65 bias=conv_layer.bias is not None,
66 padding_mode=conv_layer.padding_mode
67 )
68
69 # 随机映射函数 g: {1,...,new_width} -> {1,...,old_width}
70 # 前old_width个直接对应，后面的随机选择
71 mapping = list(range(old_width)) + [
72 np.random.randint(0, old_width) for _ in range(new_width - old_width)
73 ]
74
75 # 计算每个旧通道被复制的次数
76 replication_count = {}
77 for i, src_idx in enumerate(mapping):
78 replication_count[src_idx] = replication_count.get(src_idx, 0) + 1
79
80 # 复制权重
81 with torch.no_grad():
82 for new_idx, old_idx in enumerate(mapping):
83 new_conv.weight[new_idx] = conv_layer.weight[old_idx].clone()
84 # 添加小噪声打破对称性
85 if new_idx >= old_width: # 只对复制的权重添加噪声
86 new_conv.weight[new_idx] += torch.randn_like(new_conv.weight[new_idx]) * noise_std
87
88 if conv_layer.bias is not None:
89 for new_idx, old_idx in enumerate(mapping):
90 new_conv.bias[new_idx] = conv_layer.bias[old_idx].clone()
91 if new_idx >= old_width:
92 new_conv.bias[new_idx] += torch.randn_like(new_conv.bias[new_idx]) * noise_std
93
94 # 处理后续层
95 new_next_layer = None
96 if next_layer is not None:
97 if isinstance(next_layer, nn.Conv2d):
98 new_next_layer = self._wider_next_conv(next_layer, mapping, replication_count)
99 elif isinstance(next_layer, nn.Linear):
100 new_next_layer = self._wider_next_linear(next_layer, mapping, replication_count)
101 elif isinstance(next_layer, nn.BatchNorm2d):
102 new_next_layer = self._wider_next_batchnorm2d(next_layer, mapping, replication_count)
103
104 return new_conv, new_next_layer

106 def _wider_next_conv(self, conv_layer: nn.Conv2d, mapping: List[int],
107 replication_count: Dict[int, int]) -> nn.Conv2d:
108 """扩展后续卷积层的输入通道"""
109 old_in_channels = conv_layer.in_channels
110 new_in_channels = len(mapping)
111
112 new_conv = nn.Conv2d(
113 in_channels=new_in_channels,
114 out_channels=conv_layer.out_channels,
115 kernel_size=conv_layer.kernel_size,
116 stride=conv_layer.stride,
117 padding=conv_layer.padding,
118 dilation=conv_layer.dilation,
119 groups=conv_layer.groups,
120 bias=conv_layer.bias is not None,
121 padding_mode=conv_layer.padding_mode
122 )
123
124 with torch.no_grad():
125 for new_idx, old_idx in enumerate(mapping):
126 # 权重需要按复制次数缩放
127 scale_factor = 1.0 / replication_count[old_idx]
128 new_conv.weight[:, new_idx] = conv_layer.weight[:, old_idx] * scale_factor
129
130 if conv_layer.bias is not None:
131 new_conv.bias.copy_(conv_layer.bias)
132
133 return new_conv

135 def _wider_next_linear(self, linear_layer: nn.Linear, mapping: List[int],
136 replication_count: Dict[int, int]) -> nn.Linear:
137 """扩展后续线性层的输入特征"""
138 old_in_features = linear_layer.in_features
139 new_in_features = len(mapping)
140
141 new_linear = nn.Linear(new_in_features, linear_layer.out_features,
142 bias=linear_layer.bias is not None)
143
144 with torch.no_grad():
145 for new_idx, old_idx in enumerate(mapping):
146 scale_factor = 1.0 / replication_count[old_idx]
147 new_linear.weight[:, new_idx] = linear_layer.weight[:, old_idx] * scale_factor
148
149 if linear_layer.bias is not None:
150 new_linear.bias.copy_(linear_layer.bias)
151
152 return new_linear

154 def _wider_next_batchnorm2d(self, bn_layer: nn.BatchNorm2d, mapping: List[int],
155 replication_count: Dict[int, int]) -> nn.BatchNorm2d:
156 """扩展后续BatchNorm层"""
157 new_num_features = len(mapping)
158 new_bn = nn.BatchNorm2d(new_num_features, eps=bn_layer.eps,
159 momentum=bn_layer.momentum, affine=bn_layer.affine)
160
161 if bn_layer.affine:
162 with torch.no_grad():
163 for new_idx, old_idx in enumerate(mapping):
164 new_bn.weight[new_idx] = bn_layer.weight[old_idx]
165 new_bn.bias[new_idx] = bn_layer.bias[old_idx]
166
167 # 复制运行时统计信息
168 if bn_layer.running_mean is not None:
169 new_bn.running_mean = torch.zeros(new_num_features)
170 new_bn.running_var = torch.ones(new_num_features)
171 for new_idx, old_idx in enumerate(mapping):
172 new_bn.running_mean[new_idx] = bn_layer.running_mean[old_idx]
173 new_bn.running_var[new_idx] = bn_layer.running_var[old_idx]
174
175 return new_bn

177 def net2deeper_conv(self, reference_layer: nn.Conv2d,
178 activation: str = 'relu') -> nn.Conv2d:
179 """
180 在卷积层后插入恒等映射层
181
182 Args:
183 reference_layer: 参考层(用于确定新层的通道数)
184 activation: 激活函数类型
185
186 Returns:
187 恒等映射的新卷积层
188 """
189 # 创建恒等卷积层
190 identity_conv = nn.Conv2d(
191 in_channels=reference_layer.out_channels,
192 out_channels=reference_layer.out_channels,
193 kernel_size=3, # 使用3x3卷积
194 stride=1,
195 padding=1, # 保持尺寸不变
196 bias=True
197 )
198
199 # 初始化为恒等映射
200 with torch.no_grad():
201 nn.init.zeros_(identity_conv.weight)
202 # 设置中心权重为1(恒等核)
203 for i in range(reference_layer.out_channels):
204 identity_conv.weight[i, i, 1, 1] = 1.0
205
206 # 偏置设为0
207 if identity_conv.bias is not None:
208 nn.init.zeros_(identity_conv.bias)
209
210 return identity_conv

212 def net2deeper_linear(self, reference_layer: nn.Linear) -> nn.Linear:
213 """
214 在线性层后插入恒等映射层
215 """
216 identity_linear = nn.Linear(
217 in_features=reference_layer.out_features,
218 out_features=reference_layer.out_features,
219 bias=True
220 )
221
222 with torch.no_grad():
223 # 恒等权重矩阵
224 nn.init.eye_(identity_linear.weight)
225 # 零偏置
226 nn.init.zeros_(identity_linear.bias)
227
228 return identity_linear

230 def net2branch(self, base_layer: nn.Module,
231 num_branches: int = 2) -> List[nn.Module]:
232 """
233 创建分支结构，每个分支初始化为与基础层相同
234
235 Args:
236 base_layer: 基础层
237 num_branches: 分支数量
238
239 Returns:
240 分支层列表
241 """
242 branches = []
243
244 for i in range(num_branches):
245 if isinstance(base_layer, nn.Conv2d):
246 branch = nn.Conv2d(
247 in_channels=base_layer.in_channels,
248 out_channels=base_layer.out_channels,
249 kernel_size=base_layer.kernel_size,
250 stride=base_layer.stride,
251 padding=base_layer.padding,
252 dilation=base_layer.dilation,
253 groups=base_layer.groups,
254 bias=base_layer.bias is not None,
255 padding_mode=base_layer.padding_mode
256 )
257 elif isinstance(base_layer, nn.Linear):
258 branch = nn.Linear(
259 in_features=base_layer.in_features,
260 out_features=base_layer.out_features,
261 bias=base_layer.bias is not None
262 )
263 else:
264 raise NotImplementedError(f"Branch creation not implemented for {type(base_layer)}")
265
266 # 复制权重
267 with torch.no_grad():
268 branch.weight.copy_(base_layer.weight)
269 if base_layer.bias is not None:
270 branch.bias.copy_(base_layer.bias)
271
272 # 第二个分支之后添加小噪声
273 if i > 0:
274 noise_std = 1e-7
275 branch.weight += torch.randn_like(branch.weight) * noise_std
276 if branch.bias is not None:
277 branch.bias += torch.randn_like(branch.bias) * noise_std
278
279 branches.append(branch)
280
281 return branches

283 def smooth_transition_loss(self, student_output: torch.Tensor,
284 teacher_output: torch.Tensor,
285 alpha: float = 0.1) -> torch.Tensor:
286 """
287 平滑过渡损失函数
288
289 Args:
290 student_output: 学生网络输出
291 teacher_output: 教师网络输出
292 alpha: 过渡损失权重
293
294 Returns:
295 平滑过渡损失
296 """
297 mse_loss = F.mse_loss(student_output, teacher_output.detach())
298 return alpha * mse_loss

300 def verify_function_preserving(self, teacher_model: nn.Module,
301 student_model: nn.Module,
302 test_input: torch.Tensor,
303 tolerance: float = 1e-5) -> bool:
304 """
305 验证函数保持性质
306
307 Args:
308 teacher_model: 教师模型
309 student_model: 学生模型
310 test_input: 测试输入
311 tolerance: 容差
312
313 Returns:
314 是否保持函数相等
315 """
316 teacher_model.eval()
317 student_model.eval()
318
319 with torch.no_grad():
320 teacher_output = teacher_model(test_input)
321 student_output = student_model(test_input)
322
323 max_diff = torch.max(torch.abs(teacher_output - student_output))
324 return max_diff.item() < tolerance
