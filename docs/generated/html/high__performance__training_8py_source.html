83class CosineAnnealingWarmUpRestarts(optim.lr_scheduler._LRScheduler):
84 """带热身的余弦退火重启学习率调度器"""
85
86 def __init__(self, optimizer, T_0, T_mult=1, eta_max=0.1, T_up=0, gamma=1., last_epoch=-1):
87 if T_0 <= 0 or not isinstance(T_0, int):
88 raise ValueError("Expected positive integer T_0, but got {}".format(T_0))
89 if T_mult < 1 or not isinstance(T_mult, int):
90 raise ValueError("Expected integer T_mult >= 1, but got {}".format(T_mult))
91 if T_up < 0 or not isinstance(T_up, int):
92 raise ValueError("Expected non-negative integer T_up, but got {}".format(T_up))
93
94 self.T_0 = T_0
95 self.T_mult = T_mult
96 self.base_eta_max = eta_max
97 self.eta_max = eta_max
98 self.T_up = T_up
99 self.T_i = T_0
100 self.gamma = gamma
101 self.cycle = 0
102 self.T_cur = last_epoch
103
104 super(CosineAnnealingWarmUpRestarts, self).__init__(optimizer, last_epoch)
105
106 def get_lr(self):
107 if self.T_cur == -1:
108 return self.base_lrs
109 elif self.T_cur < self.T_up:
110 return [(self.eta_max - base_lr)*self.T_cur / self.T_up + base_lr for base_lr in self.base_lrs]
111 else:
112 return [base_lr + (self.eta_max - base_lr) * (1 + math.cos(math.pi * (self.T_cur-self.T_up) / (self.T_i - self.T_up))) / 2
113 for base_lr in self.base_lrs]
114
115 def step(self, epoch=None):
116 if epoch is None:
117 epoch = self.last_epoch + 1
118 self.T_cur = self.T_cur + 1
119 if self.T_cur >= self.T_i:
120 self.cycle += 1
121 self.T_cur = self.T_cur - self.T_i
122 self.T_i = (self.T_i - self.T_up) * self.T_mult + self.T_up
123 else:
124 if epoch >= self.T_0:
125 if self.T_mult == 1:
126 self.T_cur = epoch % self.T_0
127 self.cycle = epoch // self.T_0
128 else:
129 n = int(math.log((epoch / self.T_0 * (self.T_mult - 1) + 1), self.T_mult))
130 self.cycle = n
131 self.T_cur = epoch - self.T_0 * (self.T_mult ** n - 1) / (self.T_mult - 1)
132 self.T_i = self.T_0 * self.T_mult ** (n)
133 else:
134 self.T_i = self.T_0
135 self.T_cur = epoch
136
137 self.eta_max = self.base_eta_max * (self.gamma**self.cycle)
138 self.last_epoch = math.floor(epoch)
139 for param_group, lr in zip(self.optimizer.param_groups, self.get_lr()):
140 param_group['lr'] = lr

368def train_epoch_separated(separated_trainer, train_queue, valid_queue, epoch, args, scaler=None):
369 """分离训练一个epoch"""
370 start_time = time.time()
371
372 # 使用分离训练器
373 stats = separated_trainer.train_epoch(train_queue, valid_queue, epoch)
374
375 train_time = time.time() - start_time
376
377 # 获取训练统计
378 if 'accuracy' in stats:
379 train_acc = stats['accuracy']
380 train_loss = stats['loss']
381 else:
382 # 架构训练epoch，准确率需要通过验证获得
383 train_acc = 0.0
384 train_loss = stats['loss']
385
386 return train_acc, train_loss, train_time

388def validate_epoch(valid_queue, model, criterion):
389 """验证一个epoch"""
390 model.eval()
391
392 objs = AvgrageMeter()
393 top1 = AvgrageMeter()
394 top5 = AvgrageMeter()
395
396 with torch.no_grad():
397 for input, target in tqdm(valid_queue, desc="验证", leave=False):
398 input = input.cuda(non_blocking=True)
399 target = target.cuda(non_blocking=True)
400
401 logits = model(input)
402 loss = criterion(logits, target)
403
404 prec1, prec5 = accuracy(logits, target, topk=(1, 5))
405 n = input.size(0)
406 objs.update(loss.item(), n)
407 top1.update(prec1.item(), n)
408 top5.update(prec5.item(), n)
409
410 return top1.avg, top5.avg, objs.avg
