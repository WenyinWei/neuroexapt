67class Resizing(nn.Module):
68 """
69 A utility module to resize tensors to a target channel count.
70 This is used to match channel dimensions when operations with different
71 channel counts are mixed.
72 """
73 def __init__(self, C_in, C_out, affine=True):
74 super(Resizing, self).__init__()
75 self.C_in = C_in
76 self.C_out = C_out
77 self.op = nn.Sequential(
78 nn.ReLU(inplace=False),
79 nn.Conv2d(C_in, C_out, 1, stride=1, padding=0, bias=False),
80 nn.BatchNorm2d(C_out, affine=affine)
81 )
82
83 def forward(self, x):
84 if self.C_in == self.C_out:
85 return x
86 return self.op(x)

88class GatedCell(nn.Module):

90 def __init__(self, cell, C_in, C_out):

92 self.cell = cell
93 self.gate = nn.Parameter(1e-3 * torch.randn(1))
94 self.resize = Resizing(C_in, C_out) if C_in != C_out else None

96 def forward(self, s0, s1, weights):
97 cell_out = self.cell(s0, s1, weights)
98 gated_out = self.gate.sigmoid() * cell_out

101 s1_resized = self.resize(s1) if self.resize else s1
102
103 return s1_resized + gated_out

190class Network(nn.Module):
191 """
192 The full neural network model, composed of a stack of cells.
193 This class also initializes and stores the architecture parameters (alphas).
194 This version is simplified to remove all optimization flags.
195 """
196
197 def __init__(self, C, num_classes, layers, steps=4, block_multiplier=4, stem_multiplier=3, quiet=False):
198 super(Network, self).__init__()
199 self._C = C
200 self._num_classes = num_classes
201 self._layers = layers
202 self._steps = steps
203 self._block_multiplier = block_multiplier
204 self.quiet = quiet

207 print("üèóÔ∏è Building Search Network...")

209 C_curr = stem_multiplier * C
210 self.stem = nn.Sequential(
211 nn.Conv2d(3, C_curr, 3, padding=1, bias=False),
212 nn.BatchNorm2d(C_curr)
213 )
214
215 C_prev_prev, C_prev, C_curr = C_curr, C_curr, C
216 self.cells = nn.ModuleList()
217 reduction_prev = False
218 for i in range(layers):
219 if i in [layers // 3, 2 * layers // 3]:
220 C_curr *= 2
221 reduction = True
222 else:
223 reduction = False

225 cell = Cell(steps, block_multiplier, C_prev_prev, C_prev, C_curr, reduction, reduction_prev)

227 self.cells.append(cell)
228 C_prev_prev, C_prev = C_prev, self._block_multiplier * C_curr

231 self.classifier = nn.Linear(C_prev, num_classes)

242 def forward(self, input):
243 weights_normal = F.softmax(self.alphas_normal, dim=-1)
244 weights_reduce = F.softmax(self.alphas_reduce, dim=-1)

246 s0 = s1 = self.stem(input)

272 def genotype(self):
273 """
274 Decodes the learned architecture parameters (alphas) into a discrete genotype.
275 This method determines the final architecture of the network after search.
276 """
277
278 def _parse(weights):
279 gene = []
280 n = 2
281 start = 0
282 for i in range(self._steps):
283 end = start + n
284 W = weights[start:end].copy()
285
286 # For each node, find the two strongest predecessor connections
287 edges = sorted(range(i + 2), key=lambda x: -max(W[x][k] for k in range(len(W[x])) if k != PRIMITIVES.index('none')) if any(W[x][k] for k in range(len(W[x])) if k != PRIMITIVES.index('none')) else -1.0)[:2]
288
289 # For each of the two selected edges, find the best operation
290 for j in edges:
291 k_best = None
292 for k in range(len(W[j])):
293 if k != PRIMITIVES.index('none'):
294 if k_best is None or W[j][k] > W[j][k_best]:
295 k_best = k
296 # Default to a safe choice if all options are zero
297 if k_best is None:
298 k_best = 1 # 'max_pool_3x3' as a fallback
299 gene.append((PRIMITIVES[k_best], j))
300 start = end
301 n += 1
302 return gene
303
304 # Normalize the alpha values using softmax
305 gene_normal = _parse(torch.softmax(self.alphas_normal, dim=-1).data.cpu().numpy())
306 gene_reduce = _parse(torch.softmax(self.alphas_reduce, dim=-1).data.cpu().numpy())
307
308 concat = range(2 + self._steps - self._block_multiplier, self._steps + 2)
309 genotype = Genotype(
310 normal=gene_normal, normal_concat=concat,
311 reduce=gene_reduce, reduce_concat=concat
312 )
313 return genotype
