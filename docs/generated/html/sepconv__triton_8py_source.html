19 def _dwconv3x3_kernel(
20 x_ptr, # [B, C, H, W]
21 w_ptr, # [C, 1, 3, 3]
22 y_ptr,
23 B, C, H, W,
24 stride: tl.constexpr,
25 BLOCK_SIZE: tl.constexpr,
26 ):
27 """Depth-wise 3×3 conv (padding=1) with optional stride."""
28 pid = tl.program_id(0)
29
30 OH = (H + 2 - 3) // stride + 1
31 OW = (W + 2 - 3) // stride + 1
32 total_outputs = B * C * OH * OW
33
34 offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
35 mask = offsets < total_outputs
36
37 # 将1D索引转换为4D坐标 (b, c, oh, ow)
38 ow = offsets % OW
39 oh = (offsets // OW) % OH
40 c = (offsets // (OW * OH)) % C
41 b = offsets // (OW * OH * C)
42
43 acc = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
44
45 # 3x3深度卷积
46 for kh in tl.static_range(3):
47 for kw in tl.static_range(3):
48 ih = oh * stride + kh - 1 # padding=1
49 iw = ow * stride + kw - 1
50
51 # 边界检查
52 valid = mask & (ih >= 0) & (ih < H) & (iw >= 0) & (iw < W)
53
54 # 输入索引
55 input_idx = b * C * H * W + c * H * W + ih * W + iw
56 # 权重索引 (每个通道有自己的3x3权重)
57 weight_idx = c * 9 + kh * 3 + kw # 9 = 3*3
58
59 x_val = tl.load(x_ptr + input_idx, mask=valid, other=0.0)
60 w_val = tl.load(w_ptr + weight_idx, mask=(c < C), other=0.0)
61
62 acc += x_val * w_val
63
64 # 存储结果
65 output_idx = b * C * OH * OW + c * OH * OW + oh * OW + ow
66 tl.store(y_ptr + output_idx, acc, mask=mask)

79

83 mask = offsets < total_outputs
84

86 w_idx = offsets % W
87 h_idx = (offsets // W) % H

97
