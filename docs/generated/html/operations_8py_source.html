97class FactorizedReduce(nn.Module):
98 """Reduces the spatial dimensions and doubles the channel dimensions."""
99 def __init__(self, C_in, C_out, affine=True):
100 super(FactorizedReduce, self).__init__()
101 assert C_out % 2 == 0
102 self.relu = nn.ReLU(inplace=False)
103 self.conv_1 = nn.Conv2d(C_in, C_out // 2, 1, stride=2, padding=0, bias=False)
104 self.conv_2 = nn.Conv2d(C_in, C_out // 2, 1, stride=2, padding=0, bias=False)
105 self.bn = nn.BatchNorm2d(C_out, affine=affine)
106
107 def forward(self, x):
108 x = self.relu(x)
109 # Slicing to handle odd input dimensions
110 out = torch.cat([self.conv_1(x), self.conv_2(x[:, :, 1:, 1:])], dim=1)
111 out = self.bn(out)
112 return out

114class MixedOp(nn.Module):
115 """
116 A standard, simplified mixed operation.
117 It computes a weighted sum of a set of operations.
118 """
119 def __init__(self, C, stride):
120 super(MixedOp, self).__init__()
121 from .genotypes import PRIMITIVES
122
123 self._ops = nn.ModuleList()
124 for primitive in PRIMITIVES:
125 op = OPS[primitive](C, stride, False)
126 # Add batch norm after pooling operations for stability
127 if 'pool' in primitive:
128 op = nn.Sequential(op, nn.BatchNorm2d(C, affine=False))
129 self._ops.append(op)
130
131 def forward(self, x, weights):
132 """
133 Args:
134 x: input tensor
135 weights: a tensor of shape [num_ops], representing arch params.
136 Returns:
137 The weighted sum of the outputs of all operations.
138 """
139 # Calculate the weighted sum of operations
140 # This implementation avoids creating a large intermediate tensor of all outputs.
141 # It computes and adds them one by one.
142
143 # Find the first op with a non-zero weight to initialize the output tensor
144 # This handles cases where the first op might be 'none'
145 res = None
146 for i, (op, w) in enumerate(zip(self._ops, weights)):
147 if w > 1e-8: # A small threshold to avoid negligible computations
148 if res is None:
149 res = op(x) * w
150 else:
151 res += op(x) * w
152
153 # If all weights are zero, return a zero tensor
154 if res is None:
155 # For reduction cells, the output shape is halved.
156 if self._ops[1].stride == 2: # A simple way to check for reduction
157 return torch.zeros_like(x[:, :, ::2, ::2])
158 else:
159 return torch.zeros_like(x)
160
161 return res
