23class CleanCell(nn.Module):
24 """
25 完全重构的Cell实现 - 保证通道数计算正确
26 """
27 def __init__(self, steps: int, block_multiplier: int,
28 C_prev_prev: int, C_prev: int, C: int,
29 reduction: bool, reduction_prev: bool):
30 super(CleanCell, self).__init__()
31
32 self.reduction = reduction
33 self.steps = steps
34 self.block_multiplier = block_multiplier
35
36 # 🔧 关键修复：确保预处理层通道数正确
37 if reduction_prev:
38 # 前一层是reduction，需要FactorizedReduce
39 self.preprocess0 = FactorizedReduce(C_prev_prev, C, affine=False)
40 else:
41 # 前一层是normal，使用1x1 conv调整通道数
42 self.preprocess0 = ReLUConvBN(C_prev_prev, C, 1, 1, 0, affine=False)
43
44 # s1的预处理 - 从C_prev调整到C
45 self.preprocess1 = ReLUConvBN(C_prev, C, 1, 1, 0, affine=False)
46
47 # 创建MixedOp - 所有操作的输入输出通道数都是C
48 self._ops = nn.ModuleList()
49 for i in range(steps):
50 for j in range(2 + i): # 每个节点连接到前面所有节点
51 # 在reduction cell中，前两个节点的操作使用stride=2
52 stride = 2 if reduction and j < 2 else 1
53 op = MixedOp(C, stride) # 使用最基础的MixedOp
54 self._ops.append(op)
55
56 def forward(self, s0: torch.Tensor, s1: torch.Tensor, weights: torch.Tensor) -> torch.Tensor:
57 """前向传播 - 确保尺寸和通道数正确"""
58
59 # 预处理：调整输入到统一的通道数C
60 s0 = self.preprocess0(s0) # [B, C_prev_prev, H, W] -> [B, C, H', W']
61 s1 = self.preprocess1(s1) # [B, C_prev, H, W] -> [B, C, H, W]
62
63 # 确保s0和s1的空间尺寸匹配
64 if s0.shape[2:] != s1.shape[2:]:
65 # 如果尺寸不匹配，调整s0到s1的尺寸
66 s0 = F.interpolate(s0, size=s1.shape[2:], mode='bilinear', align_corners=False)
67
68 states = [s0, s1] # 初始状态
69 offset = 0
70
71 # 逐步构建中间节点
72 for i in range(self.steps):
73 # 收集当前节点的所有输入
74 node_inputs = []
75 for j in range(len(states)): # 连接到前面所有状态
76 op_idx = offset + j
77 if op_idx < len(self._ops) and op_idx < len(weights):
78 h = self._ops[op_idx](states[j], weights[op_idx])
79 node_inputs.append(h)
80
81 # 求和得到新节点
82 if node_inputs:
83 # 确保所有输入尺寸匹配
84 target_size = node_inputs[0].shape[2:]
85 aligned_inputs = []
86 for inp in node_inputs:
87 if inp.shape[2:] != target_size:
88 inp = F.interpolate(inp, size=target_size, mode='bilinear', align_corners=False)
89 aligned_inputs.append(inp)
90
91 new_state = sum(aligned_inputs)
92 else:
93 # 如果没有有效输入，创建零张量
94 new_state = torch.zeros_like(states[-1])
95
96 states.append(new_state)
97 offset += len(states) - 1 # 更新offset
98
99 # 输出：连接最后block_multiplier个状态
100 output_states = states[-self.block_multiplier:]
101
102 # 确保所有输出状态尺寸匹配
103 if len(output_states) > 1:
104 target_size = output_states[0].shape[2:]
105 aligned_outputs = []
106 for state in output_states:
107 if state.shape[2:] != target_size:
108 state = F.interpolate(state, size=target_size, mode='bilinear', align_corners=False)
109 aligned_outputs.append(state)
110 output_states = aligned_outputs
111
112 result = torch.cat(output_states, dim=1) # [B, C*block_multiplier, H, W]
113 return result

115class CleanNetwork(nn.Module):
116 """
117 完全重构的Network实现 - 保证通道数流动正确
118 """
119 def __init__(self, C: int, num_classes: int, layers: int,
120 steps: int = 4, block_multiplier: int = 4):
121 super(CleanNetwork, self).__init__()
122
123 self._C = C
124 self._num_classes = num_classes
125 self._layers = layers
126 self._steps = steps
127 self._block_multiplier = block_multiplier
128
129 # Stem: 3 -> C*block_multiplier
130 stem_channels = C * block_multiplier
131 self.stem = nn.Sequential(
132 nn.Conv2d(3, stem_channels, 3, padding=1, bias=False),
133 nn.BatchNorm2d(stem_channels)
134 )
135
136 # 🔧 关键修复：正确的通道数初始化和流动
137 self.cells = nn.ModuleList()
138
139 # 初始通道设置
140 C_prev_prev = stem_channels # stem输出
141 C_prev = stem_channels # 第一个cell的输入
142 C_curr = C # 每个cell内部的通道数
143 reduction_prev = False
144
145 print(f"🔧 干净模型通道数流动:")
146 print(f" Stem: 3 -> {stem_channels}")
147 print(f" 初始: C_prev_prev={C_prev_prev}, C_prev={C_prev}, C_curr={C_curr}")
148
149 for i in range(layers):
150 # 确定是否是reduction layer
151 if i in [layers // 3, 2 * layers // 3]:
152 C_curr *= 2 # reduction layer通道数翻倍
153 reduction = True
154 print(f" Layer {i}: Reduction层, C_curr={C_curr}")
155 else:
156 reduction = False
157 print(f" Layer {i}: Normal层, C_curr={C_curr}")
158
159 # 创建Cell
160 cell = CleanCell(
161 steps=steps,
162 block_multiplier=block_multiplier,
163 C_prev_prev=C_prev_prev,
164 C_prev=C_prev,
165 C=C_curr,
166 reduction=reduction,
167 reduction_prev=reduction_prev
168 )
169 self.cells.append(cell)
170
171 # 更新下一轮的通道数
172 reduction_prev = reduction
173 C_prev_prev = C_prev
174 C_prev = C_curr * block_multiplier # cell输出的通道数
175
176 print(f" -> 输出: {C_prev}, 下一轮: C_prev_prev={C_prev_prev}, C_prev={C_prev}")
177
178 # 分类器
179 self.global_pooling = nn.AdaptiveAvgPool2d(1)
180 self.classifier = nn.Linear(C_prev, num_classes)
181
182 # 初始化架构参数
183 self._initialize_alphas()
184
185 print(f" 最终分类器输入: {C_prev} -> {num_classes}")
186 print(f"✅ 干净模型构建完成!")

188 def _initialize_alphas(self):
189 """初始化架构参数"""
190 k = sum(1 for i in range(self._steps) for n in range(2 + i))
191 num_ops = len(PRIMITIVES)
192
193 self.alphas_normal = nn.Parameter(1e-3 * torch.randn(k, num_ops))
194 self.alphas_reduce = nn.Parameter(1e-3 * torch.randn(k, num_ops))
195
196 self._arch_parameters = [self.alphas_normal, self.alphas_reduce]
197
198 def arch_parameters(self):
199 return self._arch_parameters
200
201 def forward(self, input: torch.Tensor) -> torch.Tensor:
202 """前向传播 - 确保尺寸和通道数正确"""
203
204 # 预计算架构权重
205 weights_normal = F.softmax(self.alphas_normal, dim=-1)
206 weights_reduce = F.softmax(self.alphas_reduce, dim=-1)
207
208 # Stem处理
209 s0 = s1 = self.stem(input) # [B, 3, 32, 32] -> [B, C*4, 32, 32]
210
211 # 逐层处理
212 for i, cell in enumerate(self.cells):
213 # 选择权重
214 if cell.reduction:
215 weights = weights_reduce
216 else:
217 weights = weights_normal
218
219 # Cell前向传播
220 s0, s1 = s1, cell(s0, s1, weights)
221
222 # 全局池化和分类
223 out = self.global_pooling(s1) # [B, C_final, H, W] -> [B, C_final, 1, 1]
224 logits = self.classifier(out.view(out.size(0), -1)) # [B, C_final] -> [B, num_classes]
225
226 return logits

228 def genotype(self):
229 """解码架构"""
230 def _parse(weights):
231 gene = []
232 n = 2
233 start = 0
234 for i in range(self._steps):
235 end = start + n
236 W = weights[start:end].copy()
237 edges = sorted(range(i + 2), key=lambda x: -max(W[x][k] for k in range(len(W[x]))
238 if k != PRIMITIVES.index('none')))[:2]
239 for j in edges:
240 k_best = None
241 for k in range(len(W[j])):
242 if k != PRIMITIVES.index('none'):
243 if k_best is None or W[j][k] > W[j][k_best]:
244 k_best = k
245 # 确保k_best不为None
246 if k_best is None:
247 k_best = 1 # 默认使用skip_connect
248 gene.append((PRIMITIVES[k_best], j))
249 start = end
250 n += 1
251 return gene
252
253 gene_normal = _parse(F.softmax(self.alphas_normal, dim=-1).data.cpu().numpy())
254 gene_reduce = _parse(F.softmax(self.alphas_reduce, dim=-1).data.cpu().numpy())
255
256 concat = range(2 + self._steps - self._block_multiplier, self._steps + 2)
257 return Genotype(
258 normal=gene_normal, normal_concat=concat,
259 reduce=gene_reduce, reduce_concat=concat
260 )
