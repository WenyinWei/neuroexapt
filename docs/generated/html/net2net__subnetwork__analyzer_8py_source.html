73 def _build_layer_dependency_graph(self, model: nn.Module) -> Dict[str, List[str]]:
74 """构建层依赖关系图"""
75 layer_graph = defaultdict(list)
76 named_modules = dict(model.named_modules())
77
78 # 分析ResNet风格的前向传播依赖
79 for name, module in named_modules.items():
80 if name == '': # 跳过根模块
81 continue
82
83 # 解析层的逻辑位置
84 parts = name.split('.')
85
86 if len(parts) >= 2:
87 # 对于层级结构，添加依赖关系
88 parent_parts = parts[:-1]
89 parent_name = '.'.join(parent_parts)
90
91 if parent_name in named_modules:
92 layer_graph[parent_name].append(name)
93
94 # 添加特殊的依赖关系（基于ResNet架构）
95 self._add_resnet_dependencies(layer_graph, named_modules)
96
97 return dict(layer_graph)

99 def _add_resnet_dependencies(self, layer_graph: Dict[str, List[str]], named_modules: Dict[str, nn.Module]):
100 """添加ResNet特定的依赖关系"""
101
102 # 主干依赖：conv1 -> feature_blocks -> classifier
103 main_sequence = []
104
105 # 查找主要组件
106 if 'conv1' in named_modules:
107 main_sequence.append('conv1')
108
109 # 查找feature blocks
110 feature_blocks = []
111 for name in named_modules.keys():
112 if name.startswith('feature_block'):
113 if '.' not in name[len('feature_block'):]: # 只要顶级feature_block
114 feature_blocks.append(name)
115
116 feature_blocks.sort() # 按名称排序
117 main_sequence.extend(feature_blocks)
118
119 # 查找分类器
120 if 'classifier' in named_modules:
121 main_sequence.append('classifier')
122
123 # 建立主干依赖
124 for i in range(len(main_sequence) - 1):
125 current = main_sequence[i]
126 next_layer = main_sequence[i + 1]
127 if current not in layer_graph:
128 layer_graph[current] = []
129 layer_graph[current].append(next_layer)

131 def _find_downstream_layers(self,
132 layer_graph: Dict[str, List[str]],
133 start_layer: str,
134 include_start: bool = True) -> Set[str]:
135 """找到指定层之后的所有下游层"""
136
137 downstream_layers = set()
138
139 if include_start:
140 downstream_layers.add(start_layer)
141
142 # BFS遍历找到所有下游层
143 queue = [start_layer]
144 visited = set([start_layer])
145
146 while queue:
147 current_layer = queue.pop(0)
148
149 # 添加当前层的所有下游层
150 if current_layer in layer_graph:
151 for next_layer in layer_graph[current_layer]:
152 if next_layer not in visited:
153 downstream_layers.add(next_layer)
154 queue.append(next_layer)
155 visited.add(next_layer)
156
157 return downstream_layers

323 def _compute_parameter_utilization(self, subnetwork: nn.Module, gradients: torch.Tensor) -> float:
324 """计算参数利用率 - 有多少参数在积极参与学习"""
325
326 total_params = sum(p.numel() for p in subnetwork.parameters())
327 if total_params == 0:
328 return 0.0
329
330 # 计算有效梯度的参数数量
331 active_params = 0
332 for param in subnetwork.parameters():
333 if param.grad is not None:
334 # 梯度显著非零的参数被认为是活跃的
335 significant_grads = torch.abs(param.grad) > 1e-6
336 active_params += significant_grads.sum().item()
337
338 utilization = active_params / total_params
339 return min(utilization, 1.0)

341 def _compute_representation_capacity(self, activations: torch.Tensor, targets: torch.Tensor) -> float:
342 """计算表示能力 - 网络能多好地区分不同类别"""
343
344 try:
345 # 计算类间分离度
346 if len(activations.shape) > 2:
347 # 对于卷积层输出，取平均池化
348 activations_flat = F.adaptive_avg_pool2d(activations, (1, 1)).flatten(1)
349 else:
350 activations_flat = activations
351
352 # 计算不同类别的激活分布
353 unique_targets = torch.unique(targets)
354 if len(unique_targets) < 2:
355 return 0.0
356
357 class_centers = []
358 for target_class in unique_targets:
359 mask = targets == target_class
360 if mask.sum() > 0:
361 class_center = activations_flat[mask].mean(dim=0)
362 class_centers.append(class_center)
363
364 if len(class_centers) < 2:
365 return 0.0
366
367 # 计算类间距离
368 class_centers = torch.stack(class_centers)
369 distances = torch.pdist(class_centers)
370 avg_inter_class_distance = distances.mean().item()
371
372 # 计算类内方差
373 intra_class_variance = 0.0
374 for target_class in unique_targets:
375 mask = targets == target_class
376 if mask.sum() > 1:
377 class_activations = activations_flat[mask]
378 class_center = class_activations.mean(dim=0)
379 variance = ((class_activations - class_center) ** 2).mean().item()
380 intra_class_variance += variance
381
382 intra_class_variance /= len(unique_targets)
383
384 # 表示能力 = 类间距离 / 类内方差
385 if intra_class_variance > 0:
386 representation_capacity = avg_inter_class_distance / (intra_class_variance + 1e-8)
387 else:
388 representation_capacity = avg_inter_class_distance
389
390 # 归一化到[0, 1]
391 return min(representation_capacity / 10.0, 1.0)
392
393 except Exception as e:
394 logger.warning(f"表示能力计算失败: {e}")
395 return 0.0

397 def _compute_parameter_redundancy(self, subnetwork: nn.Module, activations: torch.Tensor) -> float:
398 """计算参数冗余度"""
399
400 try:
401 redundancy_scores = []
402
403 for name, module in subnetwork.named_modules():
404 if isinstance(module, (nn.Linear, nn.Conv2d)):
405 weight = module.weight.data
406
407 # 计算权重矩阵的有效秩
408 if len(weight.shape) == 2: # Linear layer
409 rank = torch.matrix_rank(weight).item()
410 max_rank = min(weight.shape[0], weight.shape[1])
411 else: # Conv2d layer
412 # 将卷积权重重塑为2D矩阵
413 weight_2d = weight.view(weight.shape[0], -1)
414 rank = torch.matrix_rank(weight_2d).item()
415 max_rank = min(weight_2d.shape[0], weight_2d.shape[1])
416
417 if max_rank > 0:
418 rank_ratio = rank / max_rank
419 redundancy = 1.0 - rank_ratio # 秩越低，冗余度越高
420 redundancy_scores.append(redundancy)
421
422 if redundancy_scores:
423 return np.mean(redundancy_scores)
424 else:
425 return 0.0
426
427 except Exception as e:
428 logger.warning(f"参数冗余度计算失败: {e}")
429 return 0.0

431 def _estimate_feasible_parameter_space(self,
432 subnetwork: nn.Module,
433 activations: torch.Tensor,
434 targets: torch.Tensor) -> float:
435 """
436 估计可行参数空间占比
437
438 核心思想：在当前参数附近，有多大比例的参数变化能够维持或改善性能
439 """
440
441 try:
442 # 使用采样方法估计可行参数空间
443 original_params = [p.data.clone() for p in subnetwork.parameters()]
444
445 # 计算原始性能
446 with torch.no_grad():
447 original_output = subnetwork(activations)
448 if len(original_output.shape) > 1 and original_output.shape[1] > 1:
449 original_loss = F.cross_entropy(original_output, targets, reduction='mean')
450 else:
451 original_loss = F.mse_loss(original_output.squeeze(), targets.float())
452
453 # 采样测试
454 feasible_count = 0
455 total_samples = 50 # 减少采样数量以提高速度
456 noise_scale = 0.01 # 小扰动
457
458 for _ in range(total_samples):
459 # 添加随机扰动
460 for param in subnetwork.parameters():
461 noise = torch.randn_like(param.data) * noise_scale
462 param.data.add_(noise)
463
464 # 测试扰动后的性能
465 try:
466 with torch.no_grad():
467 perturbed_output = subnetwork(activations)
468 if len(perturbed_output.shape) > 1 and perturbed_output.shape[1] > 1:
469 perturbed_loss = F.cross_entropy(perturbed_output, targets, reduction='mean')
470 else:
471 perturbed_loss = F.mse_loss(perturbed_output.squeeze(), targets.float())
472
473 # 如果损失没有显著增加，认为是可行的
474 if perturbed_loss <= original_loss * 1.1: # 允许10%的性能下降
475 feasible_count += 1
476
477 except Exception:
478 pass # 扰动导致的错误视为不可行
479
480 # 恢复原始参数
481 for param, original_param in zip(subnetwork.parameters(), original_params):
482 param.data.copy_(original_param)
483
484 feasible_ratio = feasible_count / total_samples
485 return feasible_ratio
486
487 except Exception as e:
488 logger.warning(f"可行参数空间估计失败: {e}")
489 return 0.0

560 def _compute_improvement_potential(self, efficiency: float, current_accuracy: float) -> float:
561 """基于效率和当前准确率计算改进潜力"""
562
563 # 效率越低，改进空间越大
564 efficiency_factor = 1.0 - efficiency
565
566 # 准确率越高，改进越困难
567 if current_accuracy > 0.95:
568 accuracy_factor = 0.1
569 elif current_accuracy > 0.90:
570 accuracy_factor = 0.3
571 elif current_accuracy > 0.80:
572 accuracy_factor = 0.6
573 else:
574 accuracy_factor = 1.0
575
576 # 综合改进潜力
577 improvement_potential = efficiency_factor * accuracy_factor
578
579 return min(improvement_potential, 1.0)

673 def _estimate_optimal_mutation_strength(self,
674 parameter_space_analysis: Dict[str, float],
675 improvement_potential: float) -> float:
676 """估计最优变异强度"""
677
678 efficiency = parameter_space_analysis['overall_efficiency']
679 feasible_space = parameter_space_analysis['feasible_space_ratio']
680
681 # 基于效率和可行空间确定变异强度
682 base_strength = 0.1 # 基础变异强度
683
684 # 效率低的子网络可以承受更强的变异
685 efficiency_factor = (1.0 - efficiency) * 2.0
686
687 # 可行空间大的子网络可以承受更强的变异
688 feasible_factor = feasible_space * 1.5
689
690 optimal_strength = base_strength + min(efficiency_factor + feasible_factor, 0.8)

694 def _assess_mutation_risks(self,
695 subnetwork_info: Dict[str, Any],
696 parameter_space_analysis: Dict[str, float]) -> Dict[str, float]:
697 """评估变异风险"""
698
699 # 参数数量风险
700 param_count = subnetwork_info['total_params']
701 param_risk = min(param_count / 1e6, 1.0) # 参数越多风险越高
702
703 # 效率风险
704 efficiency = parameter_space_analysis['overall_efficiency']
705 efficiency_risk = efficiency # 效率高的系统变异风险高
706
707 # 可行空间风险
708 feasible_space = parameter_space_analysis['feasible_space_ratio']
709 space_risk = 1.0 - feasible_space # 可行空间小风险高
710
711 # 综合风险
712 overall_risk = (param_risk + efficiency_risk + space_risk) / 3.0
713
714 return {
715 'parameter_risk': param_risk,
716 'efficiency_risk': efficiency_risk,
717 'space_risk': space_risk,
718 'overall_risk': overall_risk
719 }

721 def _compute_prediction_confidence(self, parameter_space_analysis: Dict[str, float]) -> float:
722 """计算预测置信度"""
723
724 # 基于分析结果的完整性和一致性计算置信度
725 analysis_completeness = len([v for v in parameter_space_analysis.values() if v > 0]) / len(parameter_space_analysis)
726
727 # 结果的一致性检查
728 efficiency = parameter_space_analysis['overall_efficiency']
729 redundancy = parameter_space_analysis['redundancy_ratio']
730
731 # 效率和冗余度应该负相关
732 consistency = 1.0 - abs(efficiency + redundancy - 1.0)
733
734 confidence = (analysis_completeness + consistency) / 2.0
735
736 return min(confidence, 1.0)

819 def _generate_recommendation(self,
820 layer_name: str,
821 param_space_analysis: Dict[str, float],
822 mutation_prediction: Dict[str, Any]) -> Dict[str, Any]:
823 """生成变异建议"""
824
825 improvement_potential = mutation_prediction['improvement_potential']
826 risk_assessment = mutation_prediction['risk_assessment']
827 strategy_predictions = mutation_prediction['strategy_predictions']
828
829 # 选择最佳策略
830 best_strategy = None
831 best_score = -1.0
832
833 for strategy_name, strategy_info in strategy_predictions.items():
834 # 综合评分：期望收益 - 风险 - 成本
835 score = (
836 strategy_info['expected_accuracy_gain'] * 2.0 -
837 strategy_info['stability_risk'] -
838 strategy_info['parameter_cost'] * 0.5
839 )
840
841 if score > best_score:
842 best_score = score
843 best_strategy = strategy_name
844
845 # 生成建议
846 if improvement_potential > 0.3 and risk_assessment['overall_risk'] < 0.6:
847 action = 'mutate'
848 priority = 'high' if improvement_potential > 0.6 else 'medium'
849 elif improvement_potential > 0.1 and risk_assessment['overall_risk'] < 0.8:
850 action = 'consider'
851 priority = 'low'
852 else:
853 action = 'skip'
854 priority = 'none'
855
856 return {
857 'action': action,
858 'priority': priority,
859 'recommended_strategy': best_strategy,
860 'expected_gain': strategy_predictions.get(best_strategy, {}).get('expected_accuracy_gain', 0.0),
861 'risk_level': risk_assessment['overall_risk'],
862 'reason': f"改进潜力={improvement_potential:.3f}, 风险={risk_assessment['overall_risk']:.3f}"
863 }
