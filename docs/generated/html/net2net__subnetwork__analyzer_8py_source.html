341 def _compute_representation_capacity(self, activations: torch.Tensor, targets: torch.Tensor) -> float:
342 """è®¡ç®—è¡¨ç¤ºèƒ½åŠ› - ç½‘ç»œèƒ½å¤šå¥½åœ°åŒºåˆ†ä¸åŒç±»åˆ«"""

346 if len(activations.shape) > 2:
347 # å¯¹äºå·ç§¯å±‚è¾“å‡ºï¼Œå–å¹³å‡æ± åŒ–
348 activations_flat = F.adaptive_avg_pool2d(activations, (1, 1)).flatten(1)
349 else:
350 activations_flat = activations
351
352 # è®¡ç®—ä¸åŒç±»åˆ«çš„æ¿€æ´»åˆ†å¸ƒ
353 unique_targets = torch.unique(targets)
354 if len(unique_targets) < 2:
355 return 0.0
356
357 class_centers = []
358 for target_class in unique_targets:
359 mask = targets == target_class
360 if mask.sum() > 0:
361 class_center = activations_flat[mask].mean(dim=0)
362 class_centers.append(class_center)
363
364 if len(class_centers) < 2:
365 return 0.0
366
367 # è®¡ç®—ç±»é—´è·ç¦»
368 class_centers = torch.stack(class_centers)
369 distances = torch.pdist(class_centers)

372 # è®¡ç®—ç±»å†…æ–¹å·®

378 class_center = class_activations.mean(dim=0)
379 variance = ((class_activations - class_center) ** 2).mean().item()

381
382 intra_class_variance /= len(unique_targets)
383
384 # è¡¨ç¤ºèƒ½åŠ› = ç±»é—´è·ç¦» / ç±»å†…æ–¹å·®

388 representation_capacity = avg_inter_class_distance

390 # å½’ä¸€åŒ–åˆ°[0, 1]
