91 def _initialize_weights(self):
92 for m in self.modules():
93 if isinstance(m, nn.Conv2d):
94 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
95 elif isinstance(m, nn.BatchNorm2d):
96 nn.init.constant_(m.weight, 1)
97 nn.init.constant_(m.bias, 0)
98 elif isinstance(m, nn.Linear):
99 nn.init.normal_(m.weight, 0, 0.01)
100 if m.bias is not None:
101 nn.init.constant_(m.bias, 0)

109class ActivationHook:
110 """改进的激活值捕获钩子"""
111
112 def __init__(self):
113 self.activations = {}
114 self.hooks = []
115
116 def hook_fn(self, name):
117 def fn(module, input, output):
118 self.activations[name] = output.detach().clone()
119 return fn
120
121 def register_hooks(self, model):
122 self.remove_hooks() # 清理之前的钩子
123
124 for name, module in model.named_modules():
125 if isinstance(module, (nn.Conv2d, nn.Linear)):
126 # 为所有卷积层和线性层注册钩子
127 hook = module.register_forward_hook(self.hook_fn(name))
128 self.hooks.append(hook)
129
130 def remove_hooks(self):
131 for hook in self.hooks:
132 hook.remove()
133 self.hooks.clear()
134 self.activations.clear()

136class GradientHook:

139 def __init__(self):

144 self.gradients.clear()

172def train_epoch(model, train_loader, optimizer, criterion, device, activation_hook, gradient_hook):
173 """训练一个epoch"""
174 model.train()
175 running_loss = 0.0
176 correct = 0
177 total = 0
178
179 pbar = tqdm(train_loader, desc="Training", leave=False)
180 for batch_idx, (data, target) in enumerate(pbar):
181 data, target = data.to(device), target.to(device)
182 gradient_hook.last_targets = target
183
184 optimizer.zero_grad()
185 output = model(data)
186 loss = criterion(output, target)
187 loss.backward()
188
189 # 捕获梯度
190 gradient_hook.capture_gradients(model)
191
192 # 梯度裁剪
193 torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
194
195 optimizer.step()
196
197 running_loss += loss.item()
198 _, predicted = output.max(1)
199 total += target.size(0)
200 correct += predicted.eq(target).sum().item()
201
202 if batch_idx % 100 == 0:
203 pbar.set_postfix({
204 'Loss': f'{running_loss/(batch_idx+1):.4f}',
205 'Acc': f'{100.*correct/total:.2f}%'
206 })
207
208 # 为了观察分裂效果，只训练一部分数据
209 if batch_idx >= 200: # 限制每个epoch的批次数
210 break
211
212 return running_loss / min(len(train_loader), 201), 100. * correct / total

214def validate_epoch(model, val_loader, criterion, device):
215 """验证一个epoch"""
216 model.eval()
217 val_loss = 0.0
218 correct = 0
219 total = 0
220
221 with torch.no_grad():
222 for batch_idx, (data, target) in enumerate(val_loader):
223 data, target = data.to(device), target.to(device)
224 output = model(data)
225 val_loss += criterion(output, target).item()
226
227 _, predicted = output.max(1)
228 total += target.size(0)
229 correct += predicted.eq(target).sum().item()
230
231 # 限制验证数据量
232 if batch_idx >= 50:
233 break
234
235 return val_loss / min(len(val_loader), 51), 100. * correct / total

241def analyze_model_structure(model):
242 """分析模型结构"""
243 conv_layers = 0
