39class ResidualBlock(nn.Module):
40 """Enhanced residual block with information preservation."""
41
42 def __init__(self, in_channels: int, out_channels: int, stride: int = 1, use_attention: bool = False):
43 super().__init__()
44
45 self.conv1 = nn.Conv2d(in_channels, out_channels, 3, stride=stride, padding=1, bias=False)
46 self.bn1 = nn.BatchNorm2d(out_channels)
47 self.conv2 = nn.Conv2d(out_channels, out_channels, 3, padding=1, bias=False)
48 self.bn2 = nn.BatchNorm2d(out_channels)
49
50 # Skip connection
51 self.skip = nn.Sequential()
52 if stride != 1 or in_channels != out_channels:
53 self.skip = nn.Sequential(
54 nn.Conv2d(in_channels, out_channels, 1, stride=stride, bias=False),
55 nn.BatchNorm2d(out_channels)
56 )
57
58 # Optional attention mechanism
59 self.use_attention = use_attention
60 if use_attention:
61 self.attention = nn.Sequential(
62 nn.AdaptiveAvgPool2d(1),
63 nn.Conv2d(out_channels, out_channels // 16, 1),
64 nn.ReLU(inplace=True),
65 nn.Conv2d(out_channels // 16, out_channels, 1),
66 nn.Sigmoid()
67 )
68
69 # Dropout for regularization
70 self.dropout = nn.Dropout2d(0.1)
71
72 def forward(self, x):
73 identity = self.skip(x)
74
75 out = torch.relu(self.bn1(self.conv1(x)))
76 out = self.dropout(out)
77 out = self.bn2(self.conv2(out))
78
79 if self.use_attention:
80 attention_weights = self.attention(out)
81 out = out * attention_weights
82
83 out += identity
84 return torch.relu(out)

87class InformationBottleneckLayer(nn.Module):
88 """Information bottleneck layer for controlled information flow."""
89
90 def __init__(self, in_channels: int, bottleneck_ratio: float = 0.5):
91 super().__init__()
92
93 bottleneck_channels = max(1, int(in_channels * bottleneck_ratio))
94
95 self.compress = nn.Sequential(
96 nn.Conv2d(in_channels, bottleneck_channels, 1, bias=False),
97 nn.BatchNorm2d(bottleneck_channels),
98 nn.ReLU(inplace=True)
99 )
100
101 self.expand = nn.Sequential(
102 nn.Conv2d(bottleneck_channels, in_channels, 1, bias=False),
103 nn.BatchNorm2d(in_channels),
104 nn.ReLU(inplace=True)
105 )
106
107 # Learnable information preservation weight
108 self.preserve_weight = nn.Parameter(torch.tensor(0.8))
109
110 def forward(self, x):
111 compressed = self.compress(x)
112 expanded = self.expand(compressed)
113
114 # Weighted combination of original and processed information
115 return self.preserve_weight * x + (1 - self.preserve_weight) * expanded

214 def _initialize_weights(self):
215 """Initialize weights with improved variance scaling."""
216 for m in self.modules():

220 nn.init.constant_(m.bias, 0)

222 nn.init.constant_(m.weight, 1)
223 nn.init.constant_(m.bias, 0)
224 elif isinstance(m, nn.Linear):

238 for name, module in self.named_modules():
239 if isinstance(module, (nn.Conv2d, nn.Linear)) and len(name.split('.')) <= 3:
