63 """Add a child node and set parent relationship."""

67 def get_absolute_position(self) -> Tuple[int, int]:
68 """Get absolute position by traversing up the parent chain."""
69 x, y = self.x, self.y
70 node = self.parent
71 while node:
72 x += node.x
73 y += node.y
74 node = node.parent
75 return x, y

78 """Get the x-coordinate of the node's center."""
79 abs_x, _ = self.get_absolute_position()

93 def calculate_branch_layout(self, branches: Dict[str, List[Any]],
94 additional_width: int = 0) -> Dict[str, LayoutNode]:
95 """
96 Calculate optimal layout for branch architecture.
97
98 Args:
99 branches: Dictionary mapping branch names to layer lists
100 additional_width: Additional width to consider for each branch
101
102 Returns:
103 Dictionary mapping branch names to LayoutNode objects
104 """
105 branch_nodes = {}
106 num_branches = len(branches)
107
108 if num_branches == 0:
109 return branch_nodes
110
111 # Calculate width needed for each branch
112 branch_widths = {}
113 for branch_name, layers in branches.items():
114 # Find the maximum width needed for this branch
115 max_width = len(branch_name) + 4 # Branch name + padding
116
117 for layer in layers:
118 # Estimate layer display width
119 layer_width = self._estimate_layer_width(layer) + additional_width
120 max_width = max(max_width, layer_width)
121
122 branch_widths[branch_name] = max_width
123
124 # Calculate total width needed
125 total_width = sum(branch_widths.values()) + (num_branches - 1) * self.branch_spacing
126
127 # If total width exceeds terminal width, scale down
128 if total_width > self.terminal_width - 10: # Leave some margin
129 scale_factor = (self.terminal_width - 10) / total_width
130 branch_widths = {k: max(10, int(v * scale_factor)) for k, v in branch_widths.items()}
131 total_width = sum(branch_widths.values()) + (num_branches - 1) * self.branch_spacing
132
133 # Calculate starting position to center the layout
134 start_x = (self.terminal_width - total_width) // 2
135
136 # Create layout nodes for each branch
137 current_x = start_x
138 for branch_name in sorted(branches.keys()): # Sort for consistent ordering
139 width = branch_widths[branch_name]
140
141 # Create the branch node
142 node = LayoutNode(
143 name=branch_name,
144 width=width,
145 height=1, # Will be calculated based on content
146 x=current_x,
147 y=0
148 )
149
150 branch_nodes[branch_name] = node
151 current_x += width + self.branch_spacing
152
153 return branch_nodes

262 if junction_line[j] == ' ':
263 junction_line[j] = '‚îÄ'

266 f"{Colors.MAGENTA}{''.join(junction_line[:self.terminal_width])}{Colors.RESET}"

400 def center_text(self, text: str, width: int, fill_char: str = ' ') -> str:
401 """Center text within a given width."""
402 text_len = len(self._strip_ansi(text))
403 if text_len >= width:
404 return text
405
406 left_pad = (width - text_len) // 2
407 right_pad = width - text_len - left_pad
408
409 return fill_char * left_pad + text + fill_char * right_pad

415 self._ansi_pattern = re.compile(r'\x1B(?:[@-Z\\-_]|\[[0-?]*[ -/]*[@-~])')
416 return self._ansi_pattern.sub('', text)

971 def _get_layer_info(self, layer_name: str, model: nn.Module,
972 structure_info: Dict[str, Any],
973 previous_model: Optional[nn.Module],
974 changed_layers: List[str]) -> Dict[str, Any]:
975 """Get comprehensive information about a layer."""
976 info = {
977 'name': layer_name,
978 'simplified_name': self._simplify_layer_name(layer_name),
979 'exists_in_current': layer_name in structure_info['current_layers'],
980 'exists_in_previous': layer_name in structure_info.get('prev_layers', {}),
981 'params': structure_info['layer_params'].get(layer_name, 0),
982 'prev_params': structure_info.get('prev_params', {}).get(layer_name, 0),
983 'changed': layer_name in changed_layers,
984 'status': 'normal',
985 'layer_type': None
986 }
987
988 # Determine layer type
989 if info['exists_in_current']:
990 layer = self._get_layer_module(layer_name, model)
991 info['layer_type'] = type(layer).__name__ if layer else 'Unknown'
992 elif info['exists_in_previous'] and previous_model:
993 layer = self._get_layer_module(layer_name, previous_model)
994 info['layer_type'] = type(layer).__name__ if layer else 'Unknown'
995
996 # Use architecture tracker if available for accurate status

999 if tracker_status == 'new':
1000 info['status'] = 'new'
1001 elif tracker_status == 'removed':
1002 info['status'] = 'removed'
1003 elif tracker_status == 'modified':
1004 info['status'] = 'modified'
1005 elif tracker_status == 'unchanged':
1006 info['status'] = 'normal'
1007 else:
1008 # Fallback to original logic
1009 if previous_model:
1010 if not info['exists_in_current'] and info['exists_in_previous']:
1011 info['status'] = 'removed'
1012 elif info['exists_in_current'] and not info['exists_in_previous']:
1013 info['status'] = 'new'
1014 elif info['changed'] or info['params'] != info['prev_params']:
1015 info['status'] = 'modified'
1016 elif info['changed']:
1017 info['status'] = 'modified'
1018
1019 return info
