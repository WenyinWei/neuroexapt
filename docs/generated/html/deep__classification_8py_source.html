45class BasicBlock(nn.Module):
46 """Standard ResNet Basic Block"""
47 expansion = 1
48
49 def __init__(self, in_planes, planes, stride=1):
50 super(BasicBlock, self).__init__()
51 self.conv1 = nn.Conv2d(in_planes, planes, kernel_size=3, stride=stride, padding=1, bias=False)
52 self.bn1 = nn.BatchNorm2d(planes)
53 self.conv2 = nn.Conv2d(planes, planes, kernel_size=3, stride=1, padding=1, bias=False)
54 self.bn2 = nn.BatchNorm2d(planes)
55
56 self.shortcut = nn.Sequential()
57 if stride != 1 or in_planes != self.expansion * planes:
58 self.shortcut = nn.Sequential(
59 nn.Conv2d(in_planes, self.expansion * planes, kernel_size=1, stride=stride, bias=False),
60 nn.BatchNorm2d(self.expansion * planes)
61 )
62
63 def forward(self, x):
64 out = F.relu(self.bn1(self.conv1(x)))
65 out = self.bn2(self.conv2(out))
66 out += self.shortcut(x)
67 out = F.relu(out)
68 return out

70class Bottleneck(nn.Module):
71 """ResNet Bottleneck Block"""
72 expansion = 4
73
74 def __init__(self, in_planes, planes, stride=1):
75 super(Bottleneck, self).__init__()
76 self.conv1 = nn.Conv2d(in_planes, planes, kernel_size=1, bias=False)
77 self.bn1 = nn.BatchNorm2d(planes)
78 self.conv2 = nn.Conv2d(planes, planes, kernel_size=3, stride=stride, padding=1, bias=False)
79 self.bn2 = nn.BatchNorm2d(planes)
80 self.conv3 = nn.Conv2d(planes, self.expansion*planes, kernel_size=1, bias=False)
81 self.bn3 = nn.BatchNorm2d(self.expansion*planes)
82
83 self.shortcut = nn.Sequential()
84 if stride != 1 or in_planes != self.expansion*planes:
85 self.shortcut = nn.Sequential(
86 nn.Conv2d(in_planes, self.expansion*planes, kernel_size=1, stride=stride, bias=False),
87 nn.BatchNorm2d(self.expansion*planes)
88 )
89
90 def forward(self, x):
91 out = F.relu(self.bn1(self.conv1(x)))
92 out = F.relu(self.bn2(self.conv2(out)))
93 out = self.bn3(self.conv3(out))
94 out += self.shortcut(x)
95 out = F.relu(out)
96 return out

98class FixedResNet(nn.Module):
99 """Fixed ResNet architecture for fast training"""
100 def __init__(self, block, num_blocks, num_classes=10, base_width=64):
101 super(FixedResNet, self).__init__()
102 self.in_planes = base_width
103
104 self.conv1 = nn.Conv2d(3, base_width, kernel_size=3, stride=1, padding=1, bias=False)
105 self.bn1 = nn.BatchNorm2d(base_width)
106 self.layer1 = self._make_layer(block, base_width, num_blocks[0], stride=1)
107 self.layer2 = self._make_layer(block, base_width*2, num_blocks[1], stride=2)
108 self.layer3 = self._make_layer(block, base_width*4, num_blocks[2], stride=2)
109 self.layer4 = self._make_layer(block, base_width*8, num_blocks[3], stride=2)
110 self.linear = nn.Linear(base_width*8*block.expansion, num_classes)
111
112 def _make_layer(self, block, planes, num_blocks, stride):
113 strides = [stride] + [1]*(num_blocks-1)
114 layers = []
115 for stride in strides:
116 layers.append(block(self.in_planes, planes, stride))
117 self.in_planes = planes * block.expansion
118 return nn.Sequential(*layers)
119
120 def forward(self, x):
121 out = F.relu(self.bn1(self.conv1(x)))
122 out = self.layer1(out)
123 out = self.layer2(out)
124 out = self.layer3(out)
125 out = self.layer4(out)
126 out = F.avg_pool2d(out, 4)
127 out = out.view(out.size(0), -1)
128 out = self.linear(out)
129 return out

373def train_epoch(train_queue, valid_queue, model, architect, criterion, optimizer, epoch, args):
374 """Train one epoch"""
375 model.train()
376
377 objs = AvgrageMeter()
378 top1 = AvgrageMeter()
379 top5 = AvgrageMeter()
380
381 # For architecture search modes
382 if architect is not None:
383 architect.set_epoch(epoch)
384 valid_iter = iter(valid_queue)
385 arch_updates = 0
386
387 start_time = time.time()
388
389 for step, (input, target) in enumerate(train_queue):
390 n = input.size(0)
391 input = input.cuda(non_blocking=True)
392 target = target.cuda(non_blocking=True)
393
394 # Architecture search step
395 if architect is not None and architect.should_update_arch():
396 try:
397 input_search, target_search = next(valid_iter)
398 except StopIteration:
399 valid_iter = iter(valid_queue)
400 input_search, target_search = next(valid_iter)
401
402 input_search = input_search.cuda(non_blocking=True)
403 target_search = target_search.cuda(non_blocking=True)
404
405 architect.step(input, target, input_search, target_search,
406 optimizer.param_groups[0]['lr'], optimizer, args.use_first_order)
407 arch_updates += 1
408
409 # Weight update step
410 optimizer.zero_grad()
411 logits = model(input)
412 loss = criterion(logits, target)
413 loss.backward()
414 nn.utils.clip_grad_norm_(model.parameters(), args.grad_clip)
415 optimizer.step()
416
417 # Statistics
418 prec1, prec5 = accuracy(logits, target, topk=(1, 5))
419 objs.update(loss.item(), n)
420 top1.update(prec1.item(), n)
421 top5.update(prec5.item(), n)
422
423 # Progress report
424 if step % args.report_freq == 0:
425 arch_info = f"Arch: {arch_updates}" if architect else "Fixed"
426 print(f" Step {step:3d}: Loss={objs.avg:.4f} Top1={top1.avg:.2f}% {arch_info}")
427
428 epoch_time = time.time() - start_time
429 print(f" Epoch {epoch}: Loss={objs.avg:.4f} Top1={top1.avg:.2f}% Time={epoch_time:.1f}s")
430
431 return top1.avg, objs.avg, epoch_time

433def validate_epoch(valid_queue, model, criterion):
434 """Validate one epoch"""
435 model.eval()
436
437 objs = AvgrageMeter()
438 top1 = AvgrageMeter()
439 top5 = AvgrageMeter()
440
441 with torch.no_grad():
442 for input, target in valid_queue:
443 input = input.cuda(non_blocking=True)
444 target = target.cuda(non_blocking=True)
445
446 logits = model(input)
447 loss = criterion(logits, target)
448
449 prec1, prec5 = accuracy(logits, target, topk=(1, 5))
450 n = input.size(0)
451 objs.update(loss.item(), n)
452 top1.update(prec1.item(), n)
453 top5.update(prec5.item(), n)
454
455 return top1.avg, objs.avg

457def run_single_mode(args, mode):
458 """Run training in a single mode"""
459 print(f"\n🚀 Running {mode} mode")
460 print("=" * 60)
461
462 # Create model and components
463 model = create_model(args, mode)
464 architect = create_architect(args, model)
465 criterion = nn.CrossEntropyLoss().cuda()
466 optimizer = create_optimizer(args, model)
467
468 # Learning rate scheduler
469 if args.mode == 'fixed':
470 scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, args.epochs)
471 else:
472 scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, args.epochs, eta_min=0.001)
473
474 # Create data loaders
475 train_queue, valid_queue, test_queue = create_data_loaders(args)
476
477 # Training loop
478 best_acc = 0.0
479 total_time = 0
480
481 for epoch in range(args.epochs):
482 print(f"\n--- Epoch {epoch} ---")
483
484 # Update learning rate
485 scheduler.step()
486 current_lr = scheduler.get_last_lr()[0]
487 print(f"Learning rate: {current_lr:.6f}")
488
489 # Train
490 train_acc, train_loss, train_time = train_epoch(
491 train_queue, valid_queue, model, architect, criterion, optimizer, epoch, args
492 )
493
494 # Validate
495 val_acc, val_loss = validate_epoch(valid_queue, model, criterion)
496
497 if val_acc > best_acc:
498 best_acc = val_acc
499
500 # Test on best validation accuracy
501 test_acc, test_loss = validate_epoch(test_queue, model, criterion)
502 print(f" New best: Val={val_acc:.2f}% Test={test_acc:.2f}%")
503
504 total_time += train_time
505
506 print(f"Epoch {epoch}: Train={train_acc:.2f}% Valid={val_acc:.2f}% Best={best_acc:.2f}%")
507
508 # Final test
509 final_test_acc, final_test_loss = validate_epoch(test_queue, model, criterion)
510 avg_time = total_time / args.epochs
511
512 print(f"\n📊 Results for {mode} mode:")
513 print(f" Best validation accuracy: {best_acc:.2f}%")
514 print(f" Final test accuracy: {final_test_acc:.2f}%")
515 print(f" Average time per epoch: {avg_time:.1f}s")
516 print(f" Total training time: {total_time:.1f}s")
517
518 # Get final genotype if available
519 genotype = None
520 if hasattr(model, 'genotype') and callable(model.genotype):
521 genotype = model.genotype()
522
523 # fuse kernels for inference
524 fuse_model(model) # type: ignore[arg-type]
525
526 return {
527 'mode': mode,
528 'best_val_acc': best_acc,
529 'final_test_acc': final_test_acc,
530 'avg_time': avg_time,
531 'total_time': total_time,
532 'genotype': genotype
533 }
