26class EnhancedBottleneckDetector:
27 """增强的瓶颈检测器
28
29 基于多维度指标检测网络中的性能瓶颈：
30 1. 梯度方差分析 - 检测梯度传播瓶颈
31 2. 激活多样性 - 评估神经元激活的多样性
32 3. 信息流分析 - 基于信息论的层重要性
33 4. 层贡献度 - 通过扰动分析层的重要性
34 5. 性能敏感度 - 评估层对最终性能的影响
35 """

37 def __init__(self,
38 sensitivity_threshold: float = 0.1,
39 diversity_threshold: float = 0.3,
40 gradient_threshold: float = 1e-6,
41 info_flow_threshold: float = 0.5):
42 """初始化瓶颈检测器
43
44 Args:
45 sensitivity_threshold: 性能敏感度阈值
46 diversity_threshold: 激活多样性阈值
47 gradient_threshold: 梯度方差阈值
48 info_flow_threshold: 信息流阈值
49 """
50 self.sensitivity_threshold = sensitivity_threshold
51 self.diversity_threshold = diversity_threshold
52 self.gradient_threshold = gradient_threshold
53 self.info_flow_threshold = info_flow_threshold
54
55 # 历史数据缓存
56 self.gradient_history = defaultdict(list)
57 self.activation_history = defaultdict(list)
58 self.performance_history = []
59
60 # 评估指标权重
61 self.metric_weights = {
62 'gradient_variance': 0.25,
63 'activation_diversity': 0.20,
64 'information_flow': 0.25,
65 'layer_contribution': 0.20,
66 'performance_sensitivity': 0.10
67 }

69 def detect_bottlenecks(self,
70 model: nn.Module,
71 activations: Dict[str, torch.Tensor],
72 gradients: Dict[str, torch.Tensor],
73 targets: Optional[torch.Tensor] = None) -> Dict[str, float]:
74 """检测网络瓶颈
75
76 Args:
77 model: 神经网络模型
78 activations: 层激活值字典
79 gradients: 层梯度字典
80 targets: 目标值（可选）
81
82 Returns:
83 Dict[str, float]: 层名称到瓶颈分数的映射
84 """
85 bottleneck_scores = {}
86
87 # 获取所有评估的层
88 layer_names = self._get_evaluable_layers(model)
89
90 for layer_name in layer_names:
91 try:
92 # 计算各维度指标
93 gradient_score = self._compute_gradient_variance_score(layer_name, gradients)
94 diversity_score = self._compute_activation_diversity_score(layer_name, activations)
95 info_flow_score = self._compute_information_flow_score(layer_name, activations, targets)
96 contribution_score = self._compute_layer_contribution_score(layer_name, model, activations)
97 sensitivity_score = self._compute_performance_sensitivity_score(layer_name, gradients)
98
99 # 加权综合评分
100 total_score = (
101 self.metric_weights['gradient_variance'] * gradient_score +
102 self.metric_weights['activation_diversity'] * diversity_score +
103 self.metric_weights['information_flow'] * info_flow_score +
104 self.metric_weights['layer_contribution'] * contribution_score +
105 self.metric_weights['performance_sensitivity'] * sensitivity_score
106 )
107
108 bottleneck_scores[layer_name] = total_score
109
110 logger.debug(f"Layer {layer_name}: grad={gradient_score:.3f}, "
111 f"div={diversity_score:.3f}, info={info_flow_score:.3f}, "
112 f"contrib={contribution_score:.3f}, sens={sensitivity_score:.3f}, "
113 f"total={total_score:.3f}")
114
115 except Exception as e:
116 logger.warning(f"Error evaluating layer {layer_name}: {e}")
117 bottleneck_scores[layer_name] = 0.0
118
119 return bottleneck_scores

121 def _get_evaluable_layers(self, model: nn.Module) -> List[str]:
122 """获取可评估的层名称列表"""
123 evaluable_layers = []
124 for name, module in model.named_modules():
125 if isinstance(module, (nn.Conv2d, nn.Linear, nn.BatchNorm2d)):
126 # 排除太浅或太深的层
127 if len(name.split('.')) >= 2:
128 evaluable_layers.append(name)
129 return evaluable_layers

131 def _compute_gradient_variance_score(self,
132 layer_name: str,
133 gradients: Dict[str, torch.Tensor]) -> float:
134 """计算梯度方差评分
135
136 梯度方差低表示该层可能是瓶颈
137 """
138 if layer_name not in gradients:
139 return 0.0
140
141 grad = gradients[layer_name]
142 if grad is None or grad.numel() == 0:
143 # Assign a high score to indicate a potential bottleneck due to a dead/disconnected layer
144 return 1.0
145
146 # 计算梯度方差
147 grad_var = torch.var(grad).item()
148
149 # 更新历史记录
150 self.gradient_history[layer_name].append(grad_var)
151 if len(self.gradient_history[layer_name]) > 10:
152 self.gradient_history[layer_name].pop(0)
153
154 # 计算相对方差（相对于历史平均值）
155 if len(self.gradient_history[layer_name]) > 1:
156 historical_mean = np.mean(self.gradient_history[layer_name][:-1])
157 if historical_mean > 0:
158 relative_var = grad_var / historical_mean
159 else:
160 relative_var = 1.0
161 else:
162 relative_var = 1.0
163
164 # 低梯度方差 = 高瓶颈分数
165 if grad_var < self.gradient_threshold:
166 return 1.0
167 else:
168 return max(0.0, 1.0 - np.log10(grad_var + 1e-10) / 5.0)

170 def _compute_activation_diversity_score(self,
171 layer_name: str,
172 activations: Dict[str, torch.Tensor]) -> float:
173 """计算激活多样性评分
174
175 激活多样性低表示神经元功能相似，可能需要分裂
176 """
177 if layer_name not in activations:
178 return 0.0
179
180 activation = activations[layer_name]
181 if activation is None or activation.numel() == 0:
182 return 0.0
183
184 # 计算激活值的多样性
185 if len(activation.shape) >= 4: # Conv层 (N, C, H, W)
186 # 在空间维度上平均，获得 (N, C)
187 activation_flat = activation.mean(dim=(2, 3))
188 elif len(activation.shape) == 2: # 全连接层 (N, F)
189 activation_flat = activation
190 else:
191 return 0.0
192
193 # 计算神经元间的相关性
194 if activation_flat.shape[1] > 1 and activation_flat.shape[0] >= 2: # 至少需要2个样本
195 try:
196 # 检查批次大小，torch.corrcoef 在小批次时可能不稳定
197 if activation_flat.shape[0] < 10:
198 # 对于小批次，使用更稳定的方法计算相关性
199 correlations = []
200 for i in range(activation_flat.shape[1]):
201 for j in range(i+1, activation_flat.shape[1]):
202 corr = torch.corrcoef(torch.stack([activation_flat[:, i], activation_flat[:, j]]))[0, 1]
203 if not torch.isnan(corr):
204 correlations.append(abs(corr.item()))
205
206 mean_correlation = np.mean(correlations) if correlations else 0.0
207 else:
208 correlation_matrix = torch.corrcoef(activation_flat.T)
209 # 去除对角线元素
210 mask = ~torch.eye(correlation_matrix.shape[0], dtype=torch.bool)
211 correlations = correlation_matrix[mask]
212
213 # 过滤掉NaN值
214 valid_correlations = correlations[~torch.isnan(correlations)]
215 mean_correlation = torch.abs(valid_correlations).mean().item() if len(valid_correlations) > 0 else 0.0
216
217 diversity_score = mean_correlation # 高相关性 = 高瓶颈分数
218 except Exception as e:
219 # 如果相关系数计算失败，返回默认值
220 diversity_score = 0.0
221 else:
222 diversity_score = 0.0
223
224 return min(1.0, max(0.0, diversity_score))

226 def _compute_information_flow_score(self,
227 layer_name: str,
228 activations: Dict[str, torch.Tensor],
229 targets: Optional[torch.Tensor]) -> float:
230 """计算信息流评分
231
232 基于信息论的方法评估层的信息处理能力
233 """
234 if layer_name not in activations or targets is None:
235 return 0.0
236
237 activation = activations[layer_name]
238 if activation is None or activation.numel() == 0:
239 return 0.0
240
241 try:
242 # 简化的互信息估计
243 if len(activation.shape) >= 4:
244 activation_flat = activation.mean(dim=(2, 3))
245 elif len(activation.shape) == 2:
246 activation_flat = activation
247 else:
248 return 0.0
249
250 # 计算激活值的熵
251 activation_entropy = self._estimate_entropy(activation_flat)
252
253 # 简单的信息流评分：低熵表示信息瓶颈
254 info_score = 1.0 - min(1.0, activation_entropy / 5.0) # 归一化到[0,1]
255
256 return max(0.0, info_score)
257
258 except Exception as e:
259 logger.warning(f"Error computing information flow for {layer_name}: {e}")
260 return 0.0

262 def _estimate_entropy(self, tensor: torch.Tensor) -> float:
263 """估计张量的熵"""
264 try:
265 # 简单的熵估计方法
266 tensor_flat = tensor.flatten()
267
268 # 使用直方图估计概率分布
269 hist = torch.histc(tensor_flat, bins=50)
270 hist = hist / hist.sum()
271 hist = hist[hist > 0] # 移除零概率
272
273 # 计算熵
274 entropy = -torch.sum(hist * torch.log2(hist + 1e-10)).item()
275 return entropy
276
277 except Exception:
278 return 1.0 # 默认值

280 def _compute_layer_contribution_score(self,
281 layer_name: str,
282 model: nn.Module,
283 activations: Dict[str, torch.Tensor]) -> float:
284 """计算层贡献度评分
285
286 通过扰动分析评估层对网络输出的贡献
287 """
288 if layer_name not in activations:
289 return 0.0
290
291 activation = activations[layer_name]
292 if activation is None or activation.numel() == 0:
293 return 0.0
294
295 # 简化的贡献度评估：基于激活值的幅度
296 activation_magnitude = torch.norm(activation).item()
297
298 # 归一化处理
299 if activation_magnitude > 0:
300 # 将幅度转换为贡献度评分
301 contribution_score = min(1.0, activation_magnitude / 100.0)
302 else:
303 contribution_score = 1.0 # 零激活可能表示瓶颈
304
305 return 1.0 - contribution_score # 低贡献 = 高瓶颈分数

307 def _compute_performance_sensitivity_score(self,
308 layer_name: str,
309 gradients: Dict[str, torch.Tensor]) -> float:
310 """计算性能敏感度评分
311
312 基于梯度幅度评估层对性能的敏感度
313 """
314 if layer_name not in gradients:
315 return 0.0
316
317 grad = gradients[layer_name]
318 if grad is None or grad.numel() == 0:
319 return 0.0
320
321 # 计算梯度幅度
322 grad_magnitude = torch.norm(grad).item()
323
324 # 低梯度幅度可能表示该层对性能不敏感，可能是瓶颈
325 if grad_magnitude < self.gradient_threshold:
326 return 1.0
327 else:
328 return max(0.0, 1.0 - np.log10(grad_magnitude + 1e-10) / 3.0)

345 def should_trigger_division(self,
346 bottleneck_scores: Dict[str, float],
347 performance_trend: List[float]) -> Tuple[bool, List[str]]:
348 """判断是否应该触发神经元分裂
349
350 Args:
351 bottleneck_scores: 瓶颈评分字典
352 performance_trend: 最近的性能趋势
353
354 Returns:
355 Tuple[bool, List[str]]: (是否触发, 触发原因列表)
356 """
357 reasons = []
358
359 # 检查是否有高瓶颈分数的层
360 max_score = max(bottleneck_scores.values()) if bottleneck_scores else 0.0
361 if max_score > 0.7:
362 reasons.append(f"高瓶颈分数检测: {max_score:.3f}")
363
364 # 检查性能停滞
365 if len(performance_trend) >= 3:
366 recent_improvement = performance_trend[-1] - performance_trend[-3]
367 if recent_improvement < 0.01:
368 reasons.append(f"性能停滞: 最近改善 {recent_improvement:.3f}")
369
370 # 检查多个层的瓶颈
371 high_score_layers = [name for name, score in bottleneck_scores.items() if score > 0.5]
372 if len(high_score_layers) >= 2:
373 reasons.append(f"多层瓶颈: {len(high_score_layers)} 层")
374
375 should_trigger = len(reasons) > 0
376 return should_trigger, reasons
