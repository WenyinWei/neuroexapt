77 def _make_layer(self, in_channels, out_channels, blocks, stride=1):
78 layers = []

89 def _initialize_weights(self):

95 nn.init.constant_(m.bias, 0)
96 elif isinstance(m, nn.Linear):

119class EvolvableBasicBlock(nn.Module):
120 """å¯æ¼”åŒ–çš„åŸºç¡€æ®‹å·®å—"""
121
122 def __init__(self, in_channels, out_channels, stride=1):
123 super().__init__()
124
125 self.conv1 = nn.Conv2d(in_channels, out_channels, kernel_size=3,
126 stride=stride, padding=1, bias=False)
127 self.bn1 = nn.BatchNorm2d(out_channels)
128 self.conv2 = nn.Conv2d(out_channels, out_channels, kernel_size=3,
129 stride=1, padding=1, bias=False)
130 self.bn2 = nn.BatchNorm2d(out_channels)
131 self.relu = nn.ReLU(inplace=True)
132
133 # è·³è·ƒè¿æ¥
134 if stride != 1 or in_channels != out_channels:
135 self.shortcut = nn.Sequential(
136 nn.Conv2d(in_channels, out_channels, kernel_size=1,
137 stride=stride, bias=False),
138 nn.BatchNorm2d(out_channels)
139 )
140 else:
141 self.shortcut = nn.Identity()
142
143 def forward(self, x):
144 identity = self.shortcut(x)
145
146 out = self.conv1(x)
147 out = self.bn1(out)
148 out = self.relu(out)
149
150 out = self.conv2(out)
151 out = self.bn2(out)
152
153 out += identity
154 out = self.relu(out)
155
156 return out
