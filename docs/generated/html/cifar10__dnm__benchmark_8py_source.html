43class EvolvableResNet18(nn.Module):
44 """可演化的ResNet-18架构 - 专门设计用于DNM演化"""
45
46 def __init__(self, num_classes=10, initial_channels=64):
47 super().__init__()
48 self.initial_channels = initial_channels
49
50 # 初始卷积层
51 self.conv1 = nn.Conv2d(3, initial_channels, kernel_size=3, stride=1, padding=1, bias=False)
52 self.bn1 = nn.BatchNorm2d(initial_channels)
53 self.relu = nn.ReLU(inplace=True)
54
55 # 可演化的残差块
56 self.layer1 = self._make_layer(initial_channels, initial_channels, 2, stride=1)
57 self.layer2 = self._make_layer(initial_channels, initial_channels * 2, 2, stride=2)
58 self.layer3 = self._make_layer(initial_channels * 2, initial_channels * 4, 2, stride=2)
59 self.layer4 = self._make_layer(initial_channels * 4, initial_channels * 8, 2, stride=2)
60
61 # 可演化的分类器
62 self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
63 self.fc_layers = nn.Sequential(
64 nn.Flatten(),
65 nn.Linear(initial_channels * 8, initial_channels * 4),
66 nn.ReLU(inplace=True),
67 nn.Dropout(0.3),
68 nn.Linear(initial_channels * 4, initial_channels * 2),
69 nn.ReLU(inplace=True),
70 nn.Dropout(0.2),
71 nn.Linear(initial_channels * 2, num_classes)
72 )
73
74 # 初始化权重
75 self._initialize_weights()
76
77 def _make_layer(self, in_channels, out_channels, blocks, stride=1):
78 layers = []
79
80 # 第一个块可能需要降采样
81 layers.append(EvolvableBasicBlock(in_channels, out_channels, stride))
82
83 # 其余块
84 for _ in range(1, blocks):
85 layers.append(EvolvableBasicBlock(out_channels, out_channels, 1))
86
87 return nn.Sequential(*layers)
88
89 def _initialize_weights(self):
90 for m in self.modules():
91 if isinstance(m, nn.Conv2d):
92 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
93 elif isinstance(m, nn.BatchNorm2d):
94 nn.init.constant_(m.weight, 1)
95 nn.init.constant_(m.bias, 0)
96 elif isinstance(m, nn.Linear):
97 nn.init.normal_(m.weight, 0, 0.01)
98 nn.init.constant_(m.bias, 0)
99
100 def forward(self, x):
101 # 初始特征提取
102 x = self.conv1(x)
103 x = self.bn1(x)
104 x = self.relu(x)
105
106 # 残差层
107 x = self.layer1(x)
108 x = self.layer2(x)
109 x = self.layer3(x)
110 x = self.layer4(x)
111
112 # 分类
113 x = self.avgpool(x)
114 x = self.fc_layers(x)
115
116 return x

119class EvolvableBasicBlock(nn.Module):
120 """可演化的基础残差块"""
121
122 def __init__(self, in_channels, out_channels, stride=1):
123 super().__init__()
124
125 self.conv1 = nn.Conv2d(in_channels, out_channels, kernel_size=3,
126 stride=stride, padding=1, bias=False)
127 self.bn1 = nn.BatchNorm2d(out_channels)
128 self.conv2 = nn.Conv2d(out_channels, out_channels, kernel_size=3,
129 stride=1, padding=1, bias=False)
130 self.bn2 = nn.BatchNorm2d(out_channels)
131 self.relu = nn.ReLU(inplace=True)
132
133 # 跳跃连接
134 if stride != 1 or in_channels != out_channels:
135 self.shortcut = nn.Sequential(
136 nn.Conv2d(in_channels, out_channels, kernel_size=1,
137 stride=stride, bias=False),
138 nn.BatchNorm2d(out_channels)
139 )
140 else:
141 self.shortcut = nn.Identity()
142
143 def forward(self, x):
144 identity = self.shortcut(x)
145
146 out = self.conv1(x)
147 out = self.bn1(out)
148 out = self.relu(out)
149
150 out = self.conv2(out)
151 out = self.bn2(out)
152
153 out += identity
154 out = self.relu(out)
155
156 return out

159def prepare_cifar10_data(batch_size=128, num_workers=4):
160 """准备CIFAR-10数据集"""

162 # 数据增强策略
163 transform_train = transforms.Compose([

175 ])
176

178 trainset = torchvision.datasets.CIFAR10(

182 root='./data', train=False, download=True, transform=transform_test
183 )
184
185 train_loader = DataLoader(
186 trainset, batch_size=batch_size, shuffle=True,
187 num_workers=num_workers, pin_memory=True

195 print(f" Training samples: {len(trainset)}")
196 print(f" Test samples: {len(testset)}")
197 print(f" Batch size: {batch_size}")
198
199 return train_loader, test_loader
