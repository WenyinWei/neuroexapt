46 """Auto-detect 迅雷 installation path."""

56 ]
57
58 for path in possible_paths:
59 if os.path.exists(path):
60 return path
61
62 # Try to find from registry
63 try:
64 import winreg
65 with winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, r"SOFTWARE\Thunder Network\Thunder") as key:
66 install_path = winreg.QueryValueEx(key, "InstallPath")[0]
67 thunder_exe = os.path.join(install_path, "Program", "Thunder.exe")
68 if os.path.exists(thunder_exe):
69 return thunder_exe
70 except:
71 pass

73 elif system == "darwin": # macOS
74 possible_paths = [
75 "/Applications/Thunder.app/Contents/MacOS/Thunder",

166 def _try_xunlei_com_download(self, url: str, save_path: str, filename: str) -> bool:
167 """
168 尝试使用迅雷的COM接口进行下载并指定保存路径。
169
170 Args:
171 url: 下载URL
172 save_path: 保存路径
173 filename: 文件名
174
175 Returns:
176 bool: 是否成功启动下载
177 """
178 try:
179 import win32com.client
180
181 # 先设置迅雷的默认下载路径
182 self._set_xunlei_default_path(save_path)
183
184 # 将目标路径复制到剪贴板，方便用户粘贴
185 self._copy_path_to_clipboard(save_path)
186
187 # 尝试多种COM接口
188 com_objects = [
189 "ThunderAgent.Agent.1",
190 "ThunderAgent.Agent",
191 "ThunderAgent.Agent.2"
192 ]
193
194 thunder = None
195 for com_obj in com_objects:
196 try:
197 thunder = win32com.client.Dispatch(com_obj)
198 logger.info(f"✅ 成功连接到迅雷COM接口: {com_obj}")
199 break
200 except Exception as e:
201 logger.debug(f"⚠️ COM接口 {com_obj} 连接失败: {e}")
202 continue
203
204 if thunder is None:
205 logger.warning("⚠️ 所有COM接口都连接失败")
206 return False
207
208 # 构建完整的绝对路径（包含文件名）
209 full_path = os.path.abspath(os.path.join(save_path, filename))
210
211 # 确保目录存在
212 os.makedirs(save_path, exist_ok=True)
213
214 # 使用COM接口添加下载任务
215 # 注意：某些版本的COM接口参数顺序可能不同
216 try:
217 # 方法1: 标准参数顺序 (url, save_path, filename)
218 thunder.AddTask(url, full_path, filename)
219 thunder.CommitTasks()
220 logger.info(f"✅ 迅雷COM接口下载启动成功: {filename}")
221 logger.info(f"💡 文件将保存到: {full_path}")
222 return True
223 except Exception as e1:
224 logger.debug(f"⚠️ 标准COM参数失败，尝试其他格式: {e1}")
225 try:
226 # 方法2: 只传入URL，让迅雷自动处理路径
227 thunder.AddTask(url, "", filename)
228 thunder.CommitTasks()
229 logger.info(f"✅ 迅雷COM接口下载启动成功 (简化参数): {filename}")
230 logger.info(f"💡 文件将保存到默认路径: {save_path}")
231 return True
232 except Exception as e2:
233 logger.debug(f"⚠️ 简化COM参数也失败: {e2}")
234 return False
235
236 except ImportError:
237 logger.warning("⚠️ pywin32未安装，无法使用COM接口")
238 return False
239 except Exception as e:
240 logger.warning(f"⚠️ COM接口启动失败: {e}")
241 return False

282 if not self.is_available:

286 os.makedirs(save_path, exist_ok=True)

399 def get_download_progress(self, save_path: str, filename: str) -> Tuple[bool, float]:
400 """
401 Check download progress by monitoring file size.
402
403 Args:
404 save_path: Directory where file is being saved
405 filename: Name of the file being downloaded
406
407 Returns:
408 Tuple[bool, float]: (is_complete, progress_percentage)
409 """
410 file_path = os.path.join(save_path, filename)
411
412 if not os.path.exists(file_path):
413 return False, 0.0
414
415 # Get file size
416 current_size = os.path.getsize(file_path)
417
418 # For now, we can't easily get total size from 迅雷

425 def wait_for_completion(self, save_path: str, filename: str, timeout: int = 3600) -> bool:
426 """
427 Wait for download to complete by monitoring file size stability.
428
429 Args:
430 save_path: Directory where file is being saved
431 filename: Name of the file being downloaded
432 timeout: Maximum time to wait in seconds
433
434 Returns:
435 bool: True if download completed successfully
436 """
437 file_path = os.path.join(save_path, filename)
438 start_time = time.time()
439 last_size = 0
440 stable_count = 0
441
442 logger.info(f"⏳ Waiting for 迅雷 download to complete: {filename}")
443
444 while time.time() - start_time < timeout:
445 if os.path.exists(file_path):
446 current_size = os.path.getsize(file_path)
447
448 if current_size == last_size:
449 stable_count += 1
450 if stable_count >= 10: # File size stable for 10 checks
451 logger.info(f"✅ 迅雷 download completed: {filename}")
452 return True
453 else:
454 stable_count = 0
455 last_size = current_size
456
457 # Log progress every 30 seconds
458 if int(time.time() - start_time) % 30 == 0:
459 logger.info(f"📥 Downloading: {current_size / (1024*1024):.1f}MB")
460
461 time.sleep(2)
462
463 logger.warning(f"⏰ Download timeout: {filename}")
464 return False

504 """
505 Download dataset using 迅雷.
506
507 Args:
508 dataset_name: Name of the dataset to download
509 wait_for_completion: Whether to wait for download to complete
510
511 Returns:
512 bool: True if download started successfully
513 """

515 logger.error(f"❌ Unknown dataset: {dataset_name}")

520 url = dataset['url']
521

523 logger.info(f"📋 {dataset['description']}")
524 logger.info(f"🌐 URL: {url}")
525

528 if file_path.exists() and file_path.stat().st_size == dataset['expected_size']:
529 logger.info(f"✅ {dataset_name} already downloaded and verified")

536 filename=filename
537 )
538

540 return self.xunlei_downloader.wait_for_completion(
541 str(self.data_dir),

543 )
