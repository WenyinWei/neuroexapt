25class ModularCNN(nn.Module):
26 """
27 A modular CNN with explicit subnetworks that can be added/removed.
28
29 This model demonstrates how Neuro Exapt can:
30 - Add new feature extraction modules
31 - Remove redundant processing paths
32 - Optimize network topology based on information flow
33 """

35 def __init__(self, num_classes=10):
36 super(ModularCNN, self).__init__()
37
38 # Core feature extraction subnetworks
39 self.core_conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
40 self.core_conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
41
42 # Optional subnetworks (can be added/removed during evolution)
43 self.optional_conv1 = nn.Conv2d(64, 128, kernel_size=3, padding=1)
44 self.optional_conv2 = nn.Conv2d(128, 256, kernel_size=3, padding=1)
45
46 # Attention subnetwork (can be added when needed)
47 self.attention_conv = nn.Conv2d(256, 256, kernel_size=1)
48 self.attention_gate = nn.Sequential(
49 nn.AdaptiveAvgPool2d(1),
50 nn.Conv2d(256, 256, kernel_size=1),
51 nn.Sigmoid()
52 )
53
54 # Pooling and regularization
55 self.pool = nn.MaxPool2d(2, 2)
56 self.dropout = nn.Dropout(0.25)
57
58 # Classification subnetworks
59 self.classifier1 = nn.Linear(256 * 4 * 4, 512)
60 self.classifier2 = nn.Linear(512, 256)
61 self.classifier3 = nn.Linear(256, num_classes)
62
63 # Evolution tracking
64 self.active_subnetworks = {
65 'core': True,
66 'optional': True,
67 'attention': False, # Initially disabled
68 'classifier': True
69 }
70
71 self.evolution_history = []
72
73 def forward(self, x):
74 # Core subnetwork (always active)
75 x = self.pool(F.relu(self.core_conv1(x)))
76 x = self.pool(F.relu(self.core_conv2(x)))
77
78 # Optional subnetwork (can be pruned)
79 if self.active_subnetworks['optional']:
80 x = self.pool(F.relu(self.optional_conv1(x)))
81 x = self.pool(F.relu(self.optional_conv2(x)))
82
83 # Attention subnetwork (can be added)
84 if self.active_subnetworks['attention']:
85 attention_weights = self.attention_gate(x)
86 x = x * attention_weights
87
88 # Flatten
89 x = x.view(-1, 256 * 4 * 4)
90
91 # Classification subnetwork
92 if self.active_subnetworks['classifier']:
93 x = F.relu(self.classifier1(x))
94 x = self.dropout(x)
95 x = F.relu(self.classifier2(x))
96 x = self.classifier3(x)
97
98 return x

100 def add_subnetwork(self, subnetwork_type):
101 """Add a new subnetwork."""
102 if subnetwork_type == 'attention' and not self.active_subnetworks['attention']:
103 self.active_subnetworks['attention'] = True
104 print(f"✅ Added attention subnetwork")
105 return True
106 return False
107
108 def remove_subnetwork(self, subnetwork_type):
109 """Remove a subnetwork."""
110 if subnetwork_type == 'optional' and self.active_subnetworks['optional']:
111 self.active_subnetworks['optional'] = False
112 print(f"🗑️ Removed optional subnetwork")
113 return True
114 return False

116 def get_subnetwork_info(self):
117 """Get information about active subnetworks."""
118 info = {
119 'active_count': sum(self.active_subnetworks.values()),
120 'total_count': len(self.active_subnetworks),
121 'active_subnetworks': [k for k, v in self.active_subnetworks.items() if v],
122 'inactive_subnetworks': [k for k, v in self.active_subnetworks.items() if not v]
123 }
124 return info

162def analyze_information_flow(model, dataloader, device):
163 """Analyze information flow through different subnetworks."""
164 model.eval()
165
166 # Collect activations from different subnetworks
167 activations = {
168 'core': [],
169 'optional': [],
170 'attention': [],
171 'classifier': []
172 }
173
174 with torch.no_grad():
175 for batch_idx, (data, _) in enumerate(dataloader):
176 if batch_idx >= 5: # Analyze first 5 batches
177 break
178
179 data = data.to(device)
180
181 # Forward pass through core
182 x = F.relu(model.core_conv1(data))
183 x = F.relu(model.core_conv2(x))
184 activations['core'].append(x.mean().item())
185
186 # Forward pass through optional
187 if model.active_subnetworks['optional']:
188 x = F.relu(model.optional_conv1(x))
189 x = F.relu(model.optional_conv2(x))
190 activations['optional'].append(x.mean().item())
191
192 # Forward pass through attention
193 if model.active_subnetworks['attention']:
194 attention_weights = model.attention_gate(x)
195 x = x * attention_weights
196 activations['attention'].append(attention_weights.mean().item())
197
198 # Forward pass through classifier
199 x = x.view(-1, 256 * 4 * 4)
200 if model.active_subnetworks['classifier']:
201 x = F.relu(model.classifier1(x))
202 activations['classifier'].append(x.mean().item())
203
204 # Calculate information metrics
205 info_metrics = {}
206 for subnetwork, acts in activations.items():
207 if acts:
208 info_metrics[subnetwork] = {
209 'mean_activation': np.mean(acts),
210 'activation_variance': np.var(acts),
211 'information_content': np.mean(acts) * np.var(acts) # Simple info metric
212 }
213
214 return info_metrics
