130 def save_checkpoint(self, epoch: int, model_state: Dict, optimizer_states: Dict,
131 scheduler_states: Optional[Dict], training_stats: Dict,
132 framework_state: Dict, performance_metric: float,
133 architecture_info: Optional[Dict] = None) -> str:
134 """
135 保存训练检查点
136
137 Args:
138 epoch: 当前epoch
139 model_state: 模型状态字典
140 optimizer_states: 优化器状态字典
141 scheduler_states: 学习率调度器状态
142 training_stats: 训练统计信息
143 framework_state: ASO-SE框架状态
144 performance_metric: 性能指标（用于判断最佳模型）
145 architecture_info: 架构信息
146
147 Returns:
148 保存的检查点路径
149 """
150 checkpoint_filename = f"checkpoint_epoch_{epoch:04d}.pth"
151 checkpoint_path = self.models_dir / checkpoint_filename
152
153 # 完整的检查点数据
154 checkpoint_data = {
155 # 基本信息
156 'epoch': epoch,
157 'session_id': self.training_session_id,
158 'timestamp': datetime.now().isoformat(),
159 'performance_metric': performance_metric,
160
161 # 模型状态
162 'model_state': model_state,
163 'optimizer_states': optimizer_states,
164 'scheduler_states': scheduler_states,
165
166 # 框架状态
167 'framework_state': framework_state,
168 'training_stats': training_stats,
169 'architecture_info': architecture_info,
170
171 # 元信息
172 'checkpoint_version': '2.0',
173 'pytorch_version': torch.__version__,
174 'device_info': self._get_device_info()
175 }
176
177 try:
178 # 保存检查点
179 torch.save(checkpoint_data, checkpoint_path)
180
181 # 更新历史记录
182 checkpoint_record = {
183 'epoch': epoch,
184 'path': str(checkpoint_path),
185 'performance_metric': performance_metric,
186 'timestamp': checkpoint_data['timestamp'],
187 'session_id': self.training_session_id,
188 'size_mb': checkpoint_path.stat().st_size / (1024 * 1024),
189 'architecture_info': architecture_info
190 }
191
192 self.checkpoint_history.append(checkpoint_record)
193 self.current_checkpoint = checkpoint_record
194
195 # 检查是否是最佳性能
196 if performance_metric > self.best_performance:
197 self.best_performance = performance_metric
198 self.best_checkpoint_path = str(checkpoint_path)
199 self._save_best_model(checkpoint_data)
200 logger.info(f"🏆 New best performance: {performance_metric:.4f}")
201
202 # 清理旧检查点
203 self._cleanup_old_checkpoints()
204
205 # 保存元数据
206 self._save_metadata()
207
208 logger.info(f"💾 Checkpoint saved: epoch {epoch}, performance {performance_metric:.4f}")
209 return str(checkpoint_path)
210
211 except Exception as e:
212 logger.error(f"❌ Failed to save checkpoint: {e}")
213 raise e

254 def load_checkpoint(self, checkpoint_path: str) -> Optional[Dict]:
255 """加载指定的检查点"""
256 checkpoint_path = Path(checkpoint_path)
257
258 if not checkpoint_path.exists():
259 logger.error(f"❌ Checkpoint not found: {checkpoint_path}")
260 return None
261
262 try:
263 # PyTorch 2.6+ requires weights_only=False for complex checkpoints
264 checkpoint_data = torch.load(checkpoint_path, map_location='cpu', weights_only=False)
265
266 logger.info(f"📂 Loaded checkpoint from epoch {checkpoint_data['epoch']}")
267 logger.info(f" Performance: {checkpoint_data['performance_metric']:.4f}")
268 logger.info(f" Timestamp: {checkpoint_data['timestamp']}")
269
270 return checkpoint_data
271
272 except Exception as e:
273 logger.error(f"❌ Failed to load checkpoint: {e}")
274 return None

276 def resume_training(self, model, optimizers: Dict, schedulers: Optional[Dict],
277 framework, preferred_checkpoint: str = "latest") -> Tuple[int, Dict]:
278 """
279 恢复训练状态
280
281 Args:
282 model: 要恢复的模型
283 optimizers: 优化器字典 {'weight': optimizer1, 'arch': optimizer2}
284 schedulers: 学习率调度器字典
285 framework: ASO-SE框架实例
286 preferred_checkpoint: 'latest', 'best', 或具体路径
287
288 Returns:
289 (resume_epoch, training_stats)
290 """
291 logger.info(f"🔄 Attempting to resume training from {preferred_checkpoint} checkpoint...")
292
293 # 选择检查点
294 if preferred_checkpoint == "latest":
295 checkpoint_data = self.load_latest_checkpoint()
296 elif preferred_checkpoint == "best":
297 checkpoint_data = self.load_best_checkpoint()
298 else:
299 checkpoint_data = self.load_checkpoint(preferred_checkpoint)
300
301 if checkpoint_data is None:
302 logger.info("🆕 No checkpoint found, starting fresh training")
303 return 0, {}
304
305 try:
306 # 恢复模型状态
307 model.load_state_dict(checkpoint_data['model_state'])
308 logger.info("✅ Model state restored")
309
310 # 恢复优化器状态
311 optimizer_states = checkpoint_data.get('optimizer_states', {})
312 for name, optimizer in optimizers.items():
313 if name in optimizer_states:
314 optimizer.load_state_dict(optimizer_states[name])
315 logger.info(f"✅ {name} optimizer state restored")
316
317 # 恢复学习率调度器状态
318 if schedulers and checkpoint_data.get('scheduler_states'):
319 scheduler_states = checkpoint_data['scheduler_states']
320 for name, scheduler in schedulers.items():
321 if name in scheduler_states:
322 scheduler.load_state_dict(scheduler_states[name])
323 logger.info(f"✅ {name} scheduler state restored")
324
325 # 恢复ASO-SE框架状态
326 framework_state = checkpoint_data.get('framework_state', {})
327 if hasattr(framework, 'load_state'):
328 framework.load_state(framework_state)
329 logger.info("✅ ASO-SE framework state restored")
330
331 # 恢复训练统计
332 training_stats = checkpoint_data.get('training_stats', {})
333 resume_epoch = checkpoint_data['epoch'] + 1
334
335 logger.info(f"🎯 Training resumed from epoch {resume_epoch}")
336 logger.info(f" Previous performance: {checkpoint_data['performance_metric']:.4f}")
337
338 return resume_epoch, training_stats
339
340 except Exception as e:
341 logger.error(f"❌ Failed to resume training: {e}")
342 logger.info("🆕 Starting fresh training due to resume failure")
343 return 0, {}

345 def _cleanup_old_checkpoints(self):
346 """清理旧的检查点文件"""
347 if len(self.checkpoint_history) <= self.max_checkpoints:
348 return
349
350 # 按性能排序，保留最佳的checkpoints
351 sorted_checkpoints = sorted(
352 self.checkpoint_history,
353 key=lambda x: x['performance_metric'],
354 reverse=True
355 )
356
357 # 保留最佳的检查点
358 to_keep = sorted_checkpoints[:self.max_checkpoints]
359 to_remove = [cp for cp in self.checkpoint_history if cp not in to_keep]
360
361 for checkpoint in to_remove:
362 try:
363 checkpoint_path = Path(checkpoint['path'])
364 if checkpoint_path.exists():
365 checkpoint_path.unlink()
366 logger.debug(f"🗑️ Removed old checkpoint: {checkpoint_path.name}")
367 except Exception as e:
368 logger.warning(f"⚠️ Failed to remove checkpoint: {e}")
369
370 # 更新历史记录
371 self.checkpoint_history = to_keep

459## Performance Trend
