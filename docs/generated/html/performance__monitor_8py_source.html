96 @contextmanager
97 def timer(self, name: str, log_immediately: bool = False):
98 """计时器上下文管理器"""
99 start_time = time.time()
100 try:
101 yield
102 finally:
103 elapsed = time.time() - start_time
104 self.record_time(name, elapsed)
105 if log_immediately:
106 self.logger.info(f"⏱️ {name}: {elapsed:.4f}s")

108 def record_time(self, name: str, elapsed_time: float):
109 """记录时间统计"""
110 self.stats[name].append(elapsed_time)
111
112 # 保持最近1000个记录
113 if len(self.stats[name]) > 1000:
114 self.stats[name] = self.stats[name][-1000:]
115
116 def start_timer(self, name: str):
117 """开始计时"""
118 self.current_timers[name] = time.time()
119
120 def end_timer(self, name: str) -> float:
121 """结束计时并返回耗时"""
122 if name not in self.current_timers:
123 self.logger.warning(f"Timer '{name}' not found")
124 return 0.0
125
126 elapsed = time.time() - self.current_timers[name]
127 self.record_time(name, elapsed)
128 del self.current_timers[name]
129 return elapsed

148 def log_model_stats(self, model: torch.nn.Module, model_name: str = "model"):
149 """记录模型统计信息"""
150 total_params = sum(p.numel() for p in model.parameters())
151 trainable_params = sum(p.numel() for p in model.parameters() if p.requires_grad)
152
153 self.logger.info(f"📊 {model_name} - Total params: {total_params:,}, "
154 f"Trainable: {trainable_params:,}")
155
156 # 记录参数分布
157 param_stats = {}
158 for name, param in model.named_parameters():
159 if param.requires_grad:
160 param_stats[name] = {
161 'shape': list(param.shape),
162 'numel': param.numel(),
163 'mean': param.data.mean().item(),
164 'std': param.data.std().item(),
165 'grad_norm': param.grad.norm().item() if param.grad is not None else 0.0
166 }
167
168 return param_stats

170 def log_epoch_summary(self, epoch: int, train_acc: float, valid_acc: float,
171 train_time: float, valid_time: float, arch_time: float = 0.0):
172 """记录epoch摘要"""
173 self.real_time_stats['current_epoch'] = epoch
174
175 # 更新最佳准确率
176 if valid_acc > self.real_time_stats['best_accuracy']:
177 self.real_time_stats['best_accuracy'] = valid_acc
178 self.real_time_stats['last_improvement_epoch'] = epoch
179
180 total_time = train_time + valid_time + arch_time
181
182 self.history['epoch_times'].append(total_time)
183 self.history['train_times'].append(train_time)
184 self.history['valid_times'].append(valid_time)
185 self.history['arch_times'].append(arch_time)
186
187 # 计算统计信息
188 avg_epoch_time = np.mean(self.history['epoch_times'][-10:])
189 arch_percentage = (arch_time / total_time * 100) if total_time > 0 else 0
190
191 self.logger.info(f"🏆 EPOCH {epoch} SUMMARY:")
192 self.logger.info(f" Train Acc: {train_acc:.2f}% | Valid Acc: {valid_acc:.2f}%")
193 self.logger.info(f" Times: Train {train_time:.1f}s | Valid {valid_time:.1f}s | Arch {arch_time:.1f}s")
194 self.logger.info(f" Total: {total_time:.1f}s | Arch: {arch_percentage:.1f}% | Avg: {avg_epoch_time:.1f}s")
195 self.logger.info(f" Best Acc: {self.real_time_stats['best_accuracy']:.2f}% "
196 f"(Epoch {self.real_time_stats['last_improvement_epoch']})")

198 def log_architecture_step(self, step_type: str, duration: float, details: Optional[Dict[str, Any]] = None):
199 """记录架构搜索步骤"""
200 self.real_time_stats['total_steps'] += 1
201
202 if step_type == 'arch_update':
203 self.real_time_stats['arch_steps'] += 1
204 elif step_type == 'arch_skip':
205 self.real_time_stats['skipped_arch_steps'] += 1
206
207 skip_rate = (self.real_time_stats['skipped_arch_steps'] /
208 self.real_time_stats['total_steps'] * 100)
209
210 self.logger.info(f"🔍 Architecture Step [{step_type}]: {duration:.4f}s "
211 f"(Skip Rate: {skip_rate:.1f}%)")
212
213 if details:
214 for key, value in details.items():
215 self.logger.info(f" {key}: {value}")

258 def log_bottleneck_analysis(self):
259 """分析性能瓶颈"""
260 self.logger.info("🔍 BOTTLENECK ANALYSIS:")
261
262 # 找到最耗时的操作
263 operation_times = {}
264 for name, times in self.stats.items():
265 if times:
266 operation_times[name] = {
267 'total': sum(times),
268 'avg': np.mean(times),
269 'count': len(times)
270 }
271
272 # 按总耗时排序
273 sorted_operations = sorted(operation_times.items(),
274 key=lambda x: x[1]['total'], reverse=True)
275
276 total_time = sum(op['total'] for op in operation_times.values())
277
278 self.logger.info(f" Total measured time: {total_time:.2f}s")
279
280 for i, (name, stats) in enumerate(sorted_operations[:10]):
281 percentage = (stats['total'] / total_time * 100) if total_time > 0 else 0
282 self.logger.info(f" {i+1:2d}. {name}: {stats['total']:.2f}s "
283 f"({percentage:.1f}%) | Avg: {stats['avg']:.4f}s | "
284 f"Count: {stats['count']}")

286 def suggest_optimizations(self):
287 """建议优化策略"""
288 self.logger.info("💡 OPTIMIZATION SUGGESTIONS:")
289
290 # 分析架构搜索时间占比
291 arch_times = self.stats.get('arch_step', [])
292 total_times = self.stats.get('epoch_total', [])

294 if arch_times and total_times:
295 arch_percentage = (sum(arch_times) / sum(total_times) * 100)
296 if arch_percentage > 30:
297 self.logger.info(f" 🔧 Architecture search takes {arch_percentage:.1f}% of time")
298 self.logger.info(f" → Increase arch_update_freq to reduce frequency")
299 self.logger.info(f" → Use first-order approximation (unrolled=False)")
300
301 # 检查GPU利用率
302 gpu_memory = self.history.get('gpu_memory', [])
303 if gpu_memory:
304 avg_memory = np.mean([m['allocated'] for m in gpu_memory])
305 if avg_memory < 2.0: # 少于2GB
306 self.logger.info(f" 🔧 GPU memory usage is low ({avg_memory:.1f}GB)")
307 self.logger.info(f" → Increase batch size for better GPU utilization")
308
309 # 检查训练停滞
310 last_improvement = self.real_time_stats.get('last_improvement_epoch', 0)
311 current_epoch = self.real_time_stats.get('current_epoch', 0)
312
