38 def wider_conv2d(self, layer: nn.Conv2d, new_out_channels: int,
39 next_layer: Optional[nn.Module] = None) -> Tuple[nn.Conv2d, Optional[nn.Module]]:
40 """
41 Net2WiderNet: 扩展卷积层的输出通道数
42
43 Args:
44 layer: 原始卷积层
45 new_out_channels: 新的输出通道数
46 next_layer: 下一层(用于权重调整)
47
48 Returns:
49 (新卷积层, 调整后的下一层)
50 """
51 assert new_out_channels > layer.out_channels, "新通道数必须大于原通道数"
52
53 # 创建新的卷积层
54 new_layer = nn.Conv2d(
55 in_channels=layer.in_channels,
56 out_channels=new_out_channels,
57 kernel_size=layer.kernel_size,
58 stride=layer.stride,
59 padding=layer.padding,
60 dilation=layer.dilation,
61 groups=layer.groups,
62 bias=layer.bias is not None,
63 padding_mode=layer.padding_mode
64 )
65
66 # 初始化权重
67 with torch.no_grad():
68 # 复制原有权重
69 new_layer.weight[:layer.out_channels] = layer.weight.data
70 if layer.bias is not None:
71 new_layer.bias[:layer.out_channels] = layer.bias.data
72
73 # 为新增通道随机选择复制源
74 additional_channels = new_out_channels - layer.out_channels
75 for i in range(additional_channels):
76 # 随机选择一个原有通道进行复制
77 source_idx = np.random.randint(0, layer.out_channels)
78 target_idx = layer.out_channels + i
79
80 # 复制权重并添加小扰动
81 new_layer.weight[target_idx] = layer.weight[source_idx].clone()
82 new_layer.weight[target_idx] += torch.randn_like(new_layer.weight[target_idx]) * self.noise_scale
83
84 if layer.bias is not None:
85 new_layer.bias[target_idx] = layer.bias[source_idx].clone()
86 new_layer.bias[target_idx] += torch.randn_like(new_layer.bias[target_idx]) * self.noise_scale
87
88 # 调整下一层
89 new_next_layer = None
90 if next_layer is not None:
91 new_next_layer = self._adjust_next_layer_for_wider(next_layer, layer.out_channels, new_out_channels)
92
93 logger.info(f"Net2WiderNet: {layer.out_channels} -> {new_out_channels} channels")
94 return new_layer, new_next_layer

137 def deeper_conv2d(self, layer: nn.Conv2d, position: str = 'after') -> nn.Module:
138 """
139 Net2DeeperNet: 在卷积层后插入新层
140
141 Args:
142 layer: 原始卷积层
143 position: 'before' 或 'after'
144
145 Returns:
146 新插入的层（恒等变换）
147 """
148 if position == 'after':
149 # 在后面插入1x1恒等卷积
150 new_layer = nn.Conv2d(
151 in_channels=layer.out_channels,
152 out_channels=layer.out_channels,
153 kernel_size=1,
154 stride=1,
155 padding=0,
156 bias=True
157 )
158
159 # 初始化为恒等变换
160 with torch.no_grad():
161 nn.init.eye_(new_layer.weight.squeeze())
162 if new_layer.bias is not None:
163 nn.init.zeros_(new_layer.bias)
164
165 else: # before
166 # 在前面插入恒等卷积
167 new_layer = nn.Conv2d(
168 in_channels=layer.in_channels,
169 out_channels=layer.in_channels,
170 kernel_size=layer.kernel_size,
171 stride=1,
172 padding=layer.padding,
173 bias=True
174 )
175
176 # 初始化为恒等变换
177 with torch.no_grad():
178 # 对于3x3卷积，中心设为1，其余为0
179 nn.init.zeros_(new_layer.weight)
180 if layer.kernel_size == (3, 3):
181 for i in range(layer.in_channels):
182 new_layer.weight[i, i, 1, 1] = 1.0
183 elif layer.kernel_size == (1, 1):
184 nn.init.eye_(new_layer.weight.squeeze())
185
186 if new_layer.bias is not None:
187 nn.init.zeros_(new_layer.bias)
188
189 logger.info(f"Net2DeeperNet: Added identity layer {position} conv layer")
190 return new_layer

209 def _adjust_next_layer_for_wider(self, next_layer: nn.Module, old_channels: int, new_channels: int) -> nn.Module:
210 """调整下一层以适应扩宽的前一层"""
211
212 if isinstance(next_layer, nn.Conv2d):
213 # 调整卷积层的输入通道
214 new_next = nn.Conv2d(
215 in_channels=new_channels,
216 out_channels=next_layer.out_channels,
217 kernel_size=next_layer.kernel_size,
218 stride=next_layer.stride,
219 padding=next_layer.padding,
220 dilation=next_layer.dilation,
221 groups=next_layer.groups,
222 bias=next_layer.bias is not None,
223 padding_mode=next_layer.padding_mode
224 )
225
226 with torch.no_grad():
227 # 复制原有权重
228 new_next.weight[:, :old_channels] = next_layer.weight.data
229
230 # 新增通道的权重设为0（保持函数等价性）
231 new_next.weight[:, old_channels:] = 0
232
233 if next_layer.bias is not None:
234 new_next.bias[:] = next_layer.bias.data
235
236 return new_next
237
238 elif isinstance(next_layer, nn.Linear):
239 # 调整线性层的输入特征
240 new_next = nn.Linear(
241 in_features=new_channels,
242 out_features=next_layer.out_features,
243 bias=next_layer.bias is not None
244 )
245
246 with torch.no_grad():
247 # 复制原有权重
248 new_next.weight[:, :old_channels] = next_layer.weight.data
249
250 # 新增特征的权重设为0
251 new_next.weight[:, old_channels:] = 0
252
253 if next_layer.bias is not None:
254 new_next.bias[:] = next_layer.bias.data
255
256 return new_next
257
258 elif isinstance(next_layer, nn.BatchNorm2d):
259 # 调整BatchNorm层
260 new_next = nn.BatchNorm2d(new_channels)
261
262 with torch.no_grad():
263 # 复制原有参数
264 new_next.weight[:old_channels] = next_layer.weight.data
265 new_next.bias[:old_channels] = next_layer.bias.data
266 new_next.running_mean[:old_channels] = next_layer.running_mean.data
267 new_next.running_var[:old_channels] = next_layer.running_var.data
268
269 # 新增通道初始化
270 new_next.weight[old_channels:] = 1.0
271 new_next.bias[old_channels:] = 0.0
272 new_next.running_mean[old_channels:] = 0.0
273 new_next.running_var[old_channels:] = 1.0
274
275 new_next.num_batches_tracked = next_layer.num_batches_tracked
276
277 return new_next
278
279 return next_layer

288 def split_conv_layer(self, layer: nn.Conv2d, split_type: str = 'parallel') -> nn.Module:

300 return self._split_conv_parallel(layer)

304 raise ValueError(f"Unsupported split_type: {split_type}")

306 def _split_conv_parallel(self, layer: nn.Conv2d) -> nn.Module:
307 """将卷积层分裂成两个并行分支"""
308
309 # 计算每个分支的通道数
310 branch1_channels = layer.out_channels // 2
311 branch2_channels = layer.out_channels - branch1_channels
312
313 # 创建两个分支
314 branch1 = nn.Conv2d(
315 layer.in_channels, branch1_channels,
316 layer.kernel_size, layer.stride, layer.padding,
317 layer.dilation, layer.groups, layer.bias is not None
318 )
319
320 branch2 = nn.Conv2d(
321 layer.in_channels, branch2_channels,
322 layer.kernel_size, layer.stride, layer.padding,
323 layer.dilation, layer.groups, layer.bias is not None
324 )
325
326 # 分配权重
327 with torch.no_grad():
328 branch1.weight.data = layer.weight[:branch1_channels].clone()
329 branch2.weight.data = layer.weight[branch1_channels:].clone()
330
331 if layer.bias is not None:
332 branch1.bias.data = layer.bias[:branch1_channels].clone()
333 branch2.bias.data = layer.bias[branch1_channels:].clone()
334
335 # 创建并行模块
336 class ParallelBranches(nn.Module):
337 def __init__(self, branch1, branch2):
338 super().__init__()
339 self.branch1 = branch1
340 self.branch2 = branch2
341
342 def forward(self, x):
343 out1 = self.branch1(x)
344 out2 = self.branch2(x)
345 return torch.cat([out1, out2], dim=1)
346
347 logger.info(f"Split conv layer into parallel branches: {branch1_channels} + {branch2_channels}")
348 return ParallelBranches(branch1, branch2)
349
350 def _split_conv_sequential(self, layer: nn.Conv2d) -> nn.Module:
351 """将卷积层分裂成两个串行层"""
352
353 # 计算中间通道数
354 intermediate_channels = max(layer.out_channels, layer.in_channels)
355
356 # 第一层：输入 -> 中间通道
357 layer1 = nn.Conv2d(
358 layer.in_channels, intermediate_channels,
359 kernel_size=1, stride=1, padding=0, bias=False
360 )
361
362 # 第二层：中间通道 -> 输出
363 layer2 = nn.Conv2d(
364 intermediate_channels, layer.out_channels,
365 layer.kernel_size, layer.stride, layer.padding,
366 layer.dilation, bias=layer.bias is not None
367 )
368
369 # BatchNorm层
370 bn = nn.BatchNorm2d(intermediate_channels)
371
372 # 权重初始化
373 with torch.no_grad():
374 # 第一层使用随机初始化
375 nn.init.kaiming_normal_(layer1.weight, mode='fan_out', nonlinearity='relu')
376
377 # 第二层尽量保持原有功能
378 if intermediate_channels == layer.out_channels:
379 # 如果通道数匹配，可以直接复制权重
380 layer2.weight.data = layer.weight.data.clone()
381 if layer.bias is not None:
382 layer2.bias.data = layer.bias.data.clone()
383 else:
384 # 否则使用He初始化
385 nn.init.kaiming_normal_(layer2.weight, mode='fan_out', nonlinearity='relu')
386 if layer2.bias is not None:
387 nn.init.zeros_(layer2.bias)
388
389 # 创建串行模块
390 sequential_module = nn.Sequential(layer1, bn, nn.ReLU(inplace=True), layer2)
391
392 logger.info(f"Split conv layer into sequential: {layer.in_channels} -> {intermediate_channels} -> {layer.out_channels}")
393 return sequential_module

395 def mutate_conv_to_depthwise_separable(self, layer: nn.Conv2d) -> nn.Module:
396 """将标准卷积变异为深度可分离卷积"""
397
398 if layer.groups != 1:
399 logger.warning("Layer is already grouped, skipping depthwise separable mutation")
400 return layer
401
402 # 深度卷积
403 depthwise = nn.Conv2d(
404 layer.in_channels, layer.in_channels,
405 layer.kernel_size, layer.stride, layer.padding,
406 groups=layer.in_channels, bias=False
407 )
408
409 # 逐点卷积
410 pointwise = nn.Conv2d(
411 layer.in_channels, layer.out_channels,
412 kernel_size=1, stride=1, padding=0,
413 bias=layer.bias is not None
414 )
415
416 # 权重初始化
417 with torch.no_grad():
418 # 深度卷积：每个输入通道对应一个滤波器
419 for i in range(layer.in_channels):
420 if i < layer.out_channels:
421 # 从原层复制对应的权重
422 depthwise.weight[i, 0] = layer.weight[i % layer.out_channels, i]
423 else:
424 # 随机初始化
425 nn.init.kaiming_normal_(depthwise.weight[i:i+1], mode='fan_out', nonlinearity='relu')
426
427 # 逐点卷积初始化
428 nn.init.kaiming_normal_(pointwise.weight, mode='fan_out', nonlinearity='relu')
429 if pointwise.bias is not None and layer.bias is not None:
430 pointwise.bias.data = layer.bias.data.clone()
431
432 # 组合模块
433 module = nn.Sequential(depthwise, pointwise)
434
435 logger.info(f"Mutated conv to depthwise separable: {layer.in_channels}x{layer.out_channels}")
436 return module

438 def add_residual_connection(self, layer: nn.Module, input_channels: int, output_channels: int) -> nn.Module:
439 """为层添加残差连接"""
440
441 # 创建跳跃连接（如果通道数不匹配，使用1x1卷积调整）
442 if input_channels == output_channels:
443 shortcut = nn.Identity()
444 else:
445 shortcut = nn.Conv2d(input_channels, output_channels, 1, bias=False)
446 with torch.no_grad():
447 nn.init.kaiming_normal_(shortcut.weight, mode='fan_out', nonlinearity='relu')
448
449 # 创建残差模块
450 class ResidualBlock(nn.Module):
451 def __init__(self, main_layer, shortcut):
452 super().__init__()
453 self.main_layer = main_layer
454 self.shortcut = shortcut
455
456 def forward(self, x):
457 identity = self.shortcut(x)
458 out = self.main_layer(x)
459
460 # 确保尺寸匹配
461 if out.shape != identity.shape:
462 # 使用自适应池化调整空间尺寸
463 identity = F.adaptive_avg_pool2d(identity, out.shape[2:])
464
465 return out + identity
466
467 logger.info(f"Added residual connection: {input_channels} -> {output_channels}")
468 return ResidualBlock(layer, shortcut)
