101 def _compute_information_metrics(self, activation: torch.Tensor) -> Dict[str, float]:
102 """计算信息论指标"""
103 metrics = {}
104
105 # 展平激活值
106 act_flat = activation.view(activation.size(0), -1)
107
108 if act_flat.size(1) == 0:
109 return {'entropy': 0.0, 'mutual_info_proxy': 0.0, 'information_flow': 0.0}
110
111 # 计算激活熵
112 try:
113 # 归一化激活值
114 act_norm = F.softmax(act_flat, dim=-1) + 1e-8
115 entropy = -torch.sum(act_norm * torch.log(act_norm), dim=-1).mean().item()
116 metrics['entropy'] = entropy
117 except:
118 metrics['entropy'] = 0.0
119
120 # 计算特征间相关性（互信息代理）
121 try:
122 if act_flat.size(1) > 1:
123 correlation_matrix = torch.corrcoef(act_flat.T)
124 avg_correlation = torch.mean(torch.abs(correlation_matrix - torch.eye(correlation_matrix.size(0), device=correlation_matrix.device))).item()
125 metrics['mutual_info_proxy'] = avg_correlation
126 else:
127 metrics['mutual_info_proxy'] = 0.0
128 except:
129 metrics['mutual_info_proxy'] = 0.0
130
131 # 信息流指标（激活值的标准差）
132 metrics['information_flow'] = torch.std(act_flat).item()
133
134 return metrics

136 def _compute_gradient_health(self, gradient: torch.Tensor) -> Dict[str, float]:
137 """计算梯度健康度"""
138 metrics = {}
139
140 grad_flat = gradient.view(-1)
141
142 # 梯度范数
143 metrics['gradient_norm'] = torch.norm(grad_flat).item()
144
145 # 梯度稳定性（标准差与均值的比）
146 grad_mean = torch.mean(torch.abs(grad_flat)).item()
147 grad_std = torch.std(grad_flat).item()
148 metrics['gradient_stability'] = grad_std / (grad_mean + 1e-8)
149
150 # 梯度饱和度（接近0的梯度比例）
151 near_zero_ratio = torch.sum(torch.abs(grad_flat) < 1e-6).float() / len(grad_flat)
152 metrics['gradient_saturation'] = near_zero_ratio.item()
153
154 # 梯度健康分数
155 if metrics['gradient_norm'] < 1e-8:
156 health_score = 0.0 # 梯度消失
157 elif metrics['gradient_norm'] > 100:
158 health_score = 0.2 # 梯度爆炸
159 else:
160 health_score = 1.0 / (1.0 + metrics['gradient_stability'])
161
162 metrics['gradient_health'] = health_score
163
164 return metrics

196 def _compute_feature_separability(self, features: torch.Tensor, targets: torch.Tensor) -> float:
197 """计算特征可分离性"""
198 unique_labels = torch.unique(targets)
199
200 if len(unique_labels) < 2:
201 return 0.0
202
203 # 计算类内距离和类间距离
204 intra_class_distances = []
205 inter_class_distances = []
206
207 for label in unique_labels:
208 mask = targets == label
209 if torch.sum(mask) < 2:
210 continue
211
212 class_features = features[mask]
213 class_center = torch.mean(class_features, dim=0)
214
215 # 类内距离
216 intra_dist = torch.mean(torch.norm(class_features - class_center, dim=1)).item()
217 intra_class_distances.append(intra_dist)
218
219 # 类间距离
220 for other_label in unique_labels:
221 if other_label <= label:
222 continue
223 other_mask = targets == other_label
224 if torch.sum(other_mask) == 0:
225 continue
226
227 other_center = torch.mean(features[other_mask], dim=0)
228 inter_dist = torch.norm(class_center - other_center).item()
229 inter_class_distances.append(inter_dist)
230
231 if not intra_class_distances or not inter_class_distances:
232 return 0.0
233
234 avg_intra = np.mean(intra_class_distances)
235 avg_inter = np.mean(inter_class_distances)
236
237 # 可分离性 = 类间距离 / 类内距离
238 separability = avg_inter / (avg_intra + 1e-8)
239 return min(separability, 10.0) # 限制上界

241 def _compute_activation_diversity(self, activation: torch.Tensor) -> float:
242 """计算激活多样性"""
243 # 使用激活值的方差作为多样性指标
244 variance_per_neuron = torch.var(activation, dim=0)
245 avg_variance = torch.mean(variance_per_neuron).item()
246
247 # 计算激活模式的多样性（不同样本间的差异）
248 if activation.size(0) > 1:
249 pairwise_distances = torch.pdist(activation)
250 avg_pairwise_distance = torch.mean(pairwise_distances).item()
251 else:
252 avg_pairwise_distance = 0.0
253
254 # 综合多样性分数
255 diversity = np.sqrt(avg_variance * avg_pairwise_distance)
256 return diversity

258 def _compute_representation_efficiency(self, activation: torch.Tensor) -> float:
259 """计算表示效率"""
260 # 有效维度 vs 总维度
261 variance_per_neuron = torch.var(activation, dim=0)
262 active_neurons = torch.sum(variance_per_neuron > 1e-6).item()
263 total_neurons = activation.size(1)
264
265 efficiency = active_neurons / (total_neurons + 1e-8)
266 return efficiency

268 def _compute_learning_efficiency(self, layer_name: str, current_accuracy: float) -> Dict[str, float]:
269 """计算学习效率"""
270 metrics = {}
271
272 history = self.layer_metrics_history[layer_name]
273
274 if len(history) < 3:
275 metrics['learning_rate'] = 0.0
276 metrics['improvement_trend'] = 0.0
277 return metrics
278
279 # 计算最近几个epoch的改进趋势
280 recent_accuracies = [current_accuracy] + [h.get('accuracy_impact', 0) for h in list(history)[-3:]]
281
282 if len(recent_accuracies) > 1:
283 # 计算改进趋势
284 improvements = np.diff(recent_accuracies)
285 avg_improvement = np.mean(improvements)
286 metrics['improvement_trend'] = avg_improvement
287
288 # 学习率（改进的一致性）
289 consistency = 1.0 - np.std(improvements)
290 metrics['learning_rate'] = max(0.0, consistency)
291 else:
292 metrics['learning_rate'] = 0.0
293 metrics['improvement_trend'] = 0.0
294
295 return metrics

297 def _compute_bottleneck_score(self, metrics: Dict[str, float]) -> float:
298 """计算综合瓶颈分数（越高越需要改进）"""
299
300 # 权重配置
301 weights = {
302 'gradient_health': 0.25, # 梯度健康度（越低越需要改进）
303 'feature_separability': 0.20, # 特征可分离性（越低越需要改进）
304 'representation_efficiency': 0.15, # 表示效率
305 'information_flow': 0.15, # 信息流
306 'learning_rate': 0.15, # 学习效率
307 'activation_diversity': 0.10 # 激活多样性
308 }
309
310 score = 0.0
311 total_weight = 0.0
312
313 for metric, weight in weights.items():
314 if metric in metrics:
315 value = metrics[metric]
316
317 # 将所有指标转换为"需要改进"的分数（越高越需要改进）
318 if metric in ['gradient_health', 'feature_separability', 'representation_efficiency',
319 'information_flow', 'learning_rate', 'activation_diversity']:
320 # 这些指标越低越需要改进
321 improvement_need = 1.0 - min(1.0, value)
322 else:
323 improvement_need = min(1.0, value)
324
325 score += improvement_need * weight
326 total_weight += weight
327
328 if total_weight > 0:
329 score /= total_weight
330
331 return score

352 def _identify_main_issue(self, metrics: Dict[str, float]) -> str:
353 """识别层的主要问题"""
354
355 issues = []
356
357 # 梯度问题
358 if metrics.get('gradient_health', 1.0) < 0.3:
359 if metrics.get('gradient_norm', 0) < 1e-6:
360 issues.append("梯度消失")
361 elif metrics.get('gradient_norm', 0) > 100:
362 issues.append("梯度爆炸")
363 else:
364 issues.append("梯度不稳定")
365
366 # 特征表示问题
367 if metrics.get('feature_separability', 1.0) < 0.5:
368 issues.append("特征分离度低")
369
370 # 表示效率问题
371 if metrics.get('representation_efficiency', 1.0) < 0.3:
372 issues.append("表示效率低")
373
374 # 信息流问题
375 if metrics.get('information_flow', 1.0) < 0.1:
376 issues.append("信息流受阻")
377
378 # 学习效率问题
379 if metrics.get('learning_rate', 1.0) < 0.2:
380 issues.append("学习效率低")
381
382 if not issues:
383 return "性能良好"
384
385 return " + ".join(issues[:2]) # 最多显示两个主要问题

387class SmartLayerSelector:
388 """智能层选择器"""

390 def __init__(self, analyzer: LayerPerformanceAnalyzer):
391 self.analyzer = analyzer
392 self.selection_history = deque(maxlen=20)

394 def select_optimal_division_layers(self, layer_analysis: Dict[str, Dict[str, float]],
395 max_selections: int = 2) -> List[Tuple[str, float, str]]:
396 """智能选择最优分裂层"""
397
398 # 获取推荐层
399 recommendations = self.analyzer.recommend_optimal_layers(layer_analysis, top_k=10)
400
401 # 过滤掉最近已经处理过的层
402 filtered_recommendations = []
403 recent_layers = set(h['layer'] for h in list(self.selection_history)[-5:])
404
405 for layer_name, score, issue in recommendations:
406 # 避免过度分裂同一层
407 if layer_name not in recent_layers or score > 0.8: # 高分数可以重复处理
408 filtered_recommendations.append((layer_name, score, issue))
409
410 # 选择多样化的层
411 selected = self._select_diverse_layers(filtered_recommendations, max_selections)
412
413 # 记录选择历史
414 for layer_name, score, issue in selected:
415 self.selection_history.append({
416 'layer': layer_name,
417 'score': score,
418 'issue': issue,
419 'epoch': len(self.selection_history)
420 })
421
422 return selected

424 def _select_diverse_layers(self, recommendations: List[Tuple[str, float, str]],
425 max_selections: int) -> List[Tuple[str, float, str]]:
426 """选择多样化的层（避免都在同一个模块）"""
427
428 if len(recommendations) <= max_selections:
429 return recommendations
430
431 selected = []
432 used_modules = set()
433
434 # 优先选择不同模块的层
435 for layer_name, score, issue in recommendations:
436 if len(selected) >= max_selections:
437 break
438
439 # 提取模块名（如 'features', 'classifier'）
440 module_name = layer_name.split('.')[0] if '.' in layer_name else layer_name
441
442 if module_name not in used_modules or len(selected) == 0:
443 selected.append((layer_name, score, issue))
444 used_modules.add(module_name)
445
446 # 如果还没选够，继续选择高分数的层
447 while len(selected) < max_selections and len(selected) < len(recommendations):
448 for layer_name, score, issue in recommendations:
449 if (layer_name, score, issue) not in selected:
450 selected.append((layer_name, score, issue))
451 break
452
453 return selected
