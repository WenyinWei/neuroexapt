164 def _estimate_flops(self, model: nn.Module, input_tensor: torch.Tensor) -> float:
165 """ä¼°ç®—æ¨¡å‹çš„FLOPS"""

168 def flop_count_hook(module, input, output):
169 nonlocal total_flops

171 if isinstance(module, nn.Conv2d):
172 # å·ç§¯å±‚FLOPS
173 batch_size = output.shape[0]
174 output_height, output_width = output.shape[2], output.shape[3]
175 kernel_flops = module.kernel_size[0] * module.kernel_size[1]
176 output_elements = batch_size * output_height * output_width
177 filters = module.out_channels
178 flops = output_elements * kernel_flops * filters * module.in_channels
179 total_flops += flops
180
181 elif isinstance(module, nn.Linear):
182 # å…¨è¿æ¥å±‚FLOPS
183 batch_size = input[0].shape[0]
184 flops = batch_size * module.in_features * module.out_features
185 total_flops += flops

187 # æ³¨å†Œhooks
188 hooks = []
189 for module in model.modules():
190 if isinstance(module, (nn.Conv2d, nn.Linear)):
191 hook = module.register_forward_hook(flop_count_hook)

195 with torch.no_grad():
196 _ = model(input_tensor)

198 # æ¸…ç†hooks

348 def non_dominated_sort(self, population_fitness: List[FitnessMetrics]) -> List[List[int]]:
349 """
350 éæ”¯é…æ’åº
351
352 Returns:
353 æ¯ä¸ªå‰æ²¿çš„ä¸ªä½“ç´¢å¼•åˆ—è¡¨

358 fronts = [[]]

360 # è®¡ç®—æ”¯é…å…³ç³»

365 # iæ”¯é…j

390 def calculate_crowding_distance(self, front: List[int],
391 population_fitness: List[FitnessMetrics]) -> List[float]:
392 """
393 è®¡ç®—æ‹¥æŒ¤è·ç¦»

399 Returns:
400 æ¯ä¸ªä¸ªä½“çš„æ‹¥æŒ¤è·ç¦»
401 """
402 distances = [0.0] * len(front)

404 if len(front) <= 2:
405 return [float('inf')] * len(front)

407 for obj in self.objectives:
408 # æŒ‰ç›®æ ‡å€¼æ’åº
409 front_sorted = sorted(front, key=lambda x: getattr(population_fitness[x], obj))

411 # è¾¹ç•Œä¸ªä½“è®¾ç½®ä¸ºæ— ç©·å¤§
412 idx_min = front.index(front_sorted[0])

417 # è®¡ç®—ç›®æ ‡å€¼èŒƒå›´

423 # è®¡ç®—ä¸­é—´ä¸ªä½“çš„æ‹¥æŒ¤è·ç¦»

427 obj_next = getattr(population_fitness[front_sorted[i+1]], obj)

595 child = self._crossover(parent1, parent2)

607 def _tournament_selection(self,
608 population: List[nn.Module],
609 fitness: List[FitnessMetrics],
610 fronts: List[List[int]],
611 tournament_size: int = 3) -> nn.Module:
612 """é”¦æ ‡èµ›é€‰æ‹©"""
613 candidates = random.sample(range(len(population)), min(tournament_size, len(population)))
614
615 # æ‰¾åˆ°æœ€å¥½çš„å€™é€‰è€…ï¼ˆåŸºäºå‰æ²¿æ’åå’Œæ‹¥æŒ¤è·ç¦»ï¼‰
616 best_candidate = candidates[0]
617 best_front_rank = float('inf')
618
619 for front_idx, front in enumerate(fronts):
620 for candidate in candidates:
621 if candidate in front:
622 if front_idx < best_front_rank:
623 best_candidate = candidate
624 best_front_rank = front_idx
625 elif front_idx == best_front_rank:
626 # åŒä¸€å‰æ²¿ï¼Œæ¯”è¾ƒæ‹¥æŒ¤è·ç¦»
627 distances = self.optimizer.calculate_crowding_distance(front, fitness)
628 candidate_distance = distances[front.index(candidate)]
629 best_distance = distances[front.index(best_candidate)]
630 if candidate_distance > best_distance:
631 best_candidate = candidate
632
633 return copy.deepcopy(population[best_candidate])

643 # ä½¿ç”¨parent2çš„å‚æ•°
644 child_param = dict(child.named_parameters())[name1]
645 child_param.data = param2.data.clone()

780 def _calculate_diversity_score(self, fitness_list: List[FitnessMetrics]) -> float:
781 """è®¡ç®—ç§ç¾¤å¤šæ ·æ€§è¯„åˆ†"""
782 if len(fitness_list) < 2:
783 return 0.0
784
785 # è®¡ç®—ç›®æ ‡ç©ºé—´ä¸­çš„å¹³å‡è·ç¦»
786 distances = []
787 objectives = ['accuracy', 'efficiency', 'complexity']
788
789 for i in range(len(fitness_list)):
790 for j in range(i + 1, len(fitness_list)):
791 f1, f2 = fitness_list[i], fitness_list[j]
792
793 # è®¡ç®—æ¬§æ°è·ç¦»ï¼ˆæ ‡å‡†åŒ–åï¼‰
794 dist = 0.0
795 for obj in objectives:
796 val1 = getattr(f1, obj)
797 val2 = getattr(f2, obj)
798
799 # ç®€å•æ ‡å‡†åŒ–
800 if obj == 'accuracy':
801 norm_diff = abs(val1 - val2) / 100.0
802 elif obj == 'efficiency':
803 norm_diff = abs(val1 - val2) / max(val1 + val2, 1.0)
804 else: # complexity
805 norm_diff = abs(val1 - val2) / max(val1 + val2, 1.0)
806
807 dist += norm_diff ** 2
808
809 distances.append(math.sqrt(dist))
810
811 return np.mean(distances) if distances else 0.0
