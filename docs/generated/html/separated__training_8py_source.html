24class SeparatedTrainingStrategy:
25 """
26 分离训练策略类
27
28 核心思想：
29 - 主要时间训练网络权重（固定架构）→ 快速收敛
30 - 偶尔训练架构参数（指导演化方向）→ 渐进优化
31 """

54 print(f" 预热轮次: {warmup_epochs}")

57 def _create_schedule(self) -> List[str]:
58 """创建训练计划"""
59 schedule = []
60
61 # 预热阶段：只训练权重
62 for i in range(self.warmup_epochs):
63 schedule.append('weight')
64
65 # 主训练阶段：交替训练
66 remaining_epochs = self.total_epochs - self.warmup_epochs
67 cycle_length = self.weight_training_epochs + self.arch_training_epochs
68
69 current_epoch = self.warmup_epochs
70 while current_epoch < self.total_epochs:
71 # 权重训练阶段
72 for i in range(self.weight_training_epochs):
73 if current_epoch >= self.total_epochs:
74 break
75 schedule.append('weight')
76 current_epoch += 1
77
78 # 架构训练阶段
79 for i in range(self.arch_training_epochs):
80 if current_epoch >= self.total_epochs:
81 break
82 schedule.append('arch')
83 current_epoch += 1
84
85 return schedule

107class SeparatedOptimizer:
108 """
109 分离优化器
110
111 分别管理网络权重和架构参数的优化器
112 """
113
114 def __init__(self, model: nn.Module, weight_lr: float = 0.025, arch_lr: float = 3e-4,
115 weight_momentum: float = 0.9, weight_decay: float = 3e-4):
116
117 self.model = model
118
119 # 分离参数
120 self.weight_params = []
121 self.arch_params = []
122
123 # 获取架构参数
124 if hasattr(model, 'arch_parameters'):
125 self.arch_params = list(model.arch_parameters())
126
127 # 获取网络权重参数（排除架构参数）
128 arch_param_ids = {id(p) for p in self.arch_params}
129 for param in model.parameters():
130 if id(param) not in arch_param_ids:
131 self.weight_params.append(param)
132
133 # 创建分离的优化器
134 self.weight_optimizer = optim.SGD(
135 self.weight_params,
136 lr=weight_lr,
137 momentum=weight_momentum,
138 weight_decay=weight_decay
139 )
140
141 self.arch_optimizer = optim.Adam(
142 self.arch_params,
143 lr=arch_lr,
144 weight_decay=1e-3
145 )
146
147 # 统计信息
148 print(f"📊 参数统计:")
149 print(f" 网络权重参数: {sum(p.numel() for p in self.weight_params):,}")
150 print(f" 架构参数: {sum(p.numel() for p in self.arch_params):,}")
151
152 weight_params_count = sum(p.numel() for p in self.weight_params)
153 arch_params_count = sum(p.numel() for p in self.arch_params)
154 total_params = weight_params_count + arch_params_count
155
156 if total_params > 0:
157 arch_ratio = arch_params_count / total_params * 100
158 print(f" 架构参数占比: {arch_ratio:.2f}%")

196 def get_lr_schedulers(self):
197 """获取学习率调度器"""

227 def train_epoch_weights(self, train_loader, epoch: int) -> Dict[str, float]:
228 """训练网络权重（固定架构参数）"""
229 start_time = time.time()
230
231 # 冻结架构参数，解冻权重参数
232 self.optimizer.freeze_arch_params()
233 self.optimizer.unfreeze_weight_params()
234
235 self.model.train()
236
237 total_loss = 0.0
238 total_samples = 0
239 correct = 0
240
241 for batch_idx, (data, target) in enumerate(train_loader):
242 data, target = data.cuda(), target.cuda()
243
244 self.optimizer.zero_grad_weights()
245 output = self.model(data)
246 loss = self.criterion(output, target)
247 loss.backward()
248
249 # 梯度裁剪
250 torch.nn.utils.clip_grad_norm_(self.optimizer.weight_params, 5.0)
251
252 self.optimizer.step_weights()
253
254 # 统计
255 total_loss += loss.item() * data.size(0)
256 total_samples += data.size(0)
257 pred = output.argmax(dim=1, keepdim=True)
258 correct += pred.eq(target.view_as(pred)).sum().item()
259
260 epoch_time = time.time() - start_time
261 self.strategy.weight_training_time += epoch_time
262
263 avg_loss = total_loss / total_samples
264 accuracy = 100.0 * correct / total_samples
265
266 return {
267 'loss': avg_loss,
268 'accuracy': accuracy,
269 'time': epoch_time
270 }

272 def train_epoch_architecture(self, train_loader, valid_loader, epoch: int) -> Dict[str, float]:
273 """训练架构参数（固定网络权重）"""

275

277 self.optimizer.freeze_weight_params()
278 self.optimizer.unfreeze_arch_params()
279
280 self.model.train()
281
282 total_loss = 0.0
283 total_samples = 0

285
286 valid_iter = iter(valid_loader)
287
288 for batch_idx, (data, target) in enumerate(train_loader):
289 # 获取验证数据用于架构搜索
290 try:

292 except StopIteration:
293 valid_iter = iter(valid_loader)
294 valid_data, valid_target = next(valid_iter)
295
296 data, target = data.cuda(), target.cuda()
297 valid_data, valid_target = valid_data.cuda(), valid_target.cuda()
298
299 # 架构参数梯度更新
300 self.optimizer.zero_grad_arch()
301
302 # 在验证数据上评估当前架构
303 valid_output = self.model(valid_data)
304 arch_loss = self.criterion(valid_output, valid_target)
305
306 arch_loss.backward()
307 self.optimizer.step_arch()
308
309 # 统计
310 total_loss += arch_loss.item() * valid_data.size(0)
311 total_samples += valid_data.size(0)
312 arch_updates += 1
313
314 # 定期输出架构搜索进度
315 if batch_idx % 50 == 0:
316 print(f" 架构搜索步骤 {batch_idx}: 损失={arch_loss.item():.4f}")
317
318 epoch_time = time.time() - start_time
319 self.strategy.arch_training_time += epoch_time
320
321 avg_loss = total_loss / total_samples
322
323 return {
324 'loss': avg_loss,
325 'arch_updates': float(arch_updates),
326 'time': epoch_time

329 def train_epoch(self, train_loader, valid_loader, epoch: int) -> Dict[str, float]:
330 """根据策略训练一个epoch"""
331
332 # 更新学习率
333 if self.strategy.should_train_weights(epoch):
334 self.weight_scheduler.step()
335 else:
336 self.arch_scheduler.step()
337
338 # 根据训练模式选择训练方法
339 if self.strategy.should_train_weights(epoch):
340 print(f"🏋️ Epoch {epoch}: 训练网络权重参数")
341 stats = self.train_epoch_weights(train_loader, epoch)
342 else:
343 print(f"🧬 Epoch {epoch}: 训练架构参数")
344 stats = self.train_epoch_architecture(train_loader, valid_loader, epoch)
345
346 # 记录统计信息
347 self.epoch_stats[epoch] = stats
348
349 return stats
