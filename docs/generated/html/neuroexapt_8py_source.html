137 def _calculate_objective(
138 self,
139 predictions: torch.Tensor,
140 targets: torch.Tensor
141 ) -> torch.Tensor:
142 """
143 Calculates the full objective function as defined in the theoretical framework.
144 L = L_task + lambda_1 * H(layers) + lambda_2 * D_KL(posterior || prior)
145 """
146 # 1. Task Loss (e.g., Cross-Entropy)
147 task_loss = self.criterion(predictions, targets)
148
149 # 2. Entropy Penalty (Redundancy Penalty)
150 # This requires calculating the average entropy of the network.
151 # Note: This is an approximation. A more rigorous approach would sum the
152 # entropy of each layer's activation distribution.
153 avg_entropy = self.evolution_engine.entropy_metrics.get_average_entropy()
154 entropy_penalty = self.lambda_entropy * (1.0 - avg_entropy) # Penalize low entropy
155
156 # 3. Bayesian Regularization (KL Divergence)
157 # This term is typically calculated within Bayesian layers. If the model
158 # uses them, we would sum the KL terms here. For now, this is a placeholder.
159 bayesian_loss = torch.tensor(0.0, device=self.device)
160 if hasattr(self.model, 'calculate_kl_divergence'):
161 # The attribute should be a tensor, not a method to call
162 kl_div = getattr(self.model, 'calculate_kl_divergence')
163 if isinstance(kl_div, torch.Tensor):
164 bayesian_loss = kl_div
165
166 bayesian_penalty = self.lambda_bayesian * bayesian_loss
167
168 total_loss = task_loss + entropy_penalty + bayesian_penalty
169 return total_loss

171 def train_epoch(self, optimizer: torch.optim.Optimizer):
172 """
173 Runs a single epoch of training, including the calculation of the
174 full NeuroExapt objective function.
175 """
176 self.model.train()
177 total_loss = 0
178 correct = 0
179 total = 0
180
181 # Register hooks to capture activations for entropy calculation
182 self.evolution_engine.entropy_metrics.register_hooks()
183
184 try:
185 for data, targets in self.dataloader:
186 data, targets = data.to(self.device), targets.to(self.device)
187
188 optimizer.zero_grad(set_to_none=True)
189
190 with autocast(enabled=self.use_amp):
191 predictions = self.model(data)
192 loss = self._calculate_objective(predictions, targets)
193
194 if self.use_amp:
195 self.scaler.scale(loss).backward()
196 self.scaler.step(optimizer)
197 self.scaler.update()
198 else:
199 loss.backward()
200 optimizer.step()
201
202 total_loss += loss.item()
203 _, predicted = torch.max(predictions.data, 1)
204 total += targets.size(0)
205 correct += (predicted == targets).sum().item()
206
207 except Exception as e:
208 print(f"Warning: Training error: {e}")
209 finally:
210 # Always cleanup hooks after epoch
211 self.evolution_engine.entropy_metrics.remove_hooks()
212
213 accuracy = 100 * correct / total if total > 0 else 0.0
214 avg_loss = total_loss / len(self.dataloader) if len(self.dataloader) > 0 else 0.0
215 print(f"Epoch {self.current_epoch} Training: Loss={avg_loss:.4f}, Accuracy={accuracy:.2f}%")

217 def evolve(self, performance_metrics: Dict[str, Any], optimizer: torch.optim.Optimizer):
218 """
219 Triggers one step of architectural evolution with enhanced error handling.
220 """
221 try:
222 # Dynamically resolve interface to avoid static attribute errors
223 step_fn = getattr(self.evolution_engine, "step", None)
224 evolve_fn = getattr(self.evolution_engine, "evolve", None)
225
226 if callable(step_fn):
227 result = step_fn(
228 epoch=self.current_epoch,
229 performance_metrics=performance_metrics,
230 dataloader=self.dataloader,
231 criterion=self.criterion
232 )
233 elif callable(evolve_fn):
234 result = evolve_fn(
235 epoch=self.current_epoch,
236 dataloader=self.dataloader,
237 criterion=self.criterion,
238 performance_metrics=performance_metrics
239 )
240 else:
241 raise RuntimeError("Evolution engine does not expose compatible step/evolve method.")
242
243 # Normalize result to tuple (model, action_taken)
244 if isinstance(result, tuple) and len(result) == 2:
245 new_model, action_taken = result # type: ignore
246 else:
247 new_model, action_taken = result, None
248
249 if action_taken:
250 self.model = new_model
251 self.stats['evolutions'] += 1
252 self.stats['successful_evolutions'] += 1
253
254 # CRITICAL: Reset optimizer state after architecture changes
255 print("Resetting optimizer state for new architecture...")
256 optimizer.state.clear()
257
258 # Remove old parameters and add new ones
259 optimizer.param_groups.clear()
260 optimizer.add_param_group({'params': list(self.model.parameters())})
261
262 # Reset cooldown/patience depending on engine implementation
263 reset_patience = getattr(self.evolution_engine, "reset_patience", None)
264 reset_cooldown = getattr(self.evolution_engine, "reset_cooldown", None)
265 if callable(reset_patience):
266 reset_patience()
267 elif callable(reset_cooldown):
268 reset_cooldown()
269
270 return action_taken
271
272 except Exception as e:
273 print(f"Evolution error: {e}")
274 self.stats['validation_failures'] += 1
275 return None
