23class AdaptiveCNN(nn.Module):
24 """
25 A CNN designed to be evolved with intelligent operators.
26 """
27 def __init__(self, num_classes=10):
28 super().__init__()
29
30 # Initial simple architecture
31 self.features = nn.Sequential(
32 # First block - will be analyzed for complexity
33 nn.Conv2d(3, 32, kernel_size=3, padding=1),
34 nn.BatchNorm2d(32),
35 nn.ReLU(inplace=True),
36
37 # Second block - candidate for intelligent expansion
38 nn.Conv2d(32, 64, kernel_size=3, padding=1),
39 nn.BatchNorm2d(64),
40 nn.ReLU(inplace=True),
41
42 # Third block - may be replaced with specialized layers
43 nn.Conv2d(64, 128, kernel_size=3, padding=1),
44 nn.BatchNorm2d(128),
45 nn.ReLU(inplace=True),
46 )
47
48 # Global pooling to handle variable sizes
49 self.avgpool = nn.AdaptiveAvgPool2d((4, 4))
50
51 # Classifier
52 self.classifier = nn.Sequential(
53 nn.Linear(128 * 4 * 4, 256),
54 nn.ReLU(inplace=True),
55 nn.Dropout(0.5),
56 nn.Linear(256, num_classes)
57 )
58
59 def forward(self, x):
60 x = self.features(x)
61 x = self.avgpool(x)
62 x = torch.flatten(x, 1)
63 x = self.classifier(x)
64 return x

67def visualize_evolution_process(neuroexapt, save_path="evolution_analysis.png"):
68 """
69 Visualize the intelligent evolution process.
70 """
71 fig, axes = plt.subplots(2, 2, figsize=(15, 12))

73 # 1. Layer type distribution over evolution
74 ax = axes[0, 0]
75 if hasattr(neuroexapt, 'layer_type_history'):
76 layer_types = neuroexapt.layer_type_history
77 type_counts = {}
78 for epoch_types in layer_types:
79 for layer_type in epoch_types:
80 type_counts[layer_type] = type_counts.get(layer_type, 0) + 1
81
82 ax.bar(type_counts.keys(), type_counts.values())
83 ax.set_title('Layer Types Added During Evolution')
84 ax.set_xlabel('Layer Type')
85 ax.set_ylabel('Count')

88 ax = axes[0, 1]

90 epochs = list(range(len(neuroexapt.metrics_history)))
91 entropies = [m['current_entropy'] for m in neuroexapt.metrics_history]
92 redundancies = [m.get('redundancy', 0) for m in neuroexapt.metrics_history]
93
94 ax.plot(epochs, entropies, label='Entropy', marker='o')
95 ax.plot(epochs, redundancies, label='Redundancy', marker='s')
96 ax.set_title('Information Metrics Evolution')
97 ax.set_xlabel('Evolution Step')
98 ax.set_ylabel('Metric Value')
99 ax.legend()
100
101 # 3. Layer complexity analysis
102 ax = axes[1, 0]
103 if hasattr(neuroexapt, 'layer_characteristics'):
104 chars = neuroexapt.layer_characteristics
105 if chars:
106 layers = list(chars.keys())[:10] # First 10 layers
107 complexities = [chars[l].get('spatial_complexity', 0) for l in layers]
108 redundancies = [chars[l].get('channel_redundancy', 0) for l in layers]
109
110 x = np.arange(len(layers))
111 width = 0.35
112 ax.bar(x - width/2, complexities, width, label='Spatial Complexity')
113 ax.bar(x + width/2, redundancies, width, label='Channel Redundancy')
114 ax.set_xlabel('Layer')
115 ax.set_ylabel('Metric Value')
116 ax.set_title('Layer Characteristics')
117 ax.set_xticks(x)
118 ax.set_xticklabels([l.split('.')[-1] for l in layers], rotation=45)
119 ax.legend()
120
121 # 4. Evolution actions timeline
122 ax = axes[1, 1]
123 if neuroexapt.evolution_history:
124 actions = [step.action for step in neuroexapt.evolution_history]
125 action_types = list(set(actions))
126 action_colors = plt.cm.tab10(np.linspace(0, 1, len(action_types)))
127
128 for i, action in enumerate(actions):
129 color_idx = action_types.index(action)
130 ax.scatter(i, action_types.index(action),
131 color=action_colors[color_idx], s=100, alpha=0.7)
132
133 ax.set_yticks(range(len(action_types)))
134 ax.set_yticklabels(action_types)
135 ax.set_xlabel('Evolution Step')
136 ax.set_title('Evolution Actions Timeline')
137 ax.grid(True, alpha=0.3)
138
139 plt.tight_layout()
