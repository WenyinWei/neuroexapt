93 def _initialize_weights(self):
94 for m in self.modules():
95 if isinstance(m, nn.Conv2d):
96 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
97 elif isinstance(m, nn.BatchNorm2d):
98 nn.init.constant_(m.weight, 1)
99 nn.init.constant_(m.bias, 0)
100 elif isinstance(m, nn.Linear):
101 nn.init.normal_(m.weight, 0, 0.01)
102 if m.bias is not None:
103 nn.init.constant_(m.bias, 0)
104
105 def forward(self, x):
106 x = self.features(x)
107 x = x.view(x.size(0), -1)
108 x = self.classifier(x)
109 return x

111class ActivationHook:
112 """激活值捕获钩子"""
113
114 def __init__(self):
115 self.activations = {}
116 self.hooks = []
117
118 def hook_fn(self, name):
119 def fn(module, input, output):
120 self.activations[name] = output.detach()
121 return fn
122
123 def register_hooks(self, model):
124 for name, module in model.named_modules():
125 if isinstance(module, (nn.Conv2d, nn.Linear)):
126 # 只为主要层注册钩子，避免过多的激活值
127 if ('classifier' in name and isinstance(module, nn.Linear)) or \
128 ('features' in name and isinstance(module, nn.Conv2d) and 'features.17' in name):
129 hook = module.register_forward_hook(self.hook_fn(name))
130 self.hooks.append(hook)
131
132 def remove_hooks(self):
133 for hook in self.hooks:
134 hook.remove()
135 self.hooks.clear()

137class GradientHook:
138 """梯度捕获钩子"""
139
140 def __init__(self):
141 self.gradients = {}
142
143 def capture_gradients(self, model):
144 self.gradients.clear()
145 for name, param in model.named_parameters():
146 if param.grad is not None:
147 self.gradients[name] = param.grad.detach().clone()

176def train_epoch(model, train_loader, optimizer, criterion, device, activation_hook, gradient_hook):
177 """训练一个epoch"""
178 model.train()
179 running_loss = 0.0
180 correct = 0
181 total = 0
182
183 # 保存最后一个batch的target用于分析
184 last_targets = None
185
186 pbar = tqdm(train_loader, desc="Training")
187 for batch_idx, (data, target) in enumerate(pbar):
188 data, target = data.to(device), target.to(device)
189 last_targets = target # 保存用于后续分析
190
191 optimizer.zero_grad()
192 output = model(data)
193 loss = criterion(output, target)
194 loss.backward()
195
196 # 捕获梯度
197 gradient_hook.capture_gradients(model)
198
199 # 梯度裁剪
200 torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
201
202 optimizer.step()
203
204 running_loss += loss.item()
205 _, predicted = output.max(1)
206 total += target.size(0)
207 correct += predicted.eq(target).sum().item()
208
209 if batch_idx % 50 == 0:
210 pbar.set_postfix({
211 'Loss': f'{running_loss/(batch_idx+1):.4f}',
212 'Acc': f'{100.*correct/total:.2f}%'
213 })
214
215 # 将最后一个batch的targets存储到gradient_hook中供后续使用
216 gradient_hook.last_targets = last_targets
217
218 return running_loss / len(train_loader), 100. * correct / total

220def validate_epoch(model, val_loader, criterion, device):
221 """验证一个epoch"""
222 model.eval()
223 val_loss = 0.0
224 correct = 0
225 total = 0
226
227 with torch.no_grad():
228 for data, target in val_loader:
229 data, target = data.to(device), target.to(device)
230 output = model(data)
231 val_loss += criterion(output, target).item()
232
233 _, predicted = output.max(1)
234 total += target.size(0)
235 correct += predicted.eq(target).sum().item()
236
237 return val_loss / len(val_loader), 100. * correct / total
