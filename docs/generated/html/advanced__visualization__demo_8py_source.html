15class MultiLevelBranchNet(nn.Module):
16 """
17 Complex architecture with multiple levels of branching:
18 - First level: 3 main branches
19 - Second level: Each branch splits into sub-branches
20 - Fusion at multiple points
21 """
22
23 def __init__(self):
24 super().__init__()
25
26 # Primary branches (Level 1)
27 self.main_branch = nn.Sequential(
28 nn.Conv2d(3, 64, 3, padding=1),
29 nn.BatchNorm2d(64),
30 nn.ReLU(),
31 nn.Conv2d(64, 128, 3, padding=1),
32 nn.BatchNorm2d(128),
33 nn.ReLU()
34 )
35
36 self.secondary_branch = nn.Sequential(
37 nn.Conv2d(3, 32, 1), # 1x1 conv for efficiency
38 nn.BatchNorm2d(32),
39 nn.ReLU(),
40 nn.Conv2d(32, 64, 3, padding=1),
41 nn.BatchNorm2d(64),
42 nn.ReLU()
43 )
44
45 self.attention_branch = nn.Sequential(
46 nn.Conv2d(3, 16, 1),
47 nn.BatchNorm2d(16),
48 nn.ReLU(),
49 nn.Conv2d(16, 32, 1),
50 nn.BatchNorm2d(32),
51 nn.Sigmoid() # Attention weights
52 )
53
54 # Sub-branches from main_branch (Level 2)
55 self.main_branch_deep = nn.Sequential(
56 nn.Conv2d(128, 256, 3, padding=1),
57 nn.BatchNorm2d(256),
58 nn.ReLU(),
59 nn.Conv2d(256, 256, 3, padding=1),
60 nn.BatchNorm2d(256),
61 nn.ReLU()
62 )
63
64 self.main_branch_shallow = nn.Sequential(
65 nn.Conv2d(128, 128, 1), # Skip connection style
66 nn.BatchNorm2d(128),
67 nn.ReLU()
68 )
69
70 # First fusion: combine main sub-branches
71 self.main_fusion = nn.Conv2d(256 + 128, 256, 1)
72
73 # Second fusion: combine all branches
74 self.fusion = nn.Conv2d(256 + 64 + 32, 384, 1)
75
76 # Final processing
77 self.final_conv = nn.Conv2d(384, 512, 3, padding=1)
78 self.global_pool = nn.AdaptiveAvgPool2d(1)
79 self.classifier = nn.Sequential(
80 nn.Linear(512, 256),
81 nn.ReLU(),
82 nn.Dropout(0.5),
83 nn.Linear(256, 10)
84 )
85
86 def forward(self, x):
87 # Level 1 branches
88 main_out = self.main_branch(x)
89 secondary_out = self.secondary_branch(x)
90 attention_out = self.attention_branch(x)
91
92 # Level 2 branches from main
93 main_deep = self.main_branch_deep(main_out)
94 main_shallow = self.main_branch_shallow(main_out)
95
96 # First fusion
97 main_fused = self.main_fusion(torch.cat([main_deep, main_shallow], dim=1))
98
99 # Apply attention
100 secondary_out = secondary_out * attention_out
101
102 # Final fusion
103 fused = torch.cat([main_fused, secondary_out, attention_out], dim=1)
104 fused = self.fusion(fused)
105
106 # Final layers
107 out = self.final_conv(fused)
108 out = self.global_pool(out)
109 out = out.view(out.size(0), -1)
110 out = self.classifier(out)
111
112 return out

115class DynamicMultiLevelNet(nn.Module):
116 """
117 Even more complex architecture with dynamic paths and conditional execution.
118 """
119
120 def __init__(self, num_classes=10):
121 super().__init__()
122
123 # Entry branches
124 self.branch_visual = nn.Sequential(
125 nn.Conv2d(3, 48, 3, padding=1),
126 nn.BatchNorm2d(48),
127 nn.ReLU(),
128 nn.MaxPool2d(2)
129 )
130
131 self.branch_texture = nn.Sequential(
132 nn.Conv2d(3, 32, 5, padding=2),
133 nn.BatchNorm2d(32),
134 nn.ReLU(),
135 nn.MaxPool2d(2)
136 )
137
138 self.branch_edge = nn.Sequential(
139 nn.Conv2d(3, 16, 7, padding=3),
140 nn.BatchNorm2d(16),
141 nn.ReLU(),
142 nn.MaxPool2d(2)
143 )
144
145 # Mid-level branches (after first fusion)
146 self.fusion_low = nn.Conv2d(48 + 32 + 16, 96, 1)
147
148 # High-level branches
149 self.branch_semantic = nn.Sequential(
150 nn.Conv2d(96, 128, 3, padding=1),
151 nn.BatchNorm2d(128),
152 nn.ReLU(),
153 nn.Conv2d(128, 256, 3, padding=1),
154 nn.BatchNorm2d(256),
155 nn.ReLU()
156 )
157
158 self.branch_context = nn.Sequential(
159 nn.Conv2d(96, 64, 5, padding=2),
160 nn.BatchNorm2d(64),
161 nn.ReLU(),
162 nn.Conv2d(64, 128, 5, padding=2),
163 nn.BatchNorm2d(128),
164 nn.ReLU()
165 )
166
167 # Final fusion and classification
168 self.fusion_high = nn.Conv2d(256 + 128, 512, 1)
169 self.final_pool = nn.AdaptiveAvgPool2d(1)
170 self.classifier = nn.Linear(512, num_classes)
171
172 def forward(self, x):
173 # Low-level feature extraction
174 visual = self.branch_visual(x)
175 texture = self.branch_texture(x)
176 edge = self.branch_edge(x)
177
178 # First fusion
179 low_fused = torch.cat([visual, texture, edge], dim=1)
180 low_fused = self.fusion_low(low_fused)
181
182 # High-level feature extraction
183 semantic = self.branch_semantic(low_fused)
184 context = self.branch_context(low_fused)
185
186 # Final fusion
187 high_fused = torch.cat([semantic, context], dim=1)
188 high_fused = self.fusion_high(high_fused)
189
190 # Classification
191 out = self.final_pool(high_fused)
192 out = out.view(out.size(0), -1)
193 out = self.classifier(out)
194
195 return out
