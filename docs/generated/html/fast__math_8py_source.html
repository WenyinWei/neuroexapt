27class FastEntropy:
28 """高性能熵计算器"""
29
30 @staticmethod
31 @torch.jit.script
32 def entropy_jit(x: torch.Tensor, dim: int = -1, eps: float = 1e-8) -> torch.Tensor:
33 """JIT编译的熵计算"""
34 # 使用log_softmax获得数值稳定性
35 log_probs = F.log_softmax(x, dim=dim)
36 probs = torch.exp(log_probs)
37 return -torch.sum(probs * log_probs, dim=dim)
38
39 @staticmethod
40 def batch_entropy(logits_batch: torch.Tensor, dim: int = -1) -> torch.Tensor:
41 """批量熵计算，适用于架构参数"""
42 # 形状: [batch_size, num_layers, num_ops] -> [batch_size, num_layers]
43 return FastEntropy.entropy_jit(logits_batch, dim=dim)
44
45 @staticmethod
46 def mutual_information_fast(x: torch.Tensor, y: torch.Tensor,
47 bins: int = 50) -> torch.Tensor:
48 """快速互信息估计"""
49 # 使用直方图方法进行快速MI估计
50 device = x.device
51
52 # 归一化到[0, bins-1]
53 x_norm = ((x - x.min()) / (x.max() - x.min() + 1e-8) * (bins - 1)).long()
54 y_norm = ((y - y.min()) / (y.max() - y.min() + 1e-8) * (bins - 1)).long()
55
56 # 计算联合直方图
57 joint_hist = torch.zeros(bins, bins, device=device)
58 joint_hist.scatter_add_(0, x_norm.unsqueeze(0).expand(bins, -1),
59 F.one_hot(y_norm, bins).float().t())
60
61 # 归一化为概率
62 joint_prob = joint_hist / joint_hist.sum()
63 marginal_x = joint_prob.sum(dim=1)
64 marginal_y = joint_prob.sum(dim=0)
65
66 # 计算MI
67 mi = 0.0
68 for i in range(bins):
69 for j in range(bins):
70 if joint_prob[i, j] > 1e-8:
71 mi += joint_prob[i, j] * torch.log(joint_prob[i, j] /
72 (marginal_x[i] * marginal_y[j] + 1e-8))
73
74 return mi

76class FastGradients:
77 """高性能梯度计算"""
78
79 @staticmethod
80 def compute_gradient_norm_batch(parameters: List[torch.Tensor],
81 norm_type: float = 2.0) -> torch.Tensor:
82 """批量计算梯度范数"""
83 if len(parameters) == 0:
84 return torch.tensor(0.0)
85
86 device = parameters[0].device
87 if norm_type == float('inf'):
88 norms = [p.grad.abs().max() for p in parameters if p.grad is not None]
89 return torch.stack(norms).max() if norms else torch.tensor(0.0, device=device)
90 else:
91 squared_norms = [p.grad.norm(dtype=torch.float32) ** norm_type
92 for p in parameters if p.grad is not None]
93 if not squared_norms:
94 return torch.tensor(0.0, device=device)
95 return torch.stack(squared_norms).sum() ** (1.0 / norm_type)
96
97 @staticmethod
98 def gradient_similarity(params1: List[torch.Tensor],
99 params2: List[torch.Tensor]) -> torch.Tensor:
100 """计算两组参数梯度的相似性"""
101 cos_sims = []
102 for p1, p2 in zip(params1, params2):
103 if p1.grad is not None and p2.grad is not None:
104 cos_sim = F.cosine_similarity(
105 p1.grad.flatten(), p2.grad.flatten(), dim=0
106 )
107 cos_sims.append(cos_sim)
108
109 return torch.stack(cos_sims).mean() if cos_sims else torch.tensor(0.0)
110
111 @staticmethod
112 def adaptive_gradient_clipping(parameters: List[torch.Tensor],
113 max_norm: float = 5.0,
114 adaptive_factor: float = 0.1) -> float:
115 """自适应梯度裁剪"""
116 current_norm = FastGradients.compute_gradient_norm_batch(parameters)
117
118 if current_norm > max_norm:
119 # 动态调整裁剪阈值
120 adaptive_max_norm = max_norm + adaptive_factor * (current_norm - max_norm)
121 clip_coef = adaptive_max_norm / (current_norm + 1e-8)
122
123 for p in parameters:
124 if p.grad is not None:
125 p.grad.mul_(clip_coef)
126
127 return clip_coef.item()
128
129 return 1.0

131class FastSimilarity:
132 """高性能相似度计算"""
133
134 @staticmethod
135 @torch.jit.script
136 def cosine_similarity_matrix(x: torch.Tensor, y: torch.Tensor) -> torch.Tensor:
137 """JIT优化的余弦相似度矩阵"""
138 # 归一化 - JIT需要明确的float类型
139 x_norm = F.normalize(x, p=2.0, dim=-1)
140 y_norm = F.normalize(y, p=2.0, dim=-1)
141
142 # 计算相似度矩阵
143 return torch.mm(x_norm, y_norm.t())
144
145 @staticmethod
146 def architecture_similarity(arch1: torch.Tensor, arch2: torch.Tensor,
147 method: str = 'cosine') -> torch.Tensor:
148 """架构相似度计算"""
149 if method == 'cosine':
150 return F.cosine_similarity(arch1.flatten(), arch2.flatten(), dim=0)
151 elif method == 'kl':
152 # KL散度（越小越相似）
153 p = F.softmax(arch1.flatten(), dim=0)
154 q = F.softmax(arch2.flatten(), dim=0)
155 return -torch.sum(p * torch.log(q / p + 1e-8))
156 elif method == 'l2':
157 return -torch.norm(arch1 - arch2, p=2.0)
158 else:
159 raise ValueError(f"Unknown similarity method: {method}")
160
161 @staticmethod
162 def batch_architecture_similarity(arch_batch1: torch.Tensor,
163 arch_batch2: torch.Tensor) -> torch.Tensor:
164 """批量架构相似度"""
165 # 形状: [batch_size, num_layers, num_ops]
166 batch_size = arch_batch1.size(0)
167 similarities = []
168
169 for i in range(batch_size):
170 sim = FastSimilarity.architecture_similarity(
171 arch_batch1[i], arch_batch2[i]
172 )
173 similarities.append(sim)
174
175 return torch.stack(similarities)

177class FastStatistics:
178 """高性能统计计算"""

180 @staticmethod
181 @torch.jit.script
182 def running_mean_var(x: torch.Tensor, mean: torch.Tensor, var: torch.Tensor,
183 momentum: float = 0.1) -> Tuple[torch.Tensor, torch.Tensor]:
184 """运行时均值和方差更新"""
185 batch_mean = x.mean(dim=0)
186 batch_var = x.var(dim=0, unbiased=False)
187
188 new_mean = (1 - momentum) * mean + momentum * batch_mean
189 new_var = (1 - momentum) * var + momentum * batch_var
190
191 return new_mean, new_var
192
193 @staticmethod
194 def architecture_diversity(arch_batch: torch.Tensor) -> torch.Tensor:
195 """计算架构批次的多样性"""
196 # 计算批次内架构的平均相似度（越低越多样化）
197 batch_size = arch_batch.size(0)
198 similarities = []
199
200 for i in range(batch_size):
201 for j in range(i + 1, batch_size):
202 sim = FastSimilarity.architecture_similarity(
203 arch_batch[i], arch_batch[j]
204 )
205 similarities.append(sim)
206
207 if similarities:
208 return 1.0 - torch.stack(similarities).mean() # 1 - 平均相似度 = 多样性
209 else:
210 return torch.tensor(1.0) # 单个架构认为是完全多样化

214 """计算架构权重的置信度"""
215 # 使用softmax后的最大值作为置信度

219class FastNumerical:
220 """高性能数值计算"""
221
222 @staticmethod
223 @torch.jit.script
224 def stable_softmax(x: torch.Tensor, dim: int = -1,
225 temperature: float = 1.0) -> torch.Tensor:
226 """数值稳定的softmax"""
227 x_scaled = x / temperature
228 x_max = x_scaled.max(dim=dim, keepdim=True)[0]
229 exp_x = torch.exp(x_scaled - x_max)
230 return exp_x / exp_x.sum(dim=dim, keepdim=True)
231
232 @staticmethod
233 @torch.jit.script
234 def gumbel_noise(shape: List[int], device: torch.device) -> torch.Tensor:
235 """高效Gumbel噪声生成"""
236 u = torch.rand(shape, device=device)
237 return -torch.log(-torch.log(u + 1e-8) + 1e-8)
238
239 @staticmethod
240 def batch_gumbel_softmax(logits: torch.Tensor, temperature: float = 1.0,
241 hard: bool = True) -> torch.Tensor:
242 """批量化Gumbel-Softmax"""
243 # 生成Gumbel噪声
244 gumbel_noise = FastNumerical.gumbel_noise(
245 list(logits.shape), logits.device
246 )
247
248 # 加入噪声
249 noisy_logits = (logits + gumbel_noise) / temperature
250
251 # Softmax
252 soft_samples = F.softmax(noisy_logits, dim=-1)
253
254 if hard:
255 # 硬采样
256 hard_samples = F.one_hot(
257 soft_samples.argmax(dim=-1), soft_samples.size(-1)
258 ).float()
259 # Straight-through estimator
260 return hard_samples - soft_samples.detach() + soft_samples
261
262 return soft_samples
263
264 @staticmethod
265 def differentiable_topk(x: torch.Tensor, k: int,
266 temperature: float = 1.0) -> torch.Tensor:
267 """可微的top-k选择"""
268 # 使用Gumbel trick实现可微的top-k
269 batch_size, seq_len = x.shape
270
271 # 添加Gumbel噪声
272 gumbel_noise = FastNumerical.gumbel_noise([batch_size, seq_len], x.device)
273 noisy_x = x + gumbel_noise
274
275 # 获取top-k indices
276 _, topk_indices = torch.topk(noisy_x, k, dim=-1)
277
278 # 创建mask
279 mask = torch.zeros_like(x)
280 mask.scatter_(-1, topk_indices, 1.0)
281
282 # 使用softmax进行平滑
283 weights = F.softmax(x / temperature, dim=-1)
284
285 return mask * weights

287class MemoryEfficientOperations:
288 """内存高效的操作"""
289
290 @staticmethod
291 def chunked_matmul(a: torch.Tensor, b: torch.Tensor,
292 chunk_size: int = 1024) -> torch.Tensor:
293 """分块矩阵乘法，节省内存"""
294 if a.size(0) <= chunk_size:
295 return torch.matmul(a, b)
296
297 results = []
298 for i in range(0, a.size(0), chunk_size):
299 chunk = a[i:i+chunk_size]
300 result_chunk = torch.matmul(chunk, b)
301 results.append(result_chunk)
302
303 return torch.cat(results, dim=0)
304
305 @staticmethod
306 def memory_efficient_attention(query: torch.Tensor, key: torch.Tensor,
307 value: torch.Tensor, chunk_size: int = 512) -> torch.Tensor:
308 """内存高效的注意力计算"""
309 batch_size, seq_len, d_model = query.shape
310
311 if seq_len <= chunk_size:
312 # 小序列直接计算
313 scores = torch.matmul(query, key.transpose(-2, -1)) / math.sqrt(d_model)
314 attn_weights = F.softmax(scores, dim=-1)
315 return torch.matmul(attn_weights, value)
316
317 # 大序列分块计算
318 output = torch.zeros_like(query)
319
320 for i in range(0, seq_len, chunk_size):
321 end_i = min(i + chunk_size, seq_len)
322 q_chunk = query[:, i:end_i]
323
324 # 计算当前chunk与所有key的attention
325 scores = torch.matmul(q_chunk, key.transpose(-2, -1)) / math.sqrt(d_model)
326 attn_weights = F.softmax(scores, dim=-1)
327 output_chunk = torch.matmul(attn_weights, value)
328
329 output[:, i:end_i] = output_chunk
330
331 return output

333class PerformanceProfiler:
334 """性能分析器"""
335
336 def __init__(self):
337 self.operation_times = {}
338 self.memory_usage = {}
339
340 def profile_function(self, func_name: str, func, *args, **kwargs):
341 """分析函数性能"""
342 if torch.cuda.is_available():
343 torch.cuda.synchronize()
344 start_memory = torch.cuda.memory_allocated()
345
346 import time
347 start_time = time.perf_counter()
348
349 result = func(*args, **kwargs)
350
351 if torch.cuda.is_available():
352 torch.cuda.synchronize()
353 end_memory = torch.cuda.memory_allocated()
354 memory_delta = end_memory - start_memory
355 else:
356 memory_delta = 0
357
358 end_time = time.perf_counter()
359 elapsed_time = end_time - start_time
360
361 # 记录统计
362 if func_name not in self.operation_times:
363 self.operation_times[func_name] = []
364 self.memory_usage[func_name] = []
365
366 self.operation_times[func_name].append(elapsed_time)
367 self.memory_usage[func_name].append(memory_delta)
368
369 return result
370
371 def get_report(self) -> dict:
372 """获取性能报告"""
373 report = {}
374
375 for func_name in self.operation_times:
376 times = self.operation_times[func_name]
377 memory = self.memory_usage[func_name]
378
379 report[func_name] = {
380 'avg_time': sum(times) / len(times),
381 'total_time': sum(times),
382 'call_count': len(times),
383 'avg_memory': sum(memory) / len(memory) if memory else 0,
384 'peak_memory': max(memory) if memory else 0
385 }
386
387 return report
388
389# 全局分析器实例
390_global_profiler = PerformanceProfiler()
391
392def profile_op(func_name: str):
393 """装饰器：自动分析操作性能"""
394 def decorator(func):
395 def wrapper(*args, **kwargs):
396 return _global_profiler.profile_function(func_name, func, *args, **kwargs)
397 return wrapper
398 return decorator
