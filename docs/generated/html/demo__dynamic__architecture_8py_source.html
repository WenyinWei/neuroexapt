255def train(train_queue, valid_queue, model, architect, criterion, optimizer, lr, args):
256 objs = AvgrageMeter()
257 top1 = AvgrageMeter()
258 top5 = AvgrageMeter()
259
260 # Create progress bar if available
261 if TQDM_AVAILABLE:
262 train_iter = tqdm(train_queue, desc="Training", leave=False)
263 else:
264 train_iter = train_queue
265
266 for step, (input, target) in enumerate(train_iter):
267 model.train()
268 n = input.size(0)
269
270 input = Variable(input, requires_grad=False).cuda()
271 target = Variable(target, requires_grad=False).cuda(non_blocking=True)
272
273 # get a random minibatch from the search queue with replacement
274 input_search, target_search = next(iter(valid_queue))
275 input_search = Variable(input_search, requires_grad=False).cuda()
276 target_search = Variable(target_search, requires_grad=False).cuda(non_blocking=True)
277
278 # Update architecture
279 architect.step(input, target, input_search, target_search, lr, optimizer, unrolled=args.unrolled)
280
281 # Update weights (with gradient accumulation)
282 if step % args.grad_accumulation_steps == 0:
283 optimizer.zero_grad()
284
285 logits = model(input)
286 loss = criterion(logits, target)
287
288 # Scale loss by accumulation steps
289 loss = loss / args.grad_accumulation_steps
290 loss.backward()
291
292 if (step + 1) % args.grad_accumulation_steps == 0:
293 nn.utils.clip_grad_norm_(model.parameters(), args.grad_clip)
294 optimizer.step()
295
296 prec1, prec5 = accuracy(logits, target, topk=(1, 5))
297 objs.update(loss.item(), n)
298 top1.update(prec1.item(), n)
299 top5.update(prec5.item(), n)
300
301 # Update progress bar if available
302 if TQDM_AVAILABLE:
303 train_iter.set_postfix({
304 'Loss': f'{objs.avg:.3f}',
305 'Top1': f'{top1.avg:.1f}%',
306 'Top5': f'{top5.avg:.1f}%'
307 })
308
309 if step % args.report_freq == 0:
310 logging.info('train %03d %e %f %f', step, objs.avg, top1.avg, top5.avg)
311
312 return top1.avg, objs.avg
