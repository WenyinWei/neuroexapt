26class SimpleTestNet(nn.Module):
27 """简单的测试网络，模拟ResNet结构"""
28
29 def __init__(self):
30 super().__init__()
31
32 # 简单的stem结构 (类似ResNet)
33 self.stem = nn.Sequential(
34 nn.Conv2d(3, 64, kernel_size=3, padding=1, bias=False), # stem.0
35 nn.BatchNorm2d(64), # stem.1
36 nn.ReLU(inplace=True) # stem.2
37 )
38
39 # 简单的层结构
40 self.conv1 = nn.Conv2d(64, 128, kernel_size=3, padding=1, bias=False)
41 self.bn1 = nn.BatchNorm2d(128)
42
43 # 嵌套结构 (类似BasicBlock)
44 self.block = nn.Sequential(
45 nn.Conv2d(128, 256, kernel_size=3, padding=1, bias=False), # block.0
46 nn.BatchNorm2d(256), # block.1
47 nn.ReLU(inplace=True) # block.2
48 )
49
50 self.avgpool = nn.AdaptiveAvgPool2d(1)
51 self.fc = nn.Linear(256, 10)
52
53 def forward(self, x):
54 x = self.stem(x)
55 x = self.conv1(x)
56 x = self.bn1(x)
57 x = self.block(x)
58 x = self.avgpool(x)
59 x = x.view(x.size(0), -1)
60 x = self.fc(x)
61 return x
