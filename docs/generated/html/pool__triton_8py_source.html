21 def _avgpool3x3_kernel(
22 x_ptr, y_ptr,
23 B, C, H, W,
24 stride: tl.constexpr,
25 BLOCK_SIZE: tl.constexpr
26 ):
27 # 获取程序ID
28 pid = tl.program_id(0)
29
30 # 计算输出尺寸
31 pad = 1
32 OH = (H + 2 * pad - 3) // stride + 1
33 OW = (W + 2 * pad - 3) // stride + 1
34
35 # 计算当前线程负责的输出位置
36 total_outputs = B * C * OH * OW
37
38 # 每个线程处理BLOCK_SIZE个元素
39 offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
40 mask = offsets < total_outputs
41
42 # 将1D索引转换为4D坐标 (b, c, oh, ow)
43 ow = offsets % OW
44 oh = (offsets // OW) % OH
45 c = (offsets // (OW * OH)) % C
46 b = offsets // (OW * OH * C)
47
48 acc = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
49
50 # 3x3池化窗口
51 for kh in tl.static_range(3):
52 for kw in tl.static_range(3):
53 ih = oh * stride + kh - pad
54 iw = ow * stride + kw - pad
55
56 # 边界检查
57 valid = mask & (ih >= 0) & (ih < H) & (iw >= 0) & (iw < W)
58
59 # 计算输入索引
60 input_idx = b * C * H * W + c * H * W + ih * W + iw
61
62 # 加载输入值 - 边界外为0（模拟padding）
63 x_val = tl.load(x_ptr + input_idx, mask=valid, other=0.0)
64 acc += x_val
65
66 # 计算平均值 - PyTorch默认count_include_pad=True，所以分母总是9
67 result = acc / 9.0
68
69 # 存储结果
70 output_idx = b * C * OH * OW + c * OH * OW + oh * OW + ow
71 tl.store(y_ptr + output_idx, result, mask=mask)
