23// Forward: out[b,c,h,w] = sum_i w[i]*x[i,b,c,h,w]
24// shapes: x [N, B, C, H, W] contiguous, w[N]
25
26template <typename scalar_t>
27__global__ void softmax_sum_fwd(const scalar_t* __restrict__ x,
28 const scalar_t* __restrict__ w,
29 scalar_t* __restrict__ out,
30 int N, int B, int C, int H, int W)
31{
32 int idx = blockIdx.x * blockDim.x + threadIdx.x;
33 int total = B * C * H * W;
34 if (idx >= total) return;
35
36 // decode idx to (b,c,h,w)
37 int w_idx = idx % W; // w
38 int h_idx = (idx / W) % H; // h
39 int c_idx = (idx / (W*H)) % C;// c
40 int b_idx = idx / (W*H*C); // b
41
42 scalar_t acc = 0;
43 for(int n=0; n<N; ++n){
44 // linear index into x
45 size_t offset = (((size_t)n*B + b_idx)*C + c_idx)*H*W + h_idx*W + w_idx;
46 acc += w[n] * x[offset];
47 }
48 out[idx] = acc;
49}
50
51// Backward wrt x and w
52
53template <typename scalar_t>
54__global__ void softmax_sum_bwd(const scalar_t* __restrict__ grad_out,
55 const scalar_t* __restrict__ x,
56 const scalar_t* __restrict__ w,
57 scalar_t* __restrict__ grad_x,
58 scalar_t* __restrict__ grad_w,
59 int N, int B, int C, int H, int W)
60{
61 // compute grad_x in parallel over (n,rest)
62 int idx = blockIdx.x * blockDim.x + threadIdx.x;
63 int total = N * B * C * H * W;
64 if (idx >= total) return;
65
66 int w_pos = idx % W;
67 int h_pos = (idx / W) % H;
68 int c_pos = (idx / (W*H)) % C;
69 int b_pos = (idx / (W*H*C)) % B;
70 int n_pos = idx / (W*H*C*B);
71
72 size_t off = (((size_t)n_pos*B + b_pos)*C + c_pos)*H*W + h_pos*W + w_pos;
73
74 scalar_t go = grad_out[((size_t)b_pos*C + c_pos)*H*W + h_pos*W + w_pos];
75 grad_x[off] = w[n_pos] * go;
76
77 // atomic add for grad_w
78 atomicAdd(grad_w + n_pos, x[off] * go);
79}

141def softmax_sum(x: torch.Tensor, logits: torch.Tensor):
142 """Compute Σ softmax(logits)_i * x_i for x shape [N,B,C,H,W]."""
143 if not x.is_cuda or not logits.is_cuda:

145 weights = torch.softmax(logits, 0)

152class SoftmaxSumFunction(torch.autograd.Function):
153 """
154 Autograd function for fused softmax + weighted sum.
155 Forward: out = Σ softmax(logits)_i * x_i
156 Backward: efficient computation of gradients w.r.t both x and logits
157 """
158
159 @staticmethod
160 def forward(ctx, x: torch.Tensor, logits: torch.Tensor):
161 """
162 Args:
163 x: [N, B, C, H, W] - N operations, each producing [B,C,H,W] output
164 logits: [N] - raw architecture weights (before softmax)
165 Returns:
166 out: [B, C, H, W] - weighted sum of operations
167 """
168 if not x.is_cuda or not logits.is_cuda:
169 # CPU fallback
170 weights = torch.softmax(logits, 0)
171 ctx.save_for_backward(x, weights)
172 ctx.use_cuda = False
173 return (x * weights.view(-1, 1, 1, 1, 1)).sum(dim=0)
174
175 # CUDA accelerated path
176 mod: Any = _get_module()
177 out, weights = mod.softmax_sum_forward(x.contiguous(), logits.contiguous())
178 ctx.save_for_backward(x, weights)
179 ctx.use_cuda = True
180 return out
181
182 @staticmethod
183 def backward(ctx, grad_out: torch.Tensor):
184 """
185 Args:
186 grad_out: [B, C, H, W] - gradient w.r.t output
187 Returns:
188 grad_x: [N, B, C, H, W] - gradient w.r.t input operations
189 grad_logits: [N] - gradient w.r.t logits (architecture weights)
190 """
191 x, weights = ctx.saved_tensors
192
193 if not ctx.use_cuda:
194 # CPU fallback backward
195 grad_x = grad_out.unsqueeze(0) * weights.view(-1, 1, 1, 1, 1)
196 grad_w = (x * grad_out.unsqueeze(0)).sum(dim=(1, 2, 3, 4))
197
198 # Convert grad_w to grad_logits using softmax gradient
199 # d/dlogits softmax(logits) = softmax(logits) * (delta - dot(delta, softmax))
200 grad_logits = weights * (grad_w - (grad_w * weights).sum())
201 return grad_x, grad_logits
202
203 # CUDA accelerated backward
204 mod: Any = _get_module()
205 grad_x, grad_w = mod.softmax_sum_backward(grad_out.contiguous(), x, weights)
206
207 # Convert grad_w to grad_logits
208 grad_logits = weights * (grad_w - (grad_w * weights).sum())
209 return grad_x, grad_logits
