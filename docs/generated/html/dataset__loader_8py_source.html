69class P2PDownloader:
70 """P2P-accelerated downloader with chunked downloads and resume capability."""
71
72 def __init__(self, chunk_size: int = 1024 * 1024, max_workers: int = 4):
73 self.chunk_size = chunk_size
74 self.max_workers = max_workers

80 def download_file(self, url: str, filepath: Path,
81 resume: bool = True, progress_bar: bool = True) -> bool:
82 """Download file with resume capability and progress tracking."""
83 try:
84 # Check if file exists and get its size
85 existing_size = filepath.stat().st_size if filepath.exists() else 0
86
87 # Get file info from server
88 headers = {}
89 if resume and existing_size > 0:
90 headers['Range'] = f'bytes={existing_size}-'
91
92 response = self.session.head(url, headers=headers, timeout=30)
93
94 if response.status_code not in [200, 206]:
95 logger.error(f"Failed to access {url}: {response.status_code}")
96 return False
97
98 total_size = int(response.headers.get('content-length', 0))
99 if resume and existing_size > 0:
100 total_size += existing_size
101
102 # Open file for writing
103 mode = 'ab' if resume and existing_size > 0 else 'wb'
104 with open(filepath, mode) as f:
105 if progress_bar:
106 pbar = tqdm(
107 total=total_size,
108 initial=existing_size,
109 unit='B',
110 unit_scale=True,
111 desc=f"Downloading {filepath.name}"
112 )
113
114 # Download in chunks
115 headers = {}
116 if resume and existing_size > 0:
117 headers['Range'] = f'bytes={existing_size}-'
118
119 response = self.session.get(url, headers=headers, stream=True, timeout=30)
120
121 for chunk in response.iter_content(chunk_size=self.chunk_size):
122 if chunk:
123 f.write(chunk)
124 if progress_bar:
125 pbar.update(len(chunk))
126
127 if progress_bar:
128 pbar.close()
129
130 return True
131
132 except Exception as e:
133 logger.error(f"Download failed for {url}: {e}")
134 return False

137class DatasetCache:
138 """Intelligent dataset caching system."""
139
140 def __init__(self, cache_dir: Union[str, Path] = "./data_cache"):
141 self.cache_dir = Path(cache_dir)
142 self.cache_dir.mkdir(parents=True, exist_ok=True)
143 self.metadata_file = self.cache_dir / "cache_metadata.json"
144 self.metadata = self._load_metadata()
145
146 def _load_metadata(self) -> Dict:
147 """Load cache metadata."""
148 if self.metadata_file.exists():
149 try:
150 with open(self.metadata_file, 'r') as f:
151 return json.load(f)
152 except Exception as e:
153 logger.warning(f"Failed to load cache metadata: {e}")
154 return {}
155
156 def _save_metadata(self):
157 """Save cache metadata."""
158 try:
159 with open(self.metadata_file, 'w') as f:
160 json.dump(self.metadata, f, indent=2)
161 except Exception as e:
162 logger.warning(f"Failed to save cache metadata: {e}")
163
164 def get_cache_path(self, dataset_name: str, filename: str) -> Path:
165 """Get cache path for a dataset file."""
166 return self.cache_dir / dataset_name / filename
167
168 def is_cached(self, dataset_name: str, filename: str,
169 expected_size: Optional[int] = None) -> bool:
170 """Check if file is cached and valid."""
171 cache_path = self.get_cache_path(dataset_name, filename)
172
173 if not cache_path.exists():
174 return False
175
176 # Check file size if expected size is provided
177 if expected_size is not None:
178 actual_size = cache_path.stat().st_size
179 if actual_size != expected_size:
180 logger.warning(f"Cache file size mismatch for {filename}")
181 return False
182
183 # Check metadata
184 cache_key = f"{dataset_name}/{filename}"
185 if cache_key in self.metadata:
186 cache_info = self.metadata[cache_key]
187 if 'checksum' in cache_info:
188 # Verify checksum
189 actual_checksum = self._calculate_checksum(cache_path)
190 if actual_checksum != cache_info['checksum']:
191 logger.warning(f"Cache checksum mismatch for {filename}")
192 return False
193
194 return True
195
196 def _calculate_checksum(self, filepath: Path) -> str:
197 """Calculate SHA256 checksum of file."""
198 sha256_hash = hashlib.sha256()
199 with open(filepath, "rb") as f:
200 for chunk in iter(lambda: f.read(4096), b""):
201 sha256_hash.update(chunk)
202 return sha256_hash.hexdigest()
203
204 def cache_file(self, dataset_name: str, filename: str,
205 source_path: Path, expected_size: Optional[int] = None):
206 """Cache a file with metadata."""
207 cache_path = self.get_cache_path(dataset_name, filename)
208 cache_path.parent.mkdir(parents=True, exist_ok=True)
209
210 # Copy file to cache
211 import shutil
212 shutil.copy2(source_path, cache_path)
213
214 # Update metadata
215 cache_key = f"{dataset_name}/{filename}"
216 self.metadata[cache_key] = {
217 'cached_at': time.time(),
218 'size': cache_path.stat().st_size,
219 'checksum': self._calculate_checksum(cache_path)
220 }
221
222 if expected_size is not None:
223 self.metadata[cache_key]['expected_size'] = expected_size
224
225 self._save_metadata()
226 logger.info(f"Cached {filename} successfully")

359 def download_dataset(self, dataset_name: str, force_download: bool = False) -> bool:
360 """Download dataset with intelligent caching and P2P acceleration."""
361 if dataset_name not in self.dataset_configs:
362 logger.error(f"Unknown dataset: {dataset_name}")
363 return False
364
365 config = self.dataset_configs[dataset_name]
366 dataset_dir = self.download_dir / dataset_name
367 dataset_dir.mkdir(parents=True, exist_ok=True)
368
369 # Check if already downloaded
370 if not force_download and self._is_dataset_complete(dataset_name, dataset_dir):
371 logger.info(f"Dataset {dataset_name} already exists and is complete")
372 return True
373
374 # Download files with mirror fallback
375 success = True
376 for i, filename in enumerate(config['files']):
377 cache_path = self.cache.get_cache_path(dataset_name, filename)
378 target_path = dataset_dir / filename
379
380 # Check cache first
381 expected_size = config.get('expected_sizes', [config.get('expected_size')])[i] if 'expected_sizes' in config else config.get('expected_size')
382
383 if self.cache.is_cached(dataset_name, filename, expected_size):
384 logger.info(f"Using cached file: {filename}")
385 if not target_path.exists():
386 import shutil
387 shutil.copy2(cache_path, target_path)
388 continue
389
390 # Try direct URLs first (Toronto University)
391 download_success = False
392
393 if 'direct_urls' in config and i < len(config['direct_urls']):
394 direct_url = config['direct_urls'][i]
395 logger.info(f"Trying direct source: {direct_url}")
396
397 for attempt in range(self.max_retries):
398 if self._download_file_with_fallback(direct_url, target_path, filename):
399 # Verify file size to ensure it's not an HTML page
400 if target_path.exists() and target_path.stat().st_size > 10000000: # > 10MB
401 download_success = True
402 break
403 else:
404 logger.warning(f"Downloaded file too small, likely HTML page")
405 target_path.unlink() # Remove corrupted file

407 logger.warning(f"Direct download attempt {attempt + 1} failed")
408 if attempt < self.max_retries - 1:
409 time.sleep(2 ** attempt)

414 logger.info(f"Trying alternative mirror: {mirror_url}")

421 break

423 logger.warning(f"Downloaded file too small, likely HTML page")

425 else:
426 logger.warning(f"Alternative mirror download attempt {attempt + 1} failed")

428 time.sleep(2 ** attempt)
429
430 if download_success:
431 break
432
433 if download_success:
434 # Cache the downloaded file
435 self.cache.cache_file(dataset_name, filename, target_path, expected_size)
436 else:
437 logger.error(f"Failed to download {filename} from all sources")
438 success = False
439
440 # Extract dataset if download was successful
441 if success:
442 logger.info(f"Extracting {dataset_name} dataset...")
443 if not self._extract_dataset(dataset_name, dataset_dir):
444 logger.error(f"Failed to extract {dataset_name} dataset")
445 success = False
446
447 return success

482 def _download_file_with_fallback(self, url: str, filepath: Path, filename: str) -> bool:
483 """Download file with robust fallback, 迅雷 integration, and chunked downloads."""
484 # Try 迅雷 first (for Chinese users)
485 if self.xunlei_downloader and self.xunlei_downloader.is_available:
486 logger.info(f"🚀 Trying 迅雷 download: {filename}")
487 if self.xunlei_downloader.download_with_xunlei(url, str(filepath.parent), filename):
488 logger.info(f"✅ 迅雷 download started for {filename}")
489 # Wait for download to complete
490 logger.info(f"⏳ Waiting for 迅雷 download to complete...")
491 max_wait_time = 300 # 5 minutes
492 wait_interval = 5 # Check every 5 seconds
493 waited_time = 0
494
495 while waited_time < max_wait_time:
496 if filepath.exists() and filepath.stat().st_size > 10000000: # > 10MB
497 logger.info(f"✅ 迅雷 download completed: {filename}")
498 return True
499 time.sleep(wait_interval)
500 waited_time += wait_interval
501 logger.info(f"⏳ Still waiting... ({waited_time}s/{max_wait_time}s)")
502
503 logger.warning(f"⚠️ 迅雷 download timeout for {filename}, trying alternatives")
504 else:
505 logger.warning(f"⚠️ 迅雷 download failed for {filename}, trying alternatives")
506
507 # Try P2P downloader
508 if self.p2p_downloader:
509 if self.p2p_downloader.download_file(url, filepath):
510 return True
511
512 # Fallback to robust chunked download
513 return self._download_with_chunks(url, filepath, filename)

515 def _download_with_chunks(self, url: str, filepath: Path, filename: str) -> bool:
516 """
517 Download file with chunked transfer and robust error handling.
518
519 Args:
520 url (str): Download URL
521 filepath (Path): Local file path
522 filename (str): Filename for logging
523
524 Returns:
525 bool: True if download successful
526 """
527 try:
528 # Use session for connection pooling and retries
529 session = requests.Session()
530 from requests.adapters import HTTPAdapter
531 from urllib3.util.retry import Retry
532
533 retry_strategy = Retry(
534 total=3,
535 backoff_factor=1,
536 status_forcelist=[429, 500, 502, 503, 504],
537 )
538 adapter = HTTPAdapter(max_retries=retry_strategy)
539 session.mount("http://", adapter)
540 session.mount("https://", adapter)
541
542 # Head request to get file size
543 try:
544 head_response = session.head(url, timeout=10, allow_redirects=True)
545 total_size = int(head_response.headers.get('content-length', 0))
546 except:
547 total_size = 0
548
549 # Download with progress tracking
550 response = session.get(url, stream=True, timeout=30)
551 response.raise_for_status()
552
553 downloaded_size = 0
554 chunk_size = 8192 # 8KB chunks
555 last_progress_time = time.time()
556
557 with open(filepath, 'wb') as f:
558 for chunk in response.iter_content(chunk_size=chunk_size):
559 if chunk: # chunk should already be bytes from iter_content
560 f.write(chunk)
561 downloaded_size += len(chunk)
562
563 # Progress update every 5 seconds or 1MB
564 current_time = time.time()
565 if (current_time - last_progress_time > 5 or
566 downloaded_size % (1024 * 1024) == 0):
567
568 if total_size > 0:
569 progress = (downloaded_size / total_size) * 100
570 speed = downloaded_size / (current_time - last_progress_time + 1) / (1024 * 1024)
571 logger.info(f"📥 {filename}: {progress:.1f}% ({downloaded_size / (1024*1024):.1f}MB) - {speed:.1f}MB/s")
572 else:
573 speed = downloaded_size / (current_time - last_progress_time + 1) / (1024 * 1024)
574 logger.info(f"📥 {filename}: {downloaded_size / (1024*1024):.1f}MB - {speed:.1f}MB/s")
575
576 last_progress_time = current_time
577
578 # Final progress
579 if total_size > 0:
580 progress = (downloaded_size / total_size) * 100
581 logger.info(f"✅ {filename}: Download complete ({progress:.1f}%)")
582 else:
583 logger.info(f"✅ {filename}: Download complete ({downloaded_size / (1024*1024):.1f}MB)")
584
585 return True
586
587 except requests.exceptions.Timeout:
588 logger.error(f"⏰ {filename}: Download timeout")
589 if filepath.exists():
590 filepath.unlink() # Remove partial file
591 return False
592 except requests.exceptions.ConnectionError:
593 logger.error(f"🔌 {filename}: Connection error")
594 if filepath.exists():
595 filepath.unlink() # Remove partial file
596 return False
597 except requests.exceptions.HTTPError as e:
598 logger.error(f"🌐 {filename}: HTTP error {e.response.status_code}")
599 if filepath.exists():
600 filepath.unlink() # Remove partial file
601 return False
602 except Exception as e:
603 logger.error(f"❌ {filename}: Unexpected error: {str(e)}")
604 if filepath.exists():
605 filepath.unlink() # Remove partial file
606 return False
