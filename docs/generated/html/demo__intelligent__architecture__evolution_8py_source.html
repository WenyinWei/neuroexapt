159class DemoNetwork(nn.Module):
160 """A simple CNN for demonstration purposes."""
161
162 def __init__(self, num_classes: int = 10):
163 super().__init__()
164
165 self.features = nn.Sequential(
166 # First block
167 nn.Conv2d(3, 16, kernel_size=3, padding=1),
168 nn.BatchNorm2d(16),
169 nn.ReLU(inplace=True),
170 nn.MaxPool2d(kernel_size=2, stride=2),
171
172 # Second block
173 nn.Conv2d(16, 32, kernel_size=3, padding=1),
174 nn.BatchNorm2d(32),
175 nn.ReLU(inplace=True),
176 nn.MaxPool2d(kernel_size=2, stride=2),
177
178 # Third block
179 nn.Conv2d(32, 64, kernel_size=3, padding=1),
180 nn.BatchNorm2d(64),
181 nn.ReLU(inplace=True),
182 nn.MaxPool2d(kernel_size=2, stride=2),
183 )
184
185 self.classifier = nn.Sequential(
186 nn.Dropout(0.5),
187 nn.Linear(64 * 4 * 4, 128),
188 nn.ReLU(inplace=True),
189 nn.Dropout(0.5),
190 nn.Linear(128, num_classes)
191 )
192
193 def forward(self, x):
194 x = self.features(x)
195 x = x.view(x.size(0), -1)
196 x = self.classifier(x)
197 return x

234def evaluate_model(model: nn.Module, dataloader: DataLoader, criterion: nn.Module, device: torch.device) -> tuple:
235 """Evaluate model on given dataloader."""
236 model.eval()
237 total_loss = 0.0
238 correct = 0
239 total = 0
240
241 with torch.no_grad():
242 for inputs, targets in dataloader:
243 inputs, targets = inputs.to(device), targets.to(device)
244 outputs = model(inputs)
245 loss = criterion(outputs, targets)
246
247 total_loss += loss.item()
248 _, predicted = torch.max(outputs.data, 1)
249 total += targets.size(0)
250 correct += (predicted == targets).sum().item()
251
252 accuracy = 100.0 * correct / total
253 avg_loss = total_loss / len(dataloader)
254
255 return accuracy, avg_loss

258def calculate_gradient_norm(model: nn.Module) -> float:
259 """Calculate the gradient norm for the model."""
260 total_norm = 0.0
261 param_count = 0
262
263 for p in model.parameters():
264 if p.grad is not None:
265 param_norm = p.grad.data.norm(2)
266 total_norm += param_norm.item() ** 2
267 param_count += 1
268
269 return (total_norm ** 0.5) / max(param_count, 1)
