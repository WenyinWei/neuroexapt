153 def _gradient_optimized_loss(self, input_valid, target_valid):
154 """使用梯度优化的损失计算"""
155 # 使用gradient checkpointing减少内存使用
156 import torch.utils.checkpoint as cp
157
158 def forward_wrapper(x):
159 return self.modelmodel(x)
160
161 # 分批处理以减少内存峰值
162 batch_size = input_valid.size(0)
163 if batch_size > 16:
164 # 分成较小的批次
165 losses = []
166 chunk_size = 8
167 for i in range(0, batch_size, chunk_size):
168 end_idx = min(i + chunk_size, batch_size)
169 chunk_input = input_valid[i:end_idx]
170 chunk_target = target_valid[i:end_idx]
171
172 chunk_output = cp.checkpoint(forward_wrapper, chunk_input, use_reentrant=False)
173 chunk_loss = self.criterion(chunk_output, chunk_target)
174 losses.append(chunk_loss * chunk_input.size(0))
175
176 # 加权平均
177 total_loss = sum(losses) / batch_size
178 return total_loss
179 else:
180 # 小批次直接处理
181 output = cp.checkpoint(forward_wrapper, input_valid, use_reentrant=False)
182 return self.criterion(output, target_valid)
