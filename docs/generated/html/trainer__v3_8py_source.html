70 def evaluate(self) -> Dict[str, float]:
71 """Evaluates the model on the validation set."""
72 self.model.eval()
73 total_loss = 0
74 correct = 0
75 total = 0
76 with torch.no_grad():
77 for data, targets in self.val_loader:
78 data, targets = data.to(self.device), targets.to(self.device)
79 outputs = self.model(data)
80 loss = self.criterion(outputs, targets)
81 total_loss += loss.item()
82 _, predicted = torch.max(outputs.data, 1)
83 total += targets.size(0)
84 correct += (predicted == targets).sum().item()
85
86 accuracy = 100 * correct / total
87 avg_loss = total_loss / len(self.val_loader)
88 return {'loss': avg_loss, 'accuracy': accuracy}

90 def fit(self, epochs: int, evolution_frequency: int = 1):
91 """
92 Main training and evolution loop.
93
94 Args:
95 epochs: Total number of epochs to train.
96 evolution_frequency: How often to run the evolution step.
97 """
98 print(f"--- Starting TrainerV3 fit for {epochs} epochs ---")
99
100 lr_scheduler = AdaptiveLearningRateScheduler(self.optimizer, initial_lr=self.optimizer.param_groups[0]['lr'])
101
102 for epoch in range(epochs):
103 self.neuroexapt.current_epoch = epoch + 1
104
105 # 1. Train for one epoch
106 self.neuroexapt.train_epoch(self.optimizer)
107
108 # 2. Evaluate performance
109 performance = self.evaluate()
110 print(f"Epoch {self.neuroexapt.current_epoch} Validation: Loss={performance['loss']:.4f}, Accuracy={performance['accuracy']:.2f}%")
111
112 # 3. Potentially evolve the architecture
113 if self.neuroexapt.current_epoch % evolution_frequency == 0:
114 self.neuroexapt.evolve(performance, self.optimizer)
115
116 # 4. Update learning rate
117 avg_entropy = self.neuroexapt.evolution_engine.entropy_metrics.get_average_entropy()
118 # A proper KL divergence would be computed after an evolution step
119 kl_div = 0.0
120 lr_scheduler.step(entropy=avg_entropy, kl_divergence=kl_div)
121 print(f"Updated LR to: {lr_scheduler.get_lr():.6f}")
122
123 print("--- TrainerV3 fit finished ---")
124 return self.neuroexapt.get_model()
