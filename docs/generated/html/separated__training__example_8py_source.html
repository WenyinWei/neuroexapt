169def validate_epoch(valid_queue, model, criterion):
170 """验证一个epoch"""
171 model.eval()
172
173 objs = AvgrageMeter()
174 top1 = AvgrageMeter()
175
176 with torch.no_grad():
177 for input, target in valid_queue:
178 input = input.cuda(non_blocking=True)
179 target = target.cuda(non_blocking=True)
180
181 logits = model(input)
182 loss = criterion(logits, target)
183
184 prec1, _ = accuracy(logits, target, topk=(1, 5))
185 n = input.size(0)
186 objs.update(loss.item(), n)
187 top1.update(prec1.item(), n)
188
189 return top1.avg, objs.avg
