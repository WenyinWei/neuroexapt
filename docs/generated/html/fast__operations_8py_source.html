28class FastMixedOp(nn.Module):
29 """
30 高性能混合操作
31
32 核心优化：
33 1. 权重阈值过滤：只计算权重>threshold的操作
34 2. Top-K选择：只保留前K个最大权重的操作
35 3. 操作缓存：缓存昂贵操作的结果
36 4. 动态精度：根据权重大小选择计算精度
37 """
38
39 def __init__(self, C, stride, primitives=None, weight_threshold=0.01, top_k=3):
40 super().__init__()
41
42 if primitives is None:
43 from .genotypes import PRIMITIVES
44 primitives = PRIMITIVES
45
46 self.C = C
47 self.stride = stride
48 self.weight_threshold = weight_threshold
49 self.top_k = min(top_k, len(primitives))
50
51 # 构建操作字典
52 self._ops = nn.ModuleDict()
53 self._op_names = list(primitives)
54
55 # 按计算成本排序操作（便宜的操作优先）
56 self._op_costs = self._calculate_operation_costs(primitives)
57 self._sorted_ops = sorted(enumerate(primitives), key=lambda x: self._op_costs[x[1]])
58
59 for primitive in primitives:
60 op = self._create_operation(primitive, C, stride)
61 self._ops[primitive] = op
62
63 # 缓存相关
64 self._cache = {}
65 self._cache_hits = 0
66 self._cache_misses = 0
67
68 # 性能统计
69 self.forward_count = 0
70 self.active_ops_avg = 0.0

72 def _calculate_operation_costs(self, primitives):
73 """计算操作的相对成本"""
74 costs = {
75 'none': 0.1,
76 'skip_connect': 0.2,
77 'avg_pool_3x3': 0.3,
78 'max_pool_3x3': 0.3,
79 'sep_conv_3x3': 1.0,
80 'sep_conv_5x5': 1.5,
81 'sep_conv_7x7': 2.5,
82 'dil_conv_3x3': 1.2,
83 'dil_conv_5x5': 1.8,
84 'conv_7x1_1x7': 1.3,
85 }
86 return {op: costs.get(op, 1.0) for op in primitives}

88 def _create_operation(self, primitive, C, stride):
89 """创建优化的操作"""
90 from .operations import OPS
91
92 op = OPS[primitive](C, stride, False)
93
94 # 为池化操作添加BN
95 if 'pool' in primitive:
96 op = nn.Sequential(op, nn.BatchNorm2d(C, affine=False))
97
98 # 操作融合优化
99 if isinstance(op, nn.Sequential):
100 op = self._fuse_operations(op)
101
102 return op

104 def _fuse_operations(self, sequential_op):
105 """操作融合优化"""
106 # 简单的融合：合并连续的Conv+BN+ReLU
107 layers = list(sequential_op.children())
108 fused_layers = []
109
110 i = 0
111 while i < len(layers):
112 if (i < len(layers) - 2 and
113 isinstance(layers[i], nn.ReLU) and
114 isinstance(layers[i+1], nn.Conv2d) and
115 isinstance(layers[i+2], nn.BatchNorm2d)):
116
117 # 创建融合层
118 conv = layers[i+1]
119 bn = layers[i+2]
120 fused = nn.Sequential(
121 nn.ReLU(inplace=True), # 使用inplace节省内存
122 conv,
123 bn
124 )
125 fused_layers.append(fused)
126 i += 3
127 else:
128 fused_layers.append(layers[i])
129 i += 1
130
131 return nn.Sequential(*fused_layers)

133 def forward(self, x, weights, training=True):
134 """
135 高性能前向传播
136
137 Args:
138 x: 输入张量
139 weights: 架构权重 [num_ops]
140 training: 是否训练模式
141 """
142 self.forward_count += 1
143
144 if not training:
145 # 推理时只使用最大权重的操作
146 max_idx = weights.argmax()
147 op_name = self._op_names[max_idx]
148 return self._ops[op_name](x)
149
150 # 训练时的智能选择策略
151 return self._forward_training(x, weights)

153 def _forward_training(self, x, weights):
154 """训练时的优化前向传播"""
155 device = x.device
156
157 # 策略1：权重阈值过滤
158 active_indices = (weights > self.weight_threshold).nonzero(as_tuple=True)[0]
159
160 if len(active_indices) == 0:
161 # 所有权重都很小，使用权重最大的操作
162 active_indices = weights.topk(1)[1]
163 elif len(active_indices) > self.top_k:
164 # 太多活跃操作，只保留前top_k个
165 top_weights, top_indices = weights[active_indices].topk(self.top_k)
166 active_indices = active_indices[top_indices]
167
168 # 更新统计
169 self.active_ops_avg = 0.9 * self.active_ops_avg + 0.1 * len(active_indices)
170
171 # 计算活跃操作的输出
172 outputs = []
173 active_weights = []
174
175 for idx in active_indices:
176 op_name = self._op_names[idx.item()]
177 weight = weights[idx]
178
179 # 缓存键
180 cache_key = self._get_cache_key(x, op_name)
181
182 if cache_key in self._cache:
183 output = self._cache[cache_key]
184 self._cache_hits += 1
185 else:
186 output = self._ops[op_name](x)
187 # 只缓存计算成本高的操作
188 if self._op_costs[op_name] > 1.0:
189 self._cache[cache_key] = output.clone()
190 # 限制缓存大小
191 if len(self._cache) > 100:
192 self._cache.clear()
193 self._cache_misses += 1
194
195 outputs.append(output)
196 active_weights.append(weight)
197
198 # 归一化权重
199 active_weights = torch.stack(active_weights)
200 active_weights = active_weights / (active_weights.sum() + 1e-8)
201
202 # 加权求和
203 result = sum(w * out for w, out in zip(active_weights, outputs))
204
205 return result

208 """生成缓存键"""
209 # 简化的缓存键：基于输入形状和操作名
210 return f"{op_name}_{x.shape}_{x.device}"

228class BatchedArchitectureUpdate(nn.Module):
229 """
230 批量化架构参数更新
231
232 将多个架构参数的更新操作批量化，减少GPU kernel调用次数
233 """
234
235 def __init__(self, num_layers, num_ops_per_layer):
236 super().__init__()
237 self.num_layers = num_layers
238 self.num_ops_per_layer = num_ops_per_layer
239
240 # 批量化的架构参数
241 self.arch_params = nn.Parameter(
242 torch.randn(num_layers, num_ops_per_layer) * 0.1
243 )
244
245 # Gumbel-Softmax参数
246 self.temperature = 5.0
247 self.min_temperature = 0.1
248 self.anneal_rate = 0.98
249
250 def forward(self, layer_idx=None):
251 """
252 获取架构权重
253
254 Args:
255 layer_idx: 层索引，None表示返回所有层
256 """
257 if layer_idx is not None:
258 logits = self.arch_params[layer_idx]
259 else:
260 logits = self.arch_params
261
262 if self.training:
263 return self._gumbel_softmax(logits)
264 else:
265 return F.softmax(logits, dim=-1)
266
267 def _gumbel_softmax(self, logits):
268 """批量化Gumbel-Softmax采样"""
269 # 生成Gumbel噪声
270 gumbel_noise = -torch.log(-torch.log(torch.rand_like(logits) + 1e-8) + 1e-8)
271
272 # 加入噪声并除以温度
273 noisy_logits = (logits + gumbel_noise) / self.temperature
274
275 # Softmax
276 soft_weights = F.softmax(noisy_logits, dim=-1)
277
278 # 硬采样（前向时离散，反向时连续）
279 hard_weights = F.one_hot(soft_weights.argmax(dim=-1), soft_weights.size(-1)).float()
280
281 # 使用straight-through estimator
282 return hard_weights - soft_weights.detach() + soft_weights
283
284 def anneal_temperature(self):
285 """退火温度"""
286 self.temperature = max(self.min_temperature, self.temperature * self.anneal_rate)
287 return self.temperature
288
289 def get_dominant_ops(self, threshold=0.5):
290 """获取占主导地位的操作"""
291 with torch.no_grad():
292 probs = F.softmax(self.arch_params, dim=-1)
293 dominant = (probs > threshold).float()
294 return dominant

296class MemoryEfficientCell(nn.Module):
297 """
298 内存高效的Cell实现
299
300 优化策略：
301 1. 梯度检查点：权衡计算和内存
302 2. 输出缓存：避免重复计算
303 3. 动态形状：根据需要调整
304 """
305
306 def __init__(self, C_in, C_out, num_nodes=4, num_ops=8, use_checkpoint=True):
307 super().__init__()
308
309 self.C_in = C_in
310 self.C_out = C_out
311 self.num_nodes = num_nodes
312 self.num_ops = num_ops
313 self.use_checkpoint = use_checkpoint
314
315 # 预处理层
316 self.preprocess0 = self._preprocess_layer(C_in, C_out)
317 self.preprocess1 = self._preprocess_layer(C_in, C_out)
318
319 # 混合操作
320 self.ops = nn.ModuleList()
321 for i in range(num_nodes):
322 for j in range(2 + i): # 每个节点连接前面的所有节点
323 op = FastMixedOp(C_out, stride=1)
324 self.ops.append(op)
325
326 # 架构参数更新器
327 num_edges = sum(range(2, 2 + num_nodes))
328 self.arch_updater = BatchedArchitectureUpdate(num_edges, num_ops)
329
330 # 输出处理
331 self.output_conv = nn.Conv2d(num_nodes * C_out, C_out, 1, bias=False)
332 self.output_bn = nn.BatchNorm2d(C_out)
333
334 # 性能监控
335 self.memory_usage = []
336 self.compute_time = []
337
338 def _preprocess_layer(self, C_in, C_out):
339 """预处理层"""
340 if C_in == C_out:
341 return nn.Sequential(
342 nn.ReLU(inplace=True),
343 nn.BatchNorm2d(C_in)
344 )
345 else:
346 return nn.Sequential(
347 nn.ReLU(inplace=True),
348 nn.Conv2d(C_in, C_out, 1, bias=False),
349 nn.BatchNorm2d(C_out)
350 )
351
352 def forward(self, s0, s1):
353 """
354 前向传播
355
356 Args:
357 s0, s1: 前两个节点的输出
358 """
359 # 预处理
360 s0 = self.preprocess0(s0)
361 s1 = self.preprocess1(s1)
362
363 states = [s0, s1]
364
365 # 获取所有架构权重
366 arch_weights = self.arch_updater()
367
368 op_idx = 0
369 for i in range(self.num_nodes):
370 # 计算新节点
371 if self.use_checkpoint and self.training:
372 # 使用梯度检查点节省内存
373 new_state = torch.utils.checkpoint.checkpoint(
374 self._compute_node_compute_node, i, states, arch_weights, op_idx
375 )
376 else:
377 new_state = self._compute_node_compute_node(i, states, arch_weights, op_idx)
378
379 states.append(new_state)
380 op_idx += len(states) - 1
381
382 # 拼接所有中间节点的输出
383 intermediate_states = states[2:] # 排除输入节点
384 output = torch.cat(intermediate_states, dim=1)
385
386 # 输出处理
387 output = self.output_conv(output)
388 output = self.output_bn(output)
389
390 return output
391
392 def _compute_node(self, node_idx, states, arch_weights, start_op_idx):
393 """计算单个节点"""
394 node_inputs = []
395
396 for j, state in enumerate(states):
397 op_idx = start_op_idx + j
398 if op_idx < len(arch_weights) and op_idx < len(self.ops):
399 weight = arch_weights[op_idx]
400
401 # 使用FastMixedOp
402 op_output = self.ops[op_idx](state, weight, self.training)
403 node_inputs.append(op_output)
404
405 # 求和所有输入
406 return sum(node_inputs) if node_inputs else states[0]
407
408 def get_memory_usage(self):
409 """获取内存使用情况"""
410 if torch.cuda.is_available():
411 return torch.cuda.memory_allocated() / 1024**2 # MB
412 return 0

414class FastDeviceManager:
415 """
416 快速设备管理器
417
418 优化策略：
419 1. 设备亲和性：将相关操作放在同一设备
420 2. 内存池：预分配内存减少分配开销

425 def __init__(self, device=None, memory_pool_size=1024):
426 self.device = device or torch.device('cuda' if torch.cuda.is_available() else 'cpu')
427 self.memory_pool_size = memory_pool_size # MB

438 def _init_memory_pool(self):
439 """初始化内存池"""

441 # 预分配一些常用大小的张量
442 self.memory_pool = {}

490class OperationProfiler:
491 """
492 操作性能分析器
493
494 用于分析不同操作的计算成本，指导优化决策
495 """
496
497 def __init__(self):
498 self.operation_times = {}
499 self.operation_memory = {}

502 def profile_operation(self, op_name, operation, input_tensor, num_runs=10):
503 """分析操作性能"""
504 device = input_tensor.device
505
506 # 预热
507 for _ in range(3):
508 _ = operation(input_tensor)
509
510 if device.type == 'cuda':
511 torch.cuda.synchronize()
512
513 # 计时
514 start_time = time.time()
515 start_memory = torch.cuda.memory_allocated() if device.type == 'cuda' else 0
516
517 for _ in range(num_runs):
518 output = operation(input_tensor)
519
520 if device.type == 'cuda':
521 torch.cuda.synchronize()
522
523 end_time = time.time()
524 end_memory = torch.cuda.memory_allocated() if device.type == 'cuda' else 0
525
526 # 记录结果
527 avg_time = (end_time - start_time) / num_runs
528 memory_delta = end_memory - start_memory
529
530 self.operation_times[op_name] = avg_time
531 self.operation_memory[op_name] = memory_delta
532 self.operation_count[op_name] = self.operation_count.get(op_name, 0) + 1
533
534 def get_operation_ranking(self, criterion='time'):
535 """获取操作排名"""
536 if criterion == 'time':
537 data = self.operation_times
538 elif criterion == 'memory':
539 data = self.operation_memory
540 else:
541 raise ValueError(f"Unknown criterion: {criterion}")
542
543 return sorted(data.items(), key=lambda x: x[1])
