228class BatchedArchitectureUpdate(nn.Module):
229 """
230 æ‰¹é‡åŒ–æ¶æ„å‚æ•°æ›´æ–°
231
232 å°†å¤šä¸ªæ¶æ„å‚æ•°çš„æ›´æ–°æ“ä½œæ‰¹é‡åŒ–ï¼Œå‡å°‘GPU kernelè°ƒç”¨æ¬¡æ•°

236 super().__init__()

238 self.num_ops_per_layer = num_ops_per_layer
239
240 # æ‰¹é‡åŒ–çš„æ¶æ„å‚æ•°
241 self.arch_params = nn.Parameter(
242 torch.randn(num_layers, num_ops_per_layer) * 0.1
243 )
244
245 # Gumbel-Softmaxå‚æ•°
246 self.temperature = 5.0
247 self.min_temperature = 0.1
248 self.anneal_rate = 0.98

250 def forward(self, layer_idx=None):
251 """
252 è·å–æ¶æ„æƒé‡
253
254 Args:
255 layer_idx: å±‚ç´¢å¼•ï¼ŒNoneè¡¨ç¤ºè¿”å›æ‰€æœ‰å±‚
256 """
257 if layer_idx is not None:
258 logits = self.arch_params[layer_idx]
259 else:
260 logits = self.arch_params
261
262 if self.training:
263 return self._gumbel_softmax(logits)
264 else:
265 return F.softmax(logits, dim=-1)

267 def _gumbel_softmax(self, logits):
268 """æ‰¹é‡åŒ–Gumbel-Softmaxé‡‡æ ·"""
269 # ç”ŸæˆGumbelå™ªå£°
270 gumbel_noise = -torch.log(-torch.log(torch.rand_like(logits) + 1e-8) + 1e-8)
271
272 # åŠ å…¥å™ªå£°å¹¶é™¤ä»¥æ¸©åº¦
273 noisy_logits = (logits + gumbel_noise) / self.temperature
274
275 # Softmax
276 soft_weights = F.softmax(noisy_logits, dim=-1)
277
278 # ç¡¬é‡‡æ ·ï¼ˆå‰å‘æ—¶ç¦»æ•£ï¼Œåå‘æ—¶è¿ç»­ï¼‰
279 hard_weights = F.one_hot(soft_weights.argmax(dim=-1), soft_weights.size(-1)).float()
280
281 # ä½¿ç”¨straight-through estimator
282 return hard_weights - soft_weights.detach() + soft_weights

284 def anneal_temperature(self):
285 """é€€ç«æ¸©åº¦"""
286 self.temperature = max(self.min_temperature, self.temperature * self.anneal_rate)
287 return self.temperature

291 with torch.no_grad():

293 dominant = (probs > threshold).float()
