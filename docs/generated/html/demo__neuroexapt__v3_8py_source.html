19class DemoNet(nn.Module):
20 """Demo neural network for showcasing V3 features"""
21
22 def __init__(self, num_classes=10):
23 super().__init__()
24 self.features = nn.Sequential(
25 nn.Conv2d(3, 64, 3, padding=1),
26 nn.ReLU(),
27 nn.Conv2d(64, 128, 3, padding=1),
28 nn.ReLU(),
29 nn.MaxPool2d(2),
30 nn.Conv2d(128, 256, 3, padding=1),
31 nn.ReLU(),
32 nn.AdaptiveAvgPool2d((4, 4))
33 )
34
35 self.classifier = nn.Sequential(
36 nn.Linear(256 * 16, 512),
37 nn.ReLU(),
38 nn.Dropout(0.5),
39 nn.Linear(512, 128),
40 nn.ReLU(),
41 nn.Linear(128, num_classes)
42 )
43
44 def forward(self, x):
45 x = self.features(x)
46 x = x.view(x.size(0), -1)
47 x = self.classifier(x)
48 return x
