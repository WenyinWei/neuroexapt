101 def analyze_architecture(self, dataloader: DataLoader) -> Dict[str, Any]:
102 """Perform basic architecture analysis."""
103 self.model.eval()
104
105 # Simple analysis - count parameters and layers
106 total_params = sum(p.numel() for p in self.model.parameters())
107 trainable_params = sum(p.numel() for p in self.model.parameters() if p.requires_grad)
108
109 # Count layers
110 conv_layers = sum(1 for m in self.model.modules() if isinstance(m, (nn.Conv1d, nn.Conv2d, nn.Conv3d)))
111 linear_layers = sum(1 for m in self.model.modules() if isinstance(m, nn.Linear))
112
113 # Simple redundancy estimation
114 redundancy = max(0, min(1, (total_params - trainable_params) / max(total_params, 1)))
115
116 # Simple efficiency calculation
117 efficiency = min(1.0, trainable_params / max(total_params, 1))
118
119 return {
120 'total_redundancy': redundancy,
121 'computational_efficiency': efficiency,
122 'total_parameters': total_params,
123 'trainable_parameters': trainable_params,
124 'conv_layers': conv_layers,
125 'linear_layers': linear_layers,
126 'layer_importances': {},

202 def should_rollback(self, current_performance: Dict[str, float]) -> bool:

280 def _train_epoch(self, dataloader: DataLoader, optimizer: optim.Optimizer, criterion: nn.Module) -> Dict[str, float]:
281 """Train for one epoch."""
282 self.model.train()
283 total_loss = 0.0
284 correct = 0
285 total = 0
286
287 for data, targets in dataloader:
288 data, targets = data.to(self.device), targets.to(self.device)
289
290 optimizer.zero_grad()
291 outputs = self.model(data)
292 loss = criterion(outputs, targets)
293 loss.backward()
294 optimizer.step()
295
296 total_loss += loss.item()
297 _, predicted = outputs.max(1)
298 total += targets.size(0)
299 correct += predicted.eq(targets).sum().item()
300
301 avg_loss = total_loss / len(dataloader)
302 accuracy = 100. * correct / total
303
304 return {'train_loss': avg_loss, 'train_accuracy': accuracy}

306 def _validate_epoch(self, dataloader: DataLoader, criterion: nn.Module) -> Dict[str, float]:
307 """Validate for one epoch."""
308 self.model.eval()
309 total_loss = 0.0
310 correct = 0
311 total = 0
312
313 with torch.no_grad():
314 for data, targets in dataloader:
315 data, targets = data.to(self.device), targets.to(self.device)
316 outputs = self.model(data)
317 loss = criterion(outputs, targets)
318
319 total_loss += loss.item()
320 _, predicted = outputs.max(1)
321 total += targets.size(0)
322 correct += predicted.eq(targets).sum().item()
323
324 avg_loss = total_loss / len(dataloader)
325 accuracy = 100. * correct / total
326
327 return {'val_loss': avg_loss, 'val_accuracy': accuracy}
