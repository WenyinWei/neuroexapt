41 def __init__(self, num_classes=10, initial_channels=32):
42 super(RobustEvolvableCNN, self).__init__()

44 # Feature extraction with robust architecture
45 self.features = nn.Sequential(
46 # Block 1: Initial convolution
47 nn.Conv2d(3, initial_channels, kernel_size=3, padding=1),
48 nn.BatchNorm2d(initial_channels),
49 nn.ReLU(inplace=True),
50 nn.MaxPool2d(kernel_size=2, stride=2),
51
52 # Block 2: Expandable convolution
53 nn.Conv2d(initial_channels, initial_channels * 2, kernel_size=3, padding=1),
54 nn.BatchNorm2d(initial_channels * 2),
55 nn.ReLU(inplace=True),
56 nn.MaxPool2d(kernel_size=2, stride=2),
57
58 # Block 3: Deep convolution
59 nn.Conv2d(initial_channels * 2, initial_channels * 4, kernel_size=3, padding=1),
60 nn.BatchNorm2d(initial_channels * 4),
61 nn.ReLU(inplace=True),
62 nn.MaxPool2d(kernel_size=2, stride=2),
63
64 # Block 4: Final convolution
65 nn.Conv2d(initial_channels * 4, initial_channels * 8, kernel_size=3, padding=1),
66 nn.BatchNorm2d(initial_channels * 8),
67 nn.ReLU(inplace=True),
68 nn.AdaptiveAvgPool2d((1, 1))
69 )
70
71 # Robust classifier with self-validation
72 self.classifier = nn.Sequential(
73 nn.Flatten(),
74 nn.Dropout(0.5),
75 nn.Linear(initial_channels * 8, 256),
76 nn.ReLU(inplace=True),
77 nn.Dropout(0.3),
78 nn.Linear(256, 128),
79 nn.ReLU(inplace=True),
80 nn.Dropout(0.2),
81 nn.Linear(128, num_classes)
82 )
83
84 # Evolution tracking
85 self.evolution_count = 0
86 self.parameter_history = []
87 self.validation_history = []

89 # Initialize weights
90 self._initialize_weights()

92 # Validate initial architecture
93 self._validate_architecture()

95 def forward(self, x):
96 try:
97 features = self.features(x)
98 output = self.classifier(features)
99 return output
100 except RuntimeError as e:
101 print(f"‚ùå Forward pass error: {e}")
102
103 # Attempt auto-recovery
104 if self._attempt_auto_recovery():
105 print("üîß Auto-recovery successful, retrying forward pass...")
106 features = self.features(x)
107 output = self.classifier(features)
108 return output
109 else:
110 raise e

112 def _initialize_weights(self):
113 """Initialize weights with better variance."""
114 for m in self.modules():
115 if isinstance(m, nn.Conv2d):
116 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
117 if m.bias is not None:
118 nn.init.constant_(m.bias, 0)
119 elif isinstance(m, nn.BatchNorm2d):
120 nn.init.constant_(m.weight, 1)
121 nn.init.constant_(m.bias, 0)
122 elif isinstance(m, nn.Linear):
123 nn.init.normal_(m.weight, 0, 0.01)
124 nn.init.constant_(m.bias, 0)
125
126 def _validate_architecture(self):
127 """Validate current architecture."""
128 try:
129 is_valid = validate_architecture_change(self)
130 self.validation_history.append(is_valid)
131
132 if not is_valid:
133 print("‚ö†Ô∏è Architecture validation failed, attempting fix...")
134 fixed = fix_common_shape_issues(self)
135 if fixed:
136 print("‚úÖ Architecture fixed successfully")
137 else:
138 print("‚ùå Could not fix architecture issues")
139
140 return is_valid
141
142 except Exception as e:
143 print(f"Validation error: {e}")
144 return False

146 def _attempt_auto_recovery(self) -> bool:
147 """Attempt to recover from forward pass errors."""
148 print("üîÑ Attempting auto-recovery...")
149 try:
150 return fix_common_shape_issues(self)
151 except Exception as e:
152 print(f"Auto-recovery failed: {e}")

176def load_cifar10_data(batch_size=64, download=True):
177 """Load CIFAR-10 with standard augmentation."""
178
179 transform_train = transforms.Compose([
180 transforms.RandomCrop(32, padding=4),
181 transforms.RandomHorizontalFlip(p=0.5),
182 transforms.RandomRotation(10),

186 ])

191 ])
192
193 # Load datasets
194 trainset = torchvision.datasets.CIFAR10(
195 root='./data', train=True, download=download, transform=transform_train

197 trainloader = DataLoader(trainset, batch_size=batch_size, shuffle=True,
198 num_workers=2, pin_memory=True)
199
200 testset = torchvision.datasets.CIFAR10(
201 root='./data', train=False, download=download, transform=transform_test

204 num_workers=2, pin_memory=True)

209def evaluate_model(model, testloader, device):
210 """Comprehensive model evaluation."""
211 model.eval()
212 correct = 0
213 total = 0
214 total_loss = 0
215 criterion = nn.CrossEntropyLoss()
216
217 with torch.no_grad():
218 for batch_idx, (data, targets) in enumerate(testloader):
219 data, targets = data.to(device), targets.to(device)
220 try:
221 outputs = model(data)
222 loss = criterion(outputs, targets)
223 total_loss += loss.item()
224
225 _, predicted = torch.max(outputs.data, 1)
226 total += targets.size(0)
227 correct += (predicted == targets).sum().item()
228 except RuntimeError as e:
229 print(f"Warning: Evaluation batch {batch_idx} failed: {e}")
230 continue
231
232 accuracy = 100 * correct / total if total > 0 else 0.0
233 avg_loss = total_loss / len(testloader) if len(testloader) > 0 else 0.0
234
235 model.train()
236 return accuracy, avg_loss
