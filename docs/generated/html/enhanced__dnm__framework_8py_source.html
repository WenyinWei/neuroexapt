175 def _compute_complexity_score(self, activations: Dict[str, torch.Tensor],
176 gradients: Dict[str, torch.Tensor]) -> float:
177 """计算网络复杂度分数"""
178 logger.enter_section("复杂度分数计算")
179 scores = []
180
181 for name, activation in activations.items():
182 if name not in gradients or gradients[name] is None:
183 logger.warning(f"⚠️ 跳过层 {name}: 缺少梯度信息")
184 continue
185
186 gradient = gradients[name]
187 logger.log_tensor_info(activation, f"激活值[{name}]")
188 logger.log_tensor_info(gradient, f"梯度[{name}]")
189
190 # 1. 信息熵分析
191 entropy = self._compute_entropy(activation)
192 logger.debug(f"信息熵[{name}]: {entropy:.4f}")
193
194 # 2. 梯度复杂度
195 grad_complexity = self._compute_gradient_complexity(gradient)
196 logger.debug(f"梯度复杂度[{name}]: {grad_complexity:.4f}")
197
198 # 3. 激活模式复杂度
199 activation_complexity = self._compute_activation_complexity(activation)
200 logger.debug(f"激活复杂度[{name}]: {activation_complexity:.4f}")
201
202 # 综合分数
203 layer_score = 0.4 * entropy + 0.3 * grad_complexity + 0.3 * activation_complexity
204 scores.append(layer_score)
205 logger.debug(f"层分数[{name}]: {layer_score:.4f}")
206
207 final_score = np.mean(scores) if scores else 0.0
208 logger.info(f"最终复杂度分数: {final_score:.4f} (共{len(scores)}层)")
209 logger.exit_section("复杂度分数计算")
210 return final_score

212 def _compute_entropy(self, activation: torch.Tensor) -> float:
213 """计算激活值熵"""
214 if activation.numel() == 0:
215 return 0.0
216
217 activation_flat = activation.flatten()
218 activation_abs = torch.abs(activation_flat) + 1e-8
219 probs = activation_abs / torch.sum(activation_abs)
220
221 entropy = -torch.sum(probs * torch.log(probs + 1e-8))
222 # 归一化到 [0, 1]
223 max_entropy = math.log(len(probs))
224 return min(entropy.item() / max_entropy, 1.0) if max_entropy > 0 else 0.0
225
226 def _compute_gradient_complexity(self, gradient: torch.Tensor) -> float:
227 """计算梯度复杂度"""
228 if gradient is None or gradient.numel() == 0:
229 return 0.0
230
231 grad_flat = gradient.flatten()
232
233 # 梯度的标准差/均值比率
234 grad_std = torch.std(grad_flat)
235 grad_mean = torch.mean(torch.abs(grad_flat))
236
237 if grad_mean > 1e-8:
238 complexity = grad_std / grad_mean
239 return min(complexity.item(), 1.0)
240 else:
241 return 0.0
242
243 def _compute_activation_complexity(self, activation: torch.Tensor) -> float:
244 """计算激活模式复杂度"""
245 if activation.numel() == 0 or len(activation.shape) < 2:
246 return 0.0
247
248 # 计算激活模式的变异系数
249 activation_flat = activation.view(activation.shape[0], -1)
250
251 # 批次间的变异性
252 batch_means = torch.mean(activation_flat, dim=1)
253 batch_std = torch.std(batch_means)
254 batch_mean_avg = torch.mean(batch_means)
255
256 if batch_mean_avg > 1e-8:
257 complexity = batch_std / batch_mean_avg
258 return min(complexity.item(), 1.0)
259 else:
260 return 0.0

310 def _compute_maturation_score(self, performance_history: List[float]) -> float:
311 """计算发育成熟度分数"""
312 if len(performance_history) < 5:
313 return 0.0
314
315 recent_performances = performance_history[-10:]
316
317 # 1. 性能稳定性
318 stability = 1.0 - np.std(recent_performances[-5:]) / (np.mean(recent_performances[-5:]) + 1e-8)
319
320 # 2. 改进速度
321 if len(recent_performances) >= 5:
322 early_avg = np.mean(recent_performances[:5])
323 late_avg = np.mean(recent_performances[-5:])
324 improvement_rate = (late_avg - early_avg) / (early_avg + 1e-8)
325 else:
326 improvement_rate = 0.0
327
328 # 3. 性能高度
329 performance_level = recent_performances[-1] if recent_performances else 0.0
330
331 # 综合成熟度分数
332 maturation = (
333 0.4 * min(stability, 1.0) +
334 0.3 * min(improvement_rate, 1.0) +
335 0.3 * performance_level
336 )
337
338 return max(0.0, min(maturation, 1.0))

411 def _detect_catastrophic_forgetting(self, performance_history: List[float]) -> bool:
412 """检测灾难性遗忘"""
413 if len(performance_history) < 8:
414 return False
415
416 # 检查最近性能是否显著下降
417 recent_window = 5
418 past_window = 5
419
420 recent_avg = np.mean(performance_history[-recent_window:])
421 past_avg = np.mean(performance_history[-(recent_window + past_window):-recent_window])
422
423 if past_avg > 0:
424 decline_ratio = (past_avg - recent_avg) / past_avg
425 return decline_ratio > self.forgetting_threshold
426
427 return False

429 def _detect_learning_saturation(self, performance_history: List[float]) -> bool:
430 """检测学习饱和"""
431 if len(performance_history) < 10:
432 return False
433
434 # 检查最近10个epoch的改进
435 recent_performances = performance_history[-10:]
436 improvements = [recent_performances[i] - recent_performances[i-1]
437 for i in range(1, len(recent_performances))]

442 def _detect_representation_conflict(self, activations: Dict[str, torch.Tensor]) -> bool:
443 """检测特征表示冲突"""
444 if not activations:
445 return False
446
447 # 简化的冲突检测：检查激活模式的一致性
448 conflict_scores = []
449
450 for name, activation in activations.items():
451 if len(activation.shape) >= 2 and activation.shape[0] > 1:
452 # 计算批次内激活模式的一致性
453 activation_flat = activation.view(activation.shape[0], -1)
454
455 # 计算样本间的相关性
456 if activation_flat.shape[1] > 1:
457 try:
458 correlation_matrix = torch.corrcoef(activation_flat)
459 # 对角线外的相关系数
460 mask = ~torch.eye(correlation_matrix.shape[0], dtype=torch.bool)
461 off_diagonal = correlation_matrix[mask]
462
463 if len(off_diagonal) > 0:
464 consistency = torch.mean(torch.abs(off_diagonal))
465 conflict_scores.append(1.0 - consistency.item())
466 except:
467 continue
468
469 if conflict_scores:
470 avg_conflict = np.mean(conflict_scores)
471 return avg_conflict > 0.7 # 冲突阈值
472
473 return False

947 def _enhance_bottleneck_signatures_with_net2net(self,
948 bottleneck_signatures: Dict,
949 net2net_analyses: Dict) -> Dict:
950 """使用Net2Net分析结果增强瓶颈签名"""
951
952 enhanced_signatures = copy.deepcopy(bottleneck_signatures)
953
954 for layer_name, signature in enhanced_signatures.items():
955 if layer_name in net2net_analyses:
956 net2net_analysis = net2net_analyses[layer_name]
957
958 # 获取变异预测信息
959 mutation_prediction = net2net_analysis.get('mutation_prediction', {})
960 improvement_potential = mutation_prediction.get('improvement_potential', 0.0)
961 risk_assessment = mutation_prediction.get('risk_assessment', {})
962
963 # 根据Net2Net分析调整瓶颈严重程度
964 original_severity = signature.severity
965 net2net_adjustment = improvement_potential * 0.5 # Net2Net改进潜力的权重
966
967 # 综合严重程度 = 原始严重程度 + Net2Net改进潜力 - 风险调整
968 adjusted_severity = original_severity + net2net_adjustment - risk_assessment.get('overall_risk', 0) * 0.2
969 signature.severity = max(0.0, min(1.0, adjusted_severity))
970
971 # 添加Net2Net特定信息
972 signature.net2net_improvement_potential = improvement_potential
973 signature.net2net_risk = risk_assessment.get('overall_risk', 0.0)
974 signature.net2net_recommended_strategy = net2net_analysis.get('recommendation', {}).get('recommended_strategy')
975
976 logger.debug(f"层{layer_name}: 原始严重程度={original_severity:.3f} -> "
977 f"调整后严重程度={signature.severity:.3f} "
978 f"(Net2Net潜力={improvement_potential:.3f})")
979
980 return enhanced_signatures
