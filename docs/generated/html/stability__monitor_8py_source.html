29class LossOscillationDetector:
30 """
31 损失震荡检测器
32
33 专门检测架构变化引起的损失剧烈波动
34 """
35
36 def __init__(self, window_size: int = 10, oscillation_threshold: float = 0.1,
37 severity_levels: Dict[str, float] = None):
38 """
39 Args:
40 window_size: 滑动窗口大小
41 oscillation_threshold: 震荡阈值
42 severity_levels: 严重程度级别
43 """
44 self.window_size = window_size
45 self.oscillation_threshold = oscillation_threshold
46 self.severity_levels = severity_levels or {
47 'mild': 0.05,
48 'moderate': 0.1,
49 'severe': 0.2,
50 'critical': 0.5
51 }
52
53 # 历史数据
54 self.loss_history = deque(maxlen=window_size * 3) # 保留更长历史
55 self.oscillation_events = []
56
57 logger.debug(f"🌊 Loss Oscillation Detector initialized: "
58 f"window={window_size}, threshold={oscillation_threshold}")
59
60 def update(self, loss: float, epoch: int, phase: str = "unknown") -> Dict[str, Any]:
61 """
62 更新损失值并检测震荡
63
64 Args:
65 loss: 当前损失值
66 epoch: 当前epoch
67 phase: 当前训练阶段
68
69 Returns:
70 检测结果
71 """
72 self.loss_history.append({
73 'loss': loss,
74 'epoch': epoch,
75 'phase': phase
76 })
77
78 # 需要足够的历史数据才能检测
79 if len(self.loss_history) < self.window_size:
80 return {'status': 'insufficient_data', 'oscillation_detected': False}
81
82 # 分析震荡
83 oscillation_metrics = self._analyze_oscillation()
84
85 # 检测是否有震荡
86 is_oscillating = oscillation_metrics['variance'] > self.oscillation_threshold
87
88 if is_oscillating:
89 severity = self._classify_severity(oscillation_metrics['variance'])
90
91 event = {
92 'epoch': epoch,
93 'phase': phase,
94 'severity': severity,
95 'metrics': oscillation_metrics
96 }
97 self.oscillation_events.append(event)
98
99 logger.warning(f"🌊 Loss oscillation detected at epoch {epoch}: "
100 f"{severity} (variance: {oscillation_metrics['variance']:.4f})")
101
102 return {
103 'status': 'analyzed',
104 'oscillation_detected': is_oscillating,
105 'severity': self._classify_severity(oscillation_metrics['variance']) if is_oscillating else 'none',
106 'metrics': oscillation_metrics
107 }
108
109 def _analyze_oscillation(self) -> Dict[str, float]:
110 """分析损失震荡特征"""
111 recent_losses = [item['loss'] for item in list(self.loss_history)[-self.window_size:]]
112
113 # 基本统计
114 mean_loss = np.mean(recent_losses)
115 variance = np.var(recent_losses)
116 std_dev = np.std(recent_losses)
117
118 # 相对变异系数
119 cv = std_dev / mean_loss if mean_loss > 0 else 0
120
121 # 趋势分析
122 trend = np.polyfit(range(len(recent_losses)), recent_losses, 1)[0]
123
124 # 震荡频率分析
125 direction_changes = 0
126 for i in range(1, len(recent_losses) - 1):
127 prev_diff = recent_losses[i] - recent_losses[i-1]
128 curr_diff = recent_losses[i+1] - recent_losses[i]
129 if prev_diff * curr_diff < 0: # 方向改变
130 direction_changes += 1
131
132 oscillation_frequency = direction_changes / (len(recent_losses) - 2) if len(recent_losses) > 2 else 0
133
134 return {
135 'mean': mean_loss,
136 'variance': variance,
137 'std_dev': std_dev,
138 'cv': cv,
139 'trend': trend,
140 'oscillation_frequency': oscillation_frequency
141 }
142
143 def _classify_severity(self, variance: float) -> str:
144 """分类震荡严重程度"""
145 for severity, threshold in sorted(self.severity_levels.items(),
146 key=lambda x: x[1], reverse=True):
147 if variance >= threshold:
148 return severity
149 return 'mild'
150
151 def get_stability_report(self) -> Dict[str, Any]:
152 """获取稳定性报告"""
153 if not self.loss_history:
154 return {'status': 'no_data'}
155
156 total_events = len(self.oscillation_events)
157 severity_counts = {}
158
159 for event in self.oscillation_events:
160 severity = event['severity']
161 severity_counts[severity] = severity_counts.get(severity, 0) + 1
162
163 # 计算总体稳定性评分 (0-1, 1为最稳定)
164 if total_events == 0:
165 stability_score = 1.0
166 else:
167 # 根据事件严重程度加权计算
168 weight_sum = sum(len(self.severity_levels) - i for i, _ in enumerate(self.severity_levels.keys()))
169 weighted_events = sum(
170 severity_counts.get(sev, 0) * (len(self.severity_levels) - i)
171 for i, sev in enumerate(self.severity_levels.keys())
172 )
173 stability_score = max(0, 1 - weighted_events / (len(self.loss_history) * weight_sum))
174
175 return {
176 'total_oscillation_events': total_events,
177 'severity_breakdown': severity_counts,
178 'stability_score': stability_score,
179 'recent_metrics': self._analyze_oscillation() if len(self.loss_history) >= self.window_size else None
180 }

182class ConvergenceAnalyzer:
183 """
184 收敛分析器
185
186 分析训练收敛趋势和速度
187 """
188
189 def __init__(self, patience: int = 15, min_delta: float = 0.001,
190 lookback_window: int = 20):
191 """
192 Args:
193 patience: 容忍的停滞epoch数
194 min_delta: 最小改进阈值
195 lookback_window: 回看窗口大小
196 """
197 self.patience = patience
198 self.min_delta = min_delta
199 self.lookback_window = lookback_window
200
201 # 历史记录
202 self.metric_history = deque(maxlen=lookback_window * 2)
203 self.plateau_count = 0
204 self.best_metric = -float('inf')
205 self.epochs_since_improvement = 0
206
207 logger.debug(f"📈 Convergence Analyzer initialized: "
208 f"patience={patience}, min_delta={min_delta}")
209
210 def update(self, metric: float, epoch: int, is_higher_better: bool = True) -> Dict[str, Any]:
211 """
212 更新指标并分析收敛
213
214 Args:
215 metric: 监控指标（如准确率、损失等）
216 epoch: 当前epoch
217 is_higher_better: 指标是否越高越好
218
219 Returns:
220 分析结果
221 """
222 self.metric_history.append({
223 'metric': metric,
224 'epoch': epoch
225 })
226
227 # 判断是否有改进
228 if is_higher_better:
229 improved = metric > self.best_metric + self.min_delta
230 else:
231 improved = metric < self.best_metric - self.min_delta
232
233 if improved:
234 self.best_metric = metric
235 self.epochs_since_improvement = 0
236 else:
237 self.epochs_since_improvement += 1
238
239 # 分析收敛状态
240 convergence_analysis = self._analyze_convergence(is_higher_better)
241
242 # 判断是否应该早停
243 should_early_stop = self.epochs_since_improvement >= self.patience
244
245 if should_early_stop:
246 logger.info(f"📉 Convergence plateau detected: "
247 f"{self.epochs_since_improvement} epochs without improvement")
248
249 return {
250 'improved': improved,
251 'epochs_since_improvement': self.epochs_since_improvement,
252 'should_early_stop': should_early_stop,
253 'convergence_analysis': convergence_analysis,
254 'best_metric': self.best_metric
255 }
256
257 def _analyze_convergence(self, is_higher_better: bool) -> Dict[str, Any]:
258 """分析收敛特征"""
259 if len(self.metric_history) < 5:
260 return {'status': 'insufficient_data'}
261
262 metrics = [item['metric'] for item in self.metric_history]
263 epochs = [item['epoch'] for item in self.metric_history]
264
265 # 计算收敛速度
266 if len(metrics) >= 2:
267 recent_change = metrics[-1] - metrics[-2]
268 if len(metrics) >= self.lookback_window:
269 long_term_change = metrics[-1] - metrics[-self.lookback_window]
270 convergence_rate = long_term_change / self.lookback_window
271 else:
272 convergence_rate = recent_change
273 else:
274 recent_change = 0
275 convergence_rate = 0
276
277 # 趋势分析
278 if len(metrics) >= 5:
279 trend_slope = np.polyfit(range(len(metrics)), metrics, 1)[0]
280 else:
281 trend_slope = 0
282
283 # 方差分析
284 recent_variance = np.var(metrics[-min(10, len(metrics)):])
285
286 # 收敛状态分类
287 if abs(convergence_rate) < self.min_delta / 2:
288 convergence_status = 'converged'
289 elif (is_higher_better and convergence_rate > 0) or (not is_higher_better and convergence_rate < 0):
290 convergence_status = 'improving'
291 else:
292 convergence_status = 'degrading'
293
294 return {
295 'status': 'analyzed',
296 'convergence_status': convergence_status,
297 'convergence_rate': convergence_rate,
298 'trend_slope': trend_slope,
299 'recent_variance': recent_variance,
300 'recent_change': recent_change
301 }
302
303 def get_convergence_report(self) -> Dict[str, Any]:
304 """获取收敛报告"""
305 if not self.metric_history:
306 return {'status': 'no_data'}
307
308 metrics = [item['metric'] for item in self.metric_history]
309
310 return {
311 'total_epochs': len(metrics),
312 'best_metric': self.best_metric,
313 'current_metric': metrics[-1],
314 'epochs_since_improvement': self.epochs_since_improvement,
315 'improvement_percentage': ((metrics[-1] - metrics[0]) / abs(metrics[0]) * 100) if metrics[0] != 0 else 0,
316 'average_metric': np.mean(metrics),
317 'metric_std': np.std(metrics)
318 }

320class PerformanceDegradationDetector:
321 """
322 性能退化检测器
323
324 检测性能下降并提供预警
325 """
326
327 def __init__(self, degradation_threshold: float = 0.05,
328 alert_window: int = 5, recovery_window: int = 3):
329 """
330 Args:
331 degradation_threshold: 性能退化阈值
332 alert_window: 预警窗口大小
333 recovery_window: 恢复检测窗口
334 """
335 self.degradation_threshold = degradation_threshold
336 self.alert_window = alert_window
337 self.recovery_window = recovery_window
338
339 # 状态跟踪
340 self.baseline_performance = None
341 self.performance_history = deque(maxlen=alert_window * 3)
342 self.degradation_alerts = []
343 self.in_degradation = False
344
345 logger.debug(f"⚠️ Performance Degradation Detector initialized: "
346 f"threshold={degradation_threshold}")
347
348 def update(self, performance: float, epoch: int,
349 set_baseline: bool = False) -> Dict[str, Any]:
350 """
351 更新性能并检测退化
352
353 Args:
354 performance: 当前性能指标
355 epoch: 当前epoch
356 set_baseline: 是否设置为基线性能
357
358 Returns:
359 检测结果
360 """
361 if set_baseline or self.baseline_performance is None:
362 self.baseline_performance = performance
363 logger.info(f"📊 Performance baseline set: {performance:.4f}")
364
365 self.performance_history.append({
366 'performance': performance,
367 'epoch': epoch
368 })
369
370 # 检测退化
371 degradation_analysis = self._detect_degradation(performance, epoch)
372
373 # 检测恢复
374 recovery_analysis = self._detect_recovery(performance, epoch)
375
376 return {
377 'baseline_performance': self.baseline_performance,
378 'current_performance': performance,
379 'degradation_analysis': degradation_analysis,
380 'recovery_analysis': recovery_analysis,
381 'in_degradation': self.in_degradation
382 }
383
384 def _detect_degradation(self, performance: float, epoch: int) -> Dict[str, Any]:
385 """检测性能退化"""
386 if self.baseline_performance is None:
387 return {'status': 'no_baseline'}
388
389 # 计算相对退化（避免除零错误）
390 if abs(self.baseline_performance) < 1e-8: # baseline为0或非常小
391 relative_degradation = 0.0 if performance >= 0 else -1.0
392 else:
393 relative_degradation = (self.baseline_performance - performance) / abs(self.baseline_performance)
394
395 # 检查是否超过阈值
396 is_degraded = relative_degradation > self.degradation_threshold
397
398 if is_degraded and not self.in_degradation:
399 # 新的退化事件
400 alert = {
401 'epoch': epoch,
402 'baseline': self.baseline_performance,
403 'current': performance,
404 'degradation': relative_degradation,
405 'severity': self._classify_degradation_severity(relative_degradation)
406 }
407
408 self.degradation_alerts.append(alert)
409 self.in_degradation = True
410
411 logger.warning(f"⚠️ Performance degradation detected at epoch {epoch}: "
412 f"{relative_degradation:.2%} below baseline "
413 f"(severity: {alert['severity']})")
414
415 return {
416 'is_degraded': is_degraded,
417 'relative_degradation': relative_degradation,
418 'absolute_degradation': self.baseline_performance - performance,
419 'severity': self._classify_degradation_severity(relative_degradation) if is_degraded else 'none'
420 }
421
422 def _detect_recovery(self, performance: float, epoch: int) -> Dict[str, Any]:
423 """检测性能恢复"""
424 if not self.in_degradation or self.baseline_performance is None:
425 return {'status': 'not_applicable'}
426
427 # 检查最近几个epoch的性能
428 if len(self.performance_history) >= self.recovery_window:
429 recent_performances = [
430 item['performance']
431 for item in list(self.performance_history)[-self.recovery_window:]
432 ]
433
434 # 如果最近的性能都接近或超过基线
435 recovery_threshold = self.baseline_performance * (1 - self.degradation_threshold / 2)
436 all_recovered = all(p >= recovery_threshold for p in recent_performances)
437
438 if all_recovered:
439 self.in_degradation = False
440 logger.info(f"✅ Performance recovered at epoch {epoch}")
441 return {'status': 'recovered', 'recovery_epoch': epoch}
442
443 return {'status': 'still_degraded'}
444
445 def _classify_degradation_severity(self, degradation: float) -> str:
446 """分类退化严重程度"""
447 if degradation < 0.02:
448 return 'minor'
449 elif degradation < 0.05:
450 return 'moderate'
451 elif degradation < 0.1:
452 return 'major'
453 else:
454 return 'critical'
455
456 def get_degradation_report(self) -> Dict[str, Any]:
457 """获取退化报告"""
458 return {
459 'total_degradation_events': len(self.degradation_alerts),
460 'current_status': 'degraded' if self.in_degradation else 'healthy',
461 'baseline_performance': self.baseline_performance,
462 'recent_alerts': self.degradation_alerts[-5:] if self.degradation_alerts else []
463 }

465class StabilityMonitor:
466 """
467 综合稳定性监控器

469 整合各种监控组件，提供统一的稳定性监控接口

472 def __init__(self, config: Dict[str, Any] = None):
473 """
474 Args:
475 config: 监控配置
476 """
477 config = config or {}
478
479 # 初始化各个组件
480 self.oscillation_detector = LossOscillationDetector(
481 window_size=config.get('oscillation_window', 10),
482 oscillation_threshold=config.get('oscillation_threshold', 0.1)
483 )
484
485 self.convergence_analyzer = ConvergenceAnalyzer(
486 patience=config.get('convergence_patience', 15),
487 min_delta=config.get('convergence_min_delta', 0.001)
488 )
489
490 self.degradation_detector = PerformanceDegradationDetector(
491 degradation_threshold=config.get('degradation_threshold', 0.05),
492 alert_window=config.get('degradation_window', 5)
493 )
494
495 # 综合状态
496 self.training_health_score = 1.0
497 self.health_history = []
498
499 logger.info("🏥 Comprehensive Stability Monitor initialized")
500
501 def update(self, metrics: Dict[str, float], epoch: int,
502 phase: str = "unknown") -> Dict[str, Any]:
503 """
504 更新所有监控指标
505
506 Args:
507 metrics: 指标字典，需包含 'loss', 'accuracy' 等
508 epoch: 当前epoch
509 phase: 当前训练阶段
510
511 Returns:
512 综合监控结果
513 """
514 results = {
515 'epoch': epoch,
516 'phase': phase,
517 'metrics': metrics
518 }
519
520 # 更新各个监控组件
521 if 'loss' in metrics:
522 oscillation_result = self.oscillation_detector.update(
523 metrics['loss'], epoch, phase
524 )
525 results['oscillation'] = oscillation_result
526
527 if 'accuracy' in metrics:
528 convergence_result = self.convergence_analyzer.update(
529 metrics['accuracy'], epoch, is_higher_better=True
530 )
531 results['convergence'] = convergence_result
532
533 degradation_result = self.degradation_detector.update(
534 metrics['accuracy'], epoch
535 )
536 results['degradation'] = degradation_result
537
538 # 计算综合健康度评分
539 health_score = self._calculate_health_score(results)
540 self.training_health_score = health_score
541
542 self.health_history.append({
543 'epoch': epoch,
544 'health_score': health_score,
545 'phase': phase
546 })
547
548 results['health_score'] = health_score
549 results['health_status'] = self._classify_health_status(health_score)
550
551 # 生成建议
552 recommendations = self._generate_recommendations(results)
553 results['recommendations'] = recommendations
554
555 return results
556
557 def _calculate_health_score(self, results: Dict[str, Any]) -> float:
558 """计算综合健康度评分"""
559 score = 1.0
560
561 # 震荡惩罚
562 oscillation = results.get('oscillation', {})
563 if oscillation.get('oscillation_detected', False):
564 severity_penalty = {
565 'mild': 0.05,
566 'moderate': 0.15,
567 'severe': 0.3,
568 'critical': 0.5
569 }
570 severity = oscillation.get('severity', 'mild')
571 score -= severity_penalty.get(severity, 0.05)
572
573 # 收敛惩罚
574 convergence = results.get('convergence', {})
575 if convergence.get('convergence_analysis', {}).get('convergence_status') == 'degrading':
576 score -= 0.2
577
578 # 退化惩罚
579 degradation = results.get('degradation', {})
580 if degradation.get('degradation_analysis', {}).get('is_degraded', False):
581 severity_penalty = {
582 'minor': 0.1,
583 'moderate': 0.2,
584 'major': 0.4,
585 'critical': 0.6
586 }
587 severity = degradation.get('degradation_analysis', {}).get('severity', 'minor')
588 score -= severity_penalty.get(severity, 0.1)
589
590 return max(0.0, min(1.0, score))

592 def _classify_health_status(self, health_score: float) -> str:

597 return 'good'

617 recommendations.append("Monitor loss oscillations closely")

622 recommendations.append("Consider early stopping due to convergence plateau")
623
624 convergence_status = convergence.get('convergence_analysis', {}).get('convergence_status')
625 if convergence_status == 'degrading':
626 recommendations.append("Performance is degrading - consider adjusting hyperparameters")
627
628 # 基于退化的建议
629 degradation = results.get('degradation', {})
630 if degradation.get('in_degradation', False):
631 recommendations.append("Performance degradation detected - consider rollback or adjustment")
632
633 # 基于健康度的建议
634 if self.training_health_score < 0.5:
635 recommendations.append("Training health is poor - comprehensive review recommended")
