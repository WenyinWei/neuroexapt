26def demo_cuda_softmax_sum():
27 """Demo CUDA SoftmaxSum optimization"""
28 print("🔥 CUDA SoftmaxSum Optimization Demo")
29 print("=" * 50)
30
31 try:
32 from neuroexapt.cuda_ops import SoftmaxSumFunction, CUDA_AVAILABLE
33
34 # Test shapes typical for neural architecture search
35 N, B, C, H, W = 8, 4, 32, 32, 32 # 8 operations, batch 4, 32x32x32 features
36 device = "cuda" if torch.cuda.is_available() else "cpu"
37
38 print(f"Device: {device}")
39 print(f"CUDA available: {CUDA_AVAILABLE}")
40 print(f"Input shape: N={N}, B={B}, C={C}, H={H}, W={W}")
41
42 # Generate test data
43 x = torch.randn(N, B, C, H, W, device=device, requires_grad=True)
44 logits = torch.randn(N, device=device, requires_grad=True)
45
46 # Reference PyTorch implementation
47 def pytorch_softmax_sum(x, logits):
48 weights = torch.softmax(logits, 0)
49 return (x * weights.view(-1, 1, 1, 1, 1)).sum(dim=0)
50
51 # Benchmark forward pass
52 warmup = 5
53 runs = 20
54
55 # PyTorch baseline
56 for _ in range(warmup):
57 _ = pytorch_softmax_sum(x, logits)
58
59 torch.cuda.synchronize() if device == "cuda" else None
60 start = time.perf_counter()
61 for _ in range(runs):
62 _ = pytorch_softmax_sum(x, logits)
63 torch.cuda.synchronize() if device == "cuda" else None
64 pytorch_time = (time.perf_counter() - start) / runs
65
66 # CUDA implementation
67 for _ in range(warmup):
68 _ = SoftmaxSumFunction.apply(x, logits)
69
70 torch.cuda.synchronize() if device == "cuda" else None
71 start = time.perf_counter()
72 for _ in range(runs):
73 _ = SoftmaxSumFunction.apply(x, logits)
74 torch.cuda.synchronize() if device == "cuda" else None
75 cuda_time = (time.perf_counter() - start) / runs
76
77 # Results
78 speedup = pytorch_time / cuda_time
79 print(f"PyTorch time: {pytorch_time*1000:.2f}ms")
80 print(f"CUDA time: {cuda_time*1000:.2f}ms")
81 print(f"Speedup: {speedup:.2f}x")
82
83 # Correctness check
84 ref_out = pytorch_softmax_sum(x, logits)
85 cuda_out = SoftmaxSumFunction.apply(x, logits)
86 assert isinstance(cuda_out, torch.Tensor)
87 max_diff = torch.max(torch.abs(ref_out - cuda_out)).item()

95def demo_triton_sepconv():
96 """Demo Triton separable convolution optimization"""
97 print("⚡ Triton SepConv Optimization Demo")
98 print("=" * 50)
99
100 try:
101 from neuroexapt.kernels import sepconv_forward_generic, TRITON_AVAILABLE
102
103 print(f"Triton available: {TRITON_AVAILABLE}")
104
105 # Test parameters
106 B, C, H, W = 4, 32, 32, 32
107 kernel_size = 3
108 stride = 1
109 dilation = 1
110 device = "cuda" if torch.cuda.is_available() else "cpu"
111
112 print(f"Device: {device}")
113 print(f"Input shape: B={B}, C={C}, H={H}, W={W}")
114
115 # Generate test data
116 x = torch.randn(B, C, H, W, device=device)
117 dw_weight = torch.randn(C, 1, kernel_size, kernel_size, device=device)
118 pw_weight = torch.randn(C * 2, C, 1, 1, device=device)
119 bias = torch.randn(C * 2, device=device)
120
121 # Reference PyTorch implementation
122 def pytorch_sepconv(x, dw_weight, pw_weight, bias):
123 pad = ((kernel_size - 1) * dilation) // 2
124 y = torch.nn.functional.conv2d(x, dw_weight, None, stride=stride, padding=pad, dilation=dilation, groups=C)
125 y = torch.nn.functional.conv2d(y, pw_weight, bias, stride=1, padding=0)
126 return y
127
128 # Benchmark
129 warmup = 5
130 runs = 20
131
132 # PyTorch baseline
133 for _ in range(warmup):
134 _ = pytorch_sepconv(x, dw_weight, pw_weight, bias)
135
136 torch.cuda.synchronize() if device == "cuda" else None
137 start = time.perf_counter()
138 for _ in range(runs):
139 _ = pytorch_sepconv(x, dw_weight, pw_weight, bias)
140 torch.cuda.synchronize() if device == "cuda" else None
141 pytorch_time = (time.perf_counter() - start) / runs
142
143 # Triton implementation
144 for _ in range(warmup):
145 _ = sepconv_forward_generic(x, dw_weight, pw_weight, bias, kernel_size=kernel_size, stride=stride, dilation=dilation)
146
147 torch.cuda.synchronize() if device == "cuda" else None
148 start = time.perf_counter()
149 for _ in range(runs):
150 _ = sepconv_forward_generic(x, dw_weight, pw_weight, bias, kernel_size=kernel_size, stride=stride, dilation=dilation)
151 torch.cuda.synchronize() if device == "cuda" else None
152 triton_time = (time.perf_counter() - start) / runs
153
154 # Results
155 speedup = pytorch_time / triton_time
156 print(f"PyTorch time: {pytorch_time*1000:.2f}ms")
157 print(f"Triton time: {triton_time*1000:.2f}ms")
158 print(f"Speedup: {speedup:.2f}x")
159
160 # Correctness check
161 ref_out = pytorch_sepconv(x, dw_weight, pw_weight, bias)
162 triton_out = sepconv_forward_generic(x, dw_weight, pw_weight, bias, kernel_size=kernel_size, stride=stride, dilation=dilation)
