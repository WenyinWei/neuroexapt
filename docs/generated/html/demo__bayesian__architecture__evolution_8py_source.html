42class BayesianEvolvableCNN(nn.Module):
43 """
44 为贝叶斯架构演化专门设计的CNN。
45
46 这个网络会跟踪自己的架构演化历史，并与贝叶斯顾问集成。
47 """
48
49 def __init__(self, num_classes=10, initial_depth=8, initial_channels=32):
50 super(BayesianEvolvableCNN, self).__init__()
51
52 # 记录架构演化历史
53 self.architecture_history = []
54 self.performance_history = []
55 self.bayesian_insights = []
56
57 # 构建初始架构
58 self.features = nn.Sequential(
59 # 第一层：输入处理
60 nn.Conv2d(3, initial_channels, kernel_size=3, padding=1),
61 nn.BatchNorm2d(initial_channels),
62 nn.ReLU(inplace=True),
63
64 # 第二层：特征提取
65 nn.Conv2d(initial_channels, initial_channels * 2, kernel_size=3, padding=1),
66 nn.BatchNorm2d(initial_channels * 2),
67 nn.ReLU(inplace=True),
68 nn.MaxPool2d(kernel_size=2, stride=2),
69
70 # 第三层：深度特征
71 nn.Conv2d(initial_channels * 2, initial_channels * 4, kernel_size=3, padding=1),
72 nn.BatchNorm2d(initial_channels * 4),
73 nn.ReLU(inplace=True),
74
75 # 第四层：高级特征
76 nn.Conv2d(initial_channels * 4, initial_channels * 8, kernel_size=3, padding=1),
77 nn.BatchNorm2d(initial_channels * 8),
78 nn.ReLU(inplace=True),
79 nn.MaxPool2d(kernel_size=2, stride=2),
80
81 # 全局平均池化
82 nn.AdaptiveAvgPool2d((1, 1))
83 )
84
85 # 分类器
86 self.classifier = nn.Sequential(
87 nn.Dropout(0.5),
88 nn.Linear(initial_channels * 8, initial_channels * 4),
89 nn.ReLU(inplace=True),
90 nn.Dropout(0.3),
91 nn.Linear(initial_channels * 4, num_classes)
92 )
93
94 # 初始化权重
95 self._initialize_weights()
96
97 # 记录初始状态
98 self._record_architecture_state("initial")
99
100 def forward(self, x):
101 x = self.features(x)
102 x = x.view(x.size(0), -1)
103 x = self.classifier(x)
104 return x

106 def _initialize_weights(self):
107 """初始化网络权重"""
108 for m in self.modules():
109 if isinstance(m, nn.Conv2d):
110 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
111 if m.bias is not None:
112 nn.init.constant_(m.bias, 0)
113 elif isinstance(m, nn.BatchNorm2d):
114 nn.init.constant_(m.weight, 1)
115 nn.init.constant_(m.bias, 0)
116 elif isinstance(m, nn.Linear):
117 nn.init.normal_(m.weight, 0, 0.01)
118 nn.init.constant_(m.bias, 0)
119
120 def _record_architecture_state(self, event_type: str):
121 """记录架构状态"""
122
123 # 计算架构统计信息
124 total_params = sum(p.numel() for p in self.parameters())
125 conv_layers = sum(1 for m in self.modules() if isinstance(m, nn.Conv2d))
126 linear_layers = sum(1 for m in self.modules() if isinstance(m, nn.Linear))
127
128 # 计算平均通道数
129 conv_channels = [m.out_channels for m in self.modules() if isinstance(m, nn.Conv2d)]
130 avg_channels = np.mean(conv_channels) if conv_channels else 0
131
132 state = {
133 'event_type': event_type,
134 'timestamp': time.time(),
135 'total_params': total_params,
136 'conv_layers': conv_layers,
137 'linear_layers': linear_layers,
138 'avg_channels': avg_channels,
139 'max_channels': max(conv_channels) if conv_channels else 0,
140 'architecture_depth': conv_layers + linear_layers
141 }
142
143 self.architecture_history.append(state)
144
145 print(f"📊 Architecture state recorded ({event_type}):")
146 print(f" Total parameters: {total_params:,}")
147 print(f" Architecture depth: {state['architecture_depth']}")
148 print(f" Average channels: {avg_channels:.1f}")
149
150 def update_performance_record(self, performance: Dict[str, float]):
151 """更新性能记录"""
152 self.performance_history.append({
153 'timestamp': time.time(),
154 'performance': performance
155 })

157 def get_architecture_summary(self) -> Dict[str, any]:
158 """获取架构摘要信息"""
159
160 if not self.architecture_history:
161 return {}
162
163 current_state = self.architecture_history[-1]
164 initial_state = self.architecture_history[0]
165
166 return {
167 'evolution_steps': len(self.architecture_history) - 1,
168 'parameter_growth': current_state['total_params'] - initial_state['total_params'],
169 'depth_growth': current_state['architecture_depth'] - initial_state['architecture_depth'],
170 'channel_growth': current_state['avg_channels'] - initial_state['avg_channels'],
171 'current_state': current_state,
172 'bayesian_insights': self.bayesian_insights[-5:] if self.bayesian_insights else []
173 }

176def load_demo_data(batch_size=32, num_samples=2000):
177 """加载演示数据"""
178
179 transform = transforms.Compose([
180 transforms.ToTensor(),
181 transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2023, 0.1994, 0.2010)),
182 ])
183
184 trainset = torchvision.datasets.CIFAR10(root='./data', train=True, download=True, transform=transform)
185 testset = torchvision.datasets.CIFAR10(root='./data', train=False, download=True, transform=transform)
186
187 # 创建小数据集用于快速演示
188 train_indices = torch.randperm(len(trainset))[:num_samples].tolist()
189 test_indices = torch.randperm(len(testset))[:num_samples//4].tolist()
190
191 train_subset = torch.utils.data.Subset(trainset, train_indices)
192 test_subset = torch.utils.data.Subset(testset, test_indices)
193
194 trainloader = DataLoader(train_subset, batch_size=batch_size, shuffle=True)
195 testloader = DataLoader(test_subset, batch_size=batch_size, shuffle=False)
196
197 return trainloader, testloader

200def evaluate_model(model, testloader, device):
201 """评估模型性能"""
202 model.eval()
203 correct = 0
204 total = 0
205 total_loss = 0
206 criterion = nn.CrossEntropyLoss()
207
208 with torch.no_grad():
209 for data, targets in testloader:
210 data, targets = data.to(device), targets.to(device)
211 outputs = model(data)
212 loss = criterion(outputs, targets)
213 total_loss += loss.item()
214
215 _, predicted = torch.max(outputs.data, 1)
216 total += targets.size(0)
217 correct += (predicted == targets).sum().item()
218
219 accuracy = 100 * correct / total if total > 0 else 0.0
220 avg_loss = total_loss / len(testloader) if len(testloader) > 0 else 0.0
221
222 model.train()
223 return accuracy, avg_loss
