120 def _record_architecture_state(self, event_type: str):
121 """è®°å½•æ¶æ„çŠ¶æ€"""

176def load_demo_data(batch_size=32, num_samples=2000):
177 """åŠ è½½æ¼”ç¤ºæ•°æ®"""

200def evaluate_model(model, testloader, device):
201 """è¯„ä¼°æ¨¡å‹æ€§èƒ½"""
202 model.eval()
203 correct = 0
204 total = 0
205 total_loss = 0
206 criterion = nn.CrossEntropyLoss()
207
208 with torch.no_grad():
209 for data, targets in testloader:
210 data, targets = data.to(device), targets.to(device)
211 outputs = model(data)
212 loss = criterion(outputs, targets)
213 total_loss += loss.item()
214
215 _, predicted = torch.max(outputs.data, 1)
216 total += targets.size(0)
217 correct += (predicted == targets).sum().item()
218
219 accuracy = 100 * correct / total if total > 0 else 0.0
220 avg_loss = total_loss / len(testloader) if len(testloader) > 0 else 0.0
221
222 model.train()
223 return accuracy, avg_loss
